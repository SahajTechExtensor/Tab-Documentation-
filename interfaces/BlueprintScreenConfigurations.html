    saveBlueprintAndChild(screenParameters: any): void {
        const blueprint: any = {
            Name: screenParameters.event.data.Name,
            Criteria: screenParameters.event.data.Criteria,
            Description: screenParameters.event.data.Descriptions,
            AppObjectId: screenParameters.event.data.appObjectId,
            TABD_BlueprintStatus: [],
            TABD_Transitions: [],
        };
        if (screenParameters.event.type == 'addStatus') {
            blueprint.TABD_BlueprintStatus = [
                {
                    Name: screenParameters.event.data?.statusName,
                    Color: screenParameters.event.data?.statusColor,
                    Type: screenParameters.event.data?.type,
                },
            ];
        } else if (screenParameters.event.type == 'saveTransition') {
            blueprint.TABD_Transitions = [
                {
                    Name: screenParameters.event.data.transitionName,
                    FromStatus: screenParameters.event.data.fromStatus,
                    ToStatus: screenParameters.event.data.toStatus,
                    Transition_Criteria: '',
                },
            ];
        }
        screenParameters?.serviceList?.insertService
            .insertByTable('Tabd_BluePrints', blueprint)
            ?.pipe(take(1))
            .subscribe({
                next: (res) => {
                    if (res.StatusCode == '200') {
                        screenParameters?.serviceList?.toastrService.showSuccess(
                            'Blueprint Saved',
                            'Success'
                        );
                        screenParameters.event = {
                            ...screenParameters.event.data,
                            statusId: '',
                            statusName: '',
                            statusColor: '',
                            type: '',
                            transitionName: '',
                            fromStatus: '',
                            toStatus: '',
                            blueprint: {
                                blueprint:
                                    screenParameters.event.data.blueprint
                                        .blueprint,
                                changed: res.Result,
                                isRefresh: true,
                                BluePrintId: res.Result,
                            },
                            Id: res.Result,
                        };
                        this.refreshForm(screenParameters);
                    }
                },
                error: (res) => {
                    screenParameters?.serviceList?.toastrService.showError(
                        res.Errors[0],
                        'Error'
                    );
                },
            });
    }

    /**
     * Handle `addStatus` and `saveTransition` formIo events
     *
     * @param {any} screenParameters  The screen parameters.
     * @returns {void} void
     */
    handleStatusAndTransitionEvent(screenParameters: any): void {
        if (screenParameters.event.type == 'addStatus') {
            const status: any = {
                Name: screenParameters.event.data?.statusName,
                Color: screenParameters.event.data?.statusColor,
                Type: screenParameters.event.data?.type,
                BluePrintId: screenParameters.event?.data?.Id,
            };
            if (screenParameters.event.data?.statusId) {
                status.Id = screenParameters.event.data?.statusId;
                this.updateStatus(status, screenParameters);
            } else {
                this.addStatus(status, screenParameters);
            }
        }
        if (screenParameters.event.type == 'saveTransition') {
            const transition: any = {
                Name: screenParameters.event.data.transitionName,
                FromStatus: screenParameters.event.data.fromStatus,
                ToStatus: screenParameters.event.data.toStatus,
                Transition_Criteria: '',
                BluePrintId: screenParameters.event.data.Id,
            };
            if (screenParameters.event.data.transitionId) {
                transition.Id = screenParameters.event.data?.transitionId;
                this.updateTransition(transition, screenParameters);
            } else {
                this.addTransition(transition, screenParameters);
            }
        }
    }

    /**
     * fetch DSQ filters data form AppObjectId and query builder
     *
     * @param {string} appObjectId App Object Id
     * @param {any} screenParameters The screen parameters.
     * @returns {void} void
     */
    getFiltersFromAppObject = (
        appObjectId: string,
        screenParameters: any
    ): void => {
        const apiData = {
            AppObjectName: 'TABMD_DataSourceQueries',
            DSQName: 'FK_TABMD_DataSourceQueries_TABMD_AppObject',
            Reqtokens: {
                Id: appObjectId,
            },
        };
        screenParameters.serviceList.getService
            .executeRAWDSQWithName(apiData)
            ?.pipe(take(1))
            .subscribe({
                next: (res) => {
                    if (res.Result) {
                        const dsqId = res.Result.find((dsq) =>
                            dsq.QueryName.startsWith('Default_')
                        )?.ID;
                        if (dsqId) this.getDSQFilters(dsqId, screenParameters);
                        else this.clearDSQFilters(screenParameters);
                    }
                },
            });
    };

    /**
     * fetch DSQ filters data and query builder
     *
     * @param {string} dsqId
     * @param {any} screenParameters
     * @returns {void} void
     */
    getDSQFilters = (dsqId: string, screenParameters: any) => {
        const reqTokens = { ID: dsqId };
        const apiData = {
            Reqtokens: reqTokens,
            AppObjectName: 'TABMD_DataSourceQueries_Filters',
            DSQName:
                'FK_TABMD_DataSourceQueries_Filters_TABMD_DataSourceQueries',
        };
        screenParameters.serviceList.getService
            .executeRAWDSQWithName(apiData)
            ?.pipe(take(1))
            .subscribe((response) => {
                let dsqFilterList;
                if (!!response.Result && response.Result.length > 0) {
                    dsqFilterList = response.Result;
                }
                screenParameters.event = {
                    ...screenParameters.event.data,
                    dataSourceQueryId: dsqId,
                    Criteria: {
                        filterData: dsqFilterList,
                        isRefresh: true,
                        dataSourceQueryId: dsqId,
                    },
                };
                this.refreshForm(screenParameters);
            });
    };

    /**
     * Clear query builder when appObjectId remove.
     *
     * @param {any} screenParameters
     * @returns {void} void
     */
    clearDSQFilters = (screenParameters: any): void => {
        screenParameters.event = {
            ...screenParameters.event.data,
            Criteria: {
                filterData: [],
                isRefresh: true,
                dataSourceQueryId: '',
            },
        };

        this.refreshForm(screenParameters);
    };

    /**
     * refresh screen
     *
     * @param {any} screenParameters
     * @returns {void} void
     */
    refreshForm = (screenParameters: any): void => {
        screenParameters?.formIOObject.refresh.emit({
            submission: {
                data: screenParameters.event,
            },
        });
    };
