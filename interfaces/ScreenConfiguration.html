import {
    IActionExtension,
    IActionExtensionRegistry,
    IScreenExtension,
} from '@techextensor/tab-interface';
import { FormioUtils } from '@formio/angular';

export interface ScreenConfiguration {
    css: string;
    script: string;
    layout: string;
    layoutType: any;
    screenActionList: any;
}

export class ScreenDesigner implements IScreenExtension {
    activeAction: IActionExtension;
    constructor() {}

    // Replace Screen Id to add Preview in screen designer
    // Old Screen Designer Id = 'bc7d4e1d-e976-49a1-be33-e53fc6943d7c'
    // changes reverted because of duplication issue - db6cf17d-3e68-4298-ac93-8beab694f375
    screenId = 'bc7d4e1d-e976-49a1-be33-e53fc6943d7c';

    onSubmit(event) {}

    onRender(screenParameters: any) {}

    onBeforeSubmit(screenParameters: any) {
        const screenConfiguration: ScreenConfiguration = {
            css: '',
            script: '',
            layout: '',
            layoutType: '',
            screenActionList: [],
        };
        if (screenParameters.event.data.stylesheetConfiguration) {
            screenConfiguration.css =
                screenParameters.event.data.stylesheetConfiguration;
            delete screenParameters.event.data.stylesheetConfiguration;
            screenParameters.event.data['ScreenConfiguration'] =
                screenConfiguration;
        }

        if (screenParameters.event.data.layout) {
            screenConfiguration.layout = screenParameters.event.data.layout;
            delete screenParameters.event.data.layout;
            screenParameters.event.data['ScreenConfiguration'] =
                screenConfiguration;
        } else {
            delete screenConfiguration.layout;
            screenParameters.event.data['ScreenConfiguration'] =
                screenConfiguration;
        }

        if (screenParameters?.event?.data?.layoutType) {
            screenConfiguration.layoutType =
                screenParameters.event.data.layoutType;
            delete screenParameters.event.data.layoutType;
            screenParameters.event.data['ScreenConfiguration'] =
                screenConfiguration;
        }
        if (screenParameters?.event?.data?.dataGrid?.length) {
            screenConfiguration.screenActionList =
                screenParameters?.event?.data?.dataGrid;
            delete screenParameters.event.data.dataGrid;
            screenParameters.event.data['ScreenConfiguration'] =
                screenConfiguration;
        }

        //will be used for scripts in future
        // if (screenParameters.event.data.script) {
        //     screenConfiguration.css = screenParameters.event.data.script;
        //     delete screenParameters.event.data.script;
        // }
        screenParameters?.event?.callback(null, screenParameters?.event);
    }

    onChange(screenParameters: any) {
        if (
            screenParameters?.event?.changed?.component?.type == 'select' &&
            screenParameters?.event?.changed?.component?.key ==
                'DataSourceQueryID'
        ) {
            FormioUtils.eachComponent(
                screenParameters.formIOObject.formio.components,
                function (component: any, path: any) {
                    if (
                        component.type == 'customFormbuilder' &&
                        component.key == 'Container'
                    ) {
                        component.component.customOptions.dataSetting.dataSourceQueryId =
                            screenParameters?.event?.changed?.value;
                        screenParameters?.formIOObject?.formio.triggerRedraw();
                    }
                }
            );
        }
    }

    onInvalid(event) {}

    onNextPage(event) {}

    onPrevPage(event) {}

    onErrorChange(event) {}

    onFormLoad(screenParameters: any) {}

    onReady(screenParameters: any) {}

    onAfterSubmit(screenParameters: any) {
        screenParameters.serviceList.appService
            .rebuildApp({})
            .subscribe((res) => {
                if (res?.StatusCode == '200') {
                    screenParameters.serviceList.appService.getSchema({});
                }
            });
    }

    proceedObjectBeforeRender(screenParameters: any) {
        if (screenParameters?.event?.ScreenConfiguration) {
            const screenConfiguration = JSON.parse(
                screenParameters.event.ScreenConfiguration
            );
            if (screenConfiguration) {
                if (screenConfiguration?.layout) {
                    screenParameters.event['layout'] =
                        screenConfiguration?.layout;
                }

                if (screenConfiguration?.css) {
                    screenParameters.event['stylesheetConfiguration'] =
                        screenConfiguration?.css;
                }

                if (screenConfiguration?.layoutType) {
                    screenParameters.event['layoutType'] =
                        screenConfiguration?.layoutType;
                }
                if (screenConfiguration?.screenActionList) {
                    screenParameters.event['dataGrid'] =
                        screenConfiguration?.screenActionList;
                }
            }
        }
        if (screenParameters?.event?.DataSourceQueryID) {
            FormioUtils.eachComponent(
                screenParameters?.formIOObject?.formio?.component?.components,
                function (component: any, path: any) {
                    if (
                        component.type == 'customFormbuilder' &&
                        component.key == 'Container'
                    ) {
                        component.customOptions.dataSetting.dataSourceQueryId =
                            screenParameters?.event?.DataSourceQueryID;
                    }
                }
            );
        }
    }

    updateCurrentActionInstance(actionData) {
        this.activeAction = null;
        var actions = IActionExtensionRegistry.GetImplementations();
        for (var x = 0; x < actions.length; x++) {
            const action = new actions[x]();
            if (actionData.$type.includes(action.actionType)) {
                this.activeAction = action;
            }
        }
    }

    onCustomEvent(event) {
        if (
            event.event.component.action !== 'submit' &&
            event.event.component.showValidations
        ) {
        }
        switch (event.event.component.action) {
            case 'saveState':
            case 'submit':
                event.preventDefault();
                event.stopPropagation();
                break;
            case 'event':
                break;
            case 'custom':
                break;
            case 'url':
                break;
            case 'reset':
                break;
            case 'delete':
                break;
            case 'oauth':
                break;
        }
    }
    onCustomHTMLTemplateEvent(screenParameters) {}
}
