export class AppModule {
    constructor(tabPluginService: TabPluginService, injector: Injector) {
        let plugins: any = [];
        tabPluginService.getAll(4)?.subscribe((res) => {
            if (res?.Result.length > 0) {
                plugins = res.Result;
            }
            plugins.forEach((plugin) => {
                tabPluginService.getFile(plugin.ID)?.subscribe(async (res) => {
                    if (res) {
                        var blob = new Blob([res], {
                            type: 'application/javascript',
                        });
                        const url = window.URL.createObjectURL(blob);
                        SystemJS.import(url).then(async (module: any) => {
                            module.register(injector);
                        });
                    }
                });
            });
        });
    }
}
export function setPageTitle(titleService: Title): () => Promise<any> {
    this.orgData = JSON.parse(localStorage.getItem('orgInfo'));
    if (this.orgData != null) {
        this.ProjectTitle = this.orgData.OrganizationName;
    }
    return () => {
        const title = this.ProjectTitle;
        titleService.setTitle(title);
        return Promise.resolve();
    };
}
