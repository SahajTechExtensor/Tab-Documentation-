import { Component, Input, OnInit, ViewChild } from '@angular/core';
import { Transition, TransitionRecord } from './models/transition.model';
import { Observable, Subject, debounceTime, forkJoin, map, of, take } from 'rxjs';
import { TabGetService } from '@techextensor/tab-service';
import {
    ComplexHierarchicalTree,
    DataBinding,
    Diagram,
    DiagramConstraints,
    DiagramTools,
    HierarchicalTree,
    LayoutModel,
} from '@syncfusion/ej2-angular-diagrams';
import { TabBlueprintComponent } from '../tab-blueprint/tab-blueprint.component';
import { Status } from '../tab-blueprint/models/status.model';

Diagram.Inject(DataBinding, HierarchicalTree, ComplexHierarchicalTree);

@Component({
    selector: 'app-tab-transition-preview',
    templateUrl: './tab-transition-preview.component.html',
    styleUrls: ['./tab-transition-preview.component.scss'],
})
export class TabTransitionPreviewComponent implements OnInit {
    // Form.io
    @Input() dataSettings: any = {};
    private debounceSubject = new Subject();

    transitionPreviewComponentValue: {
        BlueprintId: string,
        RecordId: string,
        SLASchemeID: string | null,
        SLAValue: number | null
    };

    // Diagram Configurations
    @ViewChild('blueprint', { static: false })
    blueprintComponent: TabBlueprintComponent;

    blueprintComponentValue = {
        BluePrintId: null,
        blueprint: '', // diagram json
        changed: '', // which field changed on event
        isRefresh: false,
    };
    // automatic layout to draw diagram
    layout: LayoutModel = {
        type: 'ComplexHierarchicalTree',
        enableRouting: true,
        enableAnimation: true,
        orientation: 'LeftToRight',
        verticalAlignment: 'Center',
        horizontalAlignment: 'Left',
        horizontalSpacing: 350,
        verticalSpacing: 150,
        connectionDirection: 'Orientation',
        margin: { top: 30, left: 30, right: 30, bottom: 50 },
    };
    diagramConstraints = DiagramConstraints.PanX;

    // Component details
    statuses: Status[] = [];
    transitions: Transition<string>[] = [];
    transitionRecords: TransitionRecord<Status>[] = [];
    slaDetails: any[] = [];

    constructor(private tabGetService: TabGetService) { }

    ngOnInit(): void {
        /**
         * Observing formIo value changes
         */
        this.debounceSubject.pipe(debounceTime(1000)).subscribe(() => {
            if (this.transitionPreviewComponentValue?.BlueprintId && this.transitionPreviewComponentValue?.RecordId) {
                forkJoin({
                    transitions: this.getTransitionsObservable(),
                    slaDetails: this.getSlaSchemeDetailsObservable(),
                }).pipe(take(1))
                    .subscribe(({ slaDetails, transitions }) => {
                        this.transitionRecords = transitions;
                        this.slaDetails = slaDetails;
                        if (this.dataSettings?.previewType == 'timeline' && this.blueprintComponent)
                            this.updateDiagram();
                    });
            }
        });
    }

    @Input()
    public set value(v: any) {
        if (v && v != null) {
            this.transitionPreviewComponentValue = v;
        }
        this.debounceSubject.next(1000);
    }

    public get value(): any {
        return this.transitionPreviewComponentValue;
    }

    /**
     * Retrieves the transaction data as an observable array.
     * If `BluePrintId` is not available in `formIoValue`, it returns an empty array.
     *
     * @returns An observable emitting an array of transaction data.
     */
    getTransitionsObservable(): Observable<any[]> {
        if (!this.transitionPreviewComponentValue?.BlueprintId && !this.transitionPreviewComponentValue?.RecordId) return of([]);

        const apiData = {
            AppObjectName: 'TABD_RecordInfo_Transitions',
            DSQName: 'FK_TABD_RecordInfo_Transitions_TABD_Transitions',
            Reqtokens: {
                BluePrintId: this.transitionPreviewComponentValue?.BlueprintId,
                RecordId: this.transitionPreviewComponentValue?.RecordId,
            },
        };
        return this.tabGetService
            .executeRAWDSQWithName(apiData)
            .pipe(map((res: any) => res?.Result));
    }

    /**
     * Retrieves the SlaDetails as an observable array.
     * If `SLASchemeID` is not available in `formIoValue`, it returns an empty array.
     *
     * @returns An observable emitting an array of SLA Details data.
     */
    getSlaSchemeDetailsObservable(): Observable<any[]> {
        if (!this.transitionPreviewComponentValue?.SLASchemeID) return of([]);

        const apiData = {
            AppObjectName: 'TABD_Blueprint_SLASchemeDetails',
            DSQName: 'SLASchemes_TABD_Blueprint_SLASchemeDetails',
            Reqtokens: {
                SLASchemeID: this.transitionPreviewComponentValue?.SLASchemeID,
            },
        };
        return this.tabGetService
            .executeRAWDSQWithName(apiData)
            .pipe(map((res: any) => res?.Result));
    }

    /**
     * Updates the diagram based on the changes in the formIoValue.
     * If the 'changed' property in formIoValue is not 'diagram', no refresh is performed.
     *
     * @returns {void}
     */
    updateDiagram = (): void => {
        const transitions = [];
        const nodeList: {
            [id: string]: Status;
        } = {};

        // To keep track of every status which Id we change in order to add new node
        const statusMap: {
            [id: string]: string;
        } = {};


        for (let i = 0; i < this.transitionRecords.length; i++) {
            const transition = this.transitionRecords[i].TransitionId;
            const transitionRecordId = this.transitionRecords[i].Id;

            if (statusMap[transition.FromStatus.Id]) {
                transition.FromStatus.Id = statusMap[transition.FromStatus.Id];
            }

            if (nodeList[transition.ToStatus.Id]) {
                statusMap[transition.ToStatus.Id] = transition.ToStatus.Id + transitionRecordId;
                transition.ToStatus.Id = transition.ToStatus.Id + transitionRecordId;
            }
            nodeList[transition.FromStatus.Id] = transition.FromStatus;
            nodeList[transition.ToStatus.Id] = transition.ToStatus;

            transitions.push({
                ...transition,
                FromStatus: transition.FromStatus.Id,
                ToStatus: transition.ToStatus.Id,
                Details: {
                    PerformedBy: this.transitionRecords[i].PerformedBy,
                    PerformedDate: this.transitionRecords[i].PerformedDate,
                    SLAHitTime: this.transitionRecords[i].SLAHitTime,
                    SLATime: (this.transitionPreviewComponentValue.SLAValue * this.slaDetails.find(sla => sla.TransitionID == transition.Id)?.SLATime)/100 
                },
            });

        }

        this.transitions = transitions;
        // Data
        this.blueprintComponent.statuses = Object.values(nodeList);
        this.blueprintComponent.transitions = transitions;
        this.blueprintComponent.layout = this.layout;
        this.blueprintComponent.diagramTool = DiagramTools.ZoomPan;
        this.blueprintComponent.diagramConstraints = this.diagramConstraints;
        this.blueprintComponent.dataSettingsBlueprint = this.dataSettings;
        this.blueprintComponentValue.changed = 'diagram';
        this.blueprintComponentValue.isRefresh = true;

        // Update diagram
        this.blueprintComponent.updateDiagram();
    };
}
