    /**
     * Handles various events triggered in the comment section.
     * @param event The event to handle.
     * @param comment Optional comment object associated with the event.
     * @param scroll Flag indicating whether to scroll to the top of the container.
     */
    onEvent(event: Event, comment?: any, scroll?: boolean) {
        // selection Events
        if (event === 'reply') this.replyToComment = comment;
        if (event === 'edit') {
            this.editComment = comment;
            this.richTextEditor.value = comment.Comment;
        }

        // CRUD
        if (event === 'delete') this.deleteComment(comment);
        if (event === 'addOrUpdate') {
            if (this.editComment) this.updateComment();
            else this.addComment();
        }

        // cancel Events
        if (event === 'cancelReply') this.replyToComment = null;
        if (event === 'cancelEdit') {
            // cancel reply
            this.replyToComment = null;

            // cancel edit
            this.editComment = null;
            this.richTextEditor.value = null;
        }

        // Scroll
        if (scroll) {
            document.getElementById('comment-scroll-helper').scrollIntoView({
                behavior: 'smooth',
                block: 'nearest',
                inline: 'nearest',
            });
            // this.container.nativeElement.scrollTop = 0;
        }
    }

    /**
     * Adds a new comment.
     */
    addComment() {
        if (this.richTextEditor.value && this.richTextEditor.value.length > 0) {
            const mentions = this.getMentionList(this.richTextEditor.value);
            const comment: any = {
                RecordId: this.componentValue?.RecordId,
                FromUser: this.userId,
                Comment: this.richTextEditor.value,
                ParentId: this.replyToComment?.Id,
                TABD_Comment_Tagged: mentions,
            };
            this.tabInsertService
                .insertByTable('TABD_Comments', comment)
                ?.pipe(take(1))
                .subscribe((res) => {
                    if (res.StatusCode == '200') {
                        this.toastrNotificationService.showSuccess(
                            'Comment Saved',
                            'Success'
                        );

                        // Loading latest on add since we don't have signalR
                        this.loadComments();

                        // FOR REFERENCE
                        // comment.Id = res.Result;
                        // comment.CreatedOn = new Date().toISOString();
                        // if (comment.ParentId) {
                        //     comment.Parent = this.comments.find(
                        //         (c: any) => c.Id == comment.ParentId
                        //     );
                        // }
                        // comment.FromUser = {
                        //     PersonId: {
                        //         ...this.user?.PersonData,
                        //         ProfileImageId: {
                        //             URL: this.userProfileUrl
                        //         }
                        //     }
                        // };
                        // this.comments = [comment, ...this.comments];

                        if (this.replyToComment) this.onEvent('cancelReply');
                        this.onEvent('cancelEdit');
                    } else {
                        this.toastrNotificationService.showError(
                            res.Errors[0],
                            'Error'
                        );
                    }
                });
        }
    }

    /**
     * Updates an existing comment.
     */
    updateComment() {
        this.editComment.Comment = this.richTextEditor.value;
        let queryFieldsArray: TabClasses.UpdateQueryParameter[] = [];
        queryFieldsArray.push({
            FieldName: 'Id',
            FieldValue: this.editComment.Id,
        });
        // To Add new mentions
        const mentions = this.getNewMentionList(
            this.editComment.Comment,
            this.editComment.Id
        );

        // Delete removed mentions
        const removedMentions = this.getRemovedMentionList(
            this.editComment.Comment,
            this.editComment.Id
        );
        const payload = {
            data: {
                Comment: this.editComment.Comment,
                TABD_Comment_Tagged: mentions,
            },
        };
        this.tabUpdateService
            .updateUsingDSQ(
                'B337572C-F522-40A2-A248-6392B0275B08', // Default_TABD_Comments
                payload,
                'TABD_Comments',
                queryFieldsArray
            )
            ?.pipe(take(1))
            .subscribe((res) => {
                if (res.StatusCode == '200') {
                    this.toastrNotificationService.showSuccess(
                        'Comment updated',
                        'Success'
                    );
                    this.deleteMentions(removedMentions);

                    this.loadComments();

                    // FOR REFERENCE
                    // this.comments = this.comments.reduce((acc, crr) => {
                    //     if (crr.Id == this.editComment.Id) {
                    //         this.editComment.FromUser = crr.FromUser;
                    //         this.editComment.CreatedOn = new Date().toISOString();
                    //         crr = this.editComment;
                    //     }
                    //     acc.push(crr);
                    //     return acc;
                    // }, []);
                    this.onEvent('cancelEdit');
                } else {
                    this.toastrNotificationService.showError(
                        res.Errors[0],
                        'Error'
                    );
                }
            });
    }

    /**
     * Deletes a comment.
     *
     * @param comment - The comment object to delete.
     */
    deleteComment(comment: any) {
        this.tabDeleteService
            .deleteRecordByTable('TABD_Comments', 'Id', comment.Id)
            ?.pipe(take(1))
            .subscribe((res) => {
                if (res.StatusCode == '200') {
                    this.toastrNotificationService.showSuccess(
                        'Comment Deleted',
                        'Success'
                    );

                    if (
                        this.editComment &&
                        this.editComment?.Id === comment.Id
                    ) {
                        this.onEvent('cancelEdit');
                    }

                    if (
                        this.replyToComment &&
                        this.replyToComment?.Id === comment.Id
                    ) {
                        this.onEvent('cancelReply');
                    }

                    this.loadComments();

                    // FOR REFERENCE

        const data = {

            DSQName: 'List_TABD_Comments',

                Id: this.componentValue.RecordId,

        this.tabGetService

    /**
     * Extracts mention chips from an HTML string.
     *
     * @param html - The HTML string to extract mention chips from.
     * @returns An array of mention chip titles.
     */
    extractMentionChips(html: string) {
        const parser = new DOMParser();
        const doc = parser.parseFromString(html, 'text/html');

        const mentionChips = Array.from(
            doc.querySelectorAll('.e-mention-chip')
        );

        const mentionChipTitles = mentionChips?.map((mentionChip) => {
            // const title = mentionChip.querySelector('a')?.getAttribute('title');
            // we are getting mentionChip in this element
            //<span contenteditable="false" class="e-mention-chip">projectmanager techextensor</span>
            const title = mentionChip.innerHTML;
            const elements = this.dataSource.find(
                (element) => element.key === title
            );
            return elements.value || '';
        });

        return mentionChipTitles;
    }

    /**
     * Retrieves the list of new mentions added in an updated comment.
     *
     * @param updatedComment - The updated comment with potential new mentions.
     * @param commentId - The ID of the comment to compare with existing mentions.
     * @returns An array of new mention objects containing the tagged user ID and optional comment ID and mention ID.
     */
    getNewMentionList(updatedComment: string, commentId: string) {
        let oldMentions: any[] = this.comments.find(
            (c) => c.Id === commentId
        )?.TABD_Comment_Tagged;
        let updateMentions = this.getMentionList(updatedComment);

        let newMentions: any[] = [];
        updateMentions?.forEach((m) => {
            if (
                !oldMentions?.find((om) => om.TaggedUserId === m.TaggedUserId)
            ) {
                newMentions.push(m);
            }
        });
        return newMentions;
    }

    /**
     * Retrieves the list of removed mentions in an updated comment.
     *
     * @param updatedComment - The updated comment with potential removed mentions.
     * @param commentId - The ID of the comment to compare with existing mentions.
     * @returns An array of removed mention objects containing the tagged user ID and optional comment ID and mention ID.
     */
    getRemovedMentionList(updatedComment: string, commentId: string) {
        let oldMentions: any[] = this.comments.find(
            (c) => c.Id === commentId
        )?.TABD_Comment_Tagged;
        let updateMentions = this.getMentionList(updatedComment);
        let removedMentions: any[] = [];
        oldMentions?.forEach((m) => {
            if (
                !updateMentions?.find(
                    (um) => um.TaggedUserId === m.TaggedUserId
                )
            ) {
                removedMentions.push(m);
            }
        });
        return removedMentions;
    }

    /**
     * Retrieves the list of mentions from a comment.
     *
     * @param comment - The comment from which to extract mentions.
     * @param commentId - Optional comment ID for filtering existing mentions.
     * @returns An array of mention objects containing the tagged user ID and optional comment ID and mention ID.
     */
    getMentionList(comment: string, commentId?: string) {
        const oldMentions = this.comments.find(
            (c) => c.Id === commentId
        )?.TABD_Comment_Tagged;
        const mentions = this.extractMentionChips(comment)?.map((m) => {
            const data = {
                TaggedUserId: m,
            };
            if (commentId) {
                data['commentId'] = commentId;
            }
            if (oldMentions) {
                const Id = oldMentions?.find((om) => om.TaggedUserId === m)?.Id;
                if (Id) data['Id'] = Id;
            }
            return data;
        });
        return mentions;
    }

    /**
     * Retrieves the username based on the user object.
     * @param user The user object containing the user's information.
     * @returns The username formatted as a string.
     */
    getUserName(user: any) {
        let username = null;
        if (
            user.hasOwnProperty('PersonId') &&
            user.PersonId &&
            user.PersonId.hasOwnProperty('FirstName') &&
            user.PersonId.FirstName
        ) {
            username = user.PersonId.FirstName;
        }
        if (
            user.hasOwnProperty('PersonId') &&
            user.PersonId &&
            user.PersonId.hasOwnProperty('LastName') &&
            user.PersonId.LastName
        ) {
            username = username + ' ' + user.PersonId.LastName;
        }
        return `${username}`;
    }

    /**
     * Retrieves the files referenced in the rich text editor value.
     * It searches for 'src' attributes in the HTML content using a regular expression.
     *
     */
    getFiles() {
        let files = [];

        const regex = /src=["'](.*?)["']/g;
        const matches = this.richTextEditor.value.match(regex);

        if (matches) {
            files = matches
                .map((match) => match.replace(/src=|["']/g, ''))
                .filter((link) => link.startsWith('blob'));
        }
        this.fileBlobLinks = files;
        if (this.fileBlobLinks.length > 0 && this.activeToast) {
            this.toastrNotificationService.clearToast(this.activeToast.toastId);
            this.toastrNotificationService.showInfo(
                'File Upload Completed',
                'File Upload Progress',
                {
                    timeOut: 2000,
                }
            );
            this.isImageUploaded = true;
        }
    }

    /**
     * Creates a file from a Blob URL.
     *
     * @param blobUrl - The Blob URL of the file.
     */
    createFileFromBlobUrl(blobUrl: string) {
        this.http
            .get(blobUrl, { responseType: 'blob' })
            .pipe(take(1))
            .subscribe((blob: Blob) => {
                const fileName = Date.now() + '.' + this.getFileExtension(blob);
                const file = new File([blob], fileName);

                // this.uploadFile(file, blobUrl, toast);
                this.uploadFile(file, blobUrl, this.activeToast);
            });
    }

    /**
     * Retrieves the file extension from a given Blob object.
     * The function extracts the file type from the Blob's 'type' property,
     *
     * @param blob - The Blob object from which to extract the file extension.
     * @returns The file extension as a string, or null if the Blob type is invalid.
     */
    getFileExtension(blob: Blob): string | null {
        const parts = blob.type.split('/');
        if (parts.length === 2) {
            return parts[1];
        }
        return null;
    }

    /**
     * Uploads a file to the server.
     *
     * @param file - The File object to upload.
     * @param oldUrl - The old URL of the file, which will be replace with new URL (server url)
     * @param toast - The toast notification displayed during the file upload progress.
     */
    uploadFile(file: File, oldUrl: string, toast: ActiveToast<any>) {
        let formData = new FormData();
        formData.append('1', file);

        this.tabMediaUploadService
            ?.uploadMedia(formData)

                    let uploadedFileData = res.Result[0];

                    this.richTextData = this.richTextData.replace(
                        oldUrl,
                        uploadedFileData.Url
                    );
                }
            });
    }

        // Get appObjectDetailsOfComments using the data source query ID
        const appObjectDetailsOfComments =
            TabAppFunction.getAppObjectFromDataSourceQueryID(
                this.dataSettingsComments.dataSourceQuery
            );

        // Get the primary key field details
        const primaryKeyField = TabRequestResponseFunction.getPrimaryKeyFields(
            appObjectDetailsOfComments
        );
        // Get the Aspnetuserid field for insert in database
        const TaggedUserField = this?.dataSettingsComments?.TaggedUserID;
        // Prepare API data for executing the data source query
        const apidata = {
            AppObjectName: appObjectDetailsOfComments.ObjectName,
            DSQName: clonedsqDetailsOfCommnets.QueryName,
            Reqtokens: {},
        };
        // FOR REFERENCE
        // if (this.componentValue && this.componentValue.parameters !== undefined) {
        //     apidata.Reqtokens = this.generateRequestTokens(this.componentValue.parameters);
        // }

        this.tabGetService
            .executeRAWDSQWithName(apidata)
            ?.pipe(take(1))
            ?.subscribe((data) => {
                const uniqueIds = new Set(
                    this.dataSource.map((item) => item.value)
                );

                // Iterate through the result data
                data.Result.forEach((element) => {
                    if (element[primaryKeyField[0].FieldName]) {
                        const idValue = element[TaggedUserField];
                        if (!uniqueIds.has(idValue)) {
                            this.dataSource.push({
                                value: idValue,
                                key: element[fieldName] || 'null',
                            });
                            // Add the ID to the Set to track uniqueness
                            uniqueIds.add(idValue);
                        }
                    }
                });
            });
    }

    // Function to generate request tokens dynamically
    generateRequestTokens(parameters) {
        if (
            parameters &&
            typeof parameters === 'object' &&
            Object.keys(parameters).length > 0
        ) {
            const reqTokens = {};
            for (const key in parameters) {
                if (parameters.hasOwnProperty(key)) {
                    reqTokens[key] = parameters[key];
                }
            }
            return reqTokens;
        } else {
            return {};
        }
    }
