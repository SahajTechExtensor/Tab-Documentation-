    public customAggregateFn = (customData: any) => {
        let totalHours = 0;
        let totalMinutes = 0;
        for (const timeObject of customData.items) {
            const [hours, minutes] = timeObject.Hours.split(':').map(Number);
            totalHours += hours;
            totalMinutes += minutes;
        }
        totalHours += Math.floor(totalMinutes / 60);
        totalMinutes %= 60;
        const formattedTotal = `${totalHours
            .toString()
            .padStart(2, '0')}:${totalMinutes.toString().padStart(2, '0')}`;
        return formattedTotal.toString();
    };

    /**
     * Deselects the currently selected row in the grid.
     * It sets the selected row to a blank row by clearing the values of each column.
     * Updates the grid instance with the new selected row and data.
     * Emits the updated grid instance through the `valueChange` event.

    deselectRow() {

        this.columns.forEach((element) => {
            blankRow[element.dataField] = '';
        });

    /**
     * Marks a custom view as the default view.

    async markAsDefaultCustomView(data: any) {
        // Asynchronously update the 'IsDefault' property for other views in the same dataSourceQueryId

    /**
     * Updates the specified custom view to be the default view.
     * @param {any} customViewId - The ID of the custom view to be marked as default.
     */
    private async updateDefaultCustomView(customViewId: any) {
        // Retrieve the appObject for 'Tabd_CustomViews' from the system database
        const appObject =
            TabAppFunction.getAppObjectFromSystemDBTableName(
                'Tabd_CustomViews'
            );

        // Prepare the payload for the update request
        let payload = { data: {} };

        // Define the data to be updated, setting IsDefault to true and including the customViewId
        let data: any = {
            IsDefault: true,
            ID: customViewId,
        };
        payload.data = data;

        this.tabUpdateService
            .updateUsingScreenId('', payload, appObject.ID)
            ?.pipe(take(1))
            ?.subscribe((response) => {
                // Check the response status code
                if (response.StatusCode == '200') {
                    // Display success notification
                    this.toastrNotificationService.showSuccess(
                        'Record Updated Successfully',
                        'Success'
                    );
                    // Refresh the default custom view ID after successful update
                    this.getDefaultCustomViewId();
                } else {
                    // Display error notification if the update is unsuccessful
                    this.toastrNotificationService.showError(
                        response.Errors[0],
                        'Error'
                    );
                }
            });
    }

    /**
     * Updates the 'IsDefault' property to false for other custom views with the same DataSourceQueryID,
     * and then calls the updateDefaultCustomView function to set the 'IsDefault' property to true
     * for the specified custom view.
     * @param {any} dataSourceQueryId - The DataSourceQueryID to identify custom views to update.
     * @param {any} customViewId - The ID of the custom view to be marked as default.
     */
    private async updateIsDefaultForOtherViews(
        dataSourceQueryId: any,
        customViewId: any
    ) {
        // Prepare an array of query parameters to identify custom views with the same DataSourceQueryID
        let queryFieldsArray: TabClasses.UpdateQueryParameter[] = [];
        queryFieldsArray.push({
            FieldName: 'DataSourceQueryID',
            FieldValue: dataSourceQueryId,
        });

        let payload = { data: {} };
        let data: any = {
            IsDefault: false,
        };
        payload.data = data;

        this.tabUpdateService
            .updateUsingDSQ(
                TabAppFunction.getDsqIDFromQueryName(
                    'Tabd_CustomViews',
                    'Detail_Tabd_CustomViews'
                )?.ID,
                payload,
                'Tabd_CustomViews',
                queryFieldsArray
            )
            ?.pipe(take(1))
            ?.subscribe(async (response) => {
                // Check the response status code and errors
                if (response.StatusCode == '200' || !response.Errors.length) {
                    // Call the UpdateDefaultCustomView function after successfully updating other custom views
                    await this.updateDefaultCustomView(customViewId);
                }
            });
    }

                            );
                            return;

                    }
                } else {
                    event.cancel = true;
                    this.toastrNotificationService.showInfo(
                        'This work item can only be placed within its current project.',
                        'Info'
                    );
                    return;
                }
            }
            const newIndex = event.dropIndex;
            this.updateNestedGridSequenceValues(row, newIndex);

        } else {

            // this.data.splice(newIndex, 0, row);
            this.updateSequenceValues(row, newIndex);
        }

    /**
     * Updates the sequence value for the dropped row based on its new index.
     *
     * @param {any} droppedRow - The dropped row.
     * @param {number} newIndex - The new index of the dropped row.
     */
    private updateSequenceValues(droppedRow: any, newIndex: number) {
        if (newIndex === 0) {
            droppedRow.Sequence = this.data[0].Sequence + 1;
        } else if (newIndex === this.data.length - 1) {
            droppedRow.Sequence = this.data[this.data.length - 1].Sequence - 1;
        } else {
            const prevValue = this.data[newIndex - 1].Sequence;
            const nextValue = this.data[newIndex].Sequence;
            droppedRow.Sequence = (prevValue + nextValue) / 2;
        }
        setTimeout(() => {
            this.updateRow(droppedRow);
        }, 10);
    }

    /**
     * Updates the sequence value for the dropped row based on its new index.
     *
     * @param {any} droppedRow - The dropped row.
     * @param {number} newIndex - The new index of the dropped row.
     */
    private updateNestedGridSequenceValues(droppedRow: any, newIndex: number) {
        if (newIndex === 0) {
            droppedRow.Sequence = this.tabBasicGrid.flatData[0].Sequence + 1;
        } else if (newIndex === this.tabBasicGrid.flatData.length - 1) {
            droppedRow.Sequence =
                this.tabBasicGrid.flatData[
                    this.tabBasicGrid.flatData.length - 1
                ].Sequence - 1;
        } else {
            const prevValue = this.tabBasicGrid.flatData[newIndex].Sequence;
            const nextValue = this.tabBasicGrid.flatData[newIndex + 1].Sequence;
            droppedRow.Sequence = (prevValue + nextValue) / 2;
        }
        setTimeout(() => {
            this.updateRow(droppedRow);
        }, 10);
    }

    /**
     * Updates the row in the database with new sequence value.
     * @param {any} data - The row data to be updated.
     */

        // const primaryKeyField = TabRequestResponseFunction.getPrimaryKeyFields(
        //     this.appObjectDetailsOfGrid
        // );
        let viewAppObjectDetailsOfGrid = this.getViewAppObjectDetails();
        const primaryKeyField = TabRequestResponseFunction.getPrimaryKeyFields(
            viewAppObjectDetailsOfGrid
        );

        const primaryFieldValue = data[primaryKeyField[0].SystemDBFieldName];
        payload.data[primaryKeyField[0].SystemDBFieldName] = primaryFieldValue;
        payload.data[this.dataSettingsGrid?.sortOrderField] = data.Sequence;
        if (this.dataSettingsGrid?.isAllowNestedGrid) {
            payload.data[
                this.dataSettingsGrid?.nestedGridChildColumn?.FieldName
            ] = data[this.dataSettingsGrid?.nestedGridChildColumn?.FieldName];
        }
        this.tabUpdateService
            .updateUsingScreenId('', payload, viewAppObjectDetailsOfGrid.ID)
            ?.pipe(take(1))
            ?.subscribe((response) => {
                if (response.StatusCode == '200') {
                    this.toastrNotificationService.showSuccess(
                        'Record Updated Successfully',
                        'Success'
                    );
                    this.tabBasicGrid.refresh();
                } else {
                    this.toastrNotificationService.showError(
                        response.Errors[0],
                        'Error'
                    );
                }
            });
    }

    cloneWorkitem(args) {
        this.tabGetService
            .executeRawSql(
                'exec SP_CloneWorkItem @aspnetuserid="' +
                    JSON.parse(localStorage.getItem('loginData'))?.Attributes
                        ?.Id +
                    '",@workitemid="' +
                    args.rowInfo.rowData.ID +
                    '"'
            )
            ?.pipe(take(1))
            ?.subscribe((response) => {
                if (response.StatusCode == '200' || !response.Errors.length) {
                    this.toastrNotificationService.showSuccess(
                        'Workitem Cloned Successfully',
                        'Success'
                    );
                    let customViewID;
                    if (this.selectedcustomView) {
                        customViewID = this.customViewsList.find(

                        ).ID;
                    }
                    this.populateGrid(this.token, customViewID);

                        'Error'
                    );

            ).ID;
            apiData['CustomViewID'] = customViewID;
