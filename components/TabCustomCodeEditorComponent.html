import { Component, EventEmitter, Input, OnDestroy, Output } from '@angular/core';
import { Subject, debounceTime, takeUntil } from 'rxjs';

@Component({
    selector: 'tab-core-tab-custom-code-editor',
    templateUrl: './tab-custom-code-editor.component.html',
    styleUrls: ['./tab-custom-code-editor.component.css'],
})
export class TabCustomCodeEditorComponent implements OnDestroy {
    @Input() disabled: boolean = false;
    @Output() valueChange = new EventEmitter<any>();
    @Input() dataSetting: any;
    editorOptions = {
        theme: 'vs-dark',
        language: 'Plain',
        wordWrap: true,
    };
    data: any;

    isShowEditor = true;

    private debounceSubject = new Subject();
    private _unsubscribeAll: Subject<any> = new Subject<any>();

    @Input()
    public set value(v: any) {
        // Update codeEditor value when it's null or undefine
        // if (v && v != null) {
        if (v != this.data) {
            this.data = v ?? '';
        }
        // }
        this.debounceSubject.next(1);
    }
    public get value(): any {
        return this.data;
    }

    constructor() {}

    ngOnInit(): void {
        this.debounceSubject
        .pipe(debounceTime(1000),takeUntil(this._unsubscribeAll))
        .subscribe((res) => {
            this.editorOptions = {
                ...this.editorOptions,
                language: this.dataSetting?.language,
            };
        });

        // setTimeout(() => {
        //     this.editorOptions = {
        //         ...this.editorOptions,
        //         language: this.dataSetting?.language,
        //     };
        //     // this.isShowEditor = true;
        // }, 100);
    }

    onChange(event: any) {
        this.valueChange.emit(event);
    }

    ngOnDestroy(): void {
        // Unsubscribe from all subscriptions
        this._unsubscribeAll.next(null);
        this._unsubscribeAll.complete();
    }
    
}
