import {
    AfterViewInit,
    Component,
    OnDestroy,
    OnInit,
    ViewChild,
    ViewEncapsulation,
} from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { fuseAnimations } from '@fuse/animations';
import { FuseConfigService } from '@fuse/services/config';
import { FuseLoadingService } from '@fuse/services/loading';
import { Store } from '@ngrx/store';
import * as fromRoot from '@techextensor/tab-service';
import { TabConstant, TabScreenFunction } from '@techextensor/tab-service';
import { AppConfig } from 'app/core/config/app.config';
import { cloneDeep } from 'lodash';
import { Subject, takeUntil } from 'rxjs';
import { TabFormRendererComponent } from '../tab-form-renderer/tab-form-renderer.component';

@Component({
    selector: 'Home',
    templateUrl: './Home.component.html',
    styleUrls: ['./Home.component.css'],
    encapsulation: ViewEncapsulation.None,
    animations: fuseAnimations,
})
export class HomeComponent implements OnInit, OnDestroy, AfterViewInit {
    @ViewChild('tabFormRendererHome', { static: true })
    tabFormRendererHome: TabFormRendererComponent;
    formJson: any;
    destroy$: Subject<boolean> = new Subject<boolean>();
    sessionScreenIds: any = [];
    breadcrumbs: any = [];
    selectedTab: number = 0;
    isEditScreen = false;
    isLoadScreen = false;
    currentRouteScreenId: string = '';
    private _unsubscribeAll: Subject<any> = new Subject<any>();
    isMultipleTabs = false;
    tagCount = false;
    parameters: any;
    /**
     * Constructor
     */
    constructor(
        private _activatedRoute: ActivatedRoute,
        private store: Store,
        private router: Router,
        private fuseLoadingService: FuseLoadingService,
        private fuseConfigService: FuseConfigService
    ) {
        this.store
            .select(fromRoot.getTabData)
            .pipe(takeUntil(this.destroy$))
            .subscribe((data: any) => {
                if (data?.tabData?.res) {
                    TabConstant.Common.tabJson = data.tabData.res;
                }
            });

        this.store
            .select(fromRoot.getTabData)
            .pipe(takeUntil(this.destroy$))
            .subscribe((data: any) => {
                if (data?.tabsData) {
                    this.sessionScreenIds = data.tabsData;
                    if (this.sessionScreenIds.length > 1) {
                        this.tagCount = true;
                    }
                } else {
                    this.sessionScreenIds = [];
                }
            });

        //configuration for settings of theme
        this.fuseConfigService.config$
            .pipe(takeUntil(this._unsubscribeAll))
            .subscribe((config: AppConfig) => {
                this.isMultipleTabs = config.multipleTabs;
                if (!this.isMultipleTabs) {
                    const dummySessionScreenIds = [];
                    this.sessionScreenIds = dummySessionScreenIds;
                    this.store.dispatch(
                        fromRoot.tabsData({ tabsData: this.sessionScreenIds })
                    );
                }
            });
    }

    ngOnInit(): void {
        let urlParams: any;
        this.fuseLoadingService.show();
        this._activatedRoute.params
        .pipe(takeUntil(this._unsubscribeAll))
        .subscribe((res) => {
            urlParams = res;
            sessionStorage.setItem(
                'activeRouteParams',
                JSON.stringify(urlParams)
            );
            if (Object.keys(urlParams).length && res['screen']) {
                this.currentRouteScreenId = res['screen'];
                if (this.isMultipleTabs) {
                    //we can configure this if we want in that way or not
                    this.processTabs(res['screen']);
                    this.findScreenIndex(res['screen']);
                }
            }
        });

        this._activatedRoute.queryParams.subscribe((qparams: Params) => {
            if (Object.keys(urlParams).length) {
                if (Object.keys(qparams).length) {
                    const qp = {};
                    Object.assign(qp, qparams);
                    if (Object.keys(qparams).indexOf('mode') > -1) {
                        delete qp['mode'];
                    }
                    qparams = qp;
                    this.isEditScreen = true;
                    this.parameters = qparams;
                } else {
                    this.isEditScreen = false;
                }
            } else {
                this.isEditScreen = false;
            }
        });

        this.tabFormRendererHome.isEditScreen.subscribe((res) => {
            if (res) {
                this.isEditScreen = true;
            }
        });
    }

    ngAfterViewInit(): void {
        this.tabFormRendererHome.showLoader.subscribe((res) => {
            if (res) {
                this.fuseLoadingService.show();
            } else {
                this.fuseLoadingService.hide();
            }
        });
    }

    processTabs(screenId: string) {
        let screenIDs: any = localStorage.getItem('tabsData');
        if (screenIDs && screenIDs != 'null') {
            screenIDs = JSON.parse(screenIDs);
            if (Object.keys(screenIDs).length == 0) {
                screenIDs = [];
                screenIDs.push(screenId);
            } else {
                if (
                    !screenIDs.find(
                        (a) => a.toLowerCase() === screenId.toLowerCase()
                    )
                ) {
                    screenIDs.push(screenId);
                }
            }
        } else {
            screenIDs = [screenId];
        }
        this.sessionScreenIds = screenIDs;
        this.findScreenIndex(this.currentRouteScreenId);
        this.processBreadcrumb();
        this.store.dispatch(
            fromRoot.tabsData({ tabsData: this.sessionScreenIds })
        );
    }

    onChange(event: any) {}

    findScreenIndex(screenId: string, index = -1) {
        if (index <= -1) {
            index = this.sessionScreenIds.findIndex((x) => x === screenId);
        }
        this.selectedTab = index;
    }

    displayTab(index: number, screenId: string) {
        this.fuseLoadingService.show();
        this.findScreenIndex(screenId, index);
        screenId = this.sessionScreenIds[index];
        const editScreenParam = localStorage.getItem(
            TabConstant.Storage.editScreenQueryParam
        );
        //we can configure this if we want in that way or not
        const screenDetail = TabScreenFunction.getScreen(screenId);
        if (screenDetail.ScreenType == 1) {
            this.router.navigate(['/screen', screenId]);
        } else {
            this.isEditScreen = true;
            this.router.navigate(['/screen', screenId], {
                queryParams: JSON.parse(editScreenParam),
            });
        }
    }

    removeScreen(screenId: string) {
        const index = this.sessionScreenIds.findIndex((x) => x === screenId);
        const cloneScreenIDs = cloneDeep(this.sessionScreenIds);
        cloneScreenIDs.splice(index, 1);
        this.sessionScreenIds = cloneScreenIDs;
        this.store.dispatch(
            fromRoot.tabsData({ tabsData: this.sessionScreenIds })
        );
        this.processBreadcrumb();
        if (index > 1) {
            this.selectedTab = index - 1;
        } else {
            this.selectedTab = 0;
        }
    }

    processBreadcrumb() {
        this.breadcrumbs = [];
        this.sessionScreenIds.forEach((screenID) => {
            if (TabConstant.Common.tabJson) {
                const screen = (
                    TabConstant.Common.tabJson as any
                ).AppScreens.find(
                    (a) => a.ID.toLowerCase() == screenID.toLowerCase()
                );
                this.breadcrumbs.push({
                    name: screen?.Name,
                    url: '/screen/' + screen?.ID,
                    screenId: screen.ID,
                    type: screen.ScreenType,
                });
            }
        });
        this.isLoadScreen = true;
    }

    ngOnDestroy(): void {
        this.destroy$.next(true);
        this.destroy$.complete();
        this._unsubscribeAll.next(null);
        this._unsubscribeAll.complete();
        this.tabFormRendererHome?.showLoader.complete();
        localStorage.removeItem(TabConstant.Storage.editScreenQueryParam);
        sessionStorage.removeItem('activeRouteParams');
    }
}
