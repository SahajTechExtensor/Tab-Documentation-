    public onInsert(): void {
        if (this.richTextEditor.formatter.getUndoRedoStack().length === 0) {
            this.richTextEditor.formatter.saveData();
        }
        this.saveSelection.restore();
        var canvas = document.createElement('CANVAS');
        var ctx = (canvas as any).getContext('2d');
        var imgData: ImageData = (this.imageEditorObj as any).getImageData();
        (canvas as any).height = imgData.height;
        (canvas as any).width = imgData.width;
        ctx.putImageData(imgData, 0, 0);
        this.isLoaded = true;
        const imageBlobUrl = this.convertDataUrlToBlobUrl(
            (canvas as any).toDataURL()
        );
        this.richTextEditor.executeCommand('editImage', {
            url: imageBlobUrl,
            width: { width: (canvas as any).width },
            height: { height: (canvas as any).height },
            selection: this.saveSelection,
            cssClass: this.imageELement
                .getAttribute('class')
                .replace('e-rte-image', ''),
        });
        this.richTextEditor.formatter.saveData();
        this.richTextEditor.formatter.enableUndo(this.richTextEditor);
        this.getFiles();
        this.fileBlobLinks.forEach((blobUrl) => {
            this.createFileFromBlobUrl(blobUrl);
        });
        this.fileBlobLinks = [];
        this.dialogObj.hide();
        this.isLoaded = false;
    }

    /**
     * Creates a file from a Blob URL.
     *
     * @param blobUrl - The Blob URL of the file.
     */
    createFileFromBlobUrl(blobUrl: string) {
        // const toast = this.toastrNotificationService.showInfo(
        //     'Uploading file... Please wait',
        //     'File Upload Progress',
        //     {
        //         disableTimeOut: true,
        //     }
        // );
        this.http
            .get(blobUrl, { responseType: 'blob' })
            .pipe(take(1))
            .subscribe((blob: Blob) => {
                const fileName = Date.now() + '.' + this.getFileExtension(blob);
                const file = new File([blob], fileName);

                // this.uploadFile(file, blobUrl, toast);
                this.uploadFile(file, blobUrl, this.activeToast);
            });
    }

    /**
     * Uploads a file to the server.
     *
     * @param file - The File object to upload.
     * @param oldUrl - The old URL of the file, which will be replace with new URL (server url)
     * @param toast - The toast notification displayed during the file upload progress.
     */
    uploadFile(file: File, oldUrl: string, toast: ActiveToast<any>) {
        let formData = new FormData();
        formData.append('1', file);

        this.tabMediaUploadService
            ?.uploadMedia(formData)
            .pipe(take(1))
            .subscribe((res: any) => {
                if (!!res.Result && res.Result.length > 0) {
                    let uploadedFileData = res.Result[0];

                    this.richTextData = this.richTextData.replace(
                        oldUrl,
                        uploadedFileData.Url
                    );
                    this.valueChange.emit(this.richTextData);
                }
            });
    }

    async populateMentionData(token: any) {
        this.appObjectDetails =
            TabAppFunction.getAppObjectFromDataSourceQueryID(
                this.dataSettings?.dataSourceQueryId
            );
        let dataSource = this.generateKeyValueDataFromFieldList(
            this.appObjectDetails.Fields,
            'DisplayName',
            'ID'
        );
        dataSource.forEach((element) => {
            this.dataSource.push({
                key: element.DisplayName,
                value: element.DisplayName,
            });
        });
    }

    generateKeyValueDataFromFieldList(list, key, value) {
        let listArray = [];
        list.forEach((element) => {
            let k = element[key];
            let v = element[value];
            listArray.push({ [key]: k, [value]: v });
        });
        return listArray;
    }

     * Retrieves the files referenced in the rich text editor value.
     * It searches for 'src' attributes in the HTML content using a regular expression.
     *
     */
    getFiles() {
        let files = [];

        const regex = /src=["'](.*?)["']/g;
        const matches = this.richTextEditor.value.match(regex);

        if (matches) {
            files = matches

    /**
     * Retrieves the file extension from a given Blob object.
     * The function extracts the file type from the Blob's 'type' property,
     *
     * @param blob - The Blob object from which to extract the file extension.
     * @returns The file extension as a string, or null if the Blob type is invalid.
     */
    getFileExtension(blob: Blob): string | null {
        const parts = blob.type.split('/');
        if (parts.length === 2) {
            return parts[1];
        }
        return null;
    }

    /**
     * convert dataURL to blob
     * @param dataUrl dataURL
     * @returns blob Url
     */
    convertDataUrlToBlobUrl(dataUrl: string): string {
        var byteCharacters = atob(dataUrl.split(',')[1]);
        var byteNumbers = new Array(byteCharacters.length);
        for (var i = 0; i < byteCharacters.length; i++) {
            byteNumbers[i] = byteCharacters.charCodeAt(i);
        }
        var byteArray = new Uint8Array(byteNumbers);
        var blob = new Blob([byteArray], { type: 'image/png' });

        // Create a blob URL from the Blob
        return URL.createObjectURL(blob);
    }

        if (!data.includes('class="rich-text-preview"')) {
            return '<div class="rich-text-preview">' + data + '<div>';
        }
        return data;
    }
