import { Router } from '@angular/router';
import {
    Component,
    ElementRef,
    EventEmitter,
    HostBinding,
    Input,
    OnChanges,
    OnDestroy,
    OnInit,
    Output,
    SimpleChanges,
    ViewChild,
    ViewEncapsulation,
} from '@angular/core';
import { FormControl } from '@angular/forms';
import { HttpClient } from '@angular/common/http';
import { debounceTime, filter, map, Subject, takeUntil } from 'rxjs';
import { fuseAnimations } from '@fuse/animations/public-api';
import {
    TabConstant,
    TabAppFunction,
    TabRequestResponseFunction,
} from '@techextensor/tab-service';
import { find } from 'lodash';
import { ToastrNotificationService } from './../../../layout/common/notifications/toastr.service';
import { MAT_AUTOCOMPLETE_DEFAULT_OPTIONS } from '@angular/material/autocomplete';

@Component({
    selector: 'search',
    templateUrl: './search.component.html',
    encapsulation: ViewEncapsulation.None,
    exportAs: 'fuseSearch',
    animations: fuseAnimations,
    providers: [
        {
            provide: MAT_AUTOCOMPLETE_DEFAULT_OPTIONS,
            useValue: { overlayPanelClass: 'view-search-layout' },
        },
    ],
})
export class SearchComponent implements OnChanges, OnInit, OnDestroy {
    @Input() appearance: 'basic' | 'bar' = 'basic';
    @Input() debounce: number = 300;
    @Input() minLength: number = 2;
    @Output() search: EventEmitter<any> = new EventEmitter<any>();

    opened: boolean = false;
    resultSets: any[];
    searchControl: FormControl = new FormControl();
    private _unsubscribeAll: Subject<any> = new Subject<any>();
    viewMoreSets: any = [];
    private setTimer: any;

    /**
     * Constructor
     */
    constructor(
        private _httpClient: HttpClient,
        private router: Router,
        private toastrNotificationService: ToastrNotificationService
    ) {}

    // -----------------------------------------------------------------------------------------------------
    // @ Accessors
    // -----------------------------------------------------------------------------------------------------

    /**
     * Host binding for component classes
     */
    @HostBinding('class') get classList(): any {
        return {
            'search-appearance-bar': this.appearance === 'bar',
            'search-appearance-basic': this.appearance === 'basic',
            'search-opened': this.opened,
        };
    }

    /**
     * Setter for bar search input
     *
     * @param value
     */
    @ViewChild('barSearchInput')
    set barSearchInput(value: ElementRef) {
        // If the value exists, it means that the search input
        // is now in the DOM and we can focus on the input..
        if (value) {
            // Give Angular time to complete the change detection cycle
            this.setTimer = setTimeout(() => {
                // Focus to the input element
                value.nativeElement.focus();
            }, 0);
        }
    }

    // -----------------------------------------------------------------------------------------------------
    // @ Lifecycle hooks
    // -----------------------------------------------------------------------------------------------------

    /**
     * On changes
     *
     * @param changes
     */
    ngOnChanges(changes: SimpleChanges): void {
        // Appearance
        if ('appearance' in changes) {
            // To prevent any issues, close the
            // search after changing the appearance
            this.close();
        }
    }

    /**
     * On init
     */
    ngOnInit(): void {
        // Subscribe to the search field value changes
        this.searchControl.valueChanges
            .pipe(
                debounceTime(this.debounce),
                takeUntil(this._unsubscribeAll),
                map((value) => {
                    // Set the resultSets to null if there is no value or
                    // the length of the value is smaller than the minLength
                    // so the autocomplete panel can be closed
                    if (!value || value.length < this.minLength) {
                        this.resultSets = null;
                    }

                    // Continue
                    return value;
                }),
                // Filter out undefined/null/false statements and also
                // filter out the values that are smaller than minLength
                filter((value) => value && value.length >= this.minLength)
            )
            .subscribe((value) => {
                this._httpClient
                    .get(
                        TabConstant.API.endpoint + 'Search?searchQuery=' + value
                    )
                    .subscribe((resultSets: any) => {
                        if (
                            resultSets?.StatusCode == '200' &&
                            resultSets.Result?.length > 0
                        ) {
                            //his.tabAppService.getSchema();

                            // Store the result sets
                            this.resultSets = resultSets.Result;
                            this.viewMoreSets = resultSets.Result;

                            this.resultSets.forEach((element) => {
                                element['displayKeyField'] =
                                    this.getDisplayKeyFieldName(
                                        element.AppObjectId
                                    );
                                element['primaryKey'] = this.getPrimaryKey(
                                    element.AppObjectId
                                );
                                // element.AppObjectId =
                                //     TabAppFunction.getAppObject(
                                //         element.AppObjectId
                                //     )?.DisplayName;
                                element.appObjectName =
                                    TabAppFunction.getAppObject(
                                        element.AppObjectId
                                    )?.DisplayName;

                                for (
                                    let i = 0;
                                    i < element.SearchLog.length;
                                    i++
                                ) {
                                    element.SearchLog[i] = JSON.parse(
                                        element.SearchLog[i]
                                    );
                                    element.SearchLog[i]['displayKeyField'] =
                                        element.displayKeyField;
                                    element.SearchLog[i]['appObjectName'] =
                                        element.appObjectName;
                                    element.SearchLog[i]['AppObjectId'] =
                                        element.AppObjectId;
                                    element.SearchLog[i]['primaryKey'] =
                                        element.primaryKey;
                                }
                            });
                        } else {
                            this.resultSets = [];
                        }
                        // Execute the event
                        this.search.next(this.resultSets);
                    });
            });
    }

    getDisplayKeyFieldName(appObjectId) {
        let appObjectDetails = TabAppFunction.getAppObject(appObjectId);
        if (appObjectDetails) {
            let objectFields = appObjectDetails?.Fields;
            let displayFieldKey = find(
                objectFields,
                (t: any) => t.IsDisplayField == true
            );
            if (displayFieldKey) {
                return displayFieldKey.SystemDBFieldName;
            }
        }
    }
    getPrimaryKey(appObjectId) {
        let appObjectDetail = TabAppFunction.getAppObject(appObjectId);
        if (appObjectDetail) {
            let PrimaryKeyFields =
                TabRequestResponseFunction.getPrimaryKeyFields(appObjectDetail);

            if (PrimaryKeyFields && PrimaryKeyFields.length === 1) {
                return PrimaryKeyFields[0]?.SystemDBFieldName;
            }
        }
    }

    /**
     * On destroy
     */
    ngOnDestroy(): void {
        // Unsubscribe from all subscriptions
        this._unsubscribeAll.next(null);
        this._unsubscribeAll.complete();
        clearTimeout(this.setTimer);
    }

    // -----------------------------------------------------------------------------------------------------
    // @ Public methods
    // -----------------------------------------------------------------------------------------------------

    /**
     * On keydown of the search input
     *
     * @param event
     */
    onKeydown(event: KeyboardEvent): void {
        // Listen for escape to close the search
        // if the appearance is 'bar'
        if (this.appearance === 'bar') {
            // Escape
            if (event.code === 'Escape') {
                // Close the search
                this.close();
            }
        }
    }

    /**
     * Open the search
     * Used in 'bar'
     */
    open(): void {
        // Return if it's already opened
        if (this.opened) {
            return;
        }

        // Open the search
        this.opened = true;
    }

    /**
     * Close the search
     * * Used in 'bar'
     */
    close(): void {
        // Return if it's already closed
        if (!this.opened) {
            return;
        }

        // Clear the search input
        this.searchControl.setValue('');

        // Close the search
        this.opened = false;
    }

    /**
     * Track by function for ngFor loops
     *
     * @param index
     * @param item
     */
    trackByFn(index: number, item: any): any {
        return item.id || index;
    }

    navigateToRoute(entity: string, dataId: string) {
        switch (entity) {
            case 'screen':
                this.router.navigateByUrl(
                    '/screen/BC7D4E1D-E976-49A1-BE33-E53FC6943D7C?mode=designer&Id=' +
                        dataId
                );
                break;
            case 'menu':
                this.router.navigateByUrl(
                    '/screen/D3B95948-683C-4494-AFF9-EDEF1C363E92?ID=' + dataId
                );
                break;
            case 'viewmore':
                localStorage.setItem(
                    'search-result',
                    JSON.stringify(this.viewMoreSets)
                );
                this.router
                    .navigateByUrl('/', { skipLocationChange: true })
                    .then(() =>
                        this.router.navigate(['/view-more'], {
                            state: { result: this.viewMoreSets },
                        })
                    );
                break;
        }
    }

    navigateToDaynmicRoute(result) {
        let appObjectDetail = TabAppFunction.getAppObject(result.AppObjectId);
        if (appObjectDetail && appObjectDetail?.AppObjectConfiguration) {
            let appObjectConfig = JSON.parse(
                appObjectDetail?.AppObjectConfiguration
            );
            if (appObjectConfig) {
                if (appObjectConfig?.addScreenId) {
                    this.router.navigateByUrl(
                        '/screen/' +
                            appObjectConfig.addScreenId +
                            '?' +
                            result.primaryKey +
                            '=' +
                            result[result.primaryKey]
                    );
                } else {
                    this.toastrNotificationService.showError(
                        'You dont have detail screen for this app object',
                        'Error'
                    );
                }
            } else {
                this.toastrNotificationService.showError(
                    'You dont have detail screen for this app object',
                    'Error'
                );
            }
        } else {
            this.toastrNotificationService.showError(
                'You dont have detail screen for this app object',
                'Error'
            );
        }
    }
}
