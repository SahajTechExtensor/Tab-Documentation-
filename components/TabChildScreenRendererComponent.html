import {
    Component,
    EventEmitter,
    Input,
    OnInit,
    Output,
    ViewChild,
} from '@angular/core';
import { FormioCustomComponent } from '@formio/angular';

@Component({
    selector: 'tab-core-tab-child-screen-renderer',
    templateUrl: './tab-child-screen-renderer.component.html',
    styleUrls: ['./tab-child-screen-renderer.component.css'],
})
export class TabChildScreenRendererComponent
    implements OnInit, FormioCustomComponent<any>
{
    @ViewChild('tabChildScreenRenderer') tabChildScreenRenderer: any;

    // @Input() childScreenId = '';
    // @Input() parameters: any = {};
    @Input() disabled: boolean = false;
    // @Input() submissionData: any;
    // @Input() isAllowToSubmit = false;
    @Input() dataSetting: any;
    @Output() valueChange = new EventEmitter<any>();

    dataObj: any;
    formvalue: any;

    @Input()
    public set value(v: any) {
        // console.log('child screen setter', v, this.dataSetting);
        if (v && v != null) {
            this.formvalue = v;
            // if (this.tabChildScreenRenderer) {
            //     this.tabChildScreenRenderer.refreshFormData(this.formvalue);
            // }
            if (!this.formvalue?.hasOwnProperty('isRender')) {
                if (this.tabChildScreenRenderer) {
                    this.tabChildScreenRenderer.refreshFormData(this.formvalue);
                }
            }
            // Added condition if parameters does not come from calculated values then we will not update datasetting parameters
            //     if (this.formvalue?.parameters) {
            //         this.dataSetting.parameters = Object.assign(
            //             JSON.parse(this.dataSetting.parameters),
            //             this.formvalue?.parameters
            //         );
            //     }
            // }

            if (
                typeof this.dataSetting.ScreenConfiguration === 'string' &&
                this.dataSetting.ScreenConfiguration.trim() !== ''
            ) {
                try {
                    // Attempt to parse the JSON string into an object
                    const screenConfiguration = JSON.parse(
                        this.dataSetting.ScreenConfiguration
                    );

                    // Merge the parsed object with the form value's parameters
                    this.dataSetting.parameters = {
                        screenConfiguration: screenConfiguration, // Spread the properties of the parsed object
                        ...(this.formvalue?.parameters || {}), // Spread the properties of formvalue?.parameters or use an empty object if formvalue?.parameters is null or undefined
                    };
                } catch (error) {
                    // If parsing fails, handle the error (e.g., log, display an error message, etc.)
                    console.error(
                        'Error parsing dataSetting.parameters:',
                        error
                    );
                }
            } else if (this.formvalue?.parameters) {
                // If this.dataSetting.parameters is not a valid JSON string or empty, and formvalue?.parameters exists, directly assign it.
                this.dataSetting.parameters = this.formvalue.parameters;
            }
        }
    }

    public get value(): any {
        return this.formvalue;
    }

    constructor() {}

    ngOnInit(): void {
        setTimeout(() => {
            // console.log('child screen render ngoninit ', this.parameters);
        });
    }

    onChange(event: any) {
        // console.log('child screen render on change');
        // console.log('this.datasetting =>', this.dataSetting);
        if (
            this.tabChildScreenRenderer &&
            !this.formvalue?.hasOwnProperty('isRender')
        ) {
            this.formvalue = this.tabChildScreenRenderer.latestForm;
            //commented as creating issue for enum list and detail screen
            this.valueChange.emit(this.formvalue);
        }
    }
}
