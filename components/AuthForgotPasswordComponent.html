import { Component, OnInit, ViewChild, ViewEncapsulation } from '@angular/core';
import { FormBuilder, FormGroup, NgForm, Validators } from '@angular/forms';
import { finalize, take } from 'rxjs';
import { fuseAnimations } from '@fuse/animations';
import { FuseAlertType } from '@fuse/components/alert';
import { AuthService } from 'app/core/auth/auth.service';
import { Router } from '@angular/router';
import {
    TabConstant,
    TabGetService,
    TabWorkflowService,
} from '@techextensor/tab-service';
import { error } from 'console';

@Component({
    selector: 'auth-forgot-password',
    templateUrl: './forgot-password.component.html',
    encapsulation: ViewEncapsulation.None,
    animations: fuseAnimations,
})
export class AuthForgotPasswordComponent implements OnInit {
    @ViewChild('forgotPasswordNgForm') forgotPasswordNgForm: NgForm;

    alert: { type: FuseAlertType; message: string } = {
        type: 'success',
        message: '',
    };
    forgotPasswordForm: FormGroup;
    showAlert: boolean = false;
    orgData: any;
    menuLogo: any;
    OrganizationInfo: any;
    defaultColor = '#0a5077';
    appCode: any;
    set accessToken(token: string) {
        sessionStorage.setItem(TabConstant.Storage.token, token);
    }

    get accessToken(): string {
        return sessionStorage.getItem(TabConstant.Storage.token) ?? '';
    }
    /**
     * Constructor
     */
    constructor(
        private _authService: AuthService,
        private _formBuilder: FormBuilder,
        private router: Router,
        private tabGetService: TabGetService,
        private tabWorkFlowService: TabWorkflowService
    ) {}

    // -----------------------------------------------------------------------------------------------------
    // @ Lifecycle hooks
    // -----------------------------------------------------------------------------------------------------

    /**
     * On init
     */
    ngOnInit(): void {
        //set the applicationCode
        this.appCode = sessionStorage.getItem(
            TabConstant.Storage.applicationCode
        );

        // if (this.appCode === null) {
        //     this.appCode = 'TAB';
        // }

        // Create the form
        this.forgotPasswordForm = this._formBuilder.group({
            email: ['', [Validators.required, Validators.email]],
        });

        // get logo from orgInfo
        this.OrganizationInfo = JSON.parse(localStorage.getItem('orgInfo'));
        if (this.OrganizationInfo != null) {
            this.orgData = JSON.parse(this.OrganizationInfo.LocaleSettings);
        }
    }

    // -----------------------------------------------------------------------------------------------------
    // @ Public methods
    // -----------------------------------------------------------------------------------------------------

    /**
     * Send the reset link
     */
    sendResetLink(): void {
        // Return if the form is invalid
        if (this.forgotPasswordForm.invalid) {
            return;
        }

        // Disable the form
        this.forgotPasswordForm.disable();

        // Hide the alert
        this.showAlert = false;

        // Forgot password
        const email = this.forgotPasswordForm.get('email').value;

        this.accessToken = TabConstant.Common.staticToken;

        this._authService
            .forgotPassword(email)
            .pipe(
                finalize(() => {
                    // Re-enable the form
                    this.forgotPasswordForm.enable();

                    // Reset the form
                    this.forgotPasswordNgForm.resetForm();

                    // Show the alert
                    this.showAlert = true;
                }),
                take(1)
            )
            .subscribe(
                (response) => {
                    if (
                        response?.StatusCode == '200' &&
                        response?.Result &&
                        response['Result'].length !== 0
                    ) {
                        // Set the alert
                        this.alert = {
                            type: 'success',
                            message:
                                "Password reset sent! You'll receive an email if you are registered on our system.",
                        };

                        // Show the alert
                        this.showAlert = true;

                        // Redirect to the reset password page with the email in the URL
                        // setTimeout(() => {
                        //     this.router.navigate(['/reset-password'], {
                        //         queryParams: {
                        //             email: email,
                        //              reset: 'true'
                        //         },
                        //     });
                        // }, 2000);
                        const workflowData: any = {
                            Email: email,
                        };

                        if (this.orgData && this.orgData !== null) {
                            const workFlowEndPoint = this.orgData.workFlowName;

                            if (workFlowEndPoint) {
                                this.tabWorkFlowService
                                    .executeELSAWorkflow(
                                        workFlowEndPoint,
                                        workflowData
                                    )
                                    .subscribe(
                                        (response) => {
                                            this.accessToken = '';
                                        },
                                        (error) => {
                                            this.accessToken = '';
                                            // Set the alert
                                            this.alert = {
                                                type: 'error',
                                                message:
                                                    'Please try again later',
                                            };
                                            // Show the alert
                                            this.showAlert = true;
                                        },
                                        () => {
                                            this.accessToken = '';
                                        }
                                    );
                            } else {
                                this.alert = {
                                    type: 'error',
                                    message:
                                        'Please Configure WorkFlow In OrgInfo.',
                                };
                            }
                        }
                    } else {
                        // Set the alert
                        this.accessToken = '';
                        this.alert = {
                            type: 'error',
                            message:
                                'Email does not found! Are you sure you are already a member?',
                        };
                        // Show the alert
                        this.showAlert = true;
                    }
                },
                (error) => {
                    this.accessToken = '';
                    // Set the alert
                    this.alert = {
                        type: 'error',
                        message:
                            'Email does not found! Are you sure you are already a member?',
                    };
                    // Show the alert
                    this.showAlert = true;
                },
                () => {
                    this.accessToken = '';
                }
            );

        //     this.accessToken = TabConstant.Common.staticToken;
