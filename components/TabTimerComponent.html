import { ChangeDetectorRef, Component, Input, OnDestroy, OnInit } from '@angular/core';
import { TabAppFunction, TabGetService } from '@techextensor/tab-service';
import { cloneDeep } from 'lodash';
import moment from 'moment';
import { Subject, debounceTime, take, takeUntil } from 'rxjs';

@Component({
    selector: 'app-tab-timer',
    templateUrl: './tab-timer.component.html',
    styleUrls: ['./tab-timer.component.scss'],
})
export class TabTimerComponent implements OnInit ,OnDestroy{
    private _value: any;
    private debounceSubject = new Subject();
    _unsubscribeAll: Subject<string> = new Subject<string>();
    private timerInterval: NodeJS.Timeout | null = null;
    isTimeNegative:boolean = false;
    dsqDetailsOfTimer: any;
    clonedsqDetailsTimer: any;
    field: any;
    dataOfField: any;
    days: any;
    hours: any;
    minutes: any;
    seconds: any;
    height: string = ''; // Set your initial height value here
    width: string = '';
    endDate = '';

    constructor(private tabGetService: TabGetService ) {}

    @Input() dataSettingsTimer: any;
    @Input()
    public set value(val: any) {
        if (val && val != null) {
            this._value = val;
        }
        this.debounceSubject.next(1000);
    }
    public get value(): any {
        return this._value;
    }

    ngOnInit(): void {
        this.debounceSubject
            .pipe(debounceTime(1000))
            .pipe(takeUntil(this._unsubscribeAll))
            .subscribe(() => {
                this.popualteData();
                this.setHeightWidth();
                this.endDate = this.calculateDisplayValue();
            });
    }

    // This function populates data based on certain conditions
    popualteData() {
        // Check if there's a dataSettingsTimer available
        if (this.dataSettingsTimer) {
            // Get data details from a data source query

            this.dsqDetailsOfTimer =
                TabAppFunction.getDataFromDataSourceQueryID(
                    this.dataSettingsTimer.dataSourceQuery
                );

            // Clone the data details for modification

            this.clonedsqDetailsTimer = cloneDeep(this.dsqDetailsOfTimer);
            // Create an API data object with relevant information

            const apidata = {
                AppObjectName: TabAppFunction.getAppObjectFromDataSourceQueryID(
                    this.dataSettingsTimer.dataSourceQuery
                ).ObjectName,
                DSQName: this.clonedsqDetailsTimer.QueryName,
                Reqtokens: { Id: this?._value?.parameters?.Id },
            };
            // Execute a service to fetch raw data with the API data

            this.tabGetService
                .executeRAWDSQWithName(apidata)
                ?.pipe(take(1))
                ?.subscribe((data) => {
                    // Extract the field name from dataSettingsTimer

                    this.field =
                        this.dataSettingsTimer.ValueField.AppFieldID.FieldName;
                    // Retrieve data based on the extracted field

                    this.dataOfField = data.Result[0][this.field];
                    // Convert the data into a specific format using convertOutOfTAT function

                    const Minutes = this.dataOfField;
                    this.convertOutOfTAT(Minutes);
                    const live =  this?._value?.parameters?.isLive;
                    if(live){
                    this.endDate = this.calculateEndDate(Date.now(),Minutes);
                    }else{
                            const allocationDatetime = moment(data.Result[0].Allocationdatetime, 'DD-MM-YYYY hh:mm:ss A');
                            // Extract hours from PhoenixTAT and convert it to milliseconds
                            const phoenixTATHours = parseFloat(data.Result[0].PhoenixTAT);
                            const phoenixTATMilliseconds = phoenixTATHours * 60 * 60 * 1000;

                            // Calculate completion time in milliseconds
                            const completionTimeMilliseconds = allocationDatetime.valueOf() + phoenixTATMilliseconds;

                       
                        this.endDate = this.calculateEndDate(completionTimeMilliseconds,0);
                    }
                });
        }
    }



    // Set the height and width of an element based on values in dataSettingsTimer, if available

    setHeightWidth() {
        if (this.dataSettingsTimer.height && this.dataSettingsTimer.width) {
            // Set the height and width properties with 'px' units
            this.height = this.dataSettingsTimer.height + 'px';
            this.width = this.dataSettingsTimer.width + 'px';
        }
    }

    convertOutOfTAT(outOfTATMinutes: number) {
        if (outOfTATMinutes != null && outOfTATMinutes != undefined) {
            let isNegative = outOfTATMinutes <= 0;
            this.isTimeNegative = isNegative;
            const absoluteMinutes = Math.abs(outOfTATMinutes);

            const days = Math.floor(absoluteMinutes / (60 * 24));
            const hours = Math.floor((absoluteMinutes % (60 * 24)) / 60);
            const remainingMinutes = absoluteMinutes % 60;
            // const seconds = 0; // Initialize seconds to 0 // You can calculate seconds if needed
            const seconds = absoluteMinutes % 60;

            this.days = isNegative ? -days : days;
            this.hours = isNegative ? -hours : hours;
            this.minutes = isNegative ? -remainingMinutes : remainingMinutes;
            this.seconds = isNegative ? -seconds : seconds;

            // Clear any existing interval if it exists
            if (this.timerInterval) {
                clearInterval(this.timerInterval);
            }

            const isLive = this?._value?.parameters?.isLive;

            if (isLive) {
                // Start counting seconds dynamically when status is not closed
                this.timerInterval = setInterval(() => {
                    if (isNegative) {
                        if (this.seconds > -59) {
                            this.seconds--;
                        } else {
                            this.seconds = 0;
                            if (this.minutes > -59) {
                                this.minutes--;
                            } else {
                                this.minutes = 0;
                                if (this.hours > -23) {
                                    this.hours--;
                                } else {
                                    this.hours = 0;
                                    this.days--;
                                }
                            }
                        }
                    } else {
                        if (this.seconds > 0) {
                            this.seconds--;
                        } else {
                            if (this.minutes > 0) {
                                this.minutes--;
                                this.seconds = 59;
                            } else {
                                if (this.hours > 0) {
                                    this.hours--;
                                    this.minutes = 59;
                                    this.seconds = 59;
                                } else {
                                    if (this.days > 0) {
                                        this.days--;
                                        this.hours = 23;
                                        this.minutes = 59;
                                        this.seconds = 59;
                                    } else {
                                        // All time units are 0, handle negative time here
                                        isNegative = true;
                                        // Optionally, stop the timer or reset it to a certain value
                                        // clearInterval(this.timerInterval);
                                        // this.resetTimer();
                                    }
                                }
                            }
                        }
                    }
                    if (
                        this.days < 0 ||
                        this.minutes < 0 ||
                        this.hours < 0 ||
                        this.seconds < 0
                    ) {
                        this.isTimeNegative = true;
                        isNegative = true;
                    }
                }, 1000); // Update seconds every 1000 milliseconds (1 second)
                // Clear the interval when the component is destroyed
                this.ngOnDestroy = () => {
                    clearInterval( this.timerInterval);
                };
            }
        }
    }

    ngOnDestroy() {
        this.clearTimerInterval(); // Clear the timer on component destruction
    }

    // Clear the timer interval if it is currently running
    private clearTimerInterval() {
        if (this.timerInterval) {
            clearInterval(this.timerInterval);
            this.timerInterval = null;
        }
    }

    calculateDisplayValue() {
        // Check if 'this.days' is defined

        if (this.days != undefined) {
            // If the absolute value of 'this.days' is less than 1, display hours, minutes, and seconds

            if (Math.abs(this.days) < 1) {
                // If 'this.days' is greater than or equal to 1, display the number of days
                // Use a ternary operator to determine whether to display "Day" or "Days" based on the sign of 'this.days'
                if(this.hours !== 0){
                return `<p class='time'>${this.hours}h ${this.minutes}m ${this.seconds}s</p><p class='date'> ${this.endDate}</p>`;
                }
                else{
                      return `<p class='time'>${this.minutes}m ${this.seconds}s</p><p class='date'> ${this.endDate}</p>`;
                }
            } else {
                if(this.days < 1)
                {
                     return `${this.days} days <br>${this.endDate}`;
                }
                else{
                     return ` ${this.days} days left <br>${this.endDate}`;
                }
            }
        } else {
            // If 'this.days' is not defined, return 'Not defined'
            return 'Not defined';
        }
    }

  calculateEndDate(endMilliSeconds,minutes:number){
    const currentDate = new Date(endMilliSeconds + minutes*60*1000);
    const year = currentDate.getFullYear();
    const month = (currentDate.getMonth() + 1).toString().padStart(2, '0'); // Months are 0-indexed, so add 1
    const day = currentDate.getDate().toString().padStart(2, '0');
    const min =  currentDate.getMinutes().toString().padStart(2, '0');
    let hours :any=  currentDate.getHours();
        let ampm = 'am';

    if (hours >= 12) {
        ampm = 'pm';
        hours = hours % 12;
    }

    hours = hours === 0 ? 12 : hours; // if hours is 0, set it to 12

    // Format the date as a string in "DD-MM-YY" format
    const formattedDate = `${day}-${month}-${year} (${hours}:${min} ${ampm})`
    return formattedDate;
        }
    }
