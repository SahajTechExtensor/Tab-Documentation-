import { Component, ElementRef, Input, OnDestroy, OnInit, ViewChild } from '@angular/core';
import { DomSanitizer, SafeResourceUrl } from '@angular/platform-browser';
import { TabGetService } from '@techextensor/tab-service';
import { cloneDeep } from 'lodash';
import { Subject, debounceTime, takeUntil } from 'rxjs';
import videojs from 'video.js';

@Component({
	selector: 'app-tab-file-preview',
	templateUrl: './tab-file-preview.component.html',
	styleUrls: ['./tab-file-preview.component.scss']
})
export class TabFilePreviewComponent implements OnInit, OnDestroy {

	@ViewChild('videoPlayer', { static: true }) videoPlayerRef: ElementRef;

	componentValue: any;

	debounceSubject = new Subject();
	_unsubscribeAll = new Subject<any>();

	@Input() dataSettingFilePreview: any;
	@Input() fileData: {
		URL: SafeResourceUrl | any,
		FileType: number
	} | null = null;

	isLoading = false;
	player: videojs.Player
	innerWidth: number;
	innerHeight: number;

	@Input()
	public set value(v: any) {
		if (v && v != null) {
			this.componentValue = v;
		}
		this.debounceSubject.next(1);
	}

	public get value(): any {
		return this.componentValue;
	}

	constructor(private tabGetService: TabGetService, private sanitizer: DomSanitizer) { }

	ngOnInit(): void {
		this.isLoading = true;
		this.innerWidth = window.innerWidth;
		this.innerHeight = window.innerHeight;
		this.debounceSubject
			.pipe(debounceTime(1000))
			.pipe(takeUntil(this._unsubscribeAll)).subscribe(() => {
				if (!this.componentValue) {
					this.componentValue = cloneDeep(this.dataSettingFilePreview);
				}
				if (this.componentValue) {
					if (this.componentValue.dataSource == 'mediaId') {
						this.getMediaInfo()
					} else if (this.componentValue.dataSource == 'url') {
						this.updateFileInfo();
					}
				}

			});
	}

	/**
	 * Retrieves media information from an API and updates the component's fileData.
	 */
	getMediaInfo() {
		const apiData = {
			AppObjectName: 'MediaInfo',
			DSQName: 'Detail_MediaInfo',
			Reqtokens: {
				Id: this.componentValue.mediaId,
			},
		};
		return this.tabGetService
			.executeRAWDSQWithName(apiData)
			.subscribe(res => {
				this.isLoading = false;
				if (res.Result?.length > 0) {
					this.fileData = res.Result[0];
					this.fileData.URL = this.sanitizer.bypassSecurityTrustResourceUrl(this.fileData.URL) as string;
					if (this.fileData.FileType == 6) {
						this.initVideoPlayer();
					}
				}
			}, (err) => {
				this.isLoading = false;
				console.log('error while getting media info', err);
			});
	}

	/**
	 * Initializes a Video.js video player.
	 */
	initVideoPlayer() {
		this.videoPlayerRef.nativeElement.style.visibility = 'visible';
		this.videoPlayerRef.nativeElement.style.height = 'calc(100vh - 291px)';
		this.videoPlayerRef.nativeElement.style.width = '100%';
		const videoPlayerElement: HTMLDivElement = this.videoPlayerRef.nativeElement;
		this.player = videojs(videoPlayerElement, {
			autoplay: true,
			controls: true,
			sources: [
				{
					src: this.fileData.URL?.changingThisBreaksApplicationSecurity ?? this.fileData.URL,
					type: 'video/mp4',
				}
			]
		});
	}

	/**
	 * Updates file information in the component.
	 */
	updateFileInfo() {
		this.isLoading = false;
		this.fileData = {
			URL: this.sanitizer.bypassSecurityTrustResourceUrl(this.componentValue.url),
			FileType: this.componentValue.fileType ?? 0
		};
		if (this.fileData.FileType == 6) {
			this.initVideoPlayer();
		}
	}

	ngOnDestroy(): void {
		if (this.player){
			this.player.dispose();
		}
		this._unsubscribeAll.next(null);
		this._unsubscribeAll.complete();
	}

}
