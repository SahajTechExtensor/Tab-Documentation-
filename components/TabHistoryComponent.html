import { Component, EventEmitter, Input, OnDestroy, OnInit, Output } from '@angular/core';
import { Subject, debounceTime, take, takeUntil } from 'rxjs';
import { FieldRecord, HistoryRecord } from './model/record.model';
import { TabAppFunction, TabGetService } from '@techextensor/tab-service';
import { DatePipe } from '@angular/common';

@Component({
    selector: 'app-tab-history',
    templateUrl: './tab-history.component.html',
    styleUrls: ['./tab-history.component.scss'],
})
export class TabHistoryComponent implements OnInit,OnDestroy {
    // FormIo
    @Input() dataSettingsHistory: any;
    @Output() valueChange = new EventEmitter<any>();

    // History Component
    componentValue: any;
    private debounceSubject = new Subject();
    historyRecords: HistoryRecord[] = [];
    historyRecordsByDate: { [key: string]: HistoryRecord[] } = {};
    expandedRecords: HistoryRecord[] = [];
    private _unsubscribeAll: Subject<any> = new Subject<any>();
    isLoading = true;

    constructor(private tabGetService: TabGetService,private dataPipe: DatePipe) {}

    @Input()
    public set value(v: any) {
        if (v && v != null) {
            this.componentValue = v;
            this.debounceSubject.next(1000);
        }
    }

    ngOnInit(): void {
        this.isLoading = true;
        this.debounceSubject
        .pipe(debounceTime(1000),takeUntil(this._unsubscribeAll))
        .subscribe((res) => {
            if (
                this.componentValue.DataSourceQueryID &&
                this.componentValue.PrimaryKey
            ) {
                const primaryKey = [this.componentValue.PrimaryKey];
                const appObjectId =
                    TabAppFunction.getAppObjectFromDataSourceQueryID(
                        this.componentValue.DataSourceQueryID
                    )?.ID;
                this.tabGetService
                    .getHistory(appObjectId, primaryKey)
                    .pipe(take(1))
                    ?.subscribe((res) => {
                        this.isLoading = false;
                        this.historyRecords = res.Result;
                        this.historyRecordsByDate = this.groupByDate(
                            res.Result
                        );
                    });
            }
        });
    }

    /**
     * Retrieves the system fields from the given array of field records.
     *
     * @param {FieldRecord[]} fieldRecords - The array of field records.
     * @returns A string containing the system fields separated by commas.
     */
    getSystemFields(fieldRecords: FieldRecord[]): string {
        return fieldRecords.reduce((acc: string, crr) => {
            if (acc.length == 0) {
                acc = crr.SystemDBFieldName;
            } else {
                acc += ', ' + crr.SystemDBFieldName;
            }
            return acc;
        }, '');
    }

    /**
     * Checks if the given history record is expanded.
     *
     * @param {HistoryRecord} record - The history record to check.
     * @returns A boolean indicating whether the record is expanded or not.
     */
    isExpanded(record: HistoryRecord) {
        return this.expandedRecords.indexOf(record) != -1;
    }

    /**
     * Toggles the expand/collapse state for the given history record.
     *
     * @param {HistoryRecord} record - The history record to expand or collapse.
     */
    expandCollapse(record: HistoryRecord) {
        const index = this.expandedRecords.indexOf(record);
        if (index != -1) {
            this.expandedRecords.splice(index, 1);
        } else this.expandedRecords.push(record);
    }

    /**
     * Groups the given array of history records by date and returns the sorted grouped data.
     *
     * @param {HistoryRecord[]} records - An array of history records to be grouped by date.
     * @returns  An object containing the grouped and sorted history records, where the keys represent the dates and the values represent the corresponding records.
     */
    groupByDate(records: HistoryRecord[]) {
        const groupedData = records.reduce(
            (result: { [key: string]: HistoryRecord[] }, item: any) => {
                const date = item.OP_Datetime.split(" ")[0];
                if (!result[date]) {
                    result[date] = [];
                }
                result[date].push(item);
                return result;
            },
            {}
        );
        let sortedData = Object.entries(groupedData)
            .sort(([dateA], [dateB]) => {
                const dateObjA = new Date(dateA);
                const dateObjB = new Date(dateB);
                return dateObjA.getTime() - dateObjB.getTime();
            })
            .reduce(
                (
                    result: { [key: string]: HistoryRecord[] },
                    [date, values]
                ) => {
                    result[date] = values.sort((a, b) => {
                        const recordADate = new Date(a.OP_Datetime);
                        const recordBDate = new Date(b.OP_Datetime);
                        return recordBDate.getTime() - recordADate.getTime();
                    });
                    return result;
                },
                {}
            );

        return sortedData;
    }

    ngOnDestroy(): void {
        // Unsubscribe from all subscriptions
        this._unsubscribeAll.next(null);
        this._unsubscribeAll.complete();
    }
}
