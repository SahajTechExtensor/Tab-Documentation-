@Component({
    selector: 'app-tab-blueprint',
    templateUrl: './tab-blueprint.component.html',
    styleUrls: ['./tab-blueprint.component.scss'],
})
export class TabBlueprintComponent
    implements OnInit, FormioCustomComponent<any>, AfterViewInit, OnDestroy
{
    // Form.io
    @Input() dataSettingsBlueprint: BlueprintSettings;
    @Input() disabled: boolean;
    @Output() valueChange = new EventEmitter<any>();
    @Output() formioEvent = new EventEmitter<FormioEvent>();
    @Input()
    componentValue = {
        BluePrintId: null,
        blueprint: '', // diagram json
        changed: '', // which field changed on event
        isRefresh: false,
    };
    private debounceSubject = new Subject();

    // syncfusion diagrams
    @ViewChild('diagram', { static: false }) diagram: DiagramComponent;
    @ViewChild('symbolpalette', { static: false })
    symbolpalette: SymbolPaletteComponent;
    selectedElements: (ConnectorModel | NodeModel)[]; // tracking selected element for deleting functionality
    diagramConstraints =
        DiagramConstraints.Default |
        DiagramConstraints.Bridging |
        DiagramConstraints.LineRouting;
    public diagramTool: DiagramTools = DiagramTools.ZoomPan;
    // diagram config fields
    showPalettes = true;
    // automatic layout to draw diagram
    @Input()
    layout: LayoutModel = {
        type: 'ComplexHierarchicalTree',
        enableRouting: true,
        enableAnimation: true,
        orientation: 'TopToBottom',
        verticalAlignment: 'Center',
        horizontalAlignment: 'Center',
        horizontalSpacing: 40,
        verticalSpacing: 55,
        margin: { top: 30, left: 30, right: 30, bottom: 0 },
    };

    snapSettings: SnapSettingsModel = {
        constraints: SnapConstraints.None,
    };

    //SymbolPalette Properties
    expandMode: ExpandMode = 'Multiple';
    symbolHeight = 40;
    symbolWidth = 90;
    symbolMargin: MarginModel = {
        left: 5,
        right: 5,
        top: 5,
        bottom: 5,
    };
    palettes: PaletteModel[] = [];

    // Blueprint details
    @Input() statuses: Status[] = [];
    @Input() transitions: Transition[] = [];
    private setTimer: any;
    private _unsubscribeAll: Subject<any> = new Subject<any>();

    constructor(
        private tabGetService: TabGetService,
        private datePipe: DatePipe
    ) {}

    ngOnInit(): void {
        /**
         * Observing formIo value changes
         */
        this.debounceSubject
            .pipe(debounceTime(1000))
            .pipe(takeUntil(this._unsubscribeAll))
            .subscribe(() => {
                if (this.componentValue?.isRefresh) {
                    forkJoin({
                        statuses: this.getStatusObservable(),
                        transitions: this.getTransitionsObservable(),
                    })
                        .pipe(take(1))
                        .subscribe(({ statuses, transitions }) => {
                            this.statuses = statuses;
                            this.transitions = transitions;
                            this.updatePalette();
                            this.updateDiagram();
                        });
                }
            });
    }

    @Input()
    public set value(v: any) {
        if (v && v != null) {
            this.componentValue = v;
        }
        this.updatePalette();
        this.debounceSubject.next(1000);
    }

    public get value(): any {
        return this.componentValue;
    }

    /**
     * configuring palette base on diagram settings
     * @returns {void}
     */
    updatePalette(): void {
        const symbols: NodeModel[] = this.statuses.map(this.getNodeModel);
        this.palettes = [
            {
                id: 'status',
                expanded: true,
                symbols: symbols,
                iconCss: 'fa fa-bars',
                title: 'Available Status',
            },
        ];
    }

    /**
     * Updates the diagram based on the changes in the formIoValue.
     * If the 'changed' property in formIoValue is not 'diagram', no refresh is performed.
     *
     * @returns {void}
     */
    updateDiagram = (): void => {
        // Only refresh if anything updated on diagram
        if (this.componentValue?.changed != 'diagram') return;

        const nodeList: {
            [id: string]: Status;
        } = {};

        for (let i = 0; i < this.transitions.length; i++) {
            const transition = this.transitions[i];

            const fromStatus = this.statuses.find(
                (s) => s.Id == transition.FromStatus
            );
            const toStatus = this.statuses.find(
                (s) => s.Id == transition.ToStatus
            );

            if (!nodeList[transition.FromStatus] && fromStatus) {
                nodeList[transition.FromStatus] = fromStatus;
            }
            if (!nodeList[transition.ToStatus] && toStatus) {
                nodeList[transition.ToStatus] = {
                    ...toStatus,
                    ParentId: transition.FromStatus,
                };
            } else {
                nodeList[transition.ToStatus]['ParentId'] =
                    transition.FromStatus;
            }
        }
        this.diagram.dataSourceSettings = {
            id: 'Id',
            parentId: 'ParentId',
            dataManager: new DataManager({
                json: Object.values(nodeList),
            }),
            doBinding: (model: NodeModel, data: Status, diagram: Diagram) => {
                // FOR REFERENCE
                // model = { ...model, ...this.getNodeModel(data) };

                // Binding for nodes (status)'
                model.id = data.Id;
                model.shape = {
                    type: 'Text',
                    content: data.Name,
                };
                model.width = this.symbolWidth;
                model.height = this.symbolHeight;
                if (
                    data.Color === undefined ||
                    data.Color === null ||
                    data.Color === ''
                ) {
                    model.borderColor = '#757575';
                } else {
                    model.borderColor = data.Color;
                }
                model.borderWidth =
                    this.dataSettingsBlueprint.nodeBorderWidth ?? 2;
                model.style = {
                    strokeColor: model.borderColor,
                    strokeWidth: 1,
                };
                // model.ports = this.getPorts(data.Color);
                // model.constraints = this.getNodeConstraints();
            },
        } as any;

        // FOR DELETE FUNCTIONALITY
        // Object.keys(nodeList).forEach((k) => {
        //     this.symbolpalette.removePaletteItem('status', k);
        // });
    };

    /**
     * Retrieves the status data as an observable array.
     * If `BluePrintId` is not available in `formIoValue`, it updates the palette and returns an empty array.
     *
     * @returns An observable emitting an array of status data.
     */
    getStatusObservable(): Observable<any[]> {
        if (!this.componentValue.BluePrintId) return of([]);

        const apiData = {
            AppObjectName: 'TABD_BlueprintStatus',
            DSQName: 'List_TABD_BlueprintStatus',
            Reqtokens: {
                // BluePrintId: '67121140-70CB-4532-87A9-69B9632BC4E6',
                BluePrintId: this.componentValue.BluePrintId,
            },
        };
        return this.tabGetService
            .executeRAWDSQWithName(apiData)
            .pipe(map((res: any) => res?.Result as any[]));
    }

    /**
     * Retrieves the transaction data as an observable array.
     * If `BluePrintId` is not available in `formIoValue`, it returns an empty array.
     *
     * @returns An observable emitting an array of transaction data.
     */
    getTransitionsObservable() {
        if (!this.componentValue.BluePrintId) return of([]);

        const apiData = {
            AppObjectName: 'TABD_Transitions',
            DSQName: 'List_TABD_Transitions',
            Reqtokens: {
                // BluePrintId: '67121140-70CB-4532-87A9-69B9632BC4E6',
                BluePrintId: this.componentValue.BluePrintId,
            },
        };
        return this.tabGetService
            .executeRAWDSQWithName(apiData)
            .pipe(map((res: any) => res?.Result));
    }

    /**
     * Creates a node model based on the provided status.
     *
     * @param {Status} status - The status object used to create the node model.
     * @returns {NodeModel} The generated node model.
     */
    getNodeModel = (status: Status): NodeModel => {
        return {
            id: status?.Id ?? status?.Name,
            width: this.symbolWidth,
            height: this.symbolHeight,
            shape: {
                type: 'Text',
                content: status.Name,
            },
            borderColor:
                this.dataSettingsBlueprint.nodeBorderColorType == 'default'
                    ? status?.Color
                    : this.dataSettingsBlueprint.nodeBorderColor,
            borderWidth: this.dataSettingsBlueprint.nodeBorderWidth ?? 2,
            style: {
                strokeWidth: 1,
                strokeColor:
                    this.dataSettingsBlueprint.nodeBorderColorType == 'default'
                        ? status?.Color
                        : this.dataSettingsBlueprint.nodeBorderColor,
            },
            ports: this.getPorts(status?.Color),
            constraints: this.getNodeConstraints(),
        };
    };

    /**
     * Retrieves the ports based on the specified color and node port settings.
     *
     * @param {string} color - The color used for the ports.
     * @returns {PointPortModel[]} - An array of PointPortModel.
     */
    getPorts(color: string): PointPortModel[] {
        const ports: PointPortModel[] = [];
        const nodePorts = this.dataSettingsBlueprint?.nodePorts;
        const portStyle = this.getPortStyle(color);
        if (nodePorts?.leftCenter) {
            ports.push({
                offset: { x: 0, y: 0.5 },
                ...portStyle,
            });
        }
        if (nodePorts?.topCenter) {
            ports.push({
                offset: { x: 0.5, y: 0 },
                ...portStyle,
            });
        }
        if (nodePorts?.rightCenter) {
            ports.push({
                offset: { x: 1, y: 0.5 },
                ...portStyle,
            });
        }
        if (nodePorts?.bottomCenter) {
            ports.push({
                offset: { x: 0.5, y: 1 },
                ...portStyle,
            });
        }

        return ports;
    }

    /**
     * Retrieves the port style base on specified color
     *
     * @param {string} color  The color used for ports
     * @returns {PointPortModel} port style
     */
    getPortStyle(color: string): PointPortModel {
        return {
            visibility: PortVisibility.Connect | PortVisibility.Hover,
            constraints: PortConstraints.Draw,
            style: {
                fill:
                    this.dataSettingsBlueprint.nodeBorderColorType === 'default'
                        ? color
                        : this.dataSettingsBlueprint.nodeBorderColor,
                strokeColor:
                    this.dataSettingsBlueprint.nodeBorderColorType === 'default'
                        ? color
                        : this.dataSettingsBlueprint.nodeBorderColor,
            },
            width: 8,
            height: 8,
            shape: 'Circle',
        };
    }

    /**
     * Retrieves node constraints
     *
     * @returns {NodeConstraints} NodeConstraints
     */
    getNodeConstraints(): NodeConstraints {
        let constraints: NodeConstraints;
        constraints =
            (NodeConstraints.Default &
                ~NodeConstraints.Resize &
                ~NodeConstraints.Rotate) |
            NodeConstraints.Select |
            NodeConstraints.Shadow;
        // Object.keys(this.dataSettingsBlueprint.nodeConstraints).forEach(
        //     (constraint) => {
        //         if (this.dataSettingsBlueprint.nodeConstraints[constraint]) {
        //             constraints =
        //                 constraints &
        //                 NodeConstraints[constraint as keyof NodeConstraints];
        //         }
        //     }
        // );
        return constraints;
    }

    /**
     * Retrieves connector constraints
     *
     * @returns {ConnectorConstraints} ConnectorConstraints
     */
    getConnectorConstraints(): ConnectorConstraints {
        let constraints: ConnectorConstraints;

    toSegment(str: string): Segments {
        switch (str) {
            case 'straight' || 'Straight':
                return 'Straight';
            case 'orthogonal' || 'Orthogonal':
                return 'Orthogonal';
            case 'polyline' || 'Polyline':
                return 'Polyline';
            case 'bezier' || 'Bezier':
                return 'Bezier';
            case 'freehand' || 'Freehand':
                return 'Freehand';
            default:
                throw new Error(`Invalid segment type: ${str}`);
        }
    }

     *
     * @param {string} action - The action to be performed.
     * @param {any} element - The element associated with the action.
     */
    callCustomEventAction(action: string, element: any) {
        let data: any = {
            action: action,
        };
        // this.showPalettes = true;
        if (element instanceof Connector) {
            // this.showPalettes = false;
            const source = this.statuses.find(
                (s) => s.Name == (element.sourceWrapper as TextElement)?.content
            );
            const target = this.statuses.find(
                (s) => s.Name == (element.targetWrapper as TextElement)?.content
            );
            data.connector = element;
            data.fromStatus = source?.Id;
            data.toStatus = target?.Id;
            data.transitionId = this.transitions?.find(
                (t) => t.ToStatus == target?.Id && t.FromStatus == source?.Id
            )?.Id;
        }
        if (element instanceof Node) {
            data.node = element;
            const status = this.statuses.find(
                (s) =>
                    s.Name ==
                    (element as any).properties?.shape?.properties?.content
            );
            data.status = status;
        }
        this.formioEvent.emit({
            eventName: 'customEvent',
            data: data,
        });
    }

    convertMinutes(minutes: number | number): string {
        if (minutes) {
            const isNegative = minutes < 0;
            minutes = Math.abs(minutes);

            const weeks = Math.floor(minutes / (60 * 24 * 7));
            const days = Math.floor((minutes % (60 * 24 * 7)) / (60 * 24));
            const hours = Math.floor((minutes % (60 * 24)) / 60);
            const remainingMinutes = Math.floor(minutes % 60);
            const timeUnits: string[] = [];
            if (weeks > 0) {
                timeUnits.push(`${weeks}W`);
            }
            if (days > 0) {
                timeUnits.push(`${days}D`);
            }
            if (hours > 0) {
                timeUnits.push(`${hours}H`);
            }
            if (remainingMinutes > 0) {
                timeUnits.push(
                    `${remainingMinutes} Min${remainingMinutes > 1 ? 's' : ''}`
                );
            }
            const result = timeUnits.join(', ');
            return isNegative ? `-${result}` : result;
        }
        return '';
    }
