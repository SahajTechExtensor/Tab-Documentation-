import {
    GlobalValueService,
    TabDeleteService,
    TabGetService,
    TabMediaUploadService,
} from '@techextensor/tab-service';
import { Component, OnInit, ViewChild } from '@angular/core';
import { Title } from '@angular/platform-browser';
import { TabUtilsEvent } from './tab-utils-events';
import { MatDialog } from '@angular/material/dialog';
import { TabDynamicSidebarComponent } from './shared/templates/tab-dynamic-sidebar/tab-dynamic-sidebar.component';
import { OnlineStatusService } from './shared/services/online-status.service';
import { ToastrNotificationService } from './layout/common/notifications/toastr.service';
import { Router } from '@angular/router';
import { MatSnackBar } from '@angular/material/snack-bar';

@Component({
    selector: 'app-root',
    templateUrl: './app.component.html',
    styleUrls: ['./app.component.scss'],
    host: { '(window:tabUtilsEvent)': 'tabUtilsEventListener($event)' },
})
export class AppComponent implements OnInit {
    alertType: string = '';
    alertHeading: string = '';
    alertMessage: string = '';
    menuLogo: any;
    favIcon: HTMLLinkElement = document.querySelector('#appIcon');
    OrganizationInfo: any;
    OrgData: any;

    @ViewChild('tabDynamicSidebarComponent')
    tabDynamicSidebarComponent: TabDynamicSidebarComponent;

    /**
     * Constructor
     */
    constructor(
        private tabGetService: TabGetService,
        private title: Title,
        private tabMediaUploadService: TabMediaUploadService,
        private dialog: MatDialog,
        public onlineStatus: OnlineStatusService,
        private tabDeleteService: TabDeleteService,
        private toastrNotificationService: ToastrNotificationService,
        private _route: Router,
        private SnackBar: MatSnackBar,
        private globalValueService: GlobalValueService
    ) {
        const apiData = {
            AppObjectName: 'TABMD_Configurations',
            DSQName: 'Detail_TABMD_Configurations',
            Reqtokens: { ID: '10052fd2-45c6-47ec-8fe9-fe0bda28830a' },
        };
        this.tabGetService
            .executeRAWDSQWithName(apiData)
            ?.subscribe((response) => {
                if (!!response.Result && response.Result.length > 0) {
                    let text = response.Result[0].Setting_Value;
                    // var css = 'h1 { background: red; }',
                    let head =
                        document.head ||
                        document.getElementsByTagName('head')[0];
                    let style = document.createElement('style');

                    head.appendChild(style);

                    // style.type = 'text/css';
                    // if (style.styleSheet) {
                    //     // This is required for IE8 and below.
                    //     style.styleSheet.cssText = css;
                    // } else {
                    //     style.appendChild(document.createTextNode(css));
                    // }
                    style.appendChild(document.createTextNode(text));

                    // $(document).ready(function () {
                    //     $('style').append(text);
                    // });
                }
            });
    }

    ngOnInit() {
        this.OrganizationInfo = JSON.parse(localStorage.getItem('orgInfo'));
        if (this.OrganizationInfo) {
            this.OrgData = JSON.parse(this.OrganizationInfo?.LocaleSettings);
            if (this.OrgData != null) {
                const logo = this.OrgData.FavIcon;
                this.favIcon.href = logo;
            } else {
                this.favIcon.href = 'favicon-16x16.png';
            }
            const title = this.OrganizationInfo.OrganizationName;
            this.title.setTitle(title);
        }
    }

    tabUtilsEventListener(tabUtilsEvent) {
        console.log(
            'tabUtilsEventListener called from AppComponent',
            tabUtilsEvent
        );

        let eventParams: any = {};
        eventParams.event = tabUtilsEvent;
        eventParams.dialogRef = this.dialog;
        eventParams.sidebarComponent = this.tabDynamicSidebarComponent;
        eventParams.serviceList = {
            mediaUploadService: this.tabMediaUploadService,
            deleteService: this.tabDeleteService,
            getService: this.tabGetService,
            toastrService: this.toastrNotificationService,
            _route: this._route,
            snackBar: this.SnackBar,
            globalValueService: this.globalValueService,
        };
        new TabUtilsEvent()[tabUtilsEvent.detail.eventName](eventParams);
    }
}
