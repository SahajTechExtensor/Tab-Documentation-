import {
    Component,
    ViewChild,
    NgZone,
    EventEmitter,
    OnInit,
    ElementRef,
    Output,
    Input,
} from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { } from 'googlemaps';
import { MapdialogComponent } from './mapdialog/mapdialog.component';

@Component({
    selector: 'app-tab-custom-place-picker-map',
    templateUrl: './tab-custom-place-picker-map.component.html',
    styleUrls: ['./tab-custom-place-picker-map.component.scss'],
})
export class TabCustomPlacePickerMapComponent implements OnInit {
    @Output() valueChange = new EventEmitter<any>();
    establishmentAddress: Object;
    formattedEstablishmentAddress: any;
    phone: string;
    map: google.maps.Map;
    lat: number = 23.051839;
    lng: number = 72.481382;
    @Input()
    dataSettingsMap;
    @ViewChild('addresstext', { static: true })
    addresstext: ElementRef;
    @ViewChild('map', { static: true })
    mapElement: ElementRef;
    isTyping: boolean = false;

    @Input()
    public set value(v: any) {
        if (v && v != null) {
            this.formattedEstablishmentAddress = v;
            this.setaddress();
        }
    }
    public get value(): any {
        return this.formattedEstablishmentAddress;
    }

    constructor(public zone: NgZone, public dialog: MatDialog) { }

    ngOnInit(): void {
        this.mapInitialize();
        this.getPlaceAutocomplete();
    }

    mapInitialize() {
        const mapProperties = {
            center: new google.maps.LatLng(this.lat, this.lng),
            zoom: 15,
            mapTypeId: google.maps.MapTypeId.ROADMAP,
        };
        this.map = new google.maps.Map(
            this.mapElement.nativeElement,
            mapProperties
        );
    }

    private getPlaceAutocomplete() {
        const autocomplete = new google.maps.places.Autocomplete(
            this.addresstext.nativeElement,
            {
                componentRestrictions: { country: 'IN' },
                types: [], // 'establishment' / 'address' / 'geocode'
            }
        );
        google.maps.event.addListener(autocomplete, 'place_changed', () => {
            const place = autocomplete.getPlace();
            var inputValue = this.addresstext.nativeElement;
            // inputValue.value = place['formatted_address'];
            this.getEstablishmentAddress(place);
        });
    }

    getEstablishmentAddress(place: object) {
        this.establishmentAddress = place['formatted_address'];
        this.getGeometry(place);
        var inputValue = this.addresstext.nativeElement;
        this.phone = this.getPhone(place);
        this.formattedEstablishmentAddress =
            // place['formatted_address'] +
            inputValue.value +
            ',' +
            'lat:' +
            +this.lat +
            ',' +
            'lng:' +
            this.lng;

        this.valueChange.emit(this.formattedEstablishmentAddress);
    }

    getAddrComponent(place, componentTemplate) {
        let result;

        for (let i = 0; i < place.address_components.length; i++) {
            const addressType = place.address_components[i].types[0];
            if (componentTemplate[addressType]) {
                result =
                    place.address_components[i][componentTemplate[addressType]];
                return result;
            }
        }
        return;
    }

    getGeometry(place) {
        const geometry = place.geometry.location;
        var glat = geometry.lat();
        var glng = geometry.lng();

        this.lat = geometry.lat();
        this.lng = geometry.lng();

        const mapProperties = {
            center: new google.maps.LatLng(glat, glng),
            zoom: 25,
            mapTypeId: google.maps.MapTypeId.ROADMAP,
        };

        if (this.dataSettingsMap?.renderMap) {
            this.map = new google.maps.Map(
                this.mapElement.nativeElement,
                mapProperties
            );

            var marker = new google.maps.Marker({
                position: { lat: glat, lng: glng },
                map: this.map,
                title: 'markers',
            });
        }
    }

    getStreetNumber(place) {
        const COMPONENT_TEMPLATE = { street_number: 'short_name' },
            streetNumber = this.getAddrComponent(place, COMPONENT_TEMPLATE);
        return streetNumber;
    }

    getStreet(place) {
        const COMPONENT_TEMPLATE = { route: 'long_name' },
            street = this.getAddrComponent(place, COMPONENT_TEMPLATE);
        return street;
    }

    getCity(place) {
        const COMPONENT_TEMPLATE = { locality: 'long_name' },
            city = this.getAddrComponent(place, COMPONENT_TEMPLATE);
        return city;
    }

    getState(place) {
        const COMPONENT_TEMPLATE = {
            administrative_area_level_1: 'short_name',
        },
            state = this.getAddrComponent(place, COMPONENT_TEMPLATE);
        return state;
    }

    getDistrict(place) {
        const COMPONENT_TEMPLATE = {
            administrative_area_level_2: 'short_name',
        },
            state = this.getAddrComponent(place, COMPONENT_TEMPLATE);
        return state;
    }

    getCountryShort(place) {
        const COMPONENT_TEMPLATE = { country: 'short_name' },
            countryShort = this.getAddrComponent(place, COMPONENT_TEMPLATE);
        return countryShort;
    }

    getCountry(place) {
        const COMPONENT_TEMPLATE = { country: 'long_name' },
            country = this.getAddrComponent(place, COMPONENT_TEMPLATE);
        return country;
    }

    getPostCode(place) {
        const COMPONENT_TEMPLATE = { postal_code: 'long_name' },
            postCode = this.getAddrComponent(place, COMPONENT_TEMPLATE);
        return postCode;
    }

    getPhone(place) {
        const COMPONENT_TEMPLATE = {
            formatted_phone_number: 'formatted_phone_number',
        },
            phone = this.getAddrComponent(place, COMPONENT_TEMPLATE);
        return phone;
    }

    setaddress() {
        let addressWithoutLatLng = this.formattedEstablishmentAddress.replace(
            /,lat:\d+\.\d+,lng:\d+\.\d+/,
            ''
        );
        var inputValue = this.addresstext.nativeElement;
        inputValue.value = addressWithoutLatLng;

        let autocompleteService = new google.maps.places.AutocompleteService();
        let request = {
            input: addressWithoutLatLng,
            // types: ['address'],
            types: [],
        };
        autocompleteService.getPlacePredictions(
            request,
            (predictionsArr, placesServiceStatus) => {
                if (predictionsArr && this.dataSettingsMap?.renderMap) {
                    let placeRequest = { placeId: predictionsArr[0].place_id };
                    let placeService = new google.maps.places.PlacesService(
                        this.map
                    );
                    placeService.getDetails(
                        placeRequest,
                        (placeResult, placeServiceStatus) => {
                            this.getGeometry(placeResult);
                        }
                    );
                }
            }
        );
    }

    directMap() {
        if (
            this.dataSettingsMap?.renderMap &&
            this.dataSettingsMap?.mode === 'direct'
        ) {
            return false;
        } else {
            return true;
        }
    }

    placeholder() {
        return this.dataSettingsMap?.placeholder
            ? this.dataSettingsMap?.placeholder
            : 'Enter a location';
    }

    mapButton() {
        if (
            this.dataSettingsMap?.renderMap &&
            this.dataSettingsMap?.mode === 'onClick'
        ) {
            return false;
        } else {
            return true;
        }
    }

    openDialog() {
        var height = parseInt(this.dataSettingsMap?.mapHeight) + 150 + 'px';
        var width = parseInt(this.dataSettingsMap?.mapWidth) + 50 + 'px';

        this.dialog.open(MapdialogComponent, {
            height: height ? height : '500px',
            width: width ? width : '500px',
            data: {
                height: this.dataSettingsMap?.mapHeight + 'px',
                width: this.dataSettingsMap?.mapWidth + 'px',
                lat: this.lat,
                lng: this.lng,
                title: this.dataSettingsMap?.title,
            },
        });
    }

    // show and hide tooltip
    onInput(): void {
        this.isTyping = true;
        var inputValue = this.addresstext.nativeElement;
        // when input is blank we need to update the value of formattedEstablishmentAddress
        if (inputValue.value === "") {
            this.formattedEstablishmentAddress = "";
            this.valueChange.emit(this.formattedEstablishmentAddress);

        }
    }

    onBlur(): void {
        this.isTyping = false;
    }
}
