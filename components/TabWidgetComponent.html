import { Component, Input, OnInit } from '@angular/core';
import {
    TabAppFunction,
    TabGetService,
    TabTemplateService,
} from '@techextensor/tab-service';
import { cloneDeep } from 'lodash';
import { Subject, debounceTime, forkJoin, of, take } from 'rxjs';
@Component({
    selector: 'app-tab-widget',
    templateUrl: './tab-widget.component.html',
    styleUrls: ['./tab-widget.component.css'],
})
export class TabWidgetComponent implements OnInit {
    @Input() dataSetting: any;
    @Input() data: any;
    dsqDetails: any;
    cloneDsqDetails: any;
    appObject: any;
    private setTimer: any;

    widgetInstance: any = {};
    debounceSubject = new Subject();
    _unsubscribeAll = new Subject<any>();

    constructor(
        private tabTemplateService: TabTemplateService,
        private tabGetService: TabGetService
    ) { }

    @Input()
    public set value(v: any) {
        if (v && v != null) {
            this.widgetInstance = v;
            // this.data = v;
        }
        this.debounceSubject.next(1);
    }
    public get value(): any {
        return this.widgetInstance;
        // return this.data;
    }

    ngOnInit(): void {
        // var token: any;
        // this.activatedRoute.queryParams.subscribe((qparams: Params) => {
        //     token = qparams;
        // });
        // setTimeout(() => {
        //     if (this.dataSetting) {
        //         this.populateData(token);
        //     }
        // }, 0);
        this.debounceSubject.pipe(debounceTime(1000)).subscribe((_) => {
            if (this.dataSetting && Object.keys(this.dataSetting).length) {
                this.populateData();
            }
        })
    }

    async populateData() {
        let apiData = {};
        if (this.dataSetting?.dataSourceMode == 'data') {
            apiData['Templatevariables'] = this.dataSetting?.Templatevariables;
        } else if (this.dataSetting?.dataSourceMode == 'dsq') {
            if (this.dataSetting?.dataSourceQueryId) {
                apiData['DataSourceQueryID'] =
                    this.dataSetting?.dataSourceQueryId;
                // For REFERENCE
                // Stop execution when there is no parameters in DSQ mode
                if (
                    !this.dataSetting?.parameters ||
                    Object.keys(this.dataSetting?.parameters ?? {}).length == 0
                )
                    return;
                apiData['Reqtokens'] = this.dataSetting?.parameters;
            }
        }
        if (this.dataSetting?.templateMode == 'customHtml') {
            apiData['DataSourceMode'] = 2;
            apiData['Value'] = await this.processTemplate(
                this.dataSetting?.htmlCode
            );
        } else if (this.dataSetting?.templateMode == 'template') {
            apiData['DataSourceMode'] = 1;
            apiData['Value'] = this.dataSetting?.template;
        }
        // this.appObject = TabAppFunction.getAppObjectFromDataSourceQueryID(
        //     this.dataSetting?.dataSourceQueryId
        // );
        this.dsqDetails = TabAppFunction.getDataFromDataSourceQueryID(
            this.dataSetting?.dataSourceQueryId
        );

        this.cloneDsqDetails = cloneDeep(this.dsqDetails);

        // apiData = {
        //     DataSourceMode: 2,
        //     DataSourceQueryID: this.dataSetting?.dataSourceQueryId,
        //     Value: this.dataSetting?.htmlCode,
        //     Reqtokens: { Id: '78a30b98-d512-4772-b070-0bc773add61e' },
        // };

        // apiData.Reqtokens = {};

        forkJoin({
            templateData: this.getTemplateObservable(),
            dsqData: this.getDSQObservable(),
        })
            .pipe(take(1))
            .subscribe(async ({ templateData, dsqData }) => {
                if (templateData?.Result?.length > 0) {
                    apiData['Value'] = await this.processTemplate(templateData.Result[0]?.TemplateBody);
                }
                if (dsqData?.Result) {
                    apiData['Templatevariables'] = JSON.stringify({
                        data: dsqData.Result,
                    });
                }
                this.tabTemplateService
                    .templateParseUniversal(apiData, true)
                    .pipe(take(1))
                    ?.subscribe((res: any) => {
                        this.data = res.Result;
                    });
            });
    }

    /**
     * Processes a template string by replacing `{{ loadTemplate() }}` 
     * placeholders with actual template content.
     * @param {string} templateString - The template string to process.
     * @returns {Promise<string>} The processed template string.
     */
    async processTemplate(templateString) {
        const regex = /{{\s*loadTemplate\((\d+)\)\s*}}/;
        const match = regex.exec(templateString);

        if (match) {
            let newTemplateString = '';
            try {
                const response = await this.getTemplateAsync(parseInt(match[1]));
                newTemplateString = templateString.replace(match[0], response.Result[0]?.TemplateBody);
            } catch (error) {
                newTemplateString = templateString.replace(match[0], `unable to load template ${match[1]}`);
                console.log('error while getting template', error);
            }
            return this.processTemplate(newTemplateString); // Recursive call with updated template
        }

        return templateString;
    }

    parseTemplate(apiData) {
        this.tabTemplateService
            .templateParseUniversal(apiData, true)
            .pipe(take(1))
            ?.subscribe((res: any) => {
                this.data = res.Result;
            });
    }

    getTemplateObservable() {
        if (this.dataSetting?.templateMode == 'customHtml') return of({});

        const data = {
            DSQName: 'Detail_TemplateInfo',
            AppObjectName: 'TemplateInfo',
            Reqtokens: {
                ID: this.dataSetting?.template,
            },
        };
        return this.tabGetService.executeRAWDSQWithName(data);
    }

    getTemplateAsync(Id: number) {
        const data = {
            DSQName: 'Detail_TemplateInfo',
            AppObjectName: 'TemplateInfo',
            Reqtokens: { 'ID': Id }
        };
        return new Promise<any>((resolve, reject) => {
            this.tabGetService.executeRAWDSQWithName(data)
                .pipe(take(1))
                .subscribe(res => {
                    const successCode = '200';
                    if (res?.Result?.length > 0 && res.StatusCode == successCode) {
                        resolve(res);
                    } else {
                        reject(res);
                    }
                },
                    err => reject(err.error));
        })
    }

    getDSQObservable() {
        if (this.dataSetting?.dataSourceMode == 'data') return of({});

        const appObject = TabAppFunction.getAppObjectFromDataSourceQueryID(
            this.dataSetting?.dataSourceQueryId
        );
        const dsqData = TabAppFunction.getDataFromDataSourceQueryID(
            this.dataSetting?.dataSourceQueryId
        );

        const apiData = {
            DSQName: dsqData.QueryName,
            AppObjectName: appObject.ObjectName,
        };
        if (this.dataSetting?.parameters)
            apiData['Reqtokens'] = this.dataSetting?.parameters;
        return this.tabGetService.executeRAWDSQWithName(apiData);
    }
    ngOnDestroy(): void {
        clearTimeout(this.setTimer);
    }
}
