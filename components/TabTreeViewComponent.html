import { ToastrNotificationService } from 'app/layout/common/notifications/toastr.service';
import {
    Component,
    EventEmitter,

    Output,
    ViewChild,
} from '@angular/core';
import { FormioCustomComponent, FormioEvent } from '@formio/angular';

import {
    BeforeOpenCloseMenuEventArgs,
    MenuEventArgs,
    MenuItemModel,
    NodeClickEventArgs,
    TreeViewComponent,
} from '@syncfusion/ej2-angular-navigations';
import { DataManager, Predicate, Query } from '@syncfusion/ej2-data';
import {
    GlobalValueService,
    TabAppFunction,
    TabClasses,
    TabDeleteService,
    TabGetService,
    TabInsertService,
    TabUpdateService,
} from '@techextensor/tab-service';
import { each, filter, find } from 'lodash';
import { FilteringEventArgs } from '@syncfusion/ej2-angular-dropdowns';
import { EmitType } from '@syncfusion/ej2-base';
import { Subject, takeUntil, take } from 'rxjs';

@Component({
    selector: 'tab-core-tab-tree-view',
    templateUrl: './tab-tree-view.component.html',
    styleUrls: ['./tab-tree-view.component.css'],
})
export class TabTreeViewComponent
    implements OnInit, FormioCustomComponent<any>
{
    @ViewChild('tabTreeView') tabTreeView: TreeViewComponent | any;
    @ViewChild('tabMaskObj') tabMaskObj: MaskedTextBoxComponent | any;
    @Input() disabled: boolean = false;
    // @Input() appObjectId: any;
    // @Input() parameters: any;
    // @Input() showCheckBox: any;
    // @Input() isGetFieldFromAppObject: any;
    @Input() dataSetting: any;
    @Output() valueChange = new EventEmitter<any>();
    @Output() formioEvent = new EventEmitter<FormioEvent>();

    dataSourceQuery: any;
    appObjectDetails: any;
    data: any[] = [];

    fields: any = {};
    fieldData: any[] = [];
    allFields: any;
    checkedNodes: any = [];
    dataSourceQueryID: any;
    dsqFields: any = [];
    displayTreeView = false;

    previousLookup: any = {};
    appObjectData: TabClasses.AppObject | undefined;
    treeViewInstance: any = {
        isReloadTreeView: true,
        selectedNode: {},
        parameters: {},
    };
    private setTimer: any;
    private _unsubscribeAll: Subject<any> = new Subject<any>();

    token: any;

    public menuItems: MenuItemModel[] = [
        { text: 'Add New Item' },
        { text: 'Rename Item' },
        { text: 'Remove Item' },
    ];

    constructor(
        private tabGetService: TabGetService,
        private tabUpdateService: TabUpdateService,
        private tabInsertService: TabInsertService,
        private tabDeleteService: TabDeleteService,
        private toastrNotificationService: ToastrNotificationService,
        private globalValueService: GlobalValueService
    ) {}

    _value: any;
    @Input()
    public set value(v: any) {
        // console.log('tree view seter', v);
        if (v && v != null) {
            this._value = v;
            this.treeViewInstance = this._value;
            if (
                this.dataSetting?.hasOwnProperty('allowAutoCheck') &&
                this.dataSetting?.allowAutoCheck
            ) {
                this.tabTreeView.autoCheck = true;
            } else {
                this.tabTreeView.autoCheck = false;
            }
            // console.log('treeViewInstance', this.treeViewInstance);
        }
        if (
            this.treeViewInstance?.isReloadTreeView &&
            !this.treeViewInstance?.fromJSON
        ) {
            this.setTimer = setTimeout(() => {
                this.populateData();
            }, 100);
        } else if (this.treeViewInstance?.fromJSON) {
            this.fieldData = this.treeViewInstance.fieldData;
            // this.checkedNodes = this.treeViewInstance.checkedNodes;
            this.fields = {
                dataSource: this.fieldData,
                id: 'id',
                text: 'name',
                child: 'child',
                parentID: 'parentID',
                hasChildren: 'hasChildren',
                selected: 'isSelected',
            };
            this.tabTreeView?.refresh();
            this.displayTreeView = true;
            this.treeViewInstance.treeViewValue = this.fields;
        }
    }
    public get value(): any {
        let abc = JSON.stringify(this.treeViewInstance);
        return JSON.parse(abc);
    }

    populateData() {
        if (this.dataSetting?.dataSourceQueryId) {
            this.dataSourceQuery = TabAppFunction.getDataFromDataSourceQueryID(
                this.dataSetting?.dataSourceQueryId
            );
            const mandatoryParams = this.checkMandatoryParamsInDSQ();
            if (mandatoryParams.length === 0) {
                let token = {};
                if (this.treeViewInstance?.parameters) {
                    token = this.treeViewInstance?.parameters;
                }
                this.executeDSQ(token);
            } else {
                if (
                    this.treeViewInstance?.parameters &&
                    Object.keys(this.treeViewInstance?.parameters).length
                ) {
                    let token = {};
                    token = this.treeViewInstance?.parameters;
                    if (this.treeViewInstance?.parameters)
                        this.executeDSQ(token);
                }
            }
        }
    }

    async executeDSQ(token) {
        // console.log('token', token);
        this.data = [];
        this.appObjectDetails =
            TabAppFunction.getAppObjectFromDataSourceQueryID(
                this.dataSetting?.dataSourceQueryId
            );

        let apiData = {
            AppObjectName: this.appObjectDetails.ObjectName,
            DSQName: this.dataSourceQuery.QueryName,
            Reqtokens: token,
        };
        this.tabGetService
            .executeRAWDSQWithName(apiData)
            .pipe(take(1))
            ?.subscribe((res: any) => {
                if (res?.Result) {
                    this.data = res.Result;
                    // console.log('data', this.data);
                    this.processTreeView();
                }
            });
    }

    nodeSelecting(event, data) {
        console.log('event-node selevted', event);
        // adding this condition when child reltion ship dsq dropdown added
        // it will help to prevent refesh when you click on dsq selection dropdown
        // it's kind of patch condition we need to make it proper in future
        // for now to dsq dropdown blink issue and menu click work will fine
        if (data && data.length) {
            this.treeViewInstance.isReloadTreeView = false;
            this.treeViewInstance.selectedNode = event.nodeData;
            this.value = this.treeViewInstance;
            let abc = JSON.stringify(this.treeViewInstance);
            this.valueChange.emit(JSON.parse(abc));
        }
    }

    public beforeopen(args: BeforeOpenCloseMenuEventArgs) {
        // console.log(this.tabTreeView.selectedNodes[0]);
    }

    public index: number = 1;
    public menuclick(args: MenuEventArgs) {
        let targetNodeId: string = this.tabTreeView.selectedNodes[0];
        if (args.item.text == 'Add New Item') {
            let nodeId: string = 'tree_' + this.index;
            let item: { [key: string]: Object } = {
                id: nodeId,
                name: '',
            };
            this.tabTreeView.addNodes([item], targetNodeId, null);
            this.index++;
            this.fieldData.push(item);
            this.tabTreeView.beginEdit(nodeId);
        } else if (args.item.text == 'Remove Item') {
            this.tabTreeView.removeNodes([targetNodeId]);
            this.removeNode(targetNodeId);
        } else if (args.item.text == 'Rename Item') {
            this.tabTreeView.beginEdit(targetNodeId);
        }
    }

    removeNode(targetNodeId) {
        let appObject: TabClasses.AppObject =
            TabAppFunction.getDataFromDataSourceQueryID(
                this.dataSetting?.dataSourceQueryId
            );
        if (appObject) {
            let data = { data: {} };
            data.data[this.dataSetting?.valueField] = targetNodeId;
            this.tabDeleteService
                .deleteRecord('', data, appObject.ObjectID_Tosave)
                .pipe(take(1))
                ?.subscribe((response) => {
                    if (response.StatusCode == '200') {
                        this.toastrNotificationService.showSuccess(
                            'Record Deleted Successfully',
                            'Success'
                        );
                    } else {
                        this.toastrNotificationService.showError(
                            response.Errors[0],
                            'Error'
                        );
                    }
                });
        }
    }

    selectFirstNode() {
        if (
            this.tabTreeView?.treeData &&
            this.tabTreeView?.treeData.length > 0
        ) {
            this.tabTreeView.selectedNodes = [
                this.tabTreeView?.treeData[0]?.id,
            ];
        }
    }

    updateNode(data, appObject) {
        this.tabUpdateService
            .updateUsingScreenId('', data, appObject.ObjectID_Tosave)
            .pipe(take(1))
            ?.subscribe((response) => {
                if (response.StatusCode == '200') {
                    this.toastrNotificationService.showSuccess(
                        'Record Updated Successfully',
                        'Success'
                    );
                } else {
                    this.toastrNotificationService.showError(
                        response.Errors[0],
                        'Error'
                    );
                }
            });
    }

    insertNode(data, appObject) {
        this.tabInsertService
            .insertByObjectId(appObject.ObjectID_Tosave, data)
            .pipe(take(1))
            ?.subscribe((response) => {
                if (response.StatusCode == '200' || !response.Errors.length) {
                    this.toastrNotificationService.showSuccess(
                        'Record Added Successfully',
                        'Success'
                    );
                    this.populateData();
                    return response.Result;
                } else {
                    this.toastrNotificationService.showError(
                        response.Errors[0],
                        'Error'
                    );
                }
            });
    }

    nodeEdited(event) {
        if (event.newText !== null && event.newText !== '') {
            let appObject: TabClasses.AppObject =
                TabAppFunction.getDataFromDataSourceQueryID(
                    this.dataSetting?.dataSourceQueryId
                );
            if (appObject) {
                let data = { data: {} };
                if (event.nodeData.id.toString().includes('tree_')) {
                    data.data[this.dataSetting?.displayField] = event.newText;
                    data.data[this.dataSetting?.groupByField] =
                        event.nodeData.parentID;
                    event.nodeData.id = this.insertNode(data, appObject);
                } else {
                    data.data[this.dataSetting?.valueField] = event.nodeData.id;
                    data.data[this.dataSetting?.displayField] = event.newText;
                    this.updateNode(data, appObject);
                }
            }
        } else {
            this.tabTreeView?.refresh();
        }
    }

    nodeDropped(event) {
        let appObject: TabClasses.AppObject =
            TabAppFunction.getDataFromDataSourceQueryID(
                this.dataSetting?.dataSourceQueryId
            );
        if (appObject) {
            let data = { data: {} };
            data.data[this.dataSetting?.valueField] = event.draggedNodeData.id;
            data.data[this.dataSetting?.sortOrderField] = event.dropIndex + 1;
            data.data[this.dataSetting?.groupByField] =
                event?.dropTarget?.dataset?.uid;
            this.updateNode(data, appObject);
        }
    }

    processTreeView() {
        this.fieldData = [];
        // console.log('before Sort', this.data);

        this.data.sort((a, b) =>
            a[this.dataSetting?.sortOrderField] !== null &&
            a[this.dataSetting?.sortOrderField] !== undefined &&
            b[this.dataSetting?.sortOrderField] !== null &&
            b[this.dataSetting?.sortOrderField] !== undefined &&
            a[this.dataSetting?.sortOrderField] >
                b[this.dataSetting?.sortOrderField]
                ? 1
                : -1
        );
        // console.log('after Sort', this.data);
        this.data.forEach((item) => {
            let fieldData = {
                id: item[this.dataSetting?.valueField],
                name: item[this.dataSetting?.displayField],
                expanded: false,
                hasChildren: this.data?.find(
                    (x) =>
                        x[this.dataSetting?.groupByField] ===
                        item[this.dataSetting?.valueField]
                )
                    ? true
                    : false,
                parentID: item[this.dataSetting?.groupByField],
                icon:
                    this.dataSetting?.displayWith &&
                    this.dataSetting?.displayWith === 'icon'
                        ? item['Icon']
                        : '',
            };
            this.fieldData.push(fieldData);
        });

        this.fields = {
            dataSource: this.fieldData,
            id: 'id',
            text: 'name',
            child: 'child',
            parentID: 'parentID',
            hasChildren: 'hasChildren',
            isChecked: 'isChecked',
        };
        this.displayTreeView = true;
        var proxy = this;
        this.setTimer = setTimeout(function (this) {
            proxy.tabTreeView.expandAll();
        }, 100);
    }

    checkMandatoryParamsInDSQ(): any[] {
        let result = [];
        if (this.dataSourceQuery) {
            // console.log('dataSourceQuery', this.dataSourceQuery);
            result = filter(this.dataSourceQuery?.Parameters, {
                IsMandatory: true,
            });
        }
        // console.log('result', result);
        return result;
    }

    ngOnInit(): void {
        // setTimeout(() => {
        //     if (this.parameters) {
        //         const isJsonString = TabCommonFunction.validateJSON(

    getLookUpFields(child: any, parentId: string, objectId: string) {
        let fds = this.allFields.filter(
            (field: any) =>
                field.AppObjectId.toLowerCase() == objectId.toLowerCase()
        );
        let fieldArray: any = [];
        if (fds && fds.length) {
            fds.forEach((element: any) => {
                if (
                    element.hasOwnProperty('LookUpDetails') &&
                    element.LookUpDetails
                ) {
                    let fjson = JSON.parse(element.LookUpDetails);
                    if (
                        child != 'first' &&
                        objectId.toLowerCase() ==
                            fjson.AppObjectId.toLowerCase()
                    ) {
                        fieldArray.push({
                            id: element.Id,
                            pid: parentId,
                            name: element.SystemDBFieldName,
                            expanded: false,
                        });
                    } else {
                        fieldArray.push({
                            id: element.Id,
                            pid: parentId,
                            name: element.SystemDBFieldName,
                            expanded: false,
                            hasChild: true,
                            // child: this.getLookUpFields('second', element.Id, fjson.AppObjectId)
                        });
                        fieldArray = fieldArray.concat(
                            this.getLookUpFields(
                                'second',
                                element.Id,
                                fjson.AppObjectId
                            )
                        );
                    }
                } else {
                    fieldArray.push({
                        id: element.Id,
                        pid: parentId,
                        name: element.SystemDBFieldName,
                    });
                }
            });
        }
        return fieldArray;
    }

    //Filtering the TreeNodes
    searchNodes() {
        let text = this.tabMaskObj.element.value;
        let predicats = [],
            array = [],
            filter = [];
        if (text == '') {
            this.changeDataSource(this.fieldData);
        } else {
            let predicate = new Predicate('name', 'contains', text, true);
            let filteredList = new DataManager(this.fieldData).executeLocal(
                new Query().where(predicate)
            );
            for (var j = 0; j < filteredList.length; j++) {
                filter.push(filteredList[j]['id']);
                var filters = this.getFilterItems(
                    filteredList[j],
                    this.fieldData
                );
                for (var i = 0; i < filters.length; i++) {
                    if (array.indexOf(filters[i]) == -1 && filters[i] != null) {
                        array.push(filters[i]);
                        predicats.push(
                            new Predicate('id', 'equal', filters[i], false)
                        );
                    }
                }
            }
            if (predicats.length == 0) {
                this.changeDataSource([]);
            } else {
                let query = new Query().where(Predicate.or(predicats));
                let newList = new DataManager(this.fieldData).executeLocal(
                    query
                );
                this.changeDataSource(newList);
                var proxy = this;
                this.setTimer = setTimeout(function (this) {
                    proxy.tabTreeView.expandAll();
                }, 100);
            }
        }
    }

    //Find the Parent Nodes for corresponding childs
    getFilterItems(fList, list) {
        let nodes = [];
        nodes.push(fList['id']);
        let query2 = new Query().where('id', 'equal', fList['parentID'], false);
        let fList1 = new DataManager(list).executeLocal(query2);
        if (fList1.length != 0) {
            let pNode = this.getFilterItems(fList1[0], list);
            for (var i = 0; i < pNode.length; i++) {
                if (nodes.indexOf(pNode[i]) == -1 && pNode[i] != null)
                    nodes.push(pNode[i]);
            }
            return nodes;
        }
        return nodes;
    }

    public onFiltering: EmitType<FilteringEventArgs> = (
        e: FilteringEventArgs
    ) => {
        let query: Query = new Query();

        query =

                : query;

        e.updateData(this.data, query);
