    /**
     * Fetches notifications using the `headlessService`. This function provides callbacks for handling
     * the state of the fetching process and errors, as well as for handling the fetched notifications.
     * It also updates the `processedNotificationData` property with the retrieved data and fetches the
     * updated unread notification count.
     *
     * @private
     * @returns {void}
     */
    private fetchNotifications(pageNumber: number) {
        // Get a reference to the headlessService.

        const headlessService = this.headlessService;

        if (headlessService) {
            // Call the fetchNotifications method of the headlessService.

            headlessService.fetchNotifications({
                listener: (result: FetchResult) => {
                    // Handle the state of the fetching process and errors here.
                    const {
                        data,
                        error,
                        isError,
                        isFetching,
                        isLoading,
                        status,
                    } = result;
                    // Handle these properties as needed
                },
                onSuccess: (response: any) => {
                    // Map the fetched data to a format suitable for the component's 'processedNotificationData' property.

                    if (response.data && response.data.length > 0) {
                        // Filter out notifications that are already loaded
                        const transformedNotifications = response.data.map(
                            (notification: any) => {
                                return {
                                    id: notification.id,
                                    content: notification.content,
                                    createdAt: notification.createdAt,
                                    updatedAt: notification.updatedAt,
                                    lastReadDate: notification.lastReadDate,
                                    lastSeenDate: notification.lastSeenDate,
                                    payload: notification.payload,
                                    read: notification.read,
                                    url: notification.cta?.data?.url, // Ensure nested properties are accessed safely
                                };
                            }
                        );

                        // Determine if there are more notifications based on the API response
                        this.hasMore = response.hasMore; // the API response has a property called 'hasMore' indicating whether there are more notifications.

                        // Update existing notifications if they exist, else add new ones
                        transformedNotifications.forEach((newNotification) => {
                            const existingNotificationIndex =
                                this.processedNotificationData.findIndex(
                                    (existingNotification) =>
                                        existingNotification.id ===
                                        newNotification.id
                                );

                            if (existingNotificationIndex !== -1) {
                                // If existing notification is found, update it
                                this.processedNotificationData[
                                    existingNotificationIndex
                                ] = newNotification;
                            } else {
                                // If not found, add it to the list
                                this.processedNotificationData.push(
                                    newNotification
                                );
                            }
                        });

                        this.loadedNotificationsCount =
                            this.processedNotificationData.length;

                        // Fetch the updated unread notification count.
                        this.fetchUnreadCount();

                        // Add new notification IDs to the uniqueNotificationIds set
                        const newNotificationIds = transformedNotifications.map(
                            (notification) => notification.id
                        );
                        newNotificationIds.forEach((id) =>
                            this.uniqueNotificationIds.add(id)
                        );
                    } else {
                        // If there are no notifications, set hasMore to false.
                        this.hasMore = false;
                    }
                },
                page: pageNumber,
                pageSize: this.pageSize,
            });
        }
    }

    getPersonIdFromLocalStorage() {
        let loginData = JSON.parse(localStorage.getItem('loginData'));
        let personId = '';
        if (loginData) {
            personId = loginData.Attributes.PersonId;
        }
        return personId;
    }

    loadMoreNotifications() {
        this.pageNum++; // Increment the page number to load the next page of notifications
        this.fetchNotifications(this.pageNum);
    }

    ngOnDestroy(): void {
        // Unsubscribe from all subscriptions

        // Dispose the overlay
        if (this._overlayRef) {
            this._overlayRef.dispose();
        }
    }

    /**
     * Open the notifications panel

    openPanel(): void {
        // Return if the notifications panel or its origin is not defined
        if (!this._notificationsPanel || !this._notificationsOrigin) {
            return;
        }

        // Create the overlay if it doesn't exist
        if (!this._overlayRef) {
            this._createOverlay();
        }

        // Attach the portal to the overlay
        this._overlayRef.attach(
            new TemplatePortal(this._notificationsPanel, this._viewContainerRef)
        );
    }

    /**
     * Close the notifications panel

    closePanel(): void {
        this._overlayRef.detach();
    }

    /**
     * Create the overlay

    private _createOverlay(): void {
        // Create the overlay
        this._overlayRef = this._overlay.create({
            hasBackdrop: true,
            backdropClass: 'fuse-backdrop-on-mobile',
            scrollStrategy: this._overlay.scrollStrategies.block(),
            positionStrategy: this._overlay
                .position()
                .flexibleConnectedTo(
                    this._notificationsOrigin._elementRef.nativeElement
                )
                .withLockedPosition(true)
                .withPush(true)
                .withPositions([
                    {
                        originX: 'start',
                        originY: 'bottom',
                        overlayX: 'start',
                        overlayY: 'top',
                    },
                    {
                        originX: 'start',
                        originY: 'top',
                        overlayX: 'start',
                        overlayY: 'bottom',
                    },
                    {
                        originX: 'end',
                        originY: 'bottom',
                        overlayX: 'end',
                        overlayY: 'top',
                    },
                    {
                        originX: 'end',
                        originY: 'top',
                        overlayX: 'end',
                        overlayY: 'bottom',
                    },
                ]),
        });

        // Detach the overlay from the portal on backdrop click
        this._overlayRef.backdropClick().subscribe(() => {
            this._overlayRef.detach();
        });
    }

    /**
     * Calculate the unread count
     *
     * @private
     */
    private _calculateUnreadCount(notifications: any[]): void {
        let count = 0;

        if (notifications && notifications.length) {
            count = notifications.filter(
                (notification) => !notification.read
            ).length;
        }

        this.unreadCount = count;
    }

    /**
     *
     * @param event
     */

    onNotificationClicked = (event: any) => {
        // Handle the notification click event here
        // Access the payload data from the event

        if (event.payload && event.payload.OnClick) {
            // Check if the payload has the 'OnClick' property and extract its value
            const { screenId, parameters } = event.payload.OnClick;

            const queryParams: Params = {};
            for (const key in parameters) {
                if (parameters.hasOwnProperty(key)) {
                    queryParams[key] = parameters[key];
                }
            }

            // Build the URL with query parameters
            const navigationExtras: NavigationExtras = {
                queryParams,
            };
            // Use the Router to navigate to the new URL with query parameters
            this.router.navigate(['/screen', screenId], navigationExtras);
        }
    };

    /**
     * Sets up listeners for incoming notifications and changes in unread notification count using the
     * `headlessService` socket. When a 'notification_received' event is emitted, it updates the
     * `processedNotificationData` with the newly received notification. When an 'unread_count_changed'
     * event is emitted, it updates the `unreadCount` property.
     *
     * @returns {void}
     */
    shouldShowReadMore(notification) {
        if (notification) {
            const maxCharacters = 125; // Adjust this value based on your preference
            return notification.content.length > maxCharacters;
        }
    }
    listenToNotifications() {
        // Set up a listener for the 'notification_received' event.

        this.headlessService.socket.on('notification_received', (res) => {
            // Create a new notification object based on the received data and prepend it to the existing data.
            this.processedNotificationData = [
                {
                    id: res.message.id,
                    content: res.message.content,
                    createdAt: res.message.createdAt,
                    updatedAt: res.message.updatedAt,
                    lastReadDate: res.message.updatedAt, // Assuming lastReadDate is same as updatedAt
                    lastSeenDate: res.message.updatedAt, // Assuming lastSeenDate is same as updatedAt
                    payload: res.message.payload,
                    read: res.message.read,
                    url: res.message.cta.data.url,
                },
                ...this.processedNotificationData,
            ];
        });
        // Set up a listener for the 'unread_count_changed' event.

        this.headlessService.socket.on('unread_count_changed', (res: any) => {
            // Update the 'unreadCount' property with the new unread count value.

            this.unreadCount = res.unreadCount;
        });
    }

    /**
     * Marks one or more notifications as read by calling the `markNotificationsAsRead` method of the
     * `headlessService` if it is available. This function can accept a single message ID or an array of
     * message IDs to mark as read. It also provides callbacks for handling success and error cases.
     *
     * @param {string | string[]} messageIds - A single message ID or an array of message IDs to mark as read.
     * @returns {void}
     */
    markNotificationsAsRead(messageIds: string | string[]): void {
        // Ensure that messageIds is an array, even if a single string is passed.
        if (!Array.isArray(messageIds)) {
            messageIds = [messageIds];
        }
        // Get a reference to the headlessService.

        const headlessService = this.headlessService;
        // Call the markNotificationsAsRead method of the headlessService.

        if (headlessService) {
            headlessService.markNotificationsAsRead({
                messageId: messageIds,
                listener: (result: any) => {
                    // After marking as read, fetch the updated unread notification count.
                    this.fetchUnreadCount();
                },
                onError: (error: any) => {
                    console.error(
                        'Error marking notifications as read:',
                        error
                    );
                },
            });
        }
    }
    /**
     * Fetches the unread notification count from the headless service, updating the `unreadCount` property
     * when the operation is successful. This function utilizes callbacks for handling success and error cases.
     *
     * @returns {void}
     */
    fetchUnreadCount(): void {
        // Get a reference to the headlessService.
        const headlessService = this.headlessService;
        // Check if the headlessService is available.

        if (headlessService) {
            // Call the fetchUnreadCount method of the headlessService.

            headlessService.fetchUnreadCount({
                listener: (result: FetchResult<{ count: number }>) => {},
                onSuccess: (data: { count: number }) => {
                    // Update the 'unreadCount' property with the retrieved count.
                    this.unreadCount = data.count;
                },
                onError: (error: any) => {
                    console.error(error);
                },
            });
        }
    }

    /**
     * Deletes a notification with the specified `messageId` using the `removeNotification` method of the
     * `headlessService`. This function provides callbacks for handling success and error cases.
     *
     * @param {string} messageId - The ID of the notification to be deleted.
     * @returns {void}
     */
    deleteNotification(messageId: string): void {
        // Get a reference to the headlessService.

        const headlessService = this.headlessService;
        // Check if the headlessService is available.

        if (headlessService) {
            // Call the removeNotification method of the headlessService to delete the notification.

            headlessService.removeNotification({
                messageId: messageId,
                listener: (result: any) => {
                    // Handle the result if needed
                },
                onSuccess: (message: any) => {
                    // Remove the deleted notification from the 'processedNotificationData' array.

                    this.processedNotificationData =
                        this.processedNotificationData.filter(
                            (n) => n.id !== messageId
                        );
                    if (this.processedNotificationData.length === 0) {
                        this.hasMore = false;
                        this.pageNum++;
                        this.fetchNotifications(this.pageNum); // Assuming you want to fetch the first page after deleting a notification.
                    }

                    // After deleting, fetch the updated unread notification count.
                    this.fetchUnreadCount();
                },
                onError: (error: any) => {
                    console.error(error);
                },
                messageIds: ['messageOne'], // Use an array for messageIds
            });
        }
    }

    /**
     * Retrieves an array of message IDs from the `processedNotificationData`. This function maps over the
     * data and extracts the IDs, returning them as an array of strings.
     *
     * @returns {string[]} An array of message IDs extracted from the `processedNotificationData`.
     */
    getIdsFromProcessedNotificationData(): string[] {
        // Use the map function to extract message IDs from each item in the 'processedNotificationData' array.
        return this.processedNotificationData.map((item) => item.id);
    }

    markAllRead() {
        // Get a reference to the headlessService.

        const headlessService = this.headlessService;
        // Check if the headlessService is available.

        if (headlessService) {
            // Call the markAllMessagesAsRead method of the headlessService to mark all messages as read.
