import {
    Component,
    ViewChild,
    Output,
    Input,
    EventEmitter,
} from '@angular/core';
import { ImageEditorComponent } from '@syncfusion/ej2-angular-image-editor';
import { createElement } from '@syncfusion/ej2-base';
import {
    DialogComponent,
    ButtonPropsModel,
    PositionDataModel,
} from '@syncfusion/ej2-angular-popups';
import { AnimationSettingsModel } from '@syncfusion/ej2-splitbuttons';
import {
    TabImageGetService,
    TabMediaUploadService,
} from '@techextensor/tab-service';
import { take } from 'rxjs';
import { ToastrService } from 'ngx-toastr';

@Component({
    selector: 'app-tab-custom-imageupload',
    templateUrl: './tab-custom-imageupload.component.html',
    styleUrls: ['./tab-custom-imageupload.component.scss'],
})
export class TabCustomImageuploadComponent {
    message;
    mediaId;
    imageUrl;
    setTimer;

    @ViewChild('Dialog')
    public DialogInstance: DialogComponent;
    @ViewChild('ImageEditor')
    public ImageEditorInstance: ImageEditorComponent;

    public Imgwidth: string = '130px';
    public Imgheight: string = '130px';
    public toolbarItems: string[] = [];
    public header: string = 'Edit Profile Image';
    public showCloseIcon: Boolean = true;
    public closeOnEscape: Boolean = true;
    public width: string = '340px';
    public height: string = '440px';
    public visible: boolean = false;
    public animationSettings: AnimationSettingsModel = { effect: 'None' };
    public position: PositionDataModel = { X: 'center', Y: 100 };
    public target: HTMLElement = document.body;
    public imgSrc: string = '';
    allowedFileExtension = ['png', 'jpg', 'jpeg', 'PNG', 'JPG', 'JPEG'];

    @Output() valueChange = new EventEmitter<any>();

    @Input() dataSettingsImage: any;

    @Input()
    public set value(v: any) {
        if (v && v != null) {
            this.mediaId = v.ID;

            if (v.URL != null) {
                this.imageUrl = v.URL;
                this.imgSrc = '';
                this.imageLoaded();
            }
        }

        this.setTimer = setTimeout(() => {
            if (this.dataSettingsImage) {
                if (this.dataSettingsImage?.imageHeight) {
                    this.Imgheight = this.dataSettingsImage?.imageHeight + 'px';
                }

                if (this.dataSettingsImage?.imageWidth) {
                    this.Imgwidth = this.dataSettingsImage?.imageWidth + 'px';
                }
            }
        }, 500);
    }

    public get value(): any {
        return this.mediaId;
    }

    constructor(
        private tabMediaUploadService: TabMediaUploadService,
        private toastrService: ToastrService,
        private tabImageGetService: TabImageGetService
    ) {}

    public created = (): void => {
        if (
            this.ImageEditorInstance.theme &&
            window.location.href.split('#')[1]
        ) {
            this.ImageEditorInstance.theme = window.location.href
                .split('#')[1]
                .split('/')[1];
        }
    };

    public imageLoaded = (): void => {
        if (this.imgSrc === '') {
            let canvas: HTMLCanvasElement =
                document.querySelector('#img-canvas');
            let image: HTMLImageElement = document.querySelector('#demo-img');
            let ctx: CanvasRenderingContext2D = canvas?.getContext('2d');
            if (image) {
                if (this.imageUrl && this.imageUrl != null) {
                    image.src = this.imageUrl;
                }
                canvas.width =
                    image.width < image.height ? image.width : image.height;
                canvas.height = canvas.width;
                ctx.drawImage(image, 0, 0, canvas.width, canvas.height);
            }
        }
    };

    public dialogOpen = (): void => {
        //Plz don't remove this commented code
        // let canvas: HTMLCanvasElement = document.querySelector(
        //     '#ej2-image-editor_0_lowerCanvas'
        // );
        // const ctx: CanvasRenderingContext2D = canvas.getContext('2d');
        // ctx.clearRect(0, 0, canvas.width, canvas.height);
        // this.ImageEditorInstance.open(canvas.toDataURL());

        const img: HTMLImageElement = document.querySelector('#demo-img');
        const image = this.tabImageGetService
            .imageGetByMediaId(this.mediaId)
            .subscribe((blob: any) => {
                // Create a FileReader instance to read the blob content
                var reader = new FileReader();

                // Event handler when the FileReader finishes loading the content
                reader.onload = (event) => {
                    // Retrieve the image URL from the loaded event data
                    const imageUrl = event.target.result as string;
                    // Set the retrieved image URL as the source of the img element
                    img.src = imageUrl;
                    // Open the Image Editor instance with the retrieved image URL
                    this.ImageEditorInstance.open(imageUrl);
                };
                // Read the blob content as a data URL (base64 encoded)
                reader.readAsDataURL(blob);
            });
    };

    public dialogClose = (): void => {
        //Plz don't remove this commented code
        // let canvas: HTMLCanvasElement = document.querySelector('#img-canvas');
        // this.ImageEditorInstance.open(canvas.toDataURL());

        const img: HTMLImageElement = document.querySelector('#demo-img');
        this.ImageEditorInstance.open('');
    };

    // canvas click event
    public onOpenDialog = function (event: any): void {
        this.DialogInstance.show();
    };

    public imageChanged = (args: any): void => {
        let fileExtension = (args.target as any)?.files[0]?.name
            ?.split('.')
            .pop();
        if (this.allowedFileExtension.includes(fileExtension)) {
            const URL = window.URL;
            const url = URL.createObjectURL((args.target as any).files[0]);
            this.ImageEditorInstance.open(url.toString());
            (document.getElementById('img-upload') as HTMLInputElement).value =
                null;
            this.imgSrc = url.toString();
        } else {
            // alert('Only png/jpg/jpeg/PNG/JPG/JPEG files are allowed.');
            this.toastrService.error(
                'Only .png , .jpg , .jpeg files are allowed.'
            );
        }
    };

    public fileOpened = (): void => {
        if (this.dataSettingsImage?.mode)
            this.ImageEditorInstance.select(this.dataSettingsImage?.mode);
        else this.ImageEditorInstance.select('Square');
    };

    public dlgOpenButtonClick = (): void => {
        document.getElementById('img-upload').click();
    };

    public dlgResetBtnClick = (): void => {
        this.ImageEditorInstance.reset();
    };

    public dlgRotateBtnClick = (): void => {
        this.ImageEditorInstance.rotate(-90);
    };

    public dlgDoneBtnClick = (): void => {
        this.ImageEditorInstance.crop();
        const croppedData: ImageData = this.ImageEditorInstance.getImageData();
        const canvas: HTMLCanvasElement = document.querySelector('#img-canvas');
        const ctx: CanvasRenderingContext2D = canvas.getContext('2d');
        const parentDiv: HTMLElement = document.querySelector('.e-profile');
        const tempCanvas: HTMLCanvasElement = parentDiv.appendChild(
            createElement('canvas') as HTMLCanvasElement
        );
        const tempContext: CanvasRenderingContext2D =
            tempCanvas.getContext('2d');
        //Plz don't remove this comment
        // let image: HTMLImageElement = document.querySelector('#demo-img');
        // image.src = '';
        tempCanvas.width = croppedData.width;
        tempCanvas.height = croppedData.height;
        tempContext.putImageData(croppedData, 0, 0);
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        ctx.drawImage(tempCanvas, 0, 0, canvas.width, canvas.height);
        var myImage = tempCanvas.toDataURL('image/png');

        tempCanvas.remove();
        parentDiv.style.borderRadius = '100%';
        canvas.style.backgroundColor = '#fff';
        this.DialogInstance.hide();

        if (this.imgSrc !== '') {
            const img: HTMLImageElement = document.querySelector('#demo-img');
            img.src = this.imgSrc;
        }

        var file = this.dataURItoBlob(myImage);
        this.onFileUpload(file);
    };

    public dlgButtons: ButtonPropsModel[] = [
        {
            click: this.dlgOpenButtonClick.bind(this),
            buttonModel: {
                content: 'Open',
                isPrimary: false,
                cssClass: 'e-custom-img-btn e-img-custom-open',
            },
        },
        {
            click: this.dlgResetBtnClick.bind(this),
            buttonModel: {
                content: 'Reset',
                isPrimary: false,
                cssClass: 'e-custom-img-btn e-img-custom-reset',
            },
        },
        {
            click: this.dlgRotateBtnClick.bind(this),
            buttonModel: {
                content: 'Rotate',
                isPrimary: false,
                cssClass: 'e-custom-img-btn e-img-custom-rotate',
            },
        },
        {
            click: this.dlgDoneBtnClick.bind(this),
            buttonModel: {
                content: 'Apply',
                isPrimary: true,
                cssClass: 'e-custom-img-btn e-img-custom-apply',
            },
        },
    ];

    public canvasClicked = (): void => {
        this.DialogInstance.show();
    };

    ngAfterViewInit(): void {
        let imageHide =
            document.getElementsByClassName('sb-desktop-wrapper')[0];
        if (imageHide) {
            (
                document.getElementsByClassName(
                    'sb-desktop-wrapper'
                )[0] as HTMLElement
            ).onclick = (args: any): void => {
                if (
                    args.target.className.indexOf('col-lg-12 control-section') >
                        -1 ||
                    args.target.className.indexOf('sb-content') > -1
                ) {
                    this.DialogInstance.hide();
                }
            };
        }
    }

    public dataURItoBlob(dataURI) {
        // convert base64/URLEncoded data component to raw binary data held in a string
        var byteString;
        if (dataURI.split(',')[0].indexOf('base64') >= 0)
            byteString = atob(dataURI.split(',')[1]);
        else byteString = unescape(dataURI.split(',')[1]);
        // separate out the mime component
        var mimeString = dataURI.split(',')[0].split(':')[1].split(';')[0];
        // write the bytes of the string to a typed array
        var ia = new Uint8Array(byteString.length);
        for (var i = 0; i < byteString.length; i++) {
            ia[i] = byteString.charCodeAt(i);
        }
        return new Blob([ia], { type: mimeString });
    }

    onFileUpload = (file: any) => {
        // To change the file name with time stamp, because database in not accept the special character file name.
        var blob = file.slice(0, file.size, file.type);
        const newFileName = Date.now() + '.png';
        const newFile = new File([blob], newFileName, { type: file.type });

        // File Readerfile
        let reader = new FileReader();
        let img = new Image();
        reader.readAsDataURL(newFile);

        if (newFile.size < 1) {
            // Check file is empty or not
            this.message = 'Empty file.';
        } else {
            reader.onload = (event: any) => {
                img.src = event.target.result;

                let formData = new FormData();
                formData.append('1', newFile);
                this.tabMediaUploadService
                    ?.uploadMedia(formData)
                    .pipe(take(1))
                    .subscribe((res: any) => {
                        if (!!res.Result && res.Result.length > 0) {
                            let resdata = res.Result[0];
                            this.mediaId = resdata.MediaId;
                            this.valueChange.emit(this.mediaId);
                        }
                    });
            };
        }
    };

    borderradius() {
        return this.dataSettingsImage?.mode === 'circle' ? '50%' : '0%';
    }
    ngOnDestroy(): void {
        clearTimeout(this.setTimer);
    }
}
