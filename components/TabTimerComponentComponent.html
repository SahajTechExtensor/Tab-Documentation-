import {
    AfterViewInit,
    Component,
    Input,
    OnDestroy,
    OnInit,
    ViewChild,
} from '@angular/core';
import { TabDynamicSidebarComponent } from '../tab-dynamic-sidebar/tab-dynamic-sidebar.component';
import moment from 'moment-timezone';

@Component({
    selector: 'app-tab-timer-component',
    templateUrl: './tab-timer-component.component.html',
    styleUrls: ['./tab-timer-component.component.scss'],
})
export class TabTimerComponentComponent implements OnInit, OnDestroy {
    @Input() data: any;

    @ViewChild('tabDynamicSidebarComponent')
    tabDynamicSidebarComponent: TabDynamicSidebarComponent;
    height: string = ''; // Set your initial height value here
    width: string = '';
    year: any;
    month: any;
    date: any;
    now: any;
    targetDate: any;
    targetTime: any;
    difference: number;
    days: any;
    hours: any;
    minutes: any;
    seconds: any;
    TAT: any;
    currentTime: any;
    private timerInterval: NodeJS.Timeout | null = null;
    isTimeNegative:boolean = false;
    constructor() {}

    ngOnInit(): void {
        //FOR REFERENCE THIS  METHOD WHEN IT ENABLE THAT TIME ALSO DOING NGAFTERVIEWINIT UNCOMMENT
        // this.extractDate();

        this.setHightandWidth(this.data);
        const field = this.data?.column?.field;
        const liveCount = this.data?.column?.columnConfig?.liveCount?.liveCount
        const dataOfColumn = this.data[field];

        const outOfTATMinutes = dataOfColumn;
        this.convertOutOfTAT(outOfTATMinutes,liveCount);
    }

    setHightandWidth(data: any) {
        if (data) {
            if (data.column.columnConfig.height) {
                this.height = data.column.columnConfig.height + 'px';
            }
            if (data.column.columnConfig.width) {
                this.width = data.column.columnConfig.width + 'px';
            }
        }
    }
    // This function extracts and processes date and time information from the 'this.data' object.
    // It initializes a timer based on the extracted data.
    extractDate() {
        // Check if the necessary data fields are available in 'this.data'
        if (this.data && this.data.Allocationdatetime && this.data.PhoenixTAT) {
            // Parse the 'Allocationdatetime' using Moment.js library with the specified format
            let mdate = moment(
                this.data.Allocationdatetime,
                'DD-MM-YYYY hh:mm:ss A'
            );
            // Convert the parsed date to a JavaScript Date object
            this.targetDate = mdate.toDate();
            // Convert 'PhoenixTAT' to milliseconds using a custom function
            this.TAT = this.convertToMillisecondsForTAT(this.data.PhoenixTAT);
            // Initialize a timer for further processing
            this.initTimer();
        }
    }
    // This function initializes a timer based on the 'this.targetDate' and 'this.TAT' values.
    initTimer() {
        // Check if both 'this.targetDate' and 'this.TAT' are defined

        if (this.targetDate && this.TAT) {
            // Check if there's an existing timer interval, and clear it to prevent overlap

            if (this.timerInterval) {
                clearInterval(this.timerInterval);
            }
            // Calculate the target time by adding 'this.TAT' to 'this.targetDate'

            this.targetTime = this.targetDate.getTime() + this.TAT;
            // Set up a new timer interval that runs every second

            this.timerInterval = setInterval(() => {
                // Calculate the time difference between the current time ('this.now') and the target time

                this.difference = this.targetTime - this.now;
                // Call the 'tickTock' function to handle further processing

                this.tickTock();
            }, 1000);
        }
    }

    // This function updates various time-related properties based on the time difference ('this.difference').
    tickTock() {
        // Get the current date and time

        this.date = new Date();
        // Get the current time in milliseconds

        this.now = this.date.getTime();
        // Calculate and store the number of days, hours, minutes, and seconds remaining
        // by calling the 'convertMilliseconds' function

        this.days = this.convertMilliseconds(this.difference).days;
        this.hours = this.convertMilliseconds(this.difference).hours;
        this.minutes = this.convertMilliseconds(this.difference).minutes;
        this.seconds = this.convertMilliseconds(this.difference).seconds;
    }

    // CALCULATE MILLISECOND
    convertMilliseconds(milliseconds) {
        // Determine the sign (positive or negative)
        const isNegative = milliseconds < 0;
        milliseconds = Math.abs(milliseconds);

        // Calculate days, hours, minutes, and seconds
        const days = Math.floor(milliseconds / (1000 * 60 * 60 * 24));
        milliseconds %= 1000 * 60 * 60 * 24;

        const hours = Math.floor(milliseconds / (1000 * 60 * 60));
        milliseconds %= 1000 * 60 * 60;

        const minutes = Math.floor(milliseconds / (1000 * 60));
        milliseconds %= 1000 * 60;

        const seconds = Math.floor(milliseconds / 1000);

        // If it was a negative value, negate the results
        if (isNegative) {
            return {
                days: -days,
                hours: -hours,
                minutes: -minutes,
                seconds: -seconds,
            };
        }

        return {
            days,
            hours,
            minutes,
            seconds,
        };
    }
    convertToMillisecondsForTAT(timeStr) {
        // Check if the input contains "Hours"
        if (timeStr.includes('Hours')) {
            // Extract the hours value and convert it to milliseconds
            const hours = parseFloat(timeStr);
            return hours * 60 * 60 * 1000; // 1 hour = 60 minutes * 60 seconds * 1000 milliseconds
        } else if (timeStr.includes('Days')) {
            // Extract the days value and convert it to milliseconds
            const days = parseFloat(timeStr);
            return days * 24 * 60 * 60 * 1000; // 1 day = 24 hours * 60 minutes * 60 seconds * 1000 milliseconds
        } else {
            // Handle other cases or invalid input
            return null;
        }
    }
    private clearTimerInterval() {
        if (this.timerInterval) {
            clearInterval(this.timerInterval);
            this.timerInterval = null;
        }
    }

    ngOnDestroy() {
        this.clearTimerInterval(); // Clear the timer on component destruction
    }

    // Add a method to convert OutOfTAT from minutes to days, hours, minutes, and seconds
    convertOutOfTAT(outOfTATMinutes: number,liveCount) {
        if (outOfTATMinutes != null && outOfTATMinutes != undefined) {
            let isNegative = outOfTATMinutes <= 0;
            this.isTimeNegative = isNegative
            const absoluteMinutes = Math.abs(outOfTATMinutes);

            const days = Math.floor(absoluteMinutes / (60 * 24));
            const hours = Math.floor((absoluteMinutes % (60 * 24)) / 60);
            const remainingMinutes = absoluteMinutes % 60;
            // const seconds = 0; // Initialize seconds to 0 // You can calculate seconds if needed
            const seconds = absoluteMinutes % 60;

            this.days = isNegative ? -days : days;
            this.hours = isNegative ? -hours : hours;
            this.minutes = isNegative ? -remainingMinutes : remainingMinutes;
            this.seconds = isNegative ? -seconds : seconds;

             // Clear any existing interval if it exists
        if (this.timerInterval) {
            clearInterval(this.timerInterval);
        }

            if(liveCount){
            // Start counting seconds dynamically
             this.timerInterval = setInterval(() => {
                if (isNegative) {
                    if (this.seconds > -59) {
                        this.seconds--;
                    } else {
                        this.seconds = 0;
                        if (this.minutes > -59) {
                            this.minutes--;
                        } else {
                            this.minutes = 0;
                            if (this.hours > -23) {
                                this.hours--;
                            } else {
                                this.hours = 0;
                                this.days--;
                            }
                        }
                    }
                } else {
                    if (this.seconds > 0) {
                            this.seconds--;
                        } else {
                            if (this.minutes > 0) {
                                this.minutes--;
                                this.seconds = 59;
                            } else {
                                if (this.hours > 0) {
                                    this.hours--;
                                    this.minutes = 59;
                                    this.seconds = 59;
                                } else {
                                    if (this.days > 0) {
                                        this.days--;
                                        this.hours = 23;
                                        this.minutes = 59;
                                        this.seconds = 59;
                                    } else {
                                        // All time units are 0, handle negative time here
                                        isNegative = true;
                                        // Optionally, stop the timer or reset it to a certain value
                                        // clearInterval(this.timerInterval);
                                        // this.resetTimer();
                                    }
                                }
                            }
                        }
                }
                    if(this.days<0 || this.minutes<0 || this.hours <0 || this.seconds<0)
                        {
                            this.isTimeNegative = true;
                            isNegative =  true
                        }
            }, 1000); // Update seconds every 1000 milliseconds (1 second)
            // Clear the interval when the component is destroyed
            this.ngOnDestroy = () => {
                clearInterval(this.timerInterval);
            };
        }
        }
    }
    // This function calculates and returns a human-readable display value

    calculateDisplayValue() {
        // Check if 'this.days' is defined

        if (this.days != undefined) {
            // If the absolute value of 'this.days' is less than 1, display hours, minutes, and seconds

            if (Math.abs(this.days) < 1) {
                // If 'this.days' is greater than or equal to 1, display the number of days
                // Use a ternary operator to determine whether to display "Day" or "Days" based on the sign of 'this.days'
                return `${this.hours}h ${this.minutes}m ${this.seconds}s`;
            } else {
                return `${this.days} ${this.days < 0 ? 'Day' : 'Day'}`;
            }
        } else {
            // If 'this.days' is not defined, return 'Not defined'
            return 'Not defined';
        }
    }
}
