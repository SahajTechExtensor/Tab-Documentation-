import { Component, EventEmitter, Input, Output } from '@angular/core';
import { FormioCustomComponent, FormioEvent } from '@formio/angular';
import { ColorPickerEventArgs } from '@syncfusion/ej2-angular-inputs';

@Component({
    selector: 'app-tab-color-picker',
    templateUrl: './tab-color-picker.component.html',
    styleUrls: ['./tab-color-picker.component.scss'],
})
export class TabColorPickerComponent implements FormioCustomComponent<any> {
    @Input() dataSettingsColorPicker: any;
    @Input() disabled: boolean;
    @Output() valueChange = new EventEmitter<any>();
    color: any;
    colorValue: string;
    defaultColor: string = '#000000';

    // Color Picker
    mode = 'Picker';
    showButtons = false;
    modeSwitcher = false;

    @Input()
    public set value(v: any) {
        if (v && v != null) {
            this.color = v;
            this.colorValue = v;
        }
    }

    public get value(): any {
        if (!this.color) {
            this.color = this.formateColor(
                this.colorValue || this.defaultColor
            );
            this.valueChange.emit(this.color);
        }
        return this.color;
    }

    /**
     * Handles the change event of the color picker.
     * @param {ColorPickerEventArgs} event - The color picker event object.
     */
    onChange(event: ColorPickerEventArgs) {
        this.color = this.formateColor(event.currentValue.hex);
        this.valueChange.emit(this.color);
    }

    // FOR REFERENCE
    /**
     * Configures the settings for the color picker.
     */
    configureSettings = () => {
        this.mode = this.dataSettingsColorPicker.mode;
        this.showButtons = this.dataSettingsColorPicker.showButtons;
        this.defaultColor = this.dataSettingsColorPicker.defaultColor;
        this.colorValue = this.dataSettingsColorPicker.defaultColor;
        this.modeSwitcher = this.dataSettingsColorPicker.modeSwitcher;
        this.onChange({
            currentValue: {
                hex: this.colorValue,
                rgba: '',
            },
            previousValue: {
                hex: '',
                rgba: '',
            },
        });
    };

    /**
     * Formats the color value to a 7-character hexadecimal format.
     * @param {string} hex - The color value in hexadecimal format.
     * @returns {string} - The formatted color value.
     */
    formateColor(hex: string): string {
        return hex.length > 7 ? hex.substring(0, 7) : hex;
    }
}
