    /**
     * Handles errors during sign-in process.
     * @param res - The response object containing error information.
     * @param customMessage - Optional custom message
     */
    handleError(res, customMessage?: string) {
        // Re-enable the form
        this.signInForm.enable();

        // Reset the form
        this.signInNgForm.resetForm();

        // Set the alert
        this.alert = {
            type: 'error',
            message: customMessage ?? res.Errors[0],
        };

        // Show the alert
        this.showAlert = true;
        this.onLoginError(res);
    }

    // before login callback
    beforeLoginCallback(err: any, data: any) {
        if (!err) {
            this.signIn();
            return;
        }
    }

    getSchemaData() {
        if (environment.name === 'DEV') {

    /**
     * Called when an error occurs during the login process. Calls the custom event 'onLoginError'.
     * @param event - The event data associated with the login error.
     */
    onLoginError(event: any) {
        this.callCustomEvent('onLoginError', event, false, (err, data) => {});
    }

    /**
     * Calls a custom event with the specified function name and event data.
     * @param functionName - The name of the custom event function to be called.
     * @param event - The event data to be passed to the custom event function.
     * @param isAsync - Specifies whether the custom event function is asynchronous or not.
     * @param callback - A callback function that will be executed after the custom event function is called. It takes two arguments: err (error object) and data (data returned by the custom event function).
     */
    callCustomEvent(
        functionName: string,
        event: any,
        isAsync: boolean,
        callback: (err: any, data: any) => void
    ): void {
        if (this.currentScreen) {
            const loginParameters: {
                serviceList: any;
                event: any;
                isAsync: boolean;
                callback: (err, data) => void | null;
            } = {
                serviceList: null,
                event: event,
                isAsync: isAsync,
                callback: callback,
            };

            loginParameters.serviceList = {
                store: this.store,
                router: this._router,
                route: this._activatedRoute,
                getService: this.tabGetService,
                appService: this.tabAppService,
                notificationService: this.toastrNotificationService,
                authenticationService: this.tabAuthenticationService,
            };
            this.currentScreen[functionName](loginParameters);
        } else {
            if (isAsync) {
                callback(null, event);
            }
        }
    }

    /**
     * Updates the current screen instance based on the provided application code.
     * @param applicationCode - The application code.
     */
    updateCurrentScreenInstance(applicationCode: string) {
        var screens = ILoginExtensionRegistry.GetImplementations();
        for (var x = 0; x < screens.length; x++) {
            const screen = new screens[x]();
            if (
                screen.applicationCode.toLowerCase() ===
                applicationCode.toLowerCase()
            ) {
                this.currentScreen = screen;
            }
        }
    }
