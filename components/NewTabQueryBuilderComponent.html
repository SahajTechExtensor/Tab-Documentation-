    public get value(): any {
        let val = JSON.stringify(this.controlValue);
        return JSON.parse(val);
    }

    @Input()
    public set value(value: any) {
        // console.log('setter new querybuilder with control value ', value);

            this.controlValue = value;
        } else {
            this.isShowBuilder = false;
        }
        this.debounceSubject.next(1000);
    }

    ngOnInit(): void {
        this.debounceSubject
            .pipe(debounceTime(1000))
            .pipe(takeUntil(this._unsubscribeAll))
            .subscribe(() => {
                if (this.controlValue?.isRefresh) {
                    this.manageQueryBuilderData();
                }
            });
    }

    /**
     * this funciton will manage the query builder data
    this will get data from dsq to appobject and from appobject to fileds and bind that fields in querybuilder
     */
    async manageQueryBuilderData() {
        // console.log('control settings', this.controlSetting);
        // console.log('control value ', this.controlValue);
        this.isShowBuilder = false;
        this.fieldList = [];
        this.properties = [];
        this.parameterList = [];
        if (
            this.controlValue &&
            this.controlValue.hasOwnProperty('dataSourceQueryId') &&
            this.controlValue['dataSourceQueryId']
        ) {
            const dsqDetails: any = await this.getDSQDetails();
            if (dsqDetails) {
                let appObjectDetailsOfDSQ = TabAppFunction.getAppObject(
                    dsqDetails?.AppObject?.ID
                );
                if (appObjectDetailsOfDSQ) {
                    this.manageAppObjectFields(appObjectDetailsOfDSQ);
                }
            }
            this.getParameterListOFDSQ(this.controlValue['dataSourceQueryId']);
            if (
                this.controlValue &&
                this.controlValue.hasOwnProperty('filterData') &&
                this.controlValue['filterData']
            ) {
                this.manageQueryBuilderRules(this.controlValue['filterData']);
            }
            this.setTimer = setTimeout(() => {
                this.isShowBuilder = true;
                this.assignAPIValueTosaveControl();
            }, 100);
        }

        // this will work when we dont have optoin to pass dsq and we have to manage through appobjet only
        if (
            this.controlValue &&
            this.controlValue.hasOwnProperty('appObjectId') &&
            this.controlValue['appObjectId']
        ) {
            let appObjectDetails = TabAppFunction.getAppObject(
                this.controlValue['appObjectId']
            );
            if (appObjectDetails) {
                this.manageAppObjectFields(appObjectDetails);
            }
            if (
                this.controlValue &&
                this.controlValue.hasOwnProperty('filterData') &&
                this.controlValue['filterData']
            ) {
                this.manageQueryBuilderRules(this.controlValue['filterData']);
            }
            this.setTimerAssignAPI = setTimeout(() => {
                this.isShowBuilder = true;
                this.assignAPIValueTosaveControl();
            }, 100);
        }
    }

    async getDSQDetails() {
        return new Promise((resolve) => {
            let paramData = {
                Reqtokens: { ID: this.controlValue['dataSourceQueryId'] },
                AppObjectName: 'TABMD_DataSourceQueries',
                DSQName: 'Detail_TABMD_DataSourceQueries',
            };
            this.tabGetService
                .executeRAWDSQWithName(paramData)
                ?.pipe(take(1))
                ?.subscribe((res) => {
                    if (!!res.Result && res.Result.length > 0) {
                        resolve(res.Result[0]);
                    } else {
                        resolve({});
                    }
                });
        });
    }

    async manageAppObjectFields(appObjectDetailsOfDSQ) {
        if (appObjectDetailsOfDSQ?.Fields) {
            each(appObjectDetailsOfDSQ.Fields, (f) => {
                this.bindAppObjectFieldsToQueryBuilder('', f, '');
            });
            // const cfl = await this.createColumns(this.fieldList);

    /**
     *
     * @param prefix prefix for lookup fields , it is field name
     * @param item fields detail
    this funciton willl create array of all the fields with configured lookup fields
     * @returns
     */
    bindAppObjectFieldsToQueryBuilder(prefix, item, parentId) {
        let fieldData = {
            field: prefix !== '' ? parentId + '___' + item.ID : item.ID,
            label: item.SystemDBFieldName,
            lookUPText: prefix
                ? prefix + '.' + item.SystemDBFieldName
                : item.SystemDBFieldName,
            type: TabCommonFunction.getFieldsType(
                item?.FieldType.DataType,
                item.hasOwnProperty('LookUpDetails')
            ),
            operators: TabCommonFunction.getOperatorListForQueryBuilder(
                TabCommonFunction.getFieldsType(
                    item?.FieldType.DataType,
                    item.hasOwnProperty('LookUpDetails')
                )
            ),
        };
        this.fieldList.push(fieldData);
        this.properties.push(fieldData.lookUPText);
        if (item.hasOwnProperty('LookUpDetails') && item.LookUpDetails) {
            let lookupAppObjectData =
                TabAppFunction.getAppObjectFromSystemDBTableName(
                    item.LookUpDetails.LookupObject
                );
            if (lookupAppObjectData) {
                if (Object.keys(this.previousLookup).length > 0) {
                    if (
                        this.previousLookup.LookupObject ==
                            item.LookUpDetails.LookupObject &&
                        !item.LookUpDetails.LookupObject.toLowerCase().includes(
                            'tabd_recordinfo'
                        )
                    ) {
                        return;
                    }
                }
                this.previousLookup = item.LookUpDetails;
                let nprefix = fieldData.lookUPText;
                if (lookupAppObjectData?.Fields) {
                    each(lookupAppObjectData.Fields, (ff) => {
                        this.bindAppObjectFieldsToQueryBuilder(
                            nprefix,
                            ff,
                            fieldData.field
                        );
                    });
                }
            }
        }
    }

    /**
     *
     * @param fields field list array
     * @returns it will return formated array which query builder required for treeview
    this function will generate formated array from all the field list and make structued as
    querybuilder required
     */
    async createColumns(fields: any[]) {
        const columns = [];
        for (const field of fields) {
            const fieldParts = field.lookUPText.split('.');
            let parentColumns = columns;
            const fid = field.field;
            const fieldType = field.type;
            for (let i = 0; i < fieldParts.length; i++) {
                const part = fieldParts[i];
                const column = parentColumns.find((c) => c.label === part);
                if (column) {
                    parentColumns = column.columns;
                } else {
                    const newColumn = {
                        field: part,
                        label: part,
                        type: fieldType,
                        columns: null,
                        operators: field.operators,
                        // template: this.filterTypeTemplate,
                    };
                    parentColumns.push(newColumn);
                    parentColumns = newColumn.columns = [];
                }
            }
        }
        return columns;
    }

    /**
     *
     * @param arr converted formated array of querybuilder
     * @param keyToDelete 'columns' field which you want to delete from json array
    this function will delete the colums key which does not have value
    so data type issue is resolved , everytime it was taking object data type for all fields
     * @returns
     */
    deleteKey(arr: any[], keyToDelete: string): any[] {
        for (let i = 0; i < arr.length; i++) {
            if (arr[i].columns.length > 0) {
                this.deleteKey(arr[i].columns, keyToDelete);
            } else {
                delete arr[i][keyToDelete];
            }
        }
        return arr;
    }

    /**
     *
     * @param dataSourceQueryID dsq id to get parameter list
     */
    getParameterListOFDSQ(dataSourceQueryID) {
        let paramData = {
            Reqtokens: { ID: dataSourceQueryID },
            AppObjectName: 'TABMD_DataSourceQueries_Parameters',
            DSQName:
                'FK_TABMD_DataSourceQueries_Parameters_TABMD_DataSourceQueries',
        };

        this.tabGetService
            .executeRAWDSQWithName(paramData)
            ?.pipe(take(1))
            ?.subscribe((res) => {
                if (!!res.Result && res.Result.length > 0) {
                    const params: any = res.Result;
                    this.parameterList = [];
                    params.forEach((element: any) => {
                        this.parameterList.push(element.MappingFieldName);
                    });
                }
            });
    }

    manageQueryBuilderRules(filterData) {
        this.finalRules = [];
        let groupByGroups: any = [];
        groupByGroups = this.groupArrayOfObjects(filterData, 'GroupID');
        Object.keys(groupByGroups).forEach((element, index) => {
            let rules = this.getGroupByRule(groupByGroups[element]);
            if (index > 0) {
                this.finalRules.rules.push(rules[0]);
            } else {
                this.finalRules = rules[0];
            }
        });
    }

    assignAPIValueTosaveControl() {
        this.controlValue.isRefresh = false;
        // let bindNewRules = { filters: this.filterda };
        // Object.assign(this.controlValue, bindNewRules);
        this.controlValue['filters'] = this.controlValue['filterData'];
        this.value = this.controlValue;
        let convertValue = JSON.stringify(this.controlValue);
        this.valueChange.emit(JSON.parse(convertValue));
    }
    groupArrayOfObjects(list: any, key: string) {
        return list.reduce(function (rv: any, x: any) {
            (rv[x[key]] = rv[x[key]] || []).push(x);
            return rv;
        }, {});
    }
    getGroupByRule(singleGroupArray: any) {
        let rules: any = [];
        let rr: any = [];
        let cnd = '';
        singleGroupArray.forEach((element: any) => {
            let field: any = {};
            // If Field is Lookup field then compare with lookuptext otherwise direct compare with field
            if (element.LookUpDetail && element.LookUpDetail.includes('.')) {
                field = this.fieldList.find(
                    (a: any) =>
                        a.lookUPText &&
                        a.lookUPText.toLowerCase() ==
                            element.LookUpDetail.toLowerCase()
                );
            } else {
                let isGuidExist = this.isGuidValid(
                    element.FieldID.toLowerCase()
                );
                if (isGuidExist) {
                    field = this.fieldList.find(
                        (a: any) =>
                            a.field.toLowerCase() ==
                            element.FieldID.toLowerCase()
                    );
                } else {
                    // changes lookuptext instead of label as we are getting lookup value in fieldid to compare
                    field = this.fieldList.find(
                        (a: any) =>
                            a.lookUPText.toLowerCase() ==
                            element.FieldID.toLowerCase()
                    );
                }
            }
            if (field) {
                let rule = {
                    label: field.label,
                    field: field.field,
                    operator:
                        TabCommonFunction.getRelationalOperatorValueToString(
                            element.RelationalOperator
                        ),
                    value: this.getTypeAndValue(element),
                    type: field.type,
                    // id: element.FieldID,
                };
                rr.push(rule);
                cnd = TabCommonFunction.getConjuctionClauseName(
                    element.ConjuctionClause
                );
            }
        });
        rules.push({ condition: cnd, rules: rr });
        return rules;
    }

    /**
     *
     * @param args args of rulechange
    this function will triggers when changing the condition(AND/OR), field, value, operator is changed.
     */
    onRuleChange(args: RuleChangeEventArgs) {
        this.finalRules = [];
        if (args?.rule && args?.rule?.rules?.length) {
            let groupCount = 1;
            args.rule.rules.forEach((element, index) => {
                if (element.hasOwnProperty('rules')) {
                    groupCount++;
                    this.updateGroupRules(element, groupCount);
                } else {
                    if (element.hasOwnProperty('value')) {
                        let sr = {
                            ConjuctionClause:
                                TabCommonFunction.getConjuctionClause(
                                    args.rule.condition
                                ),
                            FieldID: this.getAppFieldId(element.field),
                            RelationalOperator:
                                TabCommonFunction.getRelationalOperatorValueToNumber(
                                    element.operator
                                ),
                            // RelationalOperator: element.operator,
                            Value:
                                typeof element.value == 'string'
                                    ? element.value
                                          .split(',')
                                          .slice(1)
                                          .join(',')
                                    : '',
                            ValueType:
                                typeof element.value == 'string'
                                    ? element.value.split(',')[0]
                                    : '', // this should be FilterValueType like Literal,Parameter
                            // label: element.label,  Commenting this line because of Data are removing from filter if field is lookup field (scenario - if we have one filter with lookup field and after change dsq and coming back to dsq with lookup field filter and then adding new filter will not save lookup details of exiting filter.)
                            label:
                                this.getLabel(element.field) === ''
                                    ? element.label
                                    : this.getLabel(element.field),
                            group: groupCount,
                            Sequence: index + 1,
                        };
                        this.finalRules.push(sr);
                    }
                }
            });
        }
        this.controlValue.isRefresh = false;
        let bindNewRules = { filters: this.finalRules };
        Object.assign(this.controlValue, bindNewRules);
        this.value = this.controlValue;
        let convertValue = JSON.stringify(this.controlValue);
        this.valueChange.emit(JSON.parse(convertValue));
    }
    updateGroupRules(groupData: any, groupCount: number) {
        let updatedGroupCount = groupCount;
        if (
            groupData &&
            groupData.hasOwnProperty('rules') &&
            groupData.rules &&
            groupData.rules.length > 0
        ) {
            groupData.rules.forEach((element: any, index) => {
                if (element.hasOwnProperty('rules')) {
                    updatedGroupCount++;
                    this.updateGroupRules(element, updatedGroupCount);
                } else {
                    if (element.hasOwnProperty('value')) {
                        let sr = {
                            ConjuctionClause:
                                TabCommonFunction.getConjuctionClause(
                                    groupData.condition
                                ),
                            FieldID: element.field,
                            RelationalOperator:
                                TabCommonFunction.getRelationalOperatorValueToNumber(
                                    element.operator
                                ),
                            Value:
                                typeof element.value == 'string'
                                    ? element.value.split(',')[1]
                                    : '',
                            ValueType:
                                typeof element.value == 'string'
                                    ? element.value.split(',')[0]
                                    : '', // this should be FilterValueType like Literal,Parameter
                            label:
                                this.getLabel(element.field) === ''
                                    ? element.label
                                    : this.getLabel(element.field),
                            group: groupCount,
                            Sequence: index + 1,
                        };
                        this.finalRules.push(sr);
                    }
                }
            });
        }
    }

    onFilterDropdownValueChange(
        event: any,
        ruleID: string,
        type: string
    ): void {
        let element: any = document
            .getElementById(ruleID)
            ?.querySelector('.e-rule-value');
        let value;
        switch (type) {
            case 'filter':
                value = element.querySelector('#value1').textContent;
                if (value == 'Parameter') {
                    if (
                        this.controlValue &&
                        this.controlValue.hasOwnProperty('dataSourceQueryId') &&
                        this.controlValue['dataSourceQueryId']
                    ) {
                        this.getParameterListOFDSQ(
                            this.controlValue['dataSourceQueryId']
                        );
                    }
                }
                break;
            case 'parameter':
                value =
                    element.querySelector('#value1').textContent +
                    ',' +
                    element.querySelector('#value2').textContent;
                break;
            case 'property':
                value =
                    element.querySelector('#value1').textContent +
                    ',' +
                    element.querySelector('#value3').textContent;
                break;
            case 'value':
                value =
                    element.querySelector('#value1').textContent +
                    ',' +
                    event.target.value;
                break;
            case 'global':
                value =
                    element.querySelector('#value1').textContent +
                    ',' +
                    element.querySelector('#value5').textContent;
                break;
        }
        this.tabQueryBuilder?.notifyChange(value as string, element, 'value');
    }

    // Get AppField Id
    getAppFieldId(id: string): string {
        let returnId = '';

        if (id.includes('___')) {
            var splittedArray = id.split('___');

            if (splittedArray && splittedArray.length > 1) {
                returnId = splittedArray[splittedArray.length - 1].toString();
            }
        } else {
            returnId = id;
        }

        return returnId;
    }

    // Get Label
    getLabel(id: string): string {
        let label = '';

        if (id.includes('___') && this.fieldList.length > 0) {
            label = this.fieldList.find((x) => x.field === id)?.lookUPText;
        }

        return label;
    }

    isGuidValid(guid) {
        const guidPattern =
            /^[0-9a-fA-F]{8}-[0-9a-fA-F]{4}-[0-9a-fA-F]{4}-[0-9a-fA-F]{4}-[0-9a-fA-F]{12}$/;
        return guidPattern.test(guid);
    }

    getTypeAndValue(element) {
        let fs = element.hasOwnProperty('ValueType')
            ? TabCommonFunction.getFilterValueTypeInString(element.ValueType)
            : TabCommonFunction.getFilterValueTypeInString(
                  element.FilterValueType
              );
        let ss = element.hasOwnProperty('Value')
            ? element.Value
            : element.value;
        let ffs = fs + ',' + ss;
        return ffs;
    }
    ngOnDestroy(): void {
        clearTimeout(this.setTimer);
        clearTimeout(this.setTimerAssignAPI);
        // Unsubscribe from all subscriptions
        this._unsubscribeAll.next(null);
        this._unsubscribeAll.complete();
    }
}
