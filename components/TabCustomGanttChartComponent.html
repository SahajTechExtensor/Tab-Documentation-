import {
    Component,
    EventEmitter,
    Input,
    OnDestroy,
    OnInit,
    Output,
    SimpleChanges,
    ViewChild,
} from '@angular/core';
import { FormioEvent } from '@formio/angular';
import {
    ToolbarItem,
    EditSettingsModel,
    GanttComponent,
} from '@syncfusion/ej2-angular-gantt';
import { ClickEventArgs } from '@syncfusion/ej2-angular-navigations';
import {
    TabAppFunction,
    TabGetService,
    TabInsertService,
} from '@techextensor/tab-service';
import { cloneDeep } from 'lodash';
import { Subject, debounceTime, take, takeUntil } from 'rxjs';
import { MatDialog } from '@angular/material/dialog';
import { TabDynamicSidebarComponent } from '../../../shared/templates/tab-dynamic-sidebar/tab-dynamic-sidebar.component';
import { TabDynamicDialogComponent } from 'app/shared/templates/tab-dynamic-dialog/tab-dynamic-dialog.component';

@Component({
    selector: 'app-tab-custom-gantt-chart',
    templateUrl: './tab-custom-gantt-chart.component.html',
    styleUrls: ['./tab-custom-gantt-chart.component.css'],
})
export class TabCustomGanttChartComponent implements OnInit, OnDestroy {
    @ViewChild('gantt', { static: true })
    public ganttObj: GanttComponent;
    @ViewChild('tabDynamicSidebarComponent')
    tabDynamicSidebarComponent: TabDynamicSidebarComponent;
    public toolbar: ToolbarItem[];
    public data: Object[];
    public columns: object[];
    public labelSettings: object;
    public taskSettings: object;
    public editDialogFields: object[];
    public editSettings: EditSettingsModel;
    public filterSettings: object;
    public showDefaultPopup: boolean = false;
    public timelineSettings?: object;
    private debounceSubject = new Subject();
    private _value: any;
    dsqDetailsOfGanttChart: any;
    clonedsqDetailsOfGanttChart: any;
    @Output() formioEvent = new EventEmitter<FormioEvent>();
    @ViewChild('tabChildScreenRenderer') tabChildScreenRenderer: any;
    private _unsubscribeAll: Subject<any> = new Subject<any>();

    constructor(
        private tabGetService: TabGetService,
        public dialog: MatDialog
    ) {}

    @Input() dataSettingsGanttChart: any;
    @Input()
    public set value(val: any) {
        if (val && val != null) {
            this._value = val;
        }
        this.toolBarSettings();
        this.debounceSubject.next(1000);
    }
    public get value(): any {
        return this._value;
    }

    // This method is triggered when there are changes to the component's input properties.
    // It specifically listens for changes to the 'dataSettingsGanttChart' property.
    ngOnChanges(changes: SimpleChanges) {
        // Check if there are changes to the 'dataSettingsGanttChart' property and if it is not null or undefined.

        if (changes.dataSettingsGanttChart && this.dataSettingsGanttChart) {
            // If there are changes and the 'dataSettingsGanttChart' property is defined, call the 'setColumns' method.

            this.setColumns();
        }
    }

    ngOnInit(): void {
        this.debounceSubject
        .pipe(debounceTime(1000),takeUntil(this._unsubscribeAll))
        .subscribe(() => {
            //api call make here
            this.populateData();
        });

        this.editSettings = {
            allowAdding: false,
            allowEditing: false,
            allowDeleting:false,
            allowTaskbarEditing: false,
            showDeleteConfirmDialog: false,
        };
    }

    // Event handler for toolbar button clicks in the Gantt chart
    public toolbarClick(args: ClickEventArgs): void {
        if (args.item.id === 'custom_Add') {
            this.showDefaultPopup = false;

            this.dsqDetailsOfGanttChart =
                TabAppFunction.getDataFromDataSourceQueryID(
                    this.dataSettingsGanttChart.dataSourceQuery
                );

            const addScreenId = this.dataSettingsGanttChart?.addScreenId;
            const screenOpeningMode =
                this.dataSettingsGanttChart?.screenOpeningMode;

            if (this.dataSettingsGanttChart?.editingType === 'customScreen') {
                if (screenOpeningMode === 'popupModal') {
                    this.dialog.open(TabDynamicDialogComponent, {
                        height: '500px',
                        width: '900px',
                        panelClass: 'custom-modalbox',
                        data: {
                            screenID: addScreenId,
                            data: '{}',
                            title: 'Add/Edit New record',
                        },
                    });
                } else if (screenOpeningMode === 'sidebar') {
                    this.tabDynamicSidebarComponent.parameters = '';
                    this.tabDynamicSidebarComponent.tabChildScreenRenderer.refreshFormData(
                        {}
                    );
                    this.tabDynamicSidebarComponent.screenId = addScreenId;
                    this.tabDynamicSidebarComponent.title =
                        'Add/Edit New record';
                    this.tabDynamicSidebarComponent.isShowSidebarSubmit = false;
                    this.tabDynamicSidebarComponent.settingsDrawer.toggle();
                } else {
                    this.formioEvent.emit({
                        eventName: 'customEvent',
                        data: {
                            state: 'afterClick',
                            data: [],
                            action: 'add',
                            screenId: addScreenId,
                            title: 'add',
                            appObject: this.dsqDetailsOfGanttChart,
                            parameters: {},
                        },
                    });
                }
            }
        }

        if (args.item.id === 'custom_Edit') {
            const selectedTask =
                this.ganttObj.selectionModule.getSelectedRecords()[0];

            if (this.dataSettingsGanttChart?.editingType === 'customScreen') {
                const addScreenId = this.dataSettingsGanttChart?.addScreenId;
                const screenOpeningMode =
                    this.dataSettingsGanttChart?.screenOpeningMode;

                if (screenOpeningMode === 'popupModal') {
                    const data = {
                        parameters: {
                            ...selectedTask,
                            screenMode: 'Add',
                        },
                    };
                    this.dialog.open(TabDynamicDialogComponent, {
                        height: '500px',
                        width: '900px',
                        panelClass: 'custom-modalbox',
                        data: {
                            screenID: addScreenId,
                            data: JSON.stringify(data),
                            title: 'Add/Edit New record',
                        },
                    });
                } else if (screenOpeningMode === 'sidebar') {
                    const value = {
                        parameters: {
                            ...selectedTask,
                            screenMode: 'Add',
                        },
                    };

                    this.tabDynamicSidebarComponent.parameters =
                        JSON.stringify(value);
                    this.tabDynamicSidebarComponent.screenId = addScreenId;
                    this.tabDynamicSidebarComponent.title =
                        'Add/Edit New record';
                    this.tabDynamicSidebarComponent.isShowSidebarSubmit = false;
                    this.tabDynamicSidebarComponent.settingsDrawer.toggle();
                } else {
                    this.formioEvent.emit({
                        eventName: 'customEvent',
                        data: {
                            state: 'afterClick',
                            data: { ...selectedTask, Id: selectedTask['id'] },
                            action: 'edit',
                            screenId: addScreenId,
                            title: 'edit',
                        },
                    });
                }
            }
        }
        if (args.item.id === 'ganttDefault_excelexport') {
            this.ganttObj.excelExport();
        } else if (args.item.id === 'ganttDefault_csvexport') {
            this.ganttObj.csvExport();
        } else if (args.item.id === 'ganttDefault_pdfexport') {
            this.ganttObj.pdfExport();
        } else if (args.item.id === 'custom_save') {
            // this.saveData();
        }
    }

    saveData() {
        // This method will be called when the custom Save button is clicked

        const persistData = this.ganttObj ? this.ganttObj.getPersistData() : '';

        this.dataSettingsGanttChart.defaultState = persistData;
        console.log('ganttChart Settings', this.ganttObj.updatedRecords);
        let action = 'saveGanttChartState';

        this.formioEvent.emit({
            eventName: 'customEvent',
            data: {
                data: this.dataSettingsGanttChart?.defaultState,
                action,
            },
        });
    }

    toolBarSettings() {
        this.toolbar = [];
        // this.toolbar.push({ text: 'Save', id: 'custom_save' } as any);

        if (this.dataSettingsGanttChart.isAdd) {
            this.toolbar.push({ text: 'Add', id: 'custom_Add' } as any);
        }
        if (this.dataSettingsGanttChart.isEdit) {
            this.toolbar.push({ text: 'Edit', id: 'custom_Edit' } as any);
        }
        if (this.dataSettingsGanttChart.isUpdate) {
            this.toolbar.push('Update');
        }
        if (this.dataSettingsGanttChart.isDelete) {
            this.toolbar.push('Delete');
        }
        if (this.dataSettingsGanttChart.isExpandAll) {
            this.toolbar.push('ExpandAll');
        }
        if (this.dataSettingsGanttChart.isCollapseAll) {
            this.toolbar.push('CollapseAll');
        }
        if (this.dataSettingsGanttChart.isallowExcelExport) {
            this.toolbar.push('ExcelExport');
        }
        if (this.dataSettingsGanttChart.isallowPdfExport) {
            this.toolbar.push('PdfExport');
        }
    }

    // Populates data into the Gantt chart based on the configuration in 'dataSettingsGanttChart'
    populateData() {
        if (this.dataSettingsGanttChart) {
            // Retrieve the data source query details based on 'dataSourceQuery' from 'dataSettingsGanttChart'
            this.dsqDetailsOfGanttChart =
                TabAppFunction.getDataFromDataSourceQueryID(
                    this.dataSettingsGanttChart.dataSourceQuery
                );

            // Create a deep clone of the data source query details

            this.clonedsqDetailsOfGanttChart = cloneDeep(
                this.dsqDetailsOfGanttChart
            );

            // Prepare API data with the app object name, query name, and request tokens

            const apidata = {
                AppObjectName: TabAppFunction.getAppObjectFromDataSourceQueryID(
                    this.dataSettingsGanttChart.dataSourceQuery
                ).ObjectName,
                DSQName: this.clonedsqDetailsOfGanttChart.QueryName,
                Reqtokens: {},
            };
            if (this._value && this._value.parameters !== undefined) {
                apidata.Reqtokens = this.generateRequestTokens(this._value.parameters);
              }

            this.tabGetService
                .executeRAWDSQWithName(apidata)
                ?.pipe(take(1))
                ?.subscribe((data) => {
                    this.data = data.Result

                     this.data.forEach((item) => {
            item[this?.dataSettingsGanttChart?.StartDateField
                                ?.AppFieldID?.FieldName] = this.formatDate(item[this?.dataSettingsGanttChart?.StartDateField
                                ?.AppFieldID?.FieldName]); // Change to MM DD YYYY
            item[this?.dataSettingsGanttChart?.endDateField?.AppFieldID
                                ?.FieldName] = this.formatDate(item[this?.dataSettingsGanttChart?.endDateField?.AppFieldID
                                ?.FieldName] ); // Change to MM DD YYYY
        });
                        

                    // Configure the task fields of the Gantt chart based on 'dataSettingsGanttChart'

                    this.taskSettings = {
                        id: this?.dataSettingsGanttChart?.ValueField?.AppFieldID
                            ?.FieldName,
                        name: this?.dataSettingsGanttChart?.nameField?.AppFieldID
                            ?.FieldName,
                        startDate:
                            this?.dataSettingsGanttChart?.StartDateField
                                ?.AppFieldID?.FieldName,
                        endDate:
                            this?.dataSettingsGanttChart?.endDateField?.AppFieldID
                                ?.FieldName,
                        duration: 'Duration',
                        dependency: 'Predecessor',
                    };

                    // Configure the label settings for the Gantt chart based on 'dataSettingsGanttChart'
                    this.labelSettings = {
                        leftLabel:
                            this?.dataSettingsGanttChart?.nameField?.AppFieldID
                                ?.FieldName,
                    };
                   
                });
        }
    }

  formatDate(dateString: string | null): string {
    if (dateString) {
        const [day, month, year] = dateString.split('-');
        return `${month} ${day}, ${year}`;
    }
    return '';
}

    // Set up the columns for the Gantt chart based on the configuration in 'dataSettingsGanttChart'
    setColumns() {
        const startDateField =
            this.dataSettingsGanttChart.StartDateField.AppFieldID.FieldName;
        const endDateField =
            this.dataSettingsGanttChart.endDateField.AppFieldID.FieldName;

        this.columns = [
            {
                field: this?.dataSettingsGanttChart?.ValueField?.AppFieldID
                    ?.FieldName,
                headerText:
                    this?.dataSettingsGanttChart?.ValueField?.AppFieldID?.FieldName,
            },
            {
                field: this?.dataSettingsGanttChart?.nameField?.AppFieldID
                    ?.FieldName,
                headerText:
                    this?.dataSettingsGanttChart?.nameField?.AppFieldID?.FieldName,
            },
            {
                field: startDateField,
                headerText: startDateField,
                format: {
                    type: 'date',
                    format: 'dd/MM/yyyy', // Change the format according to your requirements
                },
            },
            {
                field: endDateField,
                headerText: endDateField,
                format: {
                    type: 'date',
                    format: 'dd/MM/yyyy', // Change the format according to your requirements
                },
            },
            { field: 'Duration', headerText: 'Duration' },
            // { field: 'Predecessor', headerText: 'Dependency' },
        ];
    }

    // Function to generate request tokens dynamically
generateRequestTokens(parameters) {
  if (parameters && typeof parameters === 'object' && Object.keys(parameters).length > 0) {
        const reqTokens = {};
        for (const key in parameters) {
            if (parameters.hasOwnProperty(key)) {
                reqTokens[key] = parameters[key];
            }
        }
        return reqTokens;
    } else {
        return {};
    }
}




    ngOnDestroy(): void {
         // Unsubscribe from all subscriptions
        this._unsubscribeAll.next(null);
        this._unsubscribeAll.complete();
    }
}
