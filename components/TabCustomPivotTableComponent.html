export class TabCustomPivotTableComponent implements OnInit, OnDestroy {
    public dataSourceSettings: IDataOptions;
    public gridSettings: GridSettings;
    public chartSettings: ChartSettings;
    public displayOption: DisplayOption;
    public cellTemplate: string;
    public observable = new Observable();
    private debounceSubject = new Subject();
    public toolbarOptions: CustomToolbarItems;
    appObjectDetailsOfPivot: any;
    dsqDetailsOfPivot: any;
    cloneDsqDetailsOfPivot: any;
    _value: any;
    isFirstTime: boolean = true;
    reports = [];
    private setTimer: any;
    private _unsubscribeAll: Subject<any> = new Subject<any>();

    constructor(
        private tabGetService: TabGetService,
        private route: ActivatedRoute
    ) {}

    @ViewChild('pivotview', { static: false })
    public pivotObj: PivotViewComponent;

    @Input() dataSettingsPivot: any;
    @Output() formioEvent = new EventEmitter<FormioEvent>();

    @Input()
    public set value(val: any) {
        if (val && val != null) {
            this._value = val;
        }
              this.debounceSubject.next(1000);

    }

    // used for saved state and remove state

    /*
    @param args; 
    */

    public onToolbarClick(args: ClickEventArgs): void {
        const tooltipText = args.item.tooltipText; // tooltip text like save State and Remove state that define in toolbarOptions
        const persistData = this.pivotObj ? this.pivotObj.getPersistData() : ''; // get Persist data

        //action
        let action: string;
        switch (tooltipText) {
            case 'Save State':
                //set action as savePivotState
                action = 'savePivotState';
                break;
            case 'Remove State':
                //set action as removePivotState
                action = 'removePivotState';
                break;
            default:
                return;
        }

        //saveProperties store the json data
        const savedProperties = JSON.parse(persistData);

        // set the default state  of dataSettingPivot
        this.dataSettingsPivot.defaultState = savedProperties;
        this.formioEvent.emit({
            eventName: 'customEvent',
            data: {
                data: this.dataSettingsPivot?.defaultState,
                // screenId: '01fdcc00-2683-4928-a620-589ffbbff6e5',
                action,
            },
        });
    }

    beforeToolbarRender(args: any) {
        args.customToolbar.splice(6, 0, {
            type: 'Separator',
        });
        args.customToolbar.splice(9, 0, {
            type: 'Separator',
        });
    }

    /**
     * On init
     */

    ngOnInit(): void {
        //use for change the content style for specific cell.
        this.cellTemplate = '<span class="tempwrap"></span>';
        // displayOption used for chart and table for combine use take  view:'Both'


        this.debounceSubject.pipe(debounceTime(1000)).subscribe(() => {
                                this.getPivotData();


        })

        // Pivot ChartSetting
        this.chartSetting();

        //customize the size of column and row
        this.gridSetting();

        this.displayOption = { view: 'Both' } as DisplayOption;
    }

    public gridSetting() {
        // Define the gridSettings object
        this.gridSettings = {
            columnWidth: 120,
            allowSelection: true,
            rowHeight: 36,
            selectionSettings: {
                mode: 'Cell',
                type: 'Single',
                cellSelectionMode: 'Box',
            },

            // Set the number format to undefined for cells with no value in the 'value' axis

            excelQueryCellInfo: this.observable
            .pipe(takeUntil(this._unsubscribeAll))
            .subscribe((args) => {
                if (
                    ((args as ExcelQueryCellInfoEventArgs).cell as IAxisSet)
                        .axis === 'value' &&
                    ((args as ExcelQueryCellInfoEventArgs).cell as IAxisSet)
                        .value === undefined
                ) {
                    (args as ExcelQueryCellInfoEventArgs).style.numberFormat =
                        undefined;
                }
            }) as any,
        } as GridSettings;
    }

    public chartSetting() {
        this.chartSettings = {
            chartSeries: { type: 'Column' },
        } as ChartSettings;
    }
    // restore default state of pivot table
    restoreDefaultState() {
        if (
            this.dataSettingsPivot.defaultState &&
            this.dataSettingsPivot.defaultState.length > 0
        ) {
            this.reports = this.dataSettingsPivot.defaultState;
            // this.reports = this.dataSettingsPivot.defaultState
            let report = JSON.parse(
                this.dataSettingsPivot.defaultState[0].report
            );
            this.displayOption = report.displayOption;
            this.chartSettings = report.chartSettings;
            report.dataSourceSettings.dataSource =
                this.pivotObj.dataSourceSettings.dataSource;
            this.pivotObj.dataSourceSettings = report.dataSourceSettings;
            this.pivotObj.displayOption = report.displayOption;
            this.pivotObj.currentView = this.displayOption.primary;
        }
    }

    // this function is used for toolbarOptions dynamically
    processToolbar() {
        this.toolbarOptions = [];
        if (this.dataSettingsPivot?.isReport) {
            this.toolbarOptions.push(
                'New',
                'Save',
                'SaveAs',
                'Rename',
                'Remove',
                'Load'
            );
        }
        if (this.dataSettingsPivot?.isGrid) {
            this.toolbarOptions.push('Grid');
        }
        if (this.dataSettingsPivot?.isChart) {
            this.toolbarOptions.push('Chart');
        }
        if (this.dataSettingsPivot?.isExport) {
            this.toolbarOptions.push('Export');
        }
        if (this.dataSettingsPivot?.isSubTotal) {
            this.toolbarOptions.push('SubTotal');
        }
        if (this.dataSettingsPivot?.isGrandTotal) {
            this.toolbarOptions.push('GrandTotal');
        }
        if (this.dataSettingsPivot?.isFieldList) {
            this.toolbarOptions.push('FieldList');
        }
        if (this.dataSettingsPivot?.isFormatting) {
            this.toolbarOptions.push('Formatting');
        }
    }

    //this function is used for api call and set the default rows and column and set the dataSource for pivot table.
    public getPivotData() {
        this.dsqDetailsOfPivot = TabAppFunction.getDataFromDataSourceQueryID(
            this.dataSettingsPivot.dataSourceQueryId
        );
        this.cloneDsqDetailsOfPivot = cloneDeep(this.dsqDetailsOfPivot);
        const apidata = {
            AppObjectName: TabAppFunction.getAppObjectFromDataSourceQueryID(
                this.dataSettingsPivot.dataSourceQueryId
            ).ObjectName,
            DSQName: this.cloneDsqDetailsOfPivot.QueryName,
            Reqtokens: {},
        };
        if (this._value && this._value.parameters !== undefined) {
                apidata.Reqtokens = this.generateRequestTokens(this._value.parameters);
           }
        this.tabGetService
            .executeRAWDSQWithName(apidata)
            ?.pipe(take(1))
            .subscribe((data: ApiResponse) => {
                const column = [];
                const rows = [];
                const values = [];

                this.processToolbar();

                // Iterate through the keys in the first element of the Result array and set the rows and column
                for (const key in data.Result[0]) {
                    if (
                        typeof data.Result[0][key] === 'string' ||
                        typeof data.Result[0][key] === 'number'
                    ) {
                        if (column.length === 0) {
                            column.push({ name: key, caption: key });
                        } else if (
                            column[0]['caption'] !== key &&
                            rows.length === 0
                        ) {
                            rows.push({ name: key, caption: key });
                        } else if (
                            column[0]['caption'] !== key &&
                            rows[0]['caption'] !== key &&
                            values.length <= 1
                        ) {
                            values.push({ name: key, caption: key });
                        }
                    }
                }

                const dataRes = JSON.parse(JSON.stringify(data.Result));

                // Set the dataSourceSettings for the pivotObj

                this.pivotObj.dataSourceSettings = {
                    dataSource: dataRes,
                    expandAll: false,
                    columns: column,
                    rows: rows,
                    values: values,
                    showAggregationOnValueField: false,
                };
                this.pivotObj.dataBind();
                this.restoreDefaultState();
            });
    }
    // for save chartSeries
    chartSeriesCreated() {
        this.pivotObj.chartSettings.chartSeries.legendShape =
            this.pivotObj.chartSettings.chartSeries.type === 'Polar'
                ? 'Rectangle'
                : 'SeriesType';
    }
    // create a new report
    newReport() {
        this.pivotObj.setProperties(
            {
                dataSourceSettings: {
                    columns: [],
                    rows: [],
                    values: [],
                    filters: [],
                },
            },
            false
        );
    }


/**
 * Loads a report based on the provided report name from the dataSettingsPivot.defaultState array.
 * Updates the UI and pivotObj settings with the loaded report's information.
 *
 * @param args - An object containing information about the report loading operation, including:
 *   - reportName: The name of the report to be loaded.
 */

    loadReport(args: any) {

    // Retrieve the defaultState array or initialize an empty array if it doesn't exist.

        const reportCollection = this.dataSettingsPivot.defaultState || [];

        // Find the report in the collection that matches the specified reportName.

        const matchedReport = reportCollection.find(
            (item: any) => item.reportName === args.reportName
        );
        

            // If a matching report is found, update the UI and pivotObj settings with the report information.

        if (matchedReport) {
        
        // Parse the report JSON to extract its properties.

            const report = JSON.parse(matchedReport.report);

        // Update various settings based on the loaded report.

            this.displayOption = report.displayOption;
            this.chartSettings = report.chartSettings;
            report.dataSourceSettings.dataSource =
                this.pivotObj.dataSourceSettings.dataSource;
            this.pivotObj.dataSourceSettings = report.dataSourceSettings;
            this.pivotObj.displayOption = this.displayOption;
            this.pivotObj.currentView = this.displayOption.primary;
        }
    }

/**
 * Fetches report names from the dataSettingsPivot.defaultState array and populates
 * the reportName property in the args object with the retrieved report names.
 *
 * @param args - An object containing information about the report fetching operation, including:
 *   - reportName: An array property to be populated with the report names.
 */
    fetchReport(args: any) {
    
    // Initialize an array to store report names retrieved from the defaultState.

    let reportCollection: string[] = [];
    
    // Check if the defaultState array exists and has a length defined.

    if (this.dataSettingsPivot?.defaultState?.length !== undefined) {
        reportCollection = this.dataSettingsPivot.defaultState.map(
            (item: any) => item.reportName
        );
    }
    // Populate the reportName property in the args object with the retrieved report names.

    args.reportName = reportCollection;
}
    
    /**
 * Removes a report from the dataSettingsPivot.defaultState array and triggers the 'removePivotState' action.
 *
 * @param args - An object containing information about the report to be removed, including:
 *   - reportName: The name of the report to be removed.
 */

    removeReport(args: any) {

        // Initialize an array to store the reports collection.

        let reportCollection: any[] = [];

        // Populate reportCollection with the default state if it exists.

        if (
            this.dataSettingsPivot.defaultState &&
            this.dataSettingsPivot.defaultState.length !== 0
        ) {
            reportCollection = this.dataSettingsPivot.defaultState;
        }
            // Iterate through the reports collection to find and remove the specified report.

        for (let i: number = 0; i < reportCollection.length; i++) {
            if (reportCollection[i].reportName === args.reportName) {
                reportCollection.splice(i, 1);
            }
        }

        // Update the defaultState with the modified reports collection.

        if (
            this.dataSettingsPivot.defaultState &&
            this.dataSettingsPivot.defaultState.length !== 0
        ) {
            this.dataSettingsPivot.defaultState = reportCollection;
        }
        
        // If the defaultState becomes empty, reset the dataSource in the pivotObj.

        if (this.dataSettingsPivot.defaultState == 0) {
            this.pivotObj.dataSourceSettings.dataSource = [];
        }
            // Trigger the 'removePivotState' action by emitting a customEvent with the updated data.

        let action = 'removePivotState';
        this.formioEvent.emit({
            eventName: 'customEvent',
            data: {
                data: this.dataSettingsPivot?.defaultState,
                action,
            },
        });
    }

 /**
 * Saves a report by updating or adding it to the dataSettingsPivot.defaultState array
 * and triggers the 'savePivotState' action if the report is successfully saved.
 *
 * @param args - An object containing information about the report to be saved, including:
 *   - report: The JSON representation of the report.
 *   - reportName: The name of the report.
 */

    saveReport(args: any) {
        
        // If it is the first time saving, skip the operation and mark isFirstTime as false.

        if (this.isFirstTime) {
            this.isFirstTime = false;
            return;
        }
        let isSaved: boolean = false;
          
        // Check if the necessary properties are present in the args object.

        if (args.report && args.reportName && args.reportName !== '') {
            let report = JSON.parse(args.report);
            report.dataSourceSettings.dataSource = [];
            report.pivotValues = [];
            args.report = JSON.stringify(report);

            // Iterate through existing reports to update the report if it already exists.

            this.reports.map(function (item: any): any {
                if (args.reportName === item.reportName) {
                    item.report = args.report;
                    isSaved = true;
                }
            });
        
            // If the report does not exist, add it to the reports array.

            if (!isSaved) {
                this.reports.push(args);
            }
                
            // Update the defaultState with the updated reports array.

            this.dataSettingsPivot.defaultState = this.reports;
        }
            // If the report is successfully saved, trigger the 'savePivotState' action.

        if (isSaved) {
            let action = 'savePivotState';
            this.formioEvent.emit({
                eventName: 'customEvent',
                data: {
                    data: this.dataSettingsPivot?.defaultState,
                    action,
                },
            });
        }
    }


    /**
 * Renames a report within the dataSettingsPivot.defaultState array and triggers the 'savePivotState' action.
 *
 * @param args - An object containing information about the report renaming, including:
 *   - isReportExists: A boolean indicating whether the report already exists.
 *   - rename: The new name for the report.
 *   - reportName: The current name of the report to be renamed.
 */
    renameReport(args: any) {
         // Initialize an array to store the reports collection.
        
         let reportsCollection: any[] = [];

        // Populate reportsCollection with the default state if it exists.

        if (this.dataSettingsPivot.defaultState) {
            reportsCollection = this.dataSettingsPivot.defaultState;
        }

        // If the report already exists, remove the old report with the same name.

        if (args.isReportExists) {
            for (let i: number = 0; i < reportsCollection.length; i++) {
                if (reportsCollection[i].reportName === args.rename) {
                    reportsCollection.splice(i, 1);
                }
            }
        }

        // Update the report name in the reportsCollection.

        reportsCollection.map(function (item: any): any {
            if (args.reportName === item.reportName) {
                item.reportName = args.rename;
            }
        });

        // Update the defaultStates property if the defaultState array is not empty.

        if (
            this.dataSettingsPivot.defaultState &&
            this.dataSettingsPivot.defaultState.length !== 0
        ) {
            this.dataSettingsPivot.defaultStates = reportsCollection;
        }
            
        // Emit a customEvent to trigger the 'savePivotState' action with the updated data.

        let action = 'savePivotState';
        this.formioEvent.emit({
            eventName: 'customEvent',
            data: {
                data: this.dataSettingsPivot?.defaultState,
                action,
            },
        });
    }
    ngOnDestroy(): void {
        clearTimeout(this.setTimer);
        // Unsubscribe from all subscriptions
        this._unsubscribeAll.next(null);
        this._unsubscribeAll.complete();
    }

        const reqTokens = {};
        for (const key in parameters) {
            if (parameters.hasOwnProperty(key)) {
                reqTokens[key] = parameters[key];
            }
        }
        return reqTokens;

        return {};
