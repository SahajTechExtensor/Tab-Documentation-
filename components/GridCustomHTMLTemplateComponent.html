import { DomSanitizer } from '@angular/platform-browser';
import { Component, Input, OnInit } from '@angular/core';
import { TabTemplateService } from '@techextensor/tab-service';
import { take } from 'rxjs';

// this html template which we are rendering in grid column
@Component({
    selector: 'app-grid-custom-htmltemplate',
    // template: `<div innerHTML="{{ display }}"></div>`,
    template: `<div *ngIf="display1" [innerHTML]="display1"></div>`,
})
export class GridCustomHTMLTemplateComponent implements OnInit {
    @Input() data: any;
    @Input() rowData: any;
    display = '';
    display1: any;
    private setTimer: any;

    constructor(
        private tabTemplateService: TabTemplateService,
        private sanitizer: DomSanitizer
    ) {}

    ngOnInit(): void {
        this.setTimer = setTimeout(() => {
            this.populateData(this.data?.column.columnConfig?.htmlEditor);
        }, 100);
    }

    populateData(template: any) {
        let data = [];
        data.push(this.rowData);

        let apiData = {
            DataSourceMode: 2,
            Value: this.data?.column.columnConfig?.htmlEditor,
            Templatevariables: JSON.stringify(this.rowData),
        };

        this.tabTemplateService
            .templateParseUniversal(apiData)
            ?.pipe(take(1))
            ?.subscribe((res: any) => {
                if (res.Result) {
                    this.display = res.Result;
                    this.display1 = this.sanitizer.bypassSecurityTrustHtml(
                        this.display
                    );
                }
            });
    }
    ngOnDestroy(): void {
        clearTimeout(this.setTimer);
    }
}
