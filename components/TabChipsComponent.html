    @ViewChild('drpInput') drpInput: ElementRef<HTMLInputElement>;

    chipInstance: any = {
        isRefreshComponent: true,
        deletedChips: [],
    };

    constructor(

        private tabMediaService: TabMediaUploadService,
        private tabSelectService: TabSelectService
    ) {}

    public get value(): any {

    ngOnInit(): void {
        this.debounceSubject
            .pipe(debounceTime(1000))
            .pipe(takeUntil(this._unsubscribeAll))
            .subscribe(() => {
                if (this.chipInstance?.isRefreshComponent) {
                    this.selectedChipsDataArray = [];
                    this.allChipsDataArray = [];
                    this.manageChipsComponentData();
                } else {
                    // added this condition to show validation message when value is required

    /**
     * this function will configuration and value or data for this component
     */
    manageChipsComponentData() {
        if (this.dataSetting && Object.keys(this.dataSetting).length) {
            if (this.dataSetting?.datasourceMode == 'nothing') {
                if (
                    this.chipInstance &&
                    this.chipInstance != undefined &&
                    this.chipInstance != null &&
                    Object.keys(this.chipInstance).length
                ) {
                    // if (this.chipInstance.hasOwnProperty('selectedChips')) {
                    //     let strArray = String(
                    //         this.chipInstance?.selectedChips
                    //     ).split(',');
                    //     strArray.forEach((element) => {
                    //         this.selectedChipsDataArray.push({
                    //             label: element,
                    //             value: element,
                    //         });
                    //     });
                    // }
                    if (
                        this.dataSetting.hasOwnProperty('key') &&
                        !this.chipInstance.hasOwnProperty('selectedChips')
                    ) {
                        this.getOptionListFromComponent(this.dataSetting.key);
                    }
                    // this.selectedChipsDataArray = str.split(',');
                }
            } else if (this.dataSetting?.datasourceMode == 'data') {
                this.allChipsDataArray = this.dataSetting?.customData;
            } else if (this.dataSetting?.datasourceMode == 'dsq') {
                if (this.dataSetting?.dataSourceQueryId) {
                    this.getDataFromDSQ();
                }
            }
        }
        if (
            this.chipInstance &&
            this.chipInstance.hasOwnProperty('component') &&
            this.chipInstance.component
        ) {
            this.getOptionListFromComponent(this.chipInstance.component.key);
        }
    }

    /**
     *
     * @param componentKey component key will be same as appobject name
    we will get fields from that appobject and from that get fiels and for lookup fields we will hanlde select and list query for
    option list
     */
    async getOptionListFromComponent(componentKey) {
        let splitKey = componentKey.split('.');
        let appObjectData = TabAppFunction.getAppObjectFromSystemDBTableName(
            splitKey[0]
        );
        const primaryKeyField =
            TabRequestResponseFunction.getPrimaryKeyFields(appObjectData);
        let lookUpdetails, listData;
        if (appObjectData && appObjectData?.Fields) {
            for (let fs of appObjectData?.Fields) {
                if (
                    fs.hasOwnProperty('LookUpDetails') &&
                    fs.FieldName == splitKey[1]
                ) {
                    lookUpdetails = fs.LookUpDetails;
                    listData = await this.getLookupDetails(fs.LookUpDetails);
                }
            }
            this.bindOptionList(listData, lookUpdetails);

            // manage from component value
            if (
                this.chipInstance &&
                this.chipInstance.hasOwnProperty('selectedChips') &&
                this.chipInstance.selectedChips &&
                this.chipInstance.selectedChips.length > 0
            ) {
                for (let d of this.chipInstance.selectedChips) {
                    // add selected data chips component
                    this.selectedChipsDataArray.push({
                        label: d[splitKey[1]][lookUpdetails['DisplayField']],
                        value: d[splitKey[1]][lookUpdetails['LookupField']],
                        // Id: d[primaryKeyField[0]?.FieldName],
                        [primaryKeyField[0]?.FieldName]:
                            d[primaryKeyField[0]?.FieldName],
                    });
                    // remove selected data from option list
                    const index = this.allChipsDataArray.findIndex(
                        (item) =>
                            item.value ==
                            d[splitKey[1]][lookUpdetails['LookupField']]
                    );
                    if (index > -1) {
                        this.allChipsDataArray.splice(index, 1);
                    }
                }
            }
        }
    }

    /**
     *
     * @param lookupDetails it will take lookup details and verify if select query exist in lookup detail
    it will run select query else try for list api call and get data
     * @returns return result of api with promises , list or select query execute result response
     */
    getLookupDetails(lookupDetails: any) {
        return new Promise((resolve) => {
            if (lookupDetails.hasOwnProperty('selectQuery')) {
                this.tabSelectService
                    .selectExecutor(lookupDetails.selectQuery)
                    ?.pipe(take(1))
                    ?.subscribe((res: any) => {
                        resolve(res.Result);
                    });
            } else {
                const selectedAppObject: TabClasses.AppObject =
                    TabAppFunction.getAppObjectFromSystemDBTableName(
                        lookupDetails.LookupObject
                    );
                const dsqID =
                    TabAppFunction.getDsqIdfromSelectedAppObjectDSQArray(
                        selectedAppObject.DataSourceQueries,
                        lookupDetails.LookupObject,
                        'List'
                    );
                const dsqDetails =
                    TabAppFunction.getDataFromDataSourceQueryID(dsqID);
                const apiData = {
                    AppObjectName: selectedAppObject.ObjectName,
                    DSQName: dsqDetails.QueryName,
                    Reqtokens: {},
                };
                if (
                    this.chipInstance &&
                    this.chipInstance.hasOwnProperty('parameters')
                ) {
                    apiData.Reqtokens = this.chipInstance.parameters;
                }
                this.tabGetService
                    .executeRAWDSQWithName(apiData)
                    ?.pipe(take(1))
                    ?.subscribe((res: any) => {
                        if (res?.Result) {
                            resolve(res.Result);
                        }
                    });
            }
        });
    }

    bindOptionList(listData, lookUpdetails) {
        for (let d of listData) {
            this.allChipsDataArray.push({
                label: d[lookUpdetails['DisplayField']],
                value: d[lookUpdetails['LookupField']],
            });
        }
    }

    /**
     *
     * @param event when we will add custom new chips without autocomplete
     */

    add(event: MatChipInputEvent): void {
        // this.insertedDataArray.push(event.value);
        const value = (event.value || '').trim();
        if (value) {
            this.selectedChipsDataArray.push({ label: value, value: value });
        }
        // Clear the input value
        event.chipInput!.clear();
        this.inputCtrl.setValue(null);
        // this.chipInstance.selectedChips = this.selectedChipsDataArray;
        // this.chipInstance.isRefreshComponent = false;
        // this.value = this.chipInstance;
        // let abc = JSON.stringify(this.chipInstance);
        // this.valueChange.emit(JSON.parse(abc));
    }

    /**
     *
     * @param chip when you unselect , or cancel or remove the chips this will execute
     */
    deletedDataArray = [];
    remove(chip: string): void {
        if (chip && (chip.hasOwnProperty('Id') || chip.hasOwnProperty('ID'))) {
            this.deletedDataArray.push(chip);
        }
        const index = this.selectedChipsDataArray.indexOf(chip);

        if (index >= 0) {
            this.selectedChipsDataArray.splice(index, 1);
        }
        if (this.chipInstance?.insertedChips) {
            this.chipInstance.insertedChips.splice(index, 1);
        }

        // add  option from dropdown list
        this.allChipsDataArray.push(chip);

        this.chipInstance.selectedChips = this.selectedChipsDataArray;
        this.chipInstance.deletedChips = this.deletedDataArray;
        this.chipInstance.isRefreshComponent = false;
        this.value = this.chipInstance;
        let abc = JSON.stringify(this.chipInstance);
        this.valueChange.emit(JSON.parse(abc));
    }

    /**
     *
     * @param event event when we select the option from list this event will execute
     */

    insertedDataArray = [];
    selected(event: MatAutocompleteSelectedEvent): void {
        // this.selectedChipsDataArray.push(event.option.viewValue);
        this.selectedChipsDataArray.push({
            label: event.option.value.label,
            value: event.option.value.value,
        });

        // remove option from dropdown list
        const index = this.allChipsDataArray.indexOf(event.option.value);
        if (index >= 0) {
            this.allChipsDataArray.splice(index, 1);
        }

        this.insertedDataArray.push(event?.option?.value);
        this.drpInput.nativeElement.value = '';
        this.inputCtrl.setValue(null);
        if (this.chipInstance == null || this.chipInstance == undefined) {
            this.chipInstance = {
                isRefreshComponent: true,
                deletedChips: [],
            };
        }
        this.chipInstance.insertedChips = this.insertedDataArray;
        this.chipInstance.selectedChips = this.selectedChipsDataArray;
        this.chipInstance.isRefreshComponent = false;
        this.value = this.chipInstance;
        let abc = JSON.stringify(this.chipInstance);
        this.valueChange.emit(JSON.parse(abc));
    }

    /**
     * if you have configured dsq datasource mode this will execute dsq and get data and bind to autocomplete
     */
    getDataFromDSQ() {
        let appObjectDetailsOfGrid =
            TabAppFunction.getAppObjectFromDataSourceQueryID(
                this.dataSetting?.dataSourceQueryId
            );
        let dsqDetails = TabAppFunction.getDataFromDataSourceQueryID(
            this.dataSetting?.dataSourceQueryId
        );
        const apiData = {
            AppObjectName: appObjectDetailsOfGrid?.ObjectName,
            DSQName: dsqDetails?.QueryName,
            Reqtokens: this.dataSetting?.parameters ?? {},
        };
        this.tabGetService
            .executeRAWDSQWithName(apiData)
            ?.pipe(take(1))
            ?.subscribe((res: any) => {
                if (res?.StatusCode == '200') {
                    if (res?.Result && res?.Result.length) {
                        this.allChipsDataArray = [];
                        let tdata = res?.Result;
                        for (let d of tdata) {
                            this.allChipsDataArray.push({
                                label: d[this.dataSetting?.labelField],
                                value: d[this.dataSetting?.valueField],
                            });
                        }
                        if (this.dataSetting.isDocumentPreview) {
                            this.selectedChipsDataArray =
                                this.allChipsDataArray;
                        }
                    }
                }
            });
    }

    ngOnDestroy(): void {
        // clearTimeout(this.setTimer);
        // Unsubscribe from all subscriptions
        this._unsubscribeAll.next(null);
        this._unsubscribeAll.complete();
    }

    downloadFile(chip: any) {
        const mediaId = chip.value; // Assuming the 'label' property contains the MediaId
        if (
            mediaId &&
            !!this.dataSetting.isDocumentPreview &&
            !!this.dataSetting.allowToDownload
        ) {
            this.getFileByMediaId(mediaId);
        }
    }

    getFileByMediaId(mediaId: any) {
        this.tabMediaService
            .getFileByMediaId(mediaId)
            ?.pipe(take(1))
            .subscribe((blob: any) => {
                if (blob) {
                    const url = window.URL.createObjectURL(blob);
                    const link = document.createElement('a');
                    link.href = url;
                    link.download = new File([blob], mediaId.toString()).name;
                    link.click();
                    window.URL.revokeObjectURL(url);
                }
            });
    }
}
