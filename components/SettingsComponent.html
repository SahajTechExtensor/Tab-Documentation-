import { ToastrNotificationService } from './../notifications/toastr.service';
import { Component, OnDestroy, OnInit, ViewEncapsulation } from '@angular/core';
import { Router } from '@angular/router';
import { Subject, takeUntil } from 'rxjs';
import { FuseConfigService } from '@fuse/services/config';
import { AppConfig, Scheme, Theme, Themes } from 'app/core/config/app.config';
import { Layout } from 'app/layout/layout.types';
import {
    TabGetService,
    TabUpdateService,
    TabClasses,
} from '@techextensor/tab-service';
import { AuthService } from 'app/core/auth/auth.service';

@Component({
    selector: 'settings',
    templateUrl: './settings.component.html',
    styles: [
        `
            settings {
                position: static;
                display: block;
                flex: none;
                width: auto;
            }
        `,
    ],
    encapsulation: ViewEncapsulation.None,
})
export class SettingsComponent implements OnInit, OnDestroy {
    panelOpenState = false;
    config: AppConfig;
    layout: Layout;
    scheme: 'dark' | 'light';
    theme: string;
    themes: Themes;
    private _unsubscribeAll: Subject<any> = new Subject<any>();
    private applyThemeConfigurationFromDB = true;
    isMultipleTabs = false;
    userData: any;
    /**
     * Constructor
     */
    constructor(
        private _router: Router,
        private _fuseConfigService: FuseConfigService,
        private tabUpdateService: TabUpdateService,
        private toastrNotificationService: ToastrNotificationService,
        private tabGetService: TabGetService,
        private authService: AuthService
    ) {}

    // -----------------------------------------------------------------------------------------------------
    // @ Lifecycle hooks
    // -----------------------------------------------------------------------------------------------------

    /**
     * On init
     */
    ngOnInit(): void {
        this.userData = JSON.parse(localStorage.getItem('loginData'));

        if (this.authService._authenticated) {
            const themeSettings = {
                AppObjectName: 'TABMD_Configurations',
                DSQName: 'Detail_TABMD_Configurations',
                Reqtokens: { ID: '1E8FF0CD-61B8-47D9-9D40-E7455760BDE2' },
            };
            this.tabGetService
                .executeRAWDSQWithName(themeSettings)
                .subscribe((res) => {
                    if (res?.StatusCode == '200') {
                        if (
                            res?.Result &&
                            res?.Result.length > 0 &&
                            res.Result[0].Setting_Value
                        ) {
                            this._fuseConfigService.config$
                                .pipe(takeUntil(this._unsubscribeAll))
                                .subscribe((config: AppConfig) => {
                                    // Store the config
                                    if (!this.applyThemeConfigurationFromDB) {
                                        this.config = config;
                                    } else {
                                        this.applyThemeConfigurationFromDB =
                                            false;
                                        if (
                                            res?.Result[res?.Result.length - 1]
                                                .Setting_Value
                                        ) {
                                            this.config = JSON.parse(
                                                res?.Result[
                                                    res?.Result.length - 1
                                                ].Setting_Value
                                            );
                                            this._fuseConfigService.config =
                                                this.config;
                                        }
                                    }
                                    this.isMultipleTabs =
                                        this.config.multipleTabs;
                                });
                        }
                    }
                });
        }
    }

    //Theme setting button use and access only Administrator user role
    isSettingUsercanAccess() {
        if (
            this.userData &&
            this.userData.Roles !== null &&
            this.userData.Roles.length > 0 &&
            this.userData.Roles[0]?.RoleName === 'Administrator'
        ) {
            return true;
        }

        return false;
    }

    /**
     * On destroy
     */
    ngOnDestroy(): void {
        // Unsubscribe from all subscriptions
        this._unsubscribeAll.next(null);
        this._unsubscribeAll.complete();
    }

    // -----------------------------------------------------------------------------------------------------
    // @ Public methods
    // -----------------------------------------------------------------------------------------------------

    /**
     * Set the layout on the config
     *
     * @param layout
     */
    setLayout(layout: string): void {
        // Clear the 'layout' query param to allow layout changes
        this._router
            .navigate([], {
                queryParams: {
                    layout: null,
                },
                queryParamsHandling: 'merge',
            })
            .then(() => {
                // Set the config
                this._fuseConfigService.config = { layout };
            });
    }

    setMultipleTabs(event: any): void {
        const multipleTabs = event.checked;
        this.isMultipleTabs = multipleTabs;
        this._fuseConfigService.config = { multipleTabs };
    }

    /**
     * Set the scheme on the config
     *
     * @param scheme
     */
    setScheme(scheme: Scheme): void {
        this._fuseConfigService.config = { scheme };
    }

    /**
     * Set the theme on the config
     *
     * @param theme
     */
    setTheme(theme: Theme): void {
        this._fuseConfigService.config = { theme };
    }

    saveConfiguration() {
        const themeData = {
            data: {
                Setting_Value: this._fuseConfigService._config.getValue(),
            },
        };

        //setting key id for updating theme layout
        let fieldToUpdate: TabClasses.UpdateQueryParameter[] = [];
        fieldToUpdate.push({
            FieldName: 'Id',
            FieldValue: '1E8FF0CD-61B8-47D9-9D40-E7455760BDE2',
        });

        this.tabUpdateService
            .updateUsingDSQ(
                '2750F9C1-9BDF-4374-B8D1-860E73CE59E1',
                themeData,
                'TABMD_Configurations',
                fieldToUpdate
            )
            .subscribe((res) => {
                if (res?.StatusCode == '200') {
                    this.toastrNotificationService.showSuccess(
                        'Theme Configuration updated successfully',
                        'TAB Notification'
                    );
                    location.reload();
                } else {
                    this.toastrNotificationService.showError(
                        'Sorry !!! We are not able to update theme settings',
                        'TAB Notification'
                    );
                }
            });
    }
}
