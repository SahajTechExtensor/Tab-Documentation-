import {
    Component,
    EventEmitter,
    Input,
    OnInit,
    Output,
    ViewChild,
} from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { Router } from '@angular/router';
import { TabDynamicDialogComponent } from 'app/shared/templates/tab-dynamic-dialog/tab-dynamic-dialog.component';
import { TabDynamicSidebarComponent } from 'app/shared/templates/tab-dynamic-sidebar/tab-dynamic-sidebar.component';
import {
    TabCoreHttpService,
    TabRequestResponseFunction,
} from '@techextensor/tab-service';
import { take } from 'rxjs';
// this button html template which we are rendering in grid as button
@Component({
    selector: 'button-template',
    template: `<button
        class="custom-grid-action-btn-css"
        ejs-button
        (click)="clicked($event)"
    >
        {{ data?.column?.columnConfig?.buttonLabel }}
    </button>`,
})
export class ButtonTemplateComponent implements OnInit {
    @Input() data: any;
    @Output() onSubmit = new EventEmitter<any>();
    @Output() customEvent = new EventEmitter<any>();

    @ViewChild('tabDynamicSidebarComponent')
    tabDynamicSidebarComponent: TabDynamicSidebarComponent;

    constructor(
        private router: Router,
        public dialog: MatDialog,
        private tabCoreHttpService: TabCoreHttpService
    ) {}
    ngOnInit() {}
    clicked(event) {
        if (this.data?.column?.columnConfig?.action == 'urlRedirect') {
            window.open(this.data.column.columnConfig.linkurl, '_blank');
        } else if (
            this.data?.column?.columnConfig?.action == 'screenRedirect'
        ) {
            if (
                this.data?.column?.columnConfig?.openScreenAs === 'navigation'
            ) {
                this.router.navigate(
                    ['/screen' + '/' + this.data.column.columnConfig.screen],
                    {
                        queryParams: { Id: this.data.Id },
                    }
                );
            } else if (
                this.data?.column?.columnConfig?.openScreenAs == 'modelPopup'
            ) {
                let parameters = this.data;
              if (this.data?.hasOwnProperty('ID') || this.data?.hasOwnProperty('Id')) {
                // Determine the property name dynamically
                const propertyName = this.data.hasOwnProperty('ID') ? 'ID' : 'Id';

               // Create the parameters object with the dynamically determined property name
                parameters = {[propertyName]: this.data[propertyName]};
            }
               
                const ref = this.dialog.open(TabDynamicDialogComponent, {
                    width: '1000px',
                    height: '1000px',
                    data: {
                        screenID: this.data.column.columnConfig.screen,
                        data: JSON.stringify(parameters),
                    },
                });

                ref.afterClosed().subscribe((result) => {
                    this.data[this.data?.column?.field] =
                        JSON.parse(result)?.[this.data?.column?.field];
                    this.onSubmit.emit(this.data);
                });
            } else if (
                this.data?.column?.columnConfig?.openScreenAs == 'sidebar'
            ) {
                let parameters = this.data;
                // FOR REFERENCE
                // this.tabDynamicSidebarComponent.title =
                //     this.data?.column?.field;
                // this.tabDynamicSidebarComponent.settingsDrawer.toggle();
                // if (this.data[this.data?.column?.field]) {
                //     const parsedData = JSON.stringify(
                //         this.data[this.data?.column?.field]
                //     );
                //     this.tabDynamicSidebarComponent?.tabChildScreenRenderer?.refreshFormData(
                //         parsedData ? parsedData : {}
                //     );
                // }
                // if (this.data?.hasOwnProperty('Id')) {
                //     const value = {
                //         Id: this.data.Id,
                //     };
                //     this.tabDynamicSidebarComponent.parameters =
                //         JSON.stringify(value);
                // }

                let sidebarDetails = {
                    detail: {
                        parameters: parameters,
                        pageTitle: this.data?.column?.field,
                        screenId: this.data?.column?.columnConfig?.screen,
                        screenMode: 'Add',
                        showSubmit: false,
                        eventName: 'openDefaultSidebar',
                    },
                };
                // if we have value it will be like edit mode so we are making screenmode = edit
                if (
                    this.data?.hasOwnProperty('Id') ||
                    this.data?.hasOwnProperty('ID')
                ) {
                    sidebarDetails['detail']['screenMode'] = 'Edit';
                }
                // already code written for some purpose we need to make clear this scenario
                if (this.data?.hasOwnProperty('Id')) {
                    parameters = { Id: this.data.Id };
                    sidebarDetails['detail']['screenMode'] = 'Edit';
                }
                const customEvent = new CustomEvent(
                    'tabUtilsEvent',
                    sidebarDetails
                );
                window.dispatchEvent(customEvent);
            }
        } else if (this.data?.column?.columnConfig?.action == 'postToUrl') {
            console.log(this.data?.column);
            var headers = TabRequestResponseFunction.getHeaders();
            if (
                this.data?.column?.columnConfig?.headers &&
                this.data?.column?.columnConfig?.headers.length > 0
            ) {
                this.data?.column?.columnConfig?.headers.forEach((element) => {
                    headers.append(element.header, element.value);
                });
                console.log(this.data?.column?.columnConfig?.headers);
            }
            this.tabCoreHttpService
                .get(this.data?.column?.columnConfig?.postToUrl, headers)
                .pipe(take(1))
                .subscribe();
        } else if (this.data?.column?.columnConfig?.action == 'customEvent') {
            // passing clicked row data
            event.rowData = this.data;
            this.customEvent.emit(event);
        }
    }
    onSidebarSubmit(event) {
        this.data[this.data?.column?.field] = event?.[this.data?.column?.field];
        this.onSubmit.emit(this.data);
        this.tabDynamicSidebarComponent.settingsDrawer.close();
    }
}
