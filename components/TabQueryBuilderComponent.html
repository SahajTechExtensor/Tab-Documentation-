    getAppObjectFields(appObjectId) {
        this.appObjectData = TabAppFunction.getAppObject(appObjectId);

            each(this.appObjectData.Fields, (f) => {

    // binding all the fields to query builder component with lookup field also
    bindAppObjectFieldsToQueryBuilder(prefix, item) {
        let columnData = {
            field: item.ID,
            label: prefix
                ? prefix + '.' + item.SystemDBFieldName
                : item.SystemDBFieldName,
            type: TabCommonFunction.getFieldsType(
                item?.FieldType.DataType,
                item.hasOwnProperty('LookUpDetails')
            ),
            operators: TabCommonFunction.getOperatorListForQueryBuilder(
                TabCommonFunction.getFieldsType(
                    item?.FieldType.DataType,
                    item.hasOwnProperty('LookUpDetails')
                )
            ),
        };
        this.columns.push(columnData);
        this.properties.push(columnData.label);
        if (item.hasOwnProperty('LookUpDetails') && item.LookUpDetails) {
            let lookupAppObjectData =
                TabAppFunction.getAppObjectFromSystemDBTableName(
                    item.LookUpDetails.LookupObject
                );
            if (lookupAppObjectData) {
                if (Object.keys(this.previousLookup).length > 0) {
                    if (
                        this.previousLookup.LookupObject ==
                        item.LookUpDetails.LookupObject
                    ) {
                        return;
                    }
                }
                this.previousLookup = item.LookUpDetails;
                let nprefix = prefix
                    ? prefix + '.' + item.SystemDBFieldName
                    : item.SystemDBFieldName;
                if (lookupAppObjectData?.Fields) {
                    each(lookupAppObjectData.Fields, (ff) => {
                        this.bindAppObjectFieldsToQueryBuilder(nprefix, ff);
                    });
                }
            }
        }
        // else {
        //     this.columns.push(columnData);
        // }
    }

    // get ths selected datasourcequery filter list and bind to querybuilder component
    // from app json
    getFilterListByDSQ(dsqid) {
        if (this.appObjectData?.DataSourceQueries) {
            let dsqData = this.appObjectData?.DataSourceQueries.find(
                (a: any) => {
                    return a.ID.toLowerCase() === dsqid.toLowerCase();
                }
            );
            if (dsqData) {
                if (dsqData?.Filters?.Filters) {
                    let filterList = dsqData?.Filters?.Filters;
                    if (filterList && filterList.length) {
                        this.createdControl(filterList);
                    }
                }
            }
        }
    }

    // get ths selected datasourcequery filter list and bind to querybuilder component
    // from run api so we can get latest data after update and insert
    getFilterListByDSQFromAPI(dsqid) {
        const reqtokens = { ID: dsqid };
        const apiData = {
            Reqtokens: reqtokens,
            AppObjectName: 'TABMD_DataSourceQueries_Filters',
            DSQName:
                'FK_TABMD_DataSourceQueries_Filters_TABMD_DataSourceQueries',
        };
        this.tabGetService
            .executeRAWDSQWithName(apiData)
            .pipe(take(1))
            .subscribe((response) => {
                // console.log('filterlist');
                if (!!response.Result && response.Result.length > 0) {
                    this.dsqFilterList = response.Result;
                    this.createdControl(this.dsqFilterList);
                }
            });
    }

    getFilterFromJSON(filterList) {
        if (filterList && filterList.length) {
            this.createdControl(filterList);
        }
    }

    createdControl(dsqFilterList: any): void {
        let groupByGroups: any = [];
        if (dsqFilterList.length) {
            groupByGroups = this.groupArrayOfObjects(dsqFilterList, 'GroupID');
            this.setTimer = setTimeout(() => {
                if (this.tabQueryBuilder) {
                    this.tabQueryBuilder.reset();
                    Object.keys(groupByGroups).forEach((element) => {
                        let rules = this.getGroupByRule(groupByGroups[element]);
                        this.rules = { rules: rules };
                        const groupId = 'group' + (Number(element) - 1);
                        this.tabQueryBuilder?.addGroups(rules, groupId);
                    });
                }
            }, 100);
        }
    }
    groupArrayOfObjects(list: any, key: string) {
        return list.reduce(function (rv: any, x: any) {
            (rv[x[key]] = rv[x[key]] || []).push(x);
            return rv;
        }, {});
    }
    getGroupByRule(singleGroupArray: any) {
        let rules: any = [];
        let rr: any = [];
        let cnd = '';
        singleGroupArray.forEach((element: any) => {
            // let fd = this.columnData.find((a: any) => {
            //     return a.id === element.FieldID;
            // });
            let fd = this.columns.find((a: any) => {
                // return a.field.toLowerCase() === element.Field.ID.toLowerCase(); // old code when data come from appjson
                return a.field.toLowerCase() === element.FieldID.toLowerCase(); // new code when data come from api
            });
            if (fd) {
                let rule = {
                    label: fd.field,
                    field: fd.field,
                    // operator: element.RelationalOperator,
                    operator:
                        TabCommonFunction.getRelationalOperatorValueToString(
                            element.RelationalOperator
                        ),
                    value:
                        TabCommonFunction.getFilterValueTypeInString(
                            element.FilterValueType
                        ) +
                        ',' +
                        element.value,
                    // value:
                    //     TabCommonFunction.getFilterValueTypeInString(
                    //         element.ValueType
                    //     ) +
                    //     ',' +
                    //     element.value,
                    id: element.FieldID,
                    // id: element.Field.ID,
                };
                rr.push(rule);
                cnd = TabCommonFunction.getConjuctionClauseName(
                    element.ConjuctionClause
                );
            }
        });
        rules.push({ condition: cnd, rules: rr });
        return rules;
    }

    getParameterListByDSQ(dsqid) {
        if (this.appObjectData?.DataSourceQueries) {
            let dsqData = this.appObjectData?.DataSourceQueries.find(
                (a: any) => {
                    return a.ID.toLowerCase() === dsqid.toLowerCase();
                }
            );
            if (dsqData) {
                this.params = [];
                if (dsqData?.Parameters.length) {
                    dsqData?.Parameters.forEach((element: any) => {
                        this.params.push(element.MappingFieldName);
                    });
                }
            }
        }
    }

    getParameters(dataSourceQueryID) {
        let paramData = {
            Reqtokens: { ID: dataSourceQueryID },
            AppObjectName: 'TABMD_DataSourceQueries_Parameters',
            DSQName:
                'FK_TABMD_DataSourceQueries_Parameters_TABMD_DataSourceQueries',
        };

        this.tabGetService
            .executeRAWDSQWithName(paramData)
            .pipe(take(1))
            ?.subscribe((res) => {
                if (!!res.Result && res.Result.length > 0) {
                    const params: any = res.Result;
                    this.params = [];
                    params.forEach((element: any) => {
                        this.params.push(element.MappingFieldName);
                    });
                }
            });
    }

    onRuleChange(args: any) {
        this.processRulesToQuery(args);
    }

    onValueChange(event: any, ruleID: string, type: string): void {
        let element: any = document
            .getElementById(ruleID)
            ?.querySelector('.e-rule-value');
        let value;
        switch (type) {
            case 'filter':
                value = element.querySelector('#value1').textContent;
                if (value == 'Parameter') {
                    // this.getParameters();
                    if (this.builderInstance?.parameters?.dataSourceQueryId) {
                        this.getParameters(
                            this.builderInstance?.parameters?.dataSourceQueryId
                        );
                    }
                }
                break;
            case 'parameter':
                value =
                    element.querySelector('#value1').textContent +
                    ',' +
                    element.querySelector('#value2').textContent;
                break;
            case 'property':
                value =
                    element.querySelector('#value1').textContent +
                    ',' +
                    element.querySelector('#value3').textContent;
                break;
            case 'value':
                value =
                    element.querySelector('#value1').textContent +
                    ',' +
                    event.target.value;
                break;
        }
        this.tabQueryBuilder?.notifyChange(value as string, element, 'value');
    }

    processRulesToQuery(args: RuleChangeEventArgs) {
        this.rules = [];
        if (args?.rule && args?.rule?.rules?.length) {
            let gc = 0;
            args.rule.rules.forEach((element) => {
                if (element.hasOwnProperty('rules')) {
                    gc++;
                    this.updateGroupRules(element, gc);
                } else {
                    if (element.hasOwnProperty('value')) {
                        let sr = {
                            ConjuctionClause:
                                TabCommonFunction.getConjuctionClause(
                                    args.rule.condition
                                ),
                            FieldID: element.field,
                            RelationalOperator:
                                TabCommonFunction.getRelationalOperatorValueToNumber(
                                    element.operator
                                ),
                            // RelationalOperator: element.operator,
                            Value:
                                typeof element.value == 'string'
                                    ? element.value.split(',')[1]
                                    : '',
                            ValueType:
                                typeof element.value == 'string'
                                    ? element.value.split(',')[0]
                                    : '', // this should be FilterValueType like Literal,Parameter
                            label: element.label,
                            group: 1,
                        };
                        this.rules.push(sr);
                    }
                }
            });
        }
        this.queryValue.WhereClause.Filters = this.rules;
        this.builderInstance.queryValue = this.queryValue;
        // this.valueChange.emit(this.queryValue);
    }

    updateGroupRules(groupData: any, gc: number) {
        let gcc = gc;
        if (
            groupData &&
            groupData.hasOwnProperty('rules') &&
            groupData.rules &&
            groupData.rules.length > 0
        ) {
            groupData.rules.forEach((element: any) => {
                if (element.hasOwnProperty('rules')) {
                    gcc = gc++;
                    this.updateGroupRules(element, gcc);
                } else {
                    if (element.hasOwnProperty('value')) {
                        let sr = {
                            ConjuctionClause:
                                TabCommonFunction.getConjuctionClause(
                                    groupData.condition
                                ),
                            FieldID: element.field,
                            RelationalOperator:
                                TabCommonFunction.getRelationalOperatorValueToNumber(
                                    element.operator
                                ),
                            Value:
                                typeof element.value == 'string'
                                    ? element.value.split(',')[1]
                                    : '',
                            ValueType:
                                typeof element.value == 'string'
                                    ? element.value.split(',')[0]
                                    : '', // this should be FilterValueType like Literal,Parameter
                            label: element.label,
                            group: gc,
                        };
                        this.rules.push(sr);
                    }
                }
            });
        }
    }

        let apiData = {

            AppObjectName: 'TABMD_AppFields',
            DSQName: 'Default_TABMD_AppFields',
        };
        this.tabGetService

            .pipe(take(1))
            .subscribe((response) => {
                if (!!response.Result && response.Result.length > 0) {
                    this.allFieldList = response.Result;
                }
            });
    }

    processQueryToRules() {
        let rules: any = [];
        if (this.queryValue) {
            if (
                this.queryValue.WhereClause &&
                Object.keys(this.queryValue.WhereClause).length &&
                this.queryValue.WhereClause.Filters.length
            ) {
                this.queryValue.WhereClause.Filters.forEach((element: any) => {
                    let rule = {
                        label: element.FieldID,
                        field: element.FieldID,
                        operator:
                            TabCommonFunction.getRelationalOperatorValueToString(
                                element.RelationalOperator
                            ),
                        value: TabCommonFunction.getFilterType(element),
                        id: element.FieldID,
                    };
                    rules.push(rule);
                });
            }
        }
        return rules;
    }

    saveFilters() {
        this.formioEvent.emit({
            eventName: 'customEvent',
            data: {
                filters: this.queryValue,
                action: 'saveDSQFilters',
                // dsqID: this.dataSetting?.appObjectId,
                dsqID: this.builderInstance?.parameters?.dataSourceQueryId,
            },
        });
    }

    // filtering event handler to filter a Country

    public onFiltering: EmitType<FilteringEventArgs> = (
        e: FilteringEventArgs,
        fieldType: string
    ) => {
        let query: any = new Query();
        query =
            e.text !== '' ? query.where('', 'contains', e.text, true) : query;
        if (fieldType == 'property') {
            e.updateData(this.properties, query);
        } else if (fieldType == 'parameter') {
            e.updateData(this.params, query);
        }
    };
    ngOnDestroy(): void {
        clearTimeout(this.setTimer);
    }
}
