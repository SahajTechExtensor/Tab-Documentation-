import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';
import { FormioCustomComponent, FormioUtils } from '@formio/angular';
import {
    TabAppFunction,
    TabClasses,
    TabGridFunction,
    TabRequestResponseFunction,
} from '@techextensor/tab-service';
import { cloneDeep, each } from 'lodash';

@Component({
    selector: 'tab-custom-form-builder',
    templateUrl: './tab-custom-form-builder.component.html',
    styleUrls: ['./tab-custom-form-builder.component.css'],
})
export class TabCustomFormBuilderComponent
    implements FormioCustomComponent<any>, OnInit
{
    @Output() valueChange = new EventEmitter<any>();
    rebuildEmitter = new EventEmitter<any>();
    @Input() disabled: boolean = true;
    @Input() values: {};
    @Input() dataSetting: any;
    type = 'builder';
    form: any = { display: 'form', components: [] };
    @Input() dataSourceQueryId = '';
    builderOption: any;
    appObject: TabClasses.AppObject;
    staticJson1: any = {
        builder: {
            tabBasic: {
                title: 'Custom',
                weight: 10,
            },
        },
    };
    constructor() {}

    private _value: any;
    @Input()
    public set value(v: any) {
        if (v !== null && v !== 'null' && v !== '' && v !== this.form) {
            this._value = v;
            let formJson;
            if (typeof v === 'string') {
                formJson = JSON.parse(v);
            } else {
                formJson = v;
            }
            this.form = cloneDeep(formJson);
            this._value = this.form;
        }
        if (
            this.dataSetting?.dataSourceQueryId &&
            this.dataSetting?.dataSourceQueryId !== ''
        ) {
            this.appObject = TabAppFunction.getDataFromDataSourceQueryID(
                this.dataSetting?.dataSourceQueryId
            );

            this.generateJSONForFormBuilder(this.appObject);
        }
    }
    public get value(): any {
        return this._value;
    }

    selectedType = 'form';
    ngOnInit(): void {
        this.createNewFormBuilderWithDataSourceFields(this.staticJson1);

        if (this.dataSourceQueryId) {
            this.appObject = TabAppFunction.getDataFromDataSourceQueryID(
                this.dataSourceQueryId
            );
            this.generateJSONForFormBuilder(this.appObject);
        }
    }
    generateJSONForFormBuilder(appObject: TabClasses.AppObject) {
        let masterAppObj: TabClasses.AppObject = TabAppFunction.getAppObject(
            appObject?.ObjectID_Tosave
        );
        let allfields = appObject?.Fields;
        let fields: any = [];
        allfields?.forEach((field) => {
            let doesexists;
            if (
                field.Field &&
                field.Field.FieldType.DataType &&
                field.Field.FieldType.DataType !== 6
            ) {
                let str =
                    field.Field.SystemDBFieldName.substr(0, 1).toLowerCase() +
                    field.Field.SystemDBFieldName.substr(1);
                if (this.form != null) {
                    doesexists = FormioUtils.getComponent(
                        this.form['components'],
                        str,
                        true
                    );
                }
            }
            if (doesexists == null) {
                fields.push(field);
            }
        });
        let staticJson2: any = {
            builder: {
                custom: {
                    title: 'Data Source Fields',
                    weight: 10,
                    components: {},
                },
            },
        };
        let staticJson = {
            builder: {
                ...this.staticJson1.builder,
                ...staticJson2.builder,
            },
        };

        if (fields.length) {
            each(fields, (f) => {
                if (f.Field && f.Field.DisplayName) {
                    let singleObj = {
                        title: f.Field.DisplayName,
                        key: f.Field.SystemDBFieldName,
                        icon: 'terminal',
                        schema: TabRequestResponseFunction.generateFieldsSchema(
                            f.Field
                        ),
                    };
                    staticJson.builder.custom.components[f.Field.DisplayName] =
                        singleObj;
                }
            });
        }
        let singleObj = {
            title: masterAppObj?.SystemDBTableName,
            key: 'customGrid',
            icon: 'terminal',
            schema: TabGridFunction.generateGridSchema(
                masterAppObj,
                this.dataSourceQueryId
            ),
        };
        staticJson.builder.custom.components[masterAppObj?.SystemDBTableName] =
            singleObj;
        if (masterAppObj?.hasOwnProperty('ChildRelationShips')) {
            if (masterAppObj?.ChildRelationShips.length) {
                each(masterAppObj?.ChildRelationShips, (c) => {
                    let schema = TabGridFunction.generateChildGridSchema(
                        c,
                        masterAppObj
                    );
                    if (Object.keys(schema).length) {
                        let singleObj = {
                            title: c?.childDetails?.LookupObject,
                            key: 'customGrid',
                            icon: 'terminal',
                            schema: TabGridFunction.generateChildGridSchema(
                                c,
                                masterAppObj
                            ),
                        };
                        staticJson.builder.custom.components[
                            c.childDetails.LookupObject
                        ] = singleObj;
                    }
                });
            }
        }
        this.createNewFormBuilderWithDataSourceFields(staticJson);
    }

    createNewFormBuilderWithDataSourceFields(builderJson: any) {
        this.builderOption = builderJson;
        this.rebuildEmitter.next(this.builderOption);
    }

    updateValue(event: any) {
        this.manageResourceFields();
        if (
            event.type == 'addComponent' ||
            event.type == 'deleteComponent' ||
            event.type == 'removeComponent' ||
            event.type == 'saveComponent'
        ) {
            this._value = event.form;
            this.valueChange.emit(event.form);
        }
    }

    manageResourceFields() {
        if (this.dataSourceQueryId) {
            this.generateJSONForFormBuilder(this.appObject);
        }
    }

    capitalizeFirstLetter(form: any): any {
        const updatedForm = { ...form };
        if (updatedForm.components) {
            updatedForm.components.forEach((component: any) => {
                if (component.key && typeof component.key === 'string') {
                    component.key =
                        component.key.charAt(0).toUpperCase() +
                        component.key.slice(1);
                }
            });
        }
        return updatedForm;
    }

    onDropDownChange() {
        this.form['display'] = this.selectedType;
        this.rebuildEmitter.next(this.builderOption);
        this.valueChange.emit(this.form);
    }
}
