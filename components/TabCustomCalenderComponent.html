import {
    Component,
    EventEmitter,
    Input,
    Output,
    TemplateRef,
    ViewChild,
} from '@angular/core';
import { FormioCustomComponent } from '@formio/angular';
import { isNullOrUndefined, closest, extend } from '@syncfusion/ej2-base';
import {
    DayService,
    WeekService,
    WorkWeekService,
    MonthService,
    AgendaService,
    EventSettingsModel,
    ScheduleComponent,
    ActionEventArgs,
    PopupOpenEventArgs,
    CurrentAction,
    CallbackFunction,
} from '@syncfusion/ej2-angular-schedule';
import {
    TabAppFunction,
    TabDeleteService,
    TabGetService,
} from '@techextensor/tab-service';
import { ToastrNotificationService } from 'app/layout/common/notifications/toastr.service';
import { take } from 'rxjs';

@Component({
    selector: 'tab-custom-calender',
    providers: [
        DayService,
        WeekService,
        WorkWeekService,
        MonthService,
        AgendaService,
    ],
    templateUrl: './tab-custom-calender.component.html',
    styleUrls: ['./tab-custom-calender.component.css'],
})
export class TabCustomCalenderComponent implements FormioCustomComponent<any> {
    @ViewChild('scheduleObj') public scheduleObj: ScheduleComponent;

    @ViewChild('editorTemplate') editorTemplate: TemplateRef<any>;

    scheduleData: Record<string, any>[] = [];
    data: any
    @Output() valueChange = new EventEmitter<any>();
    @Input() disabled: boolean;
    @Input() dataSetting: any;
    private setTimer: any;

    parameters: any = {
        parameters: {},
    };

    // private dataManager: DataManager = new DataManager({
    //     url: 'https://ej2services.syncfusion.com/production/web-services/api/Schedule',
    //     adaptor: new WebApiAdaptor(),
    //     crossDomain: true,
    // });
    public eventSettings: EventSettingsModel = { dataSource: extend([], this.scheduleData, null, true) as Record<string, any>[] };

    public selectedDate: Date = new Date(2021, 1, 15);

    appObjectDetails: any;
    dsqDetails: any;

    @Input()
    public set value(v: any) {
        if (v && v != null) {
            this.data = v;
        }
        this.setTimer = setTimeout(() => {
            if (this.dataSetting?.datasourceQueryId) {
                this.processdata();
            }
        }, 400);
    }
    public get value(): any {
        return this.data;
    }

    constructor(
        private tabGetService: TabGetService,
        private tabDeleteService: TabDeleteService,
        private toastrNotificationService: ToastrNotificationService
    ) { }

    ngOnInit(): void {
    }

    processdata() {
        this.appObjectDetails =
            TabAppFunction.getAppObjectFromDataSourceQueryID(
                this.dataSetting?.datasourceQueryId
            );

        this.dsqDetails = TabAppFunction.getDataFromDataSourceQueryID(
            this.dataSetting?.datasourceQueryId
        );

        let apiData = {
            AppObjectName: this.appObjectDetails.ObjectName,
            DSQName: this.dsqDetails.QueryName,
            Reqtokens: this.data.parameters,
        };

        this.tabGetService
            .executeRAWDSQWithName(apiData)
            ?.pipe(take(1))
            ?.subscribe((res: any) => {
                if (res.Result.length > 0) {
                    res.Result.forEach((element) => {
                        let i = 0;
                        var StartdateString = element[this.dataSetting?.StartDateField.AppFieldID.FieldName];
                        var enddateString = element[this.dataSetting?.endDateField.AppFieldID.FieldName];
                        let data = {
                            Id: element[this.dataSetting?.ValueField.AppFieldID.FieldName],
                            Subject: element[this.dataSetting?.TitleField.AppFieldID.FieldName],
                            StartTime: this.parseDateString(StartdateString),
                            EndTime: this.parseDateString(enddateString)
                        };
                        // this will check any dublicate meeting is in scheduleData
                        let isDuplicateId = this.scheduleData.some(item => item.Id === data.Id);
                        if (!isDuplicateId) {
                            this.scheduleData.push(data);
                        }
                    });
                    this.eventSettings = {
                        dataSource: this.scheduleData,
                    };
                }
            });
    }
    // this will convert the date and time in the format of which calender component expect
    parseDateString(dateString) {
        if (!dateString) return null;
        var parts = dateString.split(' ');
        if (parts) {
            var datePart = parts[0]?.split('-');
            if (datePart) {
                var year = parseInt(datePart[2]);
                var month = parseInt(datePart[1]) - 1;
                var month = parseInt(datePart[1]) - 1;
                var day = parseInt(datePart[0]);
            }
            var timePart = parts[1]?.split(':');

            if (timePart) {
                var hour = parseInt(timePart[0]);
                var minute = parseInt(timePart[1]);
                var second = parseInt(timePart[2]);
            }
            if (parts[2] === "PM" && hour < 12) {
                hour += 12;
            }

            return new Date(year, month, day, hour, minute, second);
        }
    }
    onPopupOpen(args: PopupOpenEventArgs): void {
        if (args.data.hasOwnProperty('Id') && !args.data.hasOwnProperty('IsAllDay')) {
            this.parameters[this.dataSetting.ValueField.AppFieldID.FieldName] = args.data?.Id;
            this.parameters.parameters = {};
        } else if (args.data.hasOwnProperty('IsAllDay')) {
            this.parameters.parameters['screenMode'] = 'Add';
            this.parameters.parameters['type'] = args.type;
            if (args.data.hasOwnProperty('StartTime')) {
                this.parameters.parameters[this.dataSetting.StartDateField.AppFieldID.FieldName] =
                    args.data?.StartTime;
            }

            if (args.data.hasOwnProperty('EndTime')) {
                this.parameters.parameters[this.dataSetting.endDateField.AppFieldID.FieldName] =
                    args.data?.EndTime;
            }
            // if we need to pass any key value in popup that we can pass that key value in customParams
            if (this.data.hasOwnProperty('customParams')) {
                this.parameters.parameters['customParams'] = this.data.customParams;
            }
        }
    }

    public buttonClickActions(e: Event): void {
        const quickPopup: HTMLElement = closest(
            e.target as HTMLElement,
            '.e-quick-popup-wrapper'
        ) as HTMLElement;
        const getSlotData: CallbackFunction = (): Record<string, any> => {
            const subject = '';
            const notes = '';
            const addObj: Record<string, any> = {};
            addObj.Id = this.scheduleObj.getEventMaxID();
            addObj.Subject = isNullOrUndefined(subject) ? 'Add title' : subject;
            addObj.StartTime = new Date(
                this.scheduleObj.activeCellsData.startTime
            );
            addObj.EndTime = new Date(this.scheduleObj.activeCellsData.endTime);
            addObj.IsAllDay = this.scheduleObj.activeCellsData.isAllDay;
            addObj.Description = isNullOrUndefined(notes) ? 'Add notes' : notes;
            // addObj.RoomId = (
            //     (quickPopup.querySelector('#eventType') as EJ2Instance)
            //         .ej2_instances[0] as DropDownListComponent
            // ).value;
            return addObj;
        };
        if ((e.target as HTMLElement).id === 'add') {
            const addObj: Record<string, any> = getSlotData();
            this.scheduleObj.addEvent(addObj);
        } else if ((e.target as HTMLElement).id === 'delete') {
            const eventDetails: Record<string, any> = this.scheduleObj
                .activeEventData.event as Record<string, any>;
            let currentAction: CurrentAction;
            if (eventDetails.RecurrenceRule) {
                currentAction = 'DeleteOccurrence';
            }
            this.scheduleObj.deleteEvent(eventDetails, currentAction);
        } else {
            const isCellPopup: boolean =
                quickPopup.firstElementChild.classList.contains('e-cell-popup');
            const eventDetails: Record<string, any> = isCellPopup
                ? getSlotData()
                : (this.scheduleObj.activeEventData.event as Record<
                    string,
                    any
                >);
            let currentAction: CurrentAction = isCellPopup ? 'Add' : 'Save';
            if (eventDetails.RecurrenceRule) {
                currentAction = 'EditOccurrence';
            }
            this.scheduleObj.openEditor(eventDetails, currentAction, true);
        }
        this.scheduleObj.closeQuickInfoPopup();
    }

    onActionComplete(args: ActionEventArgs) {
        if (args.requestType === 'eventRemoved') {
            let data = { data: {} };
            data.data[this.dataSetting?.ValueField.AppFieldID.FieldName] = args.data[0].Id;
            this.tabDeleteService
                .deleteRecord('', data, this.dsqDetails.ObjectID_Tosave)
                ?.pipe(take(1))
                ?.subscribe((response) => {
                    if (
                        response.StatusCode == '200' ||
                        !response.Errors.length
                    ) {
                        this.toastrNotificationService.showSuccess(
                            'Record Deleted Successfully',
                            'Success'
                        );
                    } else {
                        this.toastrNotificationService.showError(
                            response.Errors[0],
                            'Error'
                        );
                    }
                });
        }
    }
    ngOnDestroy(): void {
        clearTimeout(this.setTimer);
    }
}
