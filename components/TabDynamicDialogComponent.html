import { Component, ElementRef, Inject, OnInit, ViewChild } from '@angular/core';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import screenfull from 'screenfull';

@Component({
    selector: 'app-tab-dynamic-dialog',
    templateUrl: './tab-dynamic-dialog.component.html',
    styleUrls: ['./tab-dynamic-dialog.component.scss'],
})
export class TabDynamicDialogComponent implements OnInit {
    screenId = '';
    @ViewChild('tabChildScreenRenderer') tabChildScreenRenderer: any;
    @ViewChild('dynamicDialog') dynamicDialog: ElementRef;
    parsedData: any;
    private setTimer: any;
    expanded = false;

    constructor(
        public dialogRef: MatDialogRef<TabDynamicDialogComponent>,
        @Inject(MAT_DIALOG_DATA) public data: any
    ) {}

    ngOnInit(): void {
        this.setTimer = setTimeout(() => {
            if (
                this.tabChildScreenRenderer &&
                this.data?.data &&
                this.data?.data !== ''
            ) {
                this.parsedData = JSON.parse(this.data?.data);
            }
        }, 50);
    }

    onChange(event): void {}

    onSubmit(Event): void {}

    onClose() {
        this.dialogRef.close(
            JSON.stringify(this.tabChildScreenRenderer.latestForm)
        );
    }

    toggleExpand() {
        this.expanded = !this.expanded;
        if (screenfull.isEnabled) {
            screenfull.toggle(this.dynamicDialog?.nativeElement);
        }
    }

    ngOnDestroy(): void {
        clearTimeout(this.setTimer);
    }
}
