import {
    Component,
    ViewChild,
    EventEmitter,
    OnInit,
    ElementRef,
    Output,
    Input,
} from '@angular/core';
import {
    TabAppFunction,
    TabCoreHttpService,
    TabGetService,
    TabRequestResponseFunction,
    TabTemplateService,
} from '@techextensor/tab-service';
import { forkJoin, interval, of, take } from 'rxjs';
import { MatDialog } from '@angular/material/dialog';
import {} from 'googlemaps';
import cloneDeep from 'lodash-es/cloneDeep';
import { Router } from '@angular/router';
import { TabDynamicSidebarComponent } from 'app/shared/templates/tab-dynamic-sidebar/tab-dynamic-sidebar.component';
import { TabDynamicDialogComponent } from 'app/shared/templates/tab-dynamic-dialog/tab-dynamic-dialog.component';

@Component({
    selector: 'app-tab-custom-track-map',
    templateUrl: './tab-custom-track-map.component.html',
    styleUrls: ['./tab-custom-track-map.component.scss'],
})
export class TabCustomTrackMapComponent implements OnInit {
    @Output() valueChange = new EventEmitter<any>();
    @Input() dataSettingsTrackMap;
    @ViewChild('map', { static: true }) mapElement: ElementRef;
    @Output() customEvent = new EventEmitter<any>();
    @ViewChild('tabDynamicSidebarComponent')
    tabDynamicSidebarComponent: TabDynamicSidebarComponent;
    map: google.maps.Map;
    centerLocation;
    lat: number = 23.051839;
    lng: number = 72.481382;
    markers = [];
    mapStyles = [];
    defaultMapStyles = [];
    dsqDetailsOfUserLocation;
    cloneDsqDetailsOfUserLocation;
    userLocationObj;
    dsqDetails;
    cloneDsqDetails;
    templateResult;
    infoWindow;

    @Input()
    public set value(v: any) {
        if (v && v != null) {
        }

        setTimeout(() => {
            this.setInterval();
            this.populateData();
        }, 100);
    }
    public get value(): any {
        return '';
    }

    constructor(
        public dialog: MatDialog,
        private tabCoreHttpService: TabCoreHttpService,
        private router: Router,
        private tabTemplateService: TabTemplateService,
        private tabGetService: TabGetService
    ) {}

    ngOnInit(): void {
        this.mapInitialize();
    }

    /**
     * this function use to google Map initialize.
     * set style,zoom,center location
     * Add a listener for the `center_changed` and `dragend` in map
     */
    mapInitialize() {
        this.defaultMapStyles = [
            {
                featureType: 'administrative.land_parcel',
                elementType: 'labels.text.fill',
                stylers: [{ color: '#bdbdbd' }],
            },
            {

            {

                stylers: [{ color: '#e5e5e5' }],
            },
            {
                featureType: 'poi.park',
                elementType: 'labels.text.fill',
                stylers: [{ color: '#9e9e9e' }],
            },
            {
                featureType: 'road',
                elementType: 'geometry',
                stylers: [{ color: '#ffffff' }],
            },
            {
                featureType: 'road.arterial',
                elementType: 'labels.text.fill',
                stylers: [{ color: '#757575' }],
            },
            {
                featureType: 'road.highway',
                elementType: 'geometry',
                stylers: [{ color: '#dadada' }],

        this.map = new google.maps.Map(this.mapElement.nativeElement, {
            center: new google.maps.LatLng(this.lat, this.lng),
            zoom: 15,
            mapTypeId: google.maps.MapTypeId.ROADMAP,
            disableDefaultUI: true,
            styles: this.defaultMapStyles,
        });

        // Add a listener for the 'center_changed' event
        this.map.addListener('center_changed', () => {
            this.centerLocation = this.map.getCenter().toJSON();
            this.lat = this.centerLocation.lat;
            this.lng = this.centerLocation.lng;
        });

        // Add a listener for the 'dragend' event
        this.map.addListener('dragend', () => {
            setTimeout(() => {
                this.mapApiCall(); // Call your desired function here
            }, 60 * 1000); // 1 minutes expressed in milliseconds
        });
    }

    /**
     * this function use to set marker on google map.
     * set style,zoom,center location
     * Add a listener for the 'click' and `mouseover/mouseout` event marker on map
     * Add a google.maps.InfoWindow
     * @param {any} lat - set Marker position
     * @param {any} lng - set Marker position.
     */
    addMarker(lat: any, lng: any) {
        const marker = new google.maps.Marker({
            position: new google.maps.LatLng(lat, lng), // Marker position
            map: this.map,
            icon: '../assets/icons/map1x.png', // Provide the path to your custom marker icon
        });

        this.infoWindow = new google.maps.InfoWindow({
            content: this.templateResult,
        });

        // Add a listener for the 'click' event
        marker.addListener('click', () => {
            this.markerClick();
        });

        // Add a listener for the 'mouseover' event
        marker.addListener('mouseover', () => {
            if (
                this.infoWindow.getContent() == null ||
                this.infoWindow.getContent() == ''
            ) {
                this.infoWindow.setContent(this.templateResult);
            }

            if (this.templateResult) {
                this.infoWindow.open(this.map, marker);
            }
        });
        marker.addListener('mouseout', () => {
            this.infoWindow.close();
        });

        this.markers.push(marker);
    }

    /**
     * this function use to onclick marker.
     * set marker click configration like urlRedirect ,screenRedirect , postToUrl and customEvent
     */
    markerClick() {
        if (this.dataSettingsTrackMap?.action == 'urlRedirect') {
            window.open(this.dataSettingsTrackMap?.linkurl, '_blank');
        } else if (this.dataSettingsTrackMap?.action == 'screenRedirect') {
            if (this.dataSettingsTrackMap?.openScreenAs == 'navigation') {
                this.router.navigate([
                    '/screen' + '/' + this.dataSettingsTrackMap?.screen,
                ]);
            } else if (
                this.dataSettingsTrackMap?.openScreenAs == 'popupModal'
            ) {
                const ref = this.dialog.open(TabDynamicDialogComponent, {
                    width: '1000px',
                    height: '1000px',
                    data: {
                        screenID: this.dataSettingsTrackMap?.screen,
                        data: [],
                    },
                });

                ref.afterClosed().subscribe((result) => {});
            } else if (this.dataSettingsTrackMap?.openScreenAs == 'sidebar') {
                this.tabDynamicSidebarComponent.screenId =
                    this.dataSettingsTrackMap?.screen;
                this.tabDynamicSidebarComponent.settingsDrawer.toggle();
            }
        } else if (this.dataSettingsTrackMap?.action == 'postToUrl') {
            var headers = TabRequestResponseFunction.getHeaders();
            if (
                this.dataSettingsTrackMap?.headers &&
                this.dataSettingsTrackMap?.headers.length > 0
            ) {
                this.dataSettingsTrackMap?.headers.forEach((element) => {
                    headers.append(element.header, element.value);
                });
            }
            this.tabCoreHttpService
                .get(this.dataSettingsTrackMap?.postToUrl, headers)
                .subscribe();
        } else if (this.dataSettingsTrackMap?.action == 'customEvent') {
            this.customEvent.emit('');
        }
    }

    /**
     * this function use to remover all marker from the map
     * Clear the markers array
     */
    removeMarkers() {
        for (const marker of this.markers) {
            marker.setMap(null); // Remove marker from the map
        }
        this.markers = []; // Clear the markers array
    }

    /**
     * this function use to set map refresh time
     * if refreshMap == true then map is reload to interval time
     */
    setInterval() {
        if (this.dataSettingsTrackMap && this.dataSettingsTrackMap.refreshMap) {
            let tval =
                60000 *
                (this.dataSettingsTrackMap.interval == null &&
                this.dataSettingsTrackMap.interval == ''
                    ? 5
                    : this.dataSettingsTrackMap.interval);
            interval(tval).subscribe(() => {
                this.mapApiCall();
            });
        }
    }

    /**
     * this function use to call DSQ API for  from 'dataSettingsTrackMap'
     */
    mapApiCall() {
        if (this.dataSettingsTrackMap) {
            // Retrieve the data source query details based on 'dataSourceQuery' from 'dataSettingsTrackMap'
            this.dsqDetailsOfUserLocation =
                TabAppFunction.getDataFromDataSourceQueryID(
                    this.dataSettingsTrackMap?.dataSourceQuery
                );

            // Create a deep clone of the data source query details
            this.cloneDsqDetailsOfUserLocation = cloneDeep(
                this.dsqDetailsOfUserLocation
            );

            this.tabGetService
                .executeRAWDSQWithName({
                    AppObjectName:
                        TabAppFunction.getAppObjectFromDataSourceQueryID(
                            this.dataSettingsTrackMap?.dataSourceQuery
                        ).ObjectName,
                    DSQName: this.cloneDsqDetailsOfUserLocation.QueryName,
                    Reqtokens: {},
                })
                .subscribe((data) => {
                    this.userLocationObj = data.Result;
                    this.removeMarkers();

                    data.Result.forEach((element) => {
                        let lat =
                            element[
                                this.dataSettingsTrackMap?.latitude?.AppFieldID
                                    ?.FieldName
                            ];
                        let lng =
                            element[
                                this.dataSettingsTrackMap?.longitude?.AppFieldID
                                    ?.FieldName
                            ];
                        if (lat != null && lng != null) {
                            this.addMarker(lat, lng);
                        }
                    });
                });
        }
    }

    /**
     * this function use to retrieve the data for template/customHtml
     * declare getTemplateObservable for template
     * declare getDsqObservable for DSQ API
     */
    async populateData() {
        if (
            this.dataSettingsTrackMap?.dataSourceMode ||
            this.dataSettingsTrackMap?.templateMode
        ) {
            let apiData = {};
            if (this.dataSettingsTrackMap?.dataSourceMode == 'data') {
                apiData['Templatevariables'] =
                    this.dataSettingsTrackMap?.Templatevariables;
            } else if (this.dataSettingsTrackMap?.dataSourceMode == 'dsq') {
                if (this.dataSettingsTrackMap?.dataSourceQuery) {
                    apiData['DataSourceQueryID'] =
                        this.dataSettingsTrackMap?.dataSourceQuery;
                }
            }
            if (this.dataSettingsTrackMap?.templateMode == 'customHtml') {
                apiData['DataSourceMode'] = 2;
                apiData['Value'] = this.dataSettingsTrackMap?.htmlCode;
            } else if (this.dataSettingsTrackMap?.templateMode == 'template') {
                apiData['DataSourceMode'] = 1;
                apiData['Value'] = this.dataSettingsTrackMap?.template;
            }

            this.dsqDetails = TabAppFunction.getDataFromDataSourceQueryID(
                this.dataSettingsTrackMap?.dataSourceQuery
            );

            this.cloneDsqDetails = cloneDeep(this.dsqDetails);

            forkJoin({
                templateData: this.getTemplateObservable(),
                dsqData: this.getDsqObservable(),
            })
                .pipe(take(1))
                .subscribe(({ templateData, dsqData }) => {
                    if (templateData?.Result?.length > 0) {
                        apiData['Value'] = templateData.Result[0]?.TemplateBody;
                    }
                    if (dsqData?.Result) {
                        apiData['Templatevariables'] = JSON.stringify({
                            data: dsqData.Result,
                        });
                    }
                    this.tabTemplateService
                        .templateParseUniversal(apiData, true)
                        ?.subscribe((res) => {
                            this.templateResult = res.Result;
                        });
                });
        }

        this.mapApiCall();
    }

    /**
     * this function use to call DSQ API for template
     *
     * @returns {Observable}
     */
    getTemplateObservable() {
        if (this.dataSettingsTrackMap?.templateMode == 'customHtml')
            return of({});

        return this.tabGetService.executeRAWDSQWithName({
            DSQName: 'Detail_TemplateInfo',
            AppObjectName: 'TemplateInfo',
            Reqtokens: {
                ID: this.dataSettingsTrackMap?.template,
            },
        });
    }

    /**
     * this function use to call DSQ API
     *
     * @returns {Observable}
     */
    getDsqObservable() {
        if (this.dataSettingsTrackMap?.dataSourceMode) {
            if (this.dataSettingsTrackMap?.dataSourceMode == 'data')
                return of({});

            const appObject = TabAppFunction.getAppObjectFromDataSourceQueryID(
                this.dataSettingsTrackMap?.dataSourceQuery
            );
            const dsqData = TabAppFunction.getDataFromDataSourceQueryID(
                this.dataSettingsTrackMap?.dataSourceQuery
            );

            const apiData = {
                DSQName: dsqData.QueryName,
                AppObjectName: appObject.ObjectName,
            };
            if (this.dataSettingsTrackMap?.parameters)
                apiData['Reqtokens'] = this.dataSettingsTrackMap?.parameters;
            return this.tabGetService.executeRAWDSQWithName(apiData);
        }
    }
}
