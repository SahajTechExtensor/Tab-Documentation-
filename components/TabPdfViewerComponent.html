import {
    Component,
    Input,
    OnDestroy,
    OnInit,
    SimpleChanges,
    ViewChild,
} from '@angular/core';
import {
    PdfViewerComponent,
    ToolbarSettingsModel,
} from '@syncfusion/ej2-angular-pdfviewer';
import { TabMediaUploadService } from '@techextensor/tab-service';
import { Subject, debounceTime, takeUntil } from 'rxjs';

@Component({
    selector: 'app-tab-pdf-viewer',
    templateUrl: './tab-pdf-viewer.component.html',
    styleUrls: ['./tab-pdf-viewer.component.scss'],
})
export class TabPdfViewerComponent implements OnInit, OnDestroy {
    private _value: any;

    // Reference to the PdfViewerComponent
    @ViewChild('pdfviewer')
    public pdfviewerControl: PdfViewerComponent;

    // Input property to receive data setting for the PDF viewer

    @Input() dataSettingOfPdfViewer: any;

    debounceSubject = new Subject();
    _unsubscribeAll: Subject<string> = new Subject<string>();

    // Setter and getter for the value property

    @Input()
    public set value(val: any) {
        if (val && val != null) {
            this._value = val;
        }
        this.debounceSubject.next(1);
    }
    public get value(): any {
        return this._value;
    }

    constructor(private tabMediaUploadService: TabMediaUploadService) { }

    ngOnInit(): void {
        this.debounceSubject.pipe(debounceTime(1000))
            .pipe(takeUntil(this._unsubscribeAll)).subscribe(() => {
                this.load();
            })
    }

    // Service URL for the PDF viewer

    public service =
        'https://ej2services.syncfusion.com/production/web-services/api/pdfviewer';

    // Document to be displayed in the PDF viewer

    public document = '';

    // Toolbar settings for the PDF viewer

    public toolbarSettings: ToolbarSettingsModel = {
        showTooltip: true,
        toolbarItems: [],
    };

    // Lifecycle hook called when input properties change

    ngOnChanges(changes: SimpleChanges) {
        if (changes.dataSettingOfPdfViewer && this.dataSettingOfPdfViewer) {
            this.toolbarSetting();
        }
    }

    // Function to configure the toolbar settings based on the provided dataSettingOfPdfViewer
    toolbarSetting() {
        // Reset the toolbar items array

        this.toolbarSettings.toolbarItems = [];

        // Check if dataSettingOfPdfViewer has been provided

        if (this.dataSettingOfPdfViewer) {
            const {
                isOpenFile,
                isDownLoadFile,
                isNavigantion,
                isMagnification,
                isPanTool,
                isSelection,
                isSearchOption,
                isPrint,
                isUndoRedo,
                isAnnotationEdit,
                isFormDesignerTool,
                isCommentTool,
                submitForm,
            } = this.dataSettingOfPdfViewer;

            // Configure the toolbar items based on the settings

            if (isOpenFile) {
                // Add the 'OpenOption' to the toolbar items array

                this.toolbarSettings.toolbarItems.push('OpenOption');
            }

            if (isDownLoadFile) {
                // Add the 'DownloadOption' to the toolbar items array

                this.toolbarSettings.toolbarItems.push('DownloadOption');

                if (isNavigantion) {
                    // Add the 'PageNavigationTool' to the toolbar items array
                    this.toolbarSettings.toolbarItems.push(
                        'PageNavigationTool'
                    );
                }
                if (isMagnification) {
                    // Add the 'MagnificationTool' to the toolbar items array
                    this.toolbarSettings.toolbarItems.push('MagnificationTool');
                }

                if (isPanTool) {
                    // Add the 'PanTool' to the toolbar items array
                    this.toolbarSettings.toolbarItems.push('PanTool');
                }
                if (isSelection) {
                    // Add the 'SelectionTool' to the toolbar items array

                    this.toolbarSettings.toolbarItems.push('SelectionTool');
                }
                if (isSearchOption) {
                    // Add the 'PrintOption' to the toolbar items array
                    this.toolbarSettings.toolbarItems.push('SearchOption');
                }
                if (isPrint) {
                    // Add the 'UndoRedoTool' to the toolbar items array
                    this.toolbarSettings.toolbarItems.push('PrintOption');
                }
                if (isUndoRedo) {
                    // Add the 'UndoRedoTool' to the toolbar items array
                    this.toolbarSettings.toolbarItems.push('UndoRedoTool');
                }
                if (isAnnotationEdit) {
                    // Add the 'AnnotationEditTool' to the toolbar items array
                    this.toolbarSettings.toolbarItems.push(
                        'AnnotationEditTool'
                    );
                }

                if (isFormDesignerTool) {
                    // Add the 'FormDesignerEditTool' to the toolbar items array
                    this.toolbarSettings.toolbarItems.push(
                        'FormDesignerEditTool'
                    );
                }

                if (isCommentTool) {
                    // Add the 'CommentTool' to the toolbar items array
                    this.toolbarSettings.toolbarItems.push('CommentTool');
                }

                if (submitForm) {
                    // Add the 'SubmitForm' to the toolbar items array
                    this.toolbarSettings.toolbarItems.push('SubmitForm');
                }
            }
        }
    }
    // This method loads the PDF document into the PDF viewer using the mediaId provided in the _value object.

    load() {
        // Get a reference to the PDF viewer instance
        var viewer = (<any>document.getElementById('pdfViewer'))
            .ej2_instances[0];

        // Call the tabMediaUploadService to get the PDF file blob using the mediaId

        this.tabMediaUploadService
            .getFileByMediaId(this._value.mediaId)
            .subscribe((blob: any) => {
                // Create a new FileReader to read the blob as a data URL (base64)

                var reader = new FileReader();
                reader.readAsDataURL(blob);

                // When the FileReader finishes loading the data (base64 string),
                // this callback function will be executed.
                reader.onload = () => {
                    // Get the base64 string representing the PDF file
                    var base64data = reader.result;
                    // Load the base64 data into the PDF viewer
                    viewer.load(base64data, null);
                };
            });
    }

    ngOnDestroy(): void {
        this._unsubscribeAll.next(null);
        this._unsubscribeAll.complete();
    }
}
