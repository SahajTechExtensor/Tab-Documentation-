    getSuccessRateFO() {
        const { totalCase, nonGenuineCase } = this.allCaseList.reduce(
            (acc, x) => {
                if (
                    x?.BlueprintStatusId?.toLowerCase() ==
                        this.Closed.toLowerCase() ||
                    x?.BlueprintStatusId?.toLowerCase() ==
                        this.Hardcopyreceived.toLowerCase() ||
                    x?.BlueprintStatusId?.toLowerCase() ==
                        this.Paymentcleared.toLowerCase() ||
                    x?.BlueprintStatusId?.toLowerCase() ==
                        this.Closedwithpayment.toLowerCase()
                ) {
                    // Fully Assigned
                    if (x.IsFullyAssignedToFO == true) {
                        acc.totalCase++;
                        let isNonGenuine = false;
                        for (
                            let i = 0;
                            i < x?.CaseClosureMaster?.length ?? 0;
                            i++
                        ) {
                            const closure = x?.CaseClosureMaster[i];
                            if (
                                closure.IsPrimaryConclusion == true &&
                                x.FOIds.toLowerCase().includes(
                                    closure?.ClosureGivenBy?.Id?.toLowerCase()
                                )
                            ) {
                                const conclusionId = closure?.Conclusion
                                    ?.EnumDetailID as number | undefined;
                                switch (x.ClaimTypeEnumDetailId) {
                                    case 14: //MBV
                                        if (
                                            conclusionId == 2 ||
                                            conclusionId == 3
                                        )
                                            isNonGenuine = true;
                                        break;
                                    case 13: //PMJAY
                                        if (
                                            conclusionId >= 2 &&
                                            conclusionId <= 5
                                        )
                                            isNonGenuine = true;
                                        break;
                                    default: //A2G
                                        if (conclusionId == 3)
                                            isNonGenuine = true;
                                        break;
                                }
                            }
                            if (isNonGenuine) {
                                acc.nonGenuineCase++;
                                break;
                            }
                        }
                    }
                    // Partially Assigned
                    if (x.IsFullyAssignedToFO == false) {
                        // acc.totalCase++;
                        let isNonGenuine = false;
                    }
                }
                return acc;
            },
            { totalCase: 0, nonGenuineCase: 0 }
        );

        this.allCountValueJson.successRateFoCount = 0;
        if (totalCase > 0) {
            this.allCountValueJson.successRateFoCount = parseFloat(
                ((nonGenuineCase / totalCase) * 100).toFixed(2)
            );
        }
    }

    getSuccessRateCM() {
        const { totalCase, nonGenuineCase } = this.allCaseList.reduce(
            (acc, x) => {
                if (
                    x?.BlueprintStatusId?.toLowerCase() ==
                        this.Closed.toLowerCase() ||
                    x?.BlueprintStatusId?.toLowerCase() ==
                        this.Hardcopyreceived.toLowerCase() ||
                    x?.BlueprintStatusId?.toLowerCase() ==
                        this.Paymentcleared.toLowerCase() ||
                    x?.BlueprintStatusId?.toLowerCase() ==
                        this.Closedwithpayment.toLowerCase()
                ) {
                    acc.totalCase++;
                    let isNonGenuine = false;
                    for (
                        let i = 0;
                        i < x?.CaseClosureMaster?.length ?? 0;
                        i++
                    ) {
                        const closure = x?.CaseClosureMaster[i];
                        if (
                            x?.SPOCId?.Id?.toLowerCase() ==
                            closure?.ClosureGivenBy?.Id?.toLowerCase()
                        ) {
                            const conclusionId = closure?.Conclusion
                                ?.EnumDetailID as number | undefined;
                            switch (x.ClaimTypeEnumDetailId) {
                                case 14: //MBV
                                    if (conclusionId == 2 || conclusionId == 3)
                                        isNonGenuine = true;
                                    break;
                                case 13: //PMJAY
                                    if (conclusionId >= 2 && conclusionId <= 5)
                                        isNonGenuine = true;
                                    break;
                                default: //A2G
                                    if (conclusionId == 3) isNonGenuine = true;
                                    break;
                            }
                        }
                        if (isNonGenuine) {
                            acc.nonGenuineCase++;
                            break;
                        }
                    }
                }
                return acc;
            },
            { totalCase: 0, nonGenuineCase: 0 }
        );

        this.allCountValueJson.successRateCMCount = 0;
        if (totalCase > 0) {
            this.allCountValueJson.successRateCMCount = parseFloat(
                ((nonGenuineCase / totalCase) * 100).toFixed(2)
            );
        }
    }

    onCustomEvent(screenParameters) {
        let apiParams = {};
        if (screenParameters?.event?.type == 'applyDashboardFilter') {
            let loginData = JSON.parse(localStorage.getItem('loginData'));
            let roleId = loginData.Roles[0].ID;
            this.allCountValueJson.roleId = roleId;
            this.refreshForm(screenParameters, {
                ...screenParameters?.formIOObject.formio.data,
                openCase: JSON.stringify(this.allCountValueJson),
            });
            // case from date
            if (screenParameters?.event?.data?.casesFromDate1) {
                let abc = moment(screenParameters?.event?.data?.casesFromDate1)
                    .tz('Asia/Kolkata')
                    .format('yyyy-MM-DD HH:mm:ss');

                apiParams['FromDate'] = abc;
            }
            // case to date
            if (screenParameters?.event?.data?.CasesToDate1) {
                let mdate = moment(
                    screenParameters?.event?.data?.CasesToDate1,
                    'yyyy-MM-DD HH:mm:ss'
                )
                    .add('hour', 23)
                    .add('minute', 59);
                let cde = moment(mdate)
                    .tz('Asia/Kolkata')
                    .format('yyyy-MM-DD HH:mm:ss');
                apiParams['ToDate'] = cde;
            }
            // insurance company
            if (screenParameters?.event?.data?.InsuranceCompany1) {
                apiParams['InsuranceCompanyId'] =
                    screenParameters?.event?.data?.InsuranceCompany1;
            }
            // spoc ic
            if (screenParameters?.event?.data?.SPOCPhoenix1) {
                apiParams['SPOCId'] =
                    screenParameters?.event?.data?.SPOCPhoenix1;
            }

            // claim type
            if (screenParameters?.event?.data?.TypeOfClaim1) {
                apiParams['ClaimTypeEnumDetailId'] =
                    screenParameters?.event?.data?.TypeOfClaim1;
            }

            // expert
            if (screenParameters?.event?.data?.NameOfExpert1) {
                apiParams['ExpertId'] =
                    screenParameters?.event?.data?.NameOfExpert1;
            }
            // fo
            if (screenParameters?.event?.data?.NameOfFo1) {
                apiParams['FOIDS'] = screenParameters?.event?.data?.NameOfFo1;
            }

            // state o hospital
            if (screenParameters?.event?.data?.StateOfHospital1) {
                apiParams['StateId'] =
                    screenParameters?.event?.data?.StateOfHospital1;
            }
            this.getDashboardData(screenParameters, apiParams);
        }
    }
