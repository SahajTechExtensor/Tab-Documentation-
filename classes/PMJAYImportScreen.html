import { HttpErrorResponse } from '@angular/common/http';
import { ElementRef } from '@angular/core';
import { FormioUtils } from '@formio/angular';
import {
    IActionExtension,
    IScreenExtension,
} from '@techextensor/tab-interface';
import { TabConstant } from '@techextensor/tab-service';
import { TabDynamicDialogComponent } from 'app/shared/templates/tab-dynamic-dialog/tab-dynamic-dialog.component';

export class PMJAYImportScreen implements IScreenExtension {
    activeAction: IActionExtension;
    mediaInfo: any;
    constructor() {}

    screenId = '44d8b0a8-edbc-4bc0-90a5-8346eb15df20';

    onSubmit(event) {}

    onRender(screenParameters: any) {}

    async onBeforeSubmit(screenParameters: any) {}

    onChange(screenParameters: any) {
        this.setCustomheaders(screenParameters);
    }
    setCustomheaders(screenParameters) {
        FormioUtils.eachComponent(
            screenParameters?.formIOObject?.formio?.component?.components,
            async (component, path) => {
                if (component.type == 'file' && component.storage == 'url') {
                    const customOptions = {
                        headers: {
                            Accept: '*/*',
                            Applicationcode: sessionStorage.getItem(
                                TabConstant.Storage.applicationCode
                            ),
                            Authorization: sessionStorage.getItem(
                                TabConstant.Storage.token
                            ),
                        },
                    };
                    component.options = JSON.stringify(customOptions);
                }
            }
        );
    }
    onInvalid(event) {}

    onNextPage(event) {}

    onPrevPage(event) {}

    onErrorChange(event) {}

    onFormLoad(screenParameters: any) {}

    onReady(screenParameters: any) {}

    onAfterSubmit(screenParameters: any) {}

    proceedObjectBeforeRender(screenParameters: any) {}

    updateCurrentActionInstance(actionData) {}
    async onCustomEvent(screenParameters) {
        if (screenParameters.event.type == 'PMJAYimport') {
            if (screenParameters.event.data.File) {
                // commented as currently using filestorage url mode in file upload
                // this.mediaInfo = await this.uploadFileAsync(
                //     screenParameters,
                //     screenParameters.event.data.File[0]
                // );
                const Mediaid = screenParameters.event.data.MediaId;

                var data = {
                    ImportObjectMappingId:
                        'E0113D75-6D0F-480D-B1D7-F4CC5A824842',
                    File: Mediaid,
                    ImportType: 0,
                };
                var Progressdata = {
                    ImportObjectMappingId:
                        'E0113D75-6D0F-480D-B1D7-F4CC5A824842',
                    File: Mediaid,
                    ImportType: 1,
                };
                screenParameters.serviceList.getService
                    .pmjayImport(data)
                    .subscribe(
                        (res) => {
                            if (res.Result && res.Result.length > 0) {
                                this.ImportwithImportType(
                                    screenParameters,
                                    Progressdata
                                );
                            }
                        },
                        (error: HttpErrorResponse) => {
                            if (error.error.IsSuccess == false) {
                                this.showdialog(
                                    screenParameters,
                                    error.error.Errors,
                                    true
                                );
                            }
                        }
                    );
            }
        }
    }

    ImportwithImportType(screenParameters, Progressdata) {
        screenParameters.serviceList.getService
            .pmjayImport(Progressdata)
            .subscribe(
                (response) => {
                    if (response.Result && response.Result.length > 0) {
                        this.showdialog(screenParameters, response.Log, false);
                    } else {
                        screenParameters?.serviceList?.toastrService.showError(
                            response.Errors[0],
                            'Error'
                        );
                    }
                },
                (error) => {
                    if (error.error.Errors.length > 0) {
                        this.showdialog(
                            screenParameters,
                            error.error.Errors,
                            true
                        );
                    }
                }
            );
    }
    ref;
    showdialog(screenParameters, array, isError: boolean) {
        const blankarr = [];

        let data = { parameters: {} };
        data.parameters['screenMode'] = 'Add';
        data.parameters['showbutton'] = !isError;
        // for error
        if (isError == true) {
            this.arrayforError(array, blankarr);
        } else {
            // for insert
            this.arrayforLogs(array, blankarr);
        }
        data.parameters['Messages'] = blankarr;
        const target = new ElementRef(screenParameters.event.event.target);
        const rect = target.nativeElement.getBoundingClientRect();
        this.ref = screenParameters.dailogRef.open(TabDynamicDialogComponent, {
            height: '500px',
            width: '600px',
            panelClass: 'custom-modalbox',
            position: {
                left: `${rect.left - 550}px`,
                top: `${rect.top + 50}px`,
                right: `${rect.right}px`,
                bottom: `${rect.bottom}px`,
            },
            data: {
                screenID: '3be7eb24-da21-4797-95d8-5858de4a0214',
                data: JSON.stringify(data),
                title: 'PMJAY',
            },
        });
    }
    // push all the error which is from response and convert in userfriendly
    arrayforError(array, blankarr) {
        array.forEach((errorMessage) => {
            const regex = /(.+?) is not exist in lookup/;
            const match = regex.exec(errorMessage);
            if (match && match[1]) {
                const desiredString = match[1];
                const newmsg =
                    desiredString + '   provided does not match our records';
                blankarr.push(newmsg);
            }
            // dublicate NHPM id
            const duplicateKeyRegex = /System.Data.SqlClient.SqlException/i;
            const hasDuplicateKey = duplicateKeyRegex.test(errorMessage);
            if (hasDuplicateKey) {
                const indexNameRegex = /with unique index 'IX_NHPMId_NotNull'/i;
                const hasSpecificIndexName = indexNameRegex.test(errorMessage);
                if (hasDuplicateKey && hasSpecificIndexName) {
                    const match = /\(([^)]+)\)/.exec(errorMessage);
                    if (match && match[1]) {
                        const idWithinParentheses = match[1];
                        const unquieIdmessage =
                            'A record with the unique identifier ' +
                            idWithinParentheses +
                            ' NHPM ID already exists.';
                        blankarr.push(unquieIdmessage);
                    }
                }
            }
        });
    }
    // push all the insert log which is from response and convert in userfriendly
    arrayforLogs(array, blankarr) {
        array.forEach((logmessage) => {
            if (
                logmessage.indexOf('ClaimSubType from PolicySubTypeMaster') > 0
            ) {
                const match = /\(([^)]+)\)/.exec(logmessage);
                if (match && match[1]) {
                    const contentWithinParentheses = match[1];
                    let commaCount = (
                        contentWithinParentheses.match(/,/g) || []
                    ).length;
                    const incrementedCommaCount = commaCount + 1;
                    const logMessage =
                        incrementedCommaCount + ' Record added Successfully';
                    blankarr.push(logMessage);
                }
            }
        });
    }
    uploadFileAsync(screenParameters: any, file: any): Promise<any> {
        let formData = new FormData();
        const blobFile = this.dataURIToBlob(file.url);
        const uploadFile = new File([blobFile], file.originalName);
        formData.append('0', uploadFile);

        return new Promise((resolve, reject) => {
            return screenParameters?.serviceList?.mediaUploadService
                ?.uploadMedia(formData)
                .subscribe(
                    (res: any) => {
                        if (!!res.Result && res.Result.length > 0) {
                            resolve(res.Result[0]);
                        } else {
                            reject(res);
                        }
                    },
                    (error: any) => {
                        resolve(error?.error);
                    }
                );
        });
    }

    /* Converts a data URI string to a Blob object.
     *
     * @param dataURI - The data URI string to convert.
     * @returns A Blob object representing the data URI.
     */
    dataURIToBlob(dataURI: string) {
        const splitDataURI = dataURI.split(',');
        const byteString =
            splitDataURI[0].indexOf('base64') >= 0
                ? atob(splitDataURI[1])
                : decodeURI(splitDataURI[1]);
        const mimeString = splitDataURI[0]?.split(':')[1]?.split(';')[0];

        const ia = new Uint8Array(byteString.length);
        for (let i = 0; i < byteString.length; i++)
            ia[i] = byteString.charCodeAt(i);

        return new Blob([ia], { type: mimeString });
    }
}
