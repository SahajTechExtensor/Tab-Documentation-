import {
    IActionExtension,
    IActionExtensionRegistry,
    IScreenExtension,
} from '@techextensor/tab-interface';
import { TabCommonFunction, TabAppFunction } from '@techextensor/tab-service';

export class LookupConfigurationScreen implements IScreenExtension {
    activeAction: IActionExtension;
    constructor() {}

    screenId = '765b4b68-beb3-4e5f-a50b-1af14acedb5d';

    onSubmit(event) {}

    onRender(screenParameters: any) {}

    onBeforeSubmit(screenParameters: any) {
        if (
            screenParameters.event?.data?.lookUpObjectId &&
            screenParameters.event?.data?.lookUpFieldId &&
            screenParameters.event?.data?.displayFieldId
        ) {
            let selectQuery = {
                QueryObjectID: '',
                Includes: [],
                QueryType: 0,
                Joins: [],
                WhereClause: {
                    FilterLogic: 1,
                    Filters: [],
                },
            };
            let Filters = [];
            let lookUpJSON = {
                AppObjectId: screenParameters.event?.data?.lookUpObjectId,
                FieldId: screenParameters.event?.data?.lookUpFieldId,
                DisplayFieldId: screenParameters.event?.data?.displayFieldId,
            };
            if (
                screenParameters?.event?.data?.bindSelectQuery?.filters
                    ?.length > 0
            ) {
                screenParameters?.event?.data?.bindSelectQuery?.filters.forEach(
                    (element) => {
                        Filters.push({
                            Field: element.label,
                            // FieldID: element.FieldID,
                            FieldID: element.label,
                            ConjuctionClause: element.ConjuctionClause,
                            RelationalOperator: element.RelationalOperator,
                            ValueType:
                                TabCommonFunction.getFilterValueTypeInInt(
                                    element.ValueType
                                ),
                            FilterValueType:
                                TabCommonFunction.getFilterValueTypeInInt(
                                    element.ValueType
                                ),
                            value: element.Value,
                            GroupID:
                                !element.group || element.group == 0
                                    ? 1
                                    : element.group,
                            FieldType: '1',
                        });
                    }
                );
                selectQuery.QueryObjectID = TabAppFunction.getAppObject(
                    lookUpJSON.AppObjectId
                ).DisplayName;
                selectQuery.WhereClause.Filters = Filters;
                lookUpJSON['selectQuery'] = selectQuery;
            }
            screenParameters.event.data.LookUpDetails =
                JSON.stringify(lookUpJSON);
        } else {
            screenParameters.event.data.LookUpDetails = null;
        }
       screenParameters?.event?.callback(null,screenParameters?.event);
    }

    onChange(screenParameters: any) {
        if (
            screenParameters?.event?.changed?.component?.type == 'select' &&
            screenParameters?.event?.changed?.component?.key ==
                'lookUpObjectId' &&
            screenParameters?.event?.changed?.value
        ) {
            screenParameters.event.data['bindSelectQuery'] = {
                isRefresh: true,
                appObjectId: screenParameters?.event?.changed?.value,
            };
            screenParameters?.formIOObject?.formio.triggerRedraw();
            // screenParameters?.formIOObject.refresh.emit();
        }
    }

    onInvalid(event) {}

    onNextPage(event) {}

    onPrevPage(event) {}

    onErrorChange(event) {}

    onFormLoad(screenParameters: any) {}

    onReady(screenParameters: any) {}

    onAfterSubmit(screenParameters: any) {}

    updateCurrentActionInstance(actionData) {
        this.activeAction = null;
        var actions = IActionExtensionRegistry.GetImplementations();
        for (var x = 0; x < actions.length; x++) {
            const action = new actions[x]();
            if (actionData.$type.includes(action.actionType)) {
                this.activeAction = action;
            }
        }
    }
