import { FormioUtils } from '@formio/angular';
import {
    IActionExtension,
    IScreenExtension,
} from '@techextensor/tab-interface';
import { PreviousSavedScreen } from 'app/modules/admin/tab-form-renderer/tab-form-renderer.component';
import * as $ from 'jquery';

export class EditClaimScreen implements IScreenExtension {
    activeAction: IActionExtension;
    claimType: any;
    isPMJAY = false;
    constructor() {}

    screenId = 'fe4e24f0-9839-4228-9b85-af0dc0dc1e89';
    submitButtonQuery = 'button[type="submit"].form-submit-btn';

    onSubmit(event) {}

    onRender(screenParameters: any) {}

    onBeforeSubmit(screenParameters: any) {
        setTimeout(() => {
            const keys =
                'CaseMaster.activeRowsData.ClaimTypeEnumDetailId'.split('.');
            var value = JSON.parse(sessionStorage.getItem('screenContext'));
            for (const k of keys) {
                if (value.hasOwnProperty(k)) {
                    value = value[k];
                    this.claimType = value;
                } else {
                    value = undefined;
                    break;
                }
            }
            if (value != 13) {
                const redirectToScreenId =
                    'ae34f8db-6894-49c0-b89d-7f38cb4bef3f';
                const routeToNavigate = '/screen/' + redirectToScreenId;
                screenParameters.serviceList.router.navigate([routeToNavigate]);
            }
                if(this.claimType == 13){
                    this.isPMJAY = true;
                    this.onBeforeNextPage(screenParameters)                }
        }, 200);
        screenParameters?.event?.callback(null, screenParameters?.event);
    }
    onChange(screenParameters: any) {}

    onInvalid(event) {}

    onNextPage(screenParameters) {}

    onPrevPage(screenParameters) {}

    onBeforeNextPage(screenParameters) {
        console.log('EditClaimScreen[onBeforeNextPage]', screenParameters);
        // submit click
        $(this.submitButtonQuery).click();

        setTimeout(() => {
            const previousValue: PreviousSavedScreen | null = JSON.parse(
                localStorage.getItem('previousSavedScreen')
            );

            FormioUtils.eachComponent(
                this.isPMJAY ? screenParameters.formIOObject.formio.components:                
                screenParameters.event?.currentPage?.components, 
                (component) => {
                    let error: any | null = null;

                    if (component.type == 'childscreen') {
                        const childScreenId =
                            component.component?.customOptions?.dataSetting
                                ?.childScreenId;

                        const validationError = this.checkFromIsValid();
                        if (validationError) {
                            screenParameters.event.next(validationError);
                            return;
                        }

                        if (
                            childScreenId?.toLowerCase() ===
                            previousValue.screenId.toLowerCase()
                        ) {
                            // if there is an error
                            if (previousValue.state == 'error') {
                                error = { message: 'an error occur' };
                                screenParameters.event.next(error);
                                return;
                            }

                            const intervalId = setInterval(
                                () =>
                                    this.checkLocalStorageValueUpdates(
                                        screenParameters,
                                        previousValue,
                                        childScreenId,
                                        intervalId
                                    ),
                                500
                            );
                        }
                    }
                },
                true
            );
        }, 1000);
    }

    onBeforePrevPage(screenParameters) {
        // console.log('EditClaimScreen[onBeforePrevPage]', screenParameters);
        screenParameters.event.next();
    }

    onErrorChange(event) {}

    onFormLoad(screenParameters: any) {}

    onReady(screenParameters: any) {}

    onAfterSubmit(screenParameters: any) {}

    proceedObjectBeforeRender(screenParameters: any) {}
    
    redirectionAfterUpdate(screenParameters: any){
         const redirectToScreenId =
                    'ae34f8db-6894-49c0-b89d-7f38cb4bef3f';
                const routeToNavigate = '/screen/' + redirectToScreenId;
                screenParameters.serviceList.router.navigate([routeToNavigate]);
            
    }
    onCustomEvent(event) {}

    /**
     * Checks if the value in localStorage has been updated and handles .
     *
     * @param screenParameters - The parameters containing event, formIOObject, and other relevant data.
     * @param previousValue - The previous value from localStorage that needs to be compared.
     * @param childScreenId - The ID of the child screen to track changes.
     * @param intervalId - The ID of the interval timer for checking changes periodically.
     */
    checkLocalStorageValueUpdates(
        screenParameters,
        previousValue,
        childScreenId,
        intervalId
    ) {
        // Check if the value in localStorage has changed
        let error = null;
        const newValue: PreviousSavedScreen = JSON.parse(
            localStorage.getItem('previousSavedScreen')
        );

        if (childScreenId.toLowerCase() !== newValue.screenId.toLowerCase()) {
            screenParameters.event.next();
            clearInterval(intervalId);
        }

        // check localStorage values is updated
        if (newValue !== previousValue) {
            if (newValue.state == 'submitted') {
                error = null;
                if(this.claimType == 13){
                this.redirectionAfterUpdate(screenParameters)
                }else{
                screenParameters.event.next(error);
                }
                clearInterval(intervalId);
            } else if (newValue.state == 'error') {
                error = { message: 'an error occur' };
                screenParameters.event.next(error);
                clearInterval(intervalId);
            }
        }

        // if validation throw by formIo latter
        if (this.checkFailedClassExist()) {
            error = { message: 'an error occur' };
            screenParameters.event.next(error);
            clearInterval(intervalId);
        }
    }

    /**
     * Checks if the form is valid based on previous saved values and handles validation errors.
     * @param previousValue - Previous value from localStorage to be checked.
     * @returns Error object if there is a validation error, otherwise null.
     */
    checkFromIsValid(): Object | null {
        let error = null;
        // Check is their any validation error
        if (this.checkFailedClassExist()) {
            error = {
                message: 'an error occur',
            };
            return error;
        }
        // no error but no values of save screen so continue
        return;
    }

    /**
     * Checks if the submit button has a 'submit-fail' class, indicating a form submission failure.
     * @returns True if the 'submit-fail' class exists on the submit button, otherwise false.
     */
    checkFailedClassExist() {
        return $(this.submitButtonQuery).hasClass('submit-fail');
    }
}
