import {
    IActionExtension,
    IScreenExtension,
} from '@techextensor/tab-interface';
import { TabAppFunction, TabCommonFunction } from '@techextensor/tab-service';
import { TabBasicGridComponent } from 'app/modules/admin/tab-basic-grid/tab-basic-grid.component';
import { cloneDeep } from 'lodash';
import { take } from 'rxjs';

export class SLASchemesScreen implements IScreenExtension {
    activeAction: IActionExtension;
    tabBasicGrid: TabBasicGridComponent;
    constructor() { }

    roles = {
        '07540909-B95B-4702-BCAD-9A3B7646347A': 'Expert',
        '668612EB-529A-475D-9186-B5C04D271C7F': 'Claim Manager',
        '7BB52D9E-BEEF-41EE-B327-4D5F38A600FC': 'Accountant',
        '93E20DDF-5C69-4DD6-A153-4C8DA4526ABC': 'Super Admin',
        'D80071AB-BF71-4737-AB66-18D3502F272B': 'Field Officer',
        'E1578C4A-6556-480B-B8CA-FABC320DECFA': 'Back Officer',
    };

    invalidTransitionIds = [

        "92d0f4e5-48e6-4b67-aa88-3a2d67bda048",

    async onBeforeSubmit(screenParameters: any) {

        if (screenParameters?.event?.data?.ApplyCriteria) {
            await this.formateApplyCriteria(screenParameters);
        }

        if (screenParameters?.event?.data?.TABD_Blueprint_SLASchemeDetails) {
            for (const slaSchemeDetail of screenParameters?.event?.data?.TABD_Blueprint_SLASchemeDetails) {
                if (screenParameters?.event?.data?.SLASetting == 0) {
                    slaSchemeDetail.container.SLATime =
                        ((slaSchemeDetail?.container?.FixedSLATime?.Day ?? 0) * 24 * 60) +
                        ((slaSchemeDetail?.container?.FixedSLATime?.Hour ?? 0) * 60) +
                        (slaSchemeDetail?.container?.FixedSLATime?.Minute ?? 0);
                } else if (screenParameters?.event?.data?.SLASetting == 1) {
                    slaSchemeDetail.container.SLATime = slaSchemeDetail.container.PercentageSLATime;
                }
            }
        }

        if (screenParameters?.event?.data?.SLASetting != null && screenParameters?.event?.data?.SLASetting != undefined) {
            screenParameters.event.data.SLASetting = JSON.stringify({
                Type: screenParameters?.event?.data?.SLASetting
            });
        }
        screenParameters?.event?.callback(null, screenParameters?.event);
    }

    async onChange(screenParameters: any) {
        if (
            screenParameters.event?.changed?.component &&
            screenParameters.event?.changed?.component.key == 'BlueprintId.Id'
        ) {
            if (screenParameters.event.data.BlueprintId.Id) {
                try {
                    const blueprint = await this.getBlueprintDetails(
                        screenParameters.event.data.BlueprintId.Id,
                        screenParameters
                    );
                    this.getFiltersFromAppObject(
                        blueprint.AppObjectId,
                        screenParameters
                    );
                } catch (error) {
                    console.log('error while changing blueprint', error);
                }
            }
            else {
                this.clearDSQFilters(screenParameters);
            }
        }
    }

    onNextPage(event) { }

    onPrevPage(event) { }

    onErrorChange(event) { }

    onFormLoad(screenParameters: any) { }

    onReady(screenParameters: any) { }

    onAfterSubmit(screenParameters: any) { }

    proceedObjectBeforeRender(screenParameters: any) {
        if (
            screenParameters.event.hasOwnProperty('Id') &&
            screenParameters.event.Id &&
            screenParameters.event.Id !== ''
        ) {
            if (screenParameters?.event?.ApplyCriteria) {
                const applyCriteria = JSON.parse(screenParameters.event.ApplyCriteria);
                screenParameters.event['lookUpObjectId'] = applyCriteria.AppObjectId;
                let filterList = [];
                if (
                    applyCriteria &&
                    applyCriteria.hasOwnProperty('selectQuery') &&
                    applyCriteria.selectQuery
                ) {
                    filterList = applyCriteria?.selectQuery?.WhereClause?.Filters;
                }
                screenParameters.event['ApplyCriteria'] = {
                    isRefresh: true,
                    appObjectId: applyCriteria.AppObjectId,
                    dataSourceQueryId: applyCriteria.DataSourceQueryId,
                    filterData: filterList,
                };
            }

            screenParameters.event.SLASetting = JSON.parse(screenParameters.event.SLASetting)?.Type;
            if (screenParameters?.event?.TABD_Blueprint_SLASchemeDetails) {
                let configurableSLADetails = [];
                if (screenParameters?.event.BlueprintId?.Id?.toLowerCase() == '1a105f71-0ee1-45eb-b2d2-a3697c4634e2') {
                    const startingStatusId = "d98d82aa-3939-4156-9093-35f56640b915";
                    const validSlaDetails =
                        screenParameters?.event?.TABD_Blueprint_SLASchemeDetails.filter(sla => {
                            return this.invalidTransitionIds.indexOf(sla?.container?.TransitionID?.Id) == -1
                        });
                    configurableSLADetails = cloneDeep(this.sortTransitions(validSlaDetails, startingStatusId));
                } else {
                    configurableSLADetails = cloneDeep(screenParameters?.event?.TABD_Blueprint_SLASchemeDetails);
                }


                for (const slaSchemeDetail of configurableSLADetails) {

                    if (screenParameters?.event?.SLASetting == 0) {
                        slaSchemeDetail.container.FixedSLATime =
                            this.convertMinutesToDaysHoursMinutes(slaSchemeDetail?.container?.SLATime);
                    } else if (screenParameters?.event?.SLASetting == 1) {
                        slaSchemeDetail.container.PercentageSLATime = slaSchemeDetail.container.SLATime;
                    }

                    const criteria = JSON.parse(slaSchemeDetail?.container?.TransitionID?.Transition_Criteria);

                    if (criteria?.TransitionHooks_Before?.Actions) {
                        criteria?.TransitionHooks_Before?.Actions?.forEach(action => {
                            if (action?.ActionName?.toLowerCase() == "ValidateIfUserMatchesCriteria".toLowerCase()) {
                                if (action?.tokens?.Role) {
                                    slaSchemeDetail.container.RoleName =
                                        action?.tokens?.Role?.split(',').map(r => this.roles[r.toUpperCase().trim()]).join(', ');
                                    slaSchemeDetail.container.HTMLRole =
                                        action?.tokens?.Role?.split(',').map(r => this.roles[r.toUpperCase().trim()]).join(', ');
                                }
                            }
                        });
                    }
                }
                screenParameters.event.TABD_Blueprint_SLASchemeDetails = configurableSLADetails;
            }
        }
        this.refreshForm(screenParameters);
    }

    async formateApplyCriteria(screenParameters) {
        let selectQuery = {
            QueryObjectID: '',
            Includes: [],
            QueryType: 0,
            Joins: [],
            WhereClause: {
                FilterLogic: 1,
                Filters: [],
            },
        };
        let Filters = [];
        const blueprint =
            await this.getBlueprintDetails(screenParameters.event?.data?.BlueprintId.Id, screenParameters);
        let ApplyCriteria = {
            AppObjectId: blueprint?.AppObjectId,
            DataSourceQueryId: screenParameters?.event?.data?.ApplyCriteria?.dataSourceQueryId,
        };

        if (screenParameters?.event?.data?.ApplyCriteria?.filters?.length > 0) {
            screenParameters?.event?.data?.ApplyCriteria?.filters.forEach(
                (element) => {
                    Filters.push({
                        Field: element.label,
                        FieldID: element.label,
                        ConjuctionClause: element.ConjuctionClause,
                        RelationalOperator: element.RelationalOperator,
                        ValueType:
                            TabCommonFunction.getFilterValueTypeInInt(
                                element.ValueType
                            ),
                        FilterValueType:
                            TabCommonFunction.getFilterValueTypeInInt(
                                element.ValueType
                            ),
                        value: element.Value,
                        GroupID:
                            !element.group || element.group == 0
                                ? 1
                                : element.group,
                        FieldType: '1',
                        LookUpDetail: element.label,
                    });
                }
            );
            selectQuery.QueryObjectID = TabAppFunction.getAppObject(
                ApplyCriteria.AppObjectId
            ).DisplayName;
            selectQuery.WhereClause.Filters = Filters;
            ApplyCriteria['selectQuery'] = selectQuery;
        }

        screenParameters.event.data.ApplyCriteria =
            JSON.stringify(ApplyCriteria);
    }

    sortTransitions(slaDetails, startingStatusId) {
        const transitionMap = new Map();

        const validTransitions = slaDetails.filter(slaDetail => {
            if (slaDetail?.container?.TransitionID?.FromStatus?.Id
                && slaDetail?.container?.TransitionID?.ToStatus?.Id
            ) {
                transitionMap.set(slaDetail.container.TransitionID.FromStatus.Id.toLowerCase(), slaDetail);
                return true;
            }
            return false;
        });

        const sortedTransitions = [];
        let currentStatusId = startingStatusId;

        while (validTransitions.length > 0) {
            const nextTransition = transitionMap.get(currentStatusId.toLowerCase());

            if (nextTransition) {
                sortedTransitions.push(nextTransition);
                const index = validTransitions.findIndex(sla =>
                    sla.container.TransitionID.FromStatus.Id.toLowerCase() == nextTransition.container.TransitionID.FromStatus.Id.toLowerCase());
                if (index !== -1) {
                    validTransitions.splice(index, 1);
                }
                currentStatusId = nextTransition.container.TransitionID.ToStatus.Id.toLowerCase();
            } else {
                break;
            }
        }

        return sortedTransitions;
    }

    /**
     * fetch DSQ filters data form AppObjectId and query builder
     *
     * @param {string} blueprintId App Object Id
     * @param {any} screenParameters The screen parameters.
     * @returns {void} void
     */
    getBlueprintDetails = (
        blueprintId: string,
        screenParameters: any
    ): Promise<any> => {
        return new Promise((resolve, reject) => {
            const apiData = {
                AppObjectName: 'TABD_Blueprints',
                DSQName: 'Detail_TABD_BluePrints',
                Reqtokens: {
                    Id: blueprintId,
                },
            };
            screenParameters.serviceList.getService
                .executeRAWDSQWithName(apiData)
                ?.pipe(take(1))
                .subscribe({
                    next: (res) => {
                        if (res.Result && res.Result.length > 0) {
                            resolve(res.Result[0]);
                            return;
                        }
                        reject(null);
                    },
                    error: (err) => {
                        reject(null);
                    }
                });
        });
    };

    /**
     * fetch DSQ filters data form AppObjectId and query builder
     *
     * @param {string} appObjectId App Object Id
     * @param {any} screenParameters The screen parameters.
     * @returns {void} void
     */
    getFiltersFromAppObject = (
        appObjectId: string,
        screenParameters: any
    ): void => {
        const apiData = {
            AppObjectName: 'TABMD_DataSourceQueries',
            DSQName: 'FK_TABMD_DataSourceQueries_TABMD_AppObject',
            Reqtokens: {
                Id: appObjectId,
            },
        };
        screenParameters.serviceList.getService
            .executeRAWDSQWithName(apiData)
            ?.pipe(take(1))
            .subscribe({
                next: (res) => {
                    if (res.Result) {
                        const dsqId = res.Result.find((dsq) =>
                            dsq.QueryName.startsWith('Default_')
                        )?.ID;
                        if (dsqId) this.getDSQFilters(dsqId, screenParameters);
                        else this.clearDSQFilters(screenParameters);
                    }
                },
            });
    };

    /**
     * fetch DSQ filters data and query builder
     *
     * @param {string} dsqId
     * @param {any} screenParameters
     * @returns {void} void
     */
    getDSQFilters = (dsqId: string, screenParameters: any) => {
        const reqTokens = { ID: dsqId };
        const apiData = {
            Reqtokens: reqTokens,
            AppObjectName: 'TABMD_DataSourceQueries_Filters',
            DSQName:
                'FK_TABMD_DataSourceQueries_Filters_TABMD_DataSourceQueries',
        };
        screenParameters.serviceList.getService
            .executeRAWDSQWithName(apiData)
            ?.pipe(take(1))
            .subscribe((response) => {
                let dsqFilterList;
                if (!!response.Result && response.Result.length > 0) {
                    dsqFilterList = response.Result;
                }
                screenParameters.event = {
                    ...screenParameters.event.data,
                    dataSourceQueryId: dsqId,
                    ApplyCriteria: {
                        filterData: dsqFilterList,
                        isRefresh: true,
                        dataSourceQueryId: dsqId,
                    },
                };
                this.refreshForm(screenParameters);
            });
    };

    /**
     * Clear query builder when appObjectId remove.
     *
     * @param {any} screenParameters
     * @returns {void} void
     */
    clearDSQFilters = (screenParameters: any): void => {
        screenParameters.event = {
            ...screenParameters.event.data,
            ApplyCriteria: {
                filterData: [],
                isRefresh: true,
                dataSourceQueryId: '',
            },
        };

        this.refreshForm(screenParameters);
    };

    convertMinutesToDaysHoursMinutes(minutes) {
        const days = Math.floor(minutes / 1440);
        const remainingMinutes = minutes % 1440;
        const hours = Math.floor(remainingMinutes / 60);
        const minutesRemaining = remainingMinutes % 60;

        return {
            Day: days == 0 ? null : days,
            Hour: hours == 0 ? null : hours,
            Minute: minutesRemaining == 0 ? null : minutesRemaining
        };
    }

    /**
     * refresh screen
     *
     * @param {any} screenParameters
     * @returns {void} void
     */
    refreshForm = (screenParameters: any): void => {
        screenParameters?.formIOObject.refresh.emit({
            submission: {
                data: screenParameters.event,
            },
        });
    };
}
