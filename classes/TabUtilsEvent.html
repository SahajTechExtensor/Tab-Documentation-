import { TabDynamicDialogComponent } from './shared/templates/tab-dynamic-dialog/tab-dynamic-dialog.component';
import { MatSnackBarVerticalPosition } from '@angular/material/snack-bar';

export class TabUtilsEvent {

    /**
     * navigate to specific screen.
     * @param {Object} eventParams - The parameters related to the event triggering the dialog.
     * @param {Event} eventParams.event - The event that triggered the dialog opening.
     * @param {MatDialog} eventParams.dialogRef - The reference to the MatDialog service.
     */
    navigateToScreen(eventParams: any) {
        eventParams.serviceList._route.navigate(
            ['/screen/' + eventParams.event.detail.screenId],
            { queryParams: eventParams?.event?.detail?.parameters }
        );
    }

    /**
     * navigate to specific screen with state
     * @param {Object} eventParams - The parameters related to the event triggering the dialog.
     * @param {Event} eventParams.event - The event that triggered the dialog opening.
     * @param {MatDialog} eventParams.event.detail - Navigation details screenId, parameters.
     */
    navigateToScreenWithState(eventParams: any) {
        const queryParams = eventParams?.event?.detail?.parameters ?? {};
        queryParams['state'] = encodeURIComponent(
            JSON.stringify(eventParams?.event?.detail?.state ?? {})
        );
        eventParams.serviceList._route.navigate(
            ['/screen/' + eventParams.event.detail.screenId],
            { queryParams: queryParams }
        );
    }

    /**
     * Opens a default dialog using the provided event parameters.
     * @param {Object} eventParams - The parameters related to the event triggering the dialog.
     * @param {Event} eventParams.event - The event that triggered the dialog opening.
     * @param {MatDialog} eventParams.dialogRef - The reference to the MatDialog service.
     */
    openDefaultDialog(eventParams: any) {
        const fullWidth = window.innerWidth; // Get the full width of the window
        const fullHeight = window.innerHeight;
        const verticalPadding = 50;
        const top = `${verticalPadding}px`;
        const left = `${(fullWidth - 500) / 2}px`;

        const ref = eventParams.dialogRef.open(TabDynamicDialogComponent, {
            height: `${fullHeight - verticalPadding * 2}px`,
            width: '500px',
            panelClass: 'custom-modalbox',
            position: {
                left: left,
                top: top,
            },
            data: {
                data: JSON.stringify({
                    parameters: {
                        ...eventParams?.event.detail.parameters,
                        screenMode: 'Add',
                    },
                }),
                screenID: eventParams?.event.detail.screenId,
                title: eventParams?.event.detail.pageTitle,
            },
        });
    }

    /**
     * Opens a default sidebar using the provided event parameters.
     * @param {Object} eventParams - The parameters related to the event triggering the sidebar.
     * @param {Event} eventParams.event - The event that triggered the sidebar opening.
     * @param {TabDynamicSidebarComponent} eventParams.sidebarComponent - The reference to the TabDynamicSidebarComponent.
     */
    openDefaultSidebar(eventParams: any) {
        eventParams.sidebarComponent.parameters = JSON.stringify({
            parameters: {
                ...eventParams?.event.detail.parameters,
                screenMode: eventParams?.event.detail.screenMode,
                screenContext: JSON.parse(
                    sessionStorage.getItem('screenContext')
                ),
                epochTime: new Date().getTime(),
            },
        });
        eventParams.sidebarComponent.screenId =
            eventParams?.event.detail.screenId;
        eventParams.sidebarComponent.title =
            eventParams?.event.detail.pageTitle;
        eventParams.sidebarComponent.isShowSidebarSubmit =
            eventParams?.event.detail.showSubmit;
        eventParams.sidebarComponent.settingsDrawer.toggle();
    }

    /**
     * Deletes a record using the provided AppObjectId and handles the response.
     *
     * @param eventParams - An object containing necessary parameters for the operation.
     * @param eventParams.serviceList - An object containing services, such as deleteService and toastrService.
     * @param eventParams.event.detail.deleteObject - The data object representing the record to be deleted.
     * @param eventParams.event.detail.appObjectId - The identifier for the AppObject related to the record.
     *
     * @returns {void}
     */
    deleteRecordUsingAppObjectId(eventParams: any) {
        eventParams?.serviceList?.deleteService
            .deleteRecord(
                '',
                { data: eventParams?.event.detail.deleteObject },
                eventParams?.event.detail.appObjectId
            )
            ?.pipe(take(1))
            ?.subscribe((response) => {
                if (response.StatusCode == '200' || !response.Errors.length) {
                    eventParams?.serviceList?.toastrService.showSuccess(
                        'Record Deleted Successfully',
                        'Success'
                    );
                } else {
                    eventParams?.serviceList?.toastrService.showError(
                        response.Errors[0],
                        'Error'
                    );
                }
            });
    }

    viewOldCase(eventParams) {
        eventParams?.serviceList?.getService
            .executeRAWDSQWithName({
                DSQName: 'ViewOldCaseDetail_CaseView',
                AppObjectName: 'CaseView',
                Reqtokens: {
                    ID: eventParams?.event.detail.caseId,
                },
            })
            ?.pipe(take(1))
            ?.subscribe(
                (res) => {
                    //
                    if (res.StatusCode == '200' && res.Result.length > 0) {
                        const caseId = eventParams?.event.detail.caseId;
                        var currentURL = window.location.href;
                        const lastSlashIndex = currentURL.lastIndexOf('/');
                        const urlWithoutLastPath = currentURL.substring(
                            0,
                            lastSlashIndex
                        );
                        if (urlWithoutLastPath) {
                            const viewCaseScreenId =
                                '574d343a-480e-4b8b-86f1-697a3130fa7e';
                            const newURL = `${urlWithoutLastPath}/${viewCaseScreenId}?Id=${caseId}`;
                            const newTab = window.open(newURL, '_blank');
                            if (newTab) {
                                let screenContext = JSON.parse(
                                    sessionStorage.getItem('screenContext')
                                );
                                if (screenContext) {
                                    screenContext['CaseMaster'][
                                        'activeRowsData'
                                    ] = res.Result[0];
                                    newTab.sessionStorage.setItem(
                                        'screenContext',
                                        JSON.stringify(screenContext)
                                    );
                                }
                            }
                        }
                    } else {
                        // Error
                        console.log('error while navigating', res);
                    }
                },
                (error) => {
                    // Error
                    console.log('error while navigating', error);
                }
            );
    }

    /**
     * navigate to specific screen with state
     * @param {Object} eventParams - The parameters related to the event triggering the dialog.
     * @param {Event} eventParams.event - The event that triggered the dialog opening.
     * @param {MatDialog} eventParams.event.detail - Navigation details screenId, parameters.
     */
    navigateToNewWindowScreen(eventParams: any) {
        window.open(
            '/screen/' +
                eventParams.event.detail.screenId +
                '?ID=' +
                eventParams?.event?.detail?.parameters.ID,
            '_blank'
        );
    }

    copyToClipboard(eventParams: any) {
        eventParams.serviceList.snackBar.open('Copied to clipboard', '', {
            duration: 2000, // Set the duration for the snackbar
            horizontalPosition: 'center', // Set the horizontal position
            verticalPosition: 'top' as MatSnackBarVerticalPosition, // Set the vertical position
            panelClass: ['custom-snackbar-copy'],
        });
    }

    /**
     * this function is used for clone workitem it's only for pms
     * @param eventParams event params where service list is exist from appcomponent
     */
    cloneWorkitem(eventParams) {
        setTimeout(() => {
            let screenContext = sessionStorage.getItem('screenContext');
            if (screenContext) {
                let screenContext1 = JSON.parse(screenContext);
                if (
                    screenContext1 &&
                    screenContext1.hasOwnProperty('WorkItems') &&
                    screenContext1?.WorkItems
                ) {
                    let activeRowData =
                        screenContext1?.WorkItems?.activeRowsData;
                    eventParams?.serviceList?.getService
                        .executeRawSql(
                            'exec SP_CloneWorkItem @aspnetuserid="' +
                                JSON.parse(localStorage.getItem('loginData'))
                                    ?.Attributes?.Id +
                                '",@workitemid="' +
                                activeRowData.ID +
                                '"'
                        )
                        ?.pipe(take(1))
                        ?.subscribe((response) => {
                            if (
                                response.StatusCode == '200' ||
                                !response.Errors.length
                            ) {
                                eventParams?.serviceList?.globalValueService.updatedataSourceInfo(
                                    {
                                        component: {
                                            component: {
                                                customOptions: {
                                                    dataSettingsGrid: {
                                                        dataSourceQueryId:
                                                            '79931E29-C6C6-47FC-AA36-809216D9FD89'.toLowerCase(),
                                                    },
                                                },
                                            },
                                        },
                                    }
                                );
                                eventParams?.serviceList?.toastrService.showSuccess(
                                    'Workitem Cloned Successfully',
                                    'Success'
                                );
                            } else {
                                eventParams?.serviceList?.toastrService.showError(
                                    response.Errors[0],
                                    'Error'
                                );
                            }
                        });
                }
            }
        }, 500);
    }
}
