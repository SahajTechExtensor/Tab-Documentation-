import { HttpClient } from '@angular/common/http';
import {
    IActionExtension,
    IActionExtensionRegistry,
    IScreenExtension,
} from '@techextensor/tab-interface';
import { take } from 'rxjs';

import { TabConstant } from '@techextensor/tab-service';
export class CustomActivityScreen implements IScreenExtension {
    constructor(private _httpClient: HttpClient) {}
    activeAction: IActionExtension;
    screenId = '1ed35911-ce7b-44e7-9530-b618ebdbd155';
    updateCurrentActionInstance(actionData) {
        this.activeAction = null;
        var actions = IActionExtensionRegistry.GetImplementations();
        for (var x = 0; x < actions.length; x++) {
            const action = new actions[x]();
            if (actionData.$type.includes(action.actionType)) {
                this.activeAction = action;
            }
        }
    }
    onSubmit(event: any) {}
    onRender(event: any) {}
    onBeforeSubmit(screenParameters: any) {
        screenParameters?.event?.callback(null, screenParameters?.event);
    }
    onChange(event: any) {}
    onInvalid(event: any) {}
    onNextPage(event: any) {}
    onPrevPage(event: any) {}
    onCustomEvent(event: any) {
        console.log(event);

        if (event.event.type === 'SyncAppFields') {
            event.serviceList.appService.tabCoreHttpService.httpClient
                .post(
                    TabConstant.API.endpoint + 'Application/SyncAppFields',
                    {},
                    {
                        headers: {
                            ApplicationCode: sessionStorage.getItem(
                                TabConstant.Storage.applicationCode
                            ),
                            Authorization: sessionStorage.getItem(
                                TabConstant.Storage.token
                            ),
                        },
                    }
                )
                ?.pipe(take(1))
                ?.subscribe((response) => {
                    console.log(response);
                });
        } else if (event.event.type === 'RunMissingRecordInfo') {
            event?.serviceList?.getService
                .executeRawSql(
                    'exec TABD_AddMissingRecordInRecordInfo @UserId="' +
                        JSON.parse(localStorage.getItem('loginData'))
                            ?.Attributes?.Id +
                        '"'
                )
                ?.pipe(take(1))
                ?.subscribe((response) => {
                    if (
                        response.StatusCode == '200' ||
                        !response.Errors.length
                    ) {
                        event?.serviceList?.toastrService.showSuccess(
                            'Add Missing Record Info Run Successfully',
                            'Success'
                        );
                    } else {
                        event?.serviceList?.toastrService.showError(
                            response.Errors[0],
                            'Error'
                        );
                    }
                });
        } else if (event.event.type === 'RegisterEntity') {
            event?.serviceList?.getService
                .executeRawSql(
                    'exec TABMD_RegisterEntity ' +
                        event?.event?.data?.tableName +
                        ',' +
                        '@UserId="' +
                        JSON.parse(localStorage.getItem('loginData'))
                            ?.Attributes?.Id +
                        '"'
                )
                ?.pipe(take(1))
                ?.subscribe((response) => {
                    if (
                        response.StatusCode == '200' ||
                        !response.Errors.length
                    ) {
                        event?.serviceList?.toastrService.showSuccess(
                            'Entity Registered Successfully',
                            'Success'
                        );
                    } else {
                        event?.serviceList?.toastrService.showError(
                            response.Errors[0],
                            'Error'
                        );
                    }
                });
        } else if (event.event.type === 'syncAppObjectAccess') {
            event?.serviceList?.getService
                .executeRawSql(
                    'exec TABMD_SyncAppObjectAccessWithRoles @UserId="' +
                        JSON.parse(localStorage.getItem('loginData'))
                            ?.Attributes?.Id +
                        '"'
                )
                ?.pipe(take(1))
                ?.subscribe((response) => {
                    if (
                        response.StatusCode == '200' ||
                        !response.Errors.length
                    ) {
                        event?.serviceList?.toastrService.showSuccess(
                            'Sync Appobject aceess missing with Roles/Permission SP Run Successfully',
                            'Success'
                        );
                    } else {
                        event?.serviceList?.toastrService.showError(
                            response.Errors[0],
                            'Error'
                        );
                    }
                });
        }
    }
    onErrorChange(event: any) {}
    onFormLoad(event: any) {}
    onReady(event: any) {}
    proceedObjectBeforeRender(event: any) {}
    onAfterSubmit(event: any) {}
}
