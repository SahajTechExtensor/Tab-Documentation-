import { FormioUtils } from '@formio/angular';
import {
    IActionExtension,
    IScreenExtension,
} from '@techextensor/tab-interface';
import { TabAppFunction, TabConstant } from '@techextensor/tab-service';
import * as $ from 'jquery'; // Import jQuery
import { take } from 'rxjs';

export class FileUploadScreen implements IScreenExtension {
    activeAction: IActionExtension;
    storageType: any;

    constructor() {}

    screenId = '85f0002c-89fe-486b-a869-b612c95ac97e';
    addScreenId = 'ec0181c4-d210-46c8-b0dc-caf926b40cf7';

    onSubmit(screenParameters) {}

    onRender(screenParameters: any) {}

    onBeforeSubmit(screenParameters: any) {
        screenParameters?.event?.callback(null, screenParameters?.event);
    }

    onChange(screenParameters: any) {
        // code for the set dynamic reuqest header and filekey while file storage will URL
        if (screenParameters.event.changed) {
            this.setCustomheaders(screenParameters);
            this.setCustomFormKey(screenParameters);
        }
        if (screenParameters.event.changed?.component?.key == 'DocumentType') {
        }
        // if (screenParameters?.event?.changed?.value[0]?.storage) {
        //     // this.refreshForm(screenParameters, screenParameters?.event?.data);

        // }
    }
    setCustomheaders(screenParameters) {
        screenParameters.formIOObject.formio.component.components.forEach(
            (element) => {
                if (element.storage == 'url') {
                    const customOptions = {
                        headers: {
                            Accept: '*/*',
                            Applicationcode: sessionStorage.getItem(
                                TabConstant.Storage.applicationCode
                            ),
                            Authorization: sessionStorage.getItem(
                                TabConstant.Storage.token
                            ),
                        },
                    };
                    element.options = JSON.stringify(customOptions);
                }
            }
        );
    }
    setCustomFormKey(screenParameters) {
        screenParameters.formIOObject.formio.component.components.forEach(
            (element) => {
                if (element.storage == 'url') {
                    let key =
                        (screenParameters?.event?.data?.DocumentType ?? '0') +
                        ';0;1;0';
                    element.fileKey = key;
                }
            }
        );
    }
    onInvalid(screenParameters) {}

    onNextPage(screenParameters) {}

    onPrevPage(screenParameters) {}

    onErrorChange(screenParameters) {}

    onFormLoad(screenParameters: any) {}

    onReady(screenParameters: any) {}

    onAfterSubmit(screenParameters: any) {}

    proceedObjectBeforeRender(screenParameters: any) {
        const parameters = screenParameters.event.customParams;
        const screenConfiguration =
            screenParameters?.event?.customParams?.screenConfiguration;
        const viewMode = screenConfiguration?.viewMode; // grid, chip, widget
        const allowToUploadFile = screenConfiguration?.allowToUploadFile;
        const showDocumentType = screenConfiguration?.showDocumentType;
        const allowToDelete = screenConfiguration?.allowToDelete;
        const allowToDownload = screenConfiguration?.allowToDownload;

        if (screenConfiguration) {
            FormioUtils.eachComponent(
                screenParameters?.formIOObject?.formio?.component?.components,
                async (component, path) => {
                    if (
                        component.type == 'customGrid' &&
                        component.key == 'UploadedFiles'
                    ) {
                        component.hidden = viewMode != 'grid';
                        // For REFERENCE
                        // screenParameters.event['UploadedFiles'] = {
                        //     parameters: {
                        //         Id: parameters.RecordId,
                        //     },
                        //     isReloadGrid: true,
                        // };
                    }

                    if (
                        component.type == 'customWidget' &&
                        component.key == 'documentWidget'
                    ) {
                        component.hidden = viewMode != 'widget';
                        if (!component.hidden) {
                            component.customOptions.dataSetting.dataSourceMode =
                                'dsq';
                            component.customOptions.dataSetting.parameters = {
                                Id: parameters?.RecordId ?? '',
                            };
                        }
                    }

                    if (
                        component.type == 'customChips' &&
                        component.key == 'chipsComponent'
                    ) {
                        component.hidden = viewMode != 'chip';
                        if (!component.hidden) {
                            component.customOptions.dataSetting[
                                'dataSourceQueryId'
                            ] = 'A9ED889B-F9D4-491E-AACF-805ED6B4416B';
                            component.customOptions.dataSetting['labelField'] =
                                'Title';
                            component.customOptions.dataSetting.valueField =
                                'MediaId';
                            component.customOptions.dataSetting.parameters = {
                                Id: parameters?.RecordId,
                            };

                            if (allowToDownload) {
                                component.label =
                                    component.label + ' (click to download)';
                            }
                            component.customOptions.dataSetting.allowToDownload =
                                allowToDownload;
                            component.customOptions.dataSetting.isDocumentPreview =
                                true;
                            component.customOptions.dataSetting.allowToDelete =
                                allowToDelete;
                            component.customOptions.dataSetting.icon =
                                'insert_drive_file';
                            screenParameters.event['chipsComponent'] = {
                                isRefreshComponent: true,
                            };
                        }
                    }

                    if (
                        component.type == 'select' &&
                        component.key == 'DocumentType'
                    ) {
                        component.hidden = !showDocumentType;
                    }

                    if (component.type == 'file' && component.key == 'Files') {
                        component.hidden = !allowToUploadFile;
                    }

                    if (
                        component.type == 'button' &&
                        component.key == 'submit'
                    ) {
                        component.hidden = !allowToUploadFile;
                    }
                }
            );
        }

        this.refreshForm(screenParameters, screenParameters?.event);
    }

    updateCurrentScreenInstance(actionData) {}

    onCustomEvent(screenParameters: any) {
        screenParameters.formJson.components.forEach((element) => {
            if (element.storage) {
                this.storageType = element.storage;
            }
        });
        if (this.storageType != 'url') {
            if (
                screenParameters.event.type === 'uploadFile' &&
                screenParameters?.event?.data?.Files?.length > 0
            ) {
                this.uploadFiles(screenParameters);
                // For REFERENCE
                // delete screenParameters.event.data.Files;
            }
        } else {
            // code for the update the fileType in mediainfo after upload the file with type 0
            const mediaid = screenParameters.event.data.MediaId;
            if (mediaid) {
                this.uploadMediawithFileType(screenParameters);
                this.addAttachment(screenParameters, mediaid);
            }
        }
    }

    /**
     * Uploads the files from the screen parameters to the server.
     *
     * @param screenParameters - The screen parameters object containing the files to be uploaded.
     */
    uploadFiles(screenParameters) {
        const files = screenParameters?.event?.data?.Files;
        let formData = new FormData();
        const file = screenParameters?.event?.data?.Files[0];
        const blobFile = this.dataURIToBlob(file.url);
        const uploadFile = new File([blobFile], file.originalName);
        let key =
            (screenParameters?.event?.data?.DocumentType ?? '0') + ';0;1;';
        key = key + this.getFileTypeEnumValue(file);
        formData.append(key, uploadFile);

        const toast = screenParameters?.serviceList?.toastrService.showInfo(
            'Uploading file... Please wait',
            'File Upload Progress',
            {
                disableTimeOut: true,
            }
        );

        screenParameters?.serviceList?.mediaUploadService
            ?.uploadMedia(formData)
            .subscribe(
                (res: any) => {
                    screenParameters?.serviceList?.toastrService.clearToast(
                        toast.toastId
                    );
                    if (!!res.Result && res.Result.length > 0) {
                        screenParameters?.serviceList?.toastrService.showInfo(
                            'File Upload Completed',
                            'File Upload Progress'
                        );

                        for (let j = 0; j < res.Result.length; j++) {
                            const media = res.Result[j];
                            this.addAttachment(screenParameters, media.MediaId);
                        }
                    } else {
                        screenParameters?.serviceList?.toastrService.showError(
                            res.Errors[0],
                            'File Upload Failed'
                        );

                        // Loading files back on error
                        this.updateFiles(screenParameters, files);
                    }
                },
                (error: any) => {
                    console.log('failed to upload files', error);
                    screenParameters?.serviceList?.toastrService.clearToast(
                        toast.toastId
                    );
                }
            );
    }
    uploadMediawithFileType(screenParameters) {
        const Mediaid = screenParameters.event.data.MediaId;
        const file = screenParameters.event.data.Files[0];
        let appObject =
            TabAppFunction.getAppObjectFromSystemDBTableName('MediaInfo');
        let payload = { data: {} };
        let data: any = {
            ID: Mediaid,
            FileType: this.getFileTypeEnumValue(file),
        };
        payload.data = data;
        screenParameters.serviceList.updateService
            .updateUsingScreenId('', payload, appObject.ID)
            ?.pipe(take(1))
            ?.subscribe((response) => {
                if (response.StatusCode == '200' || !response.Errors.length) {
                    screenParameters?.serviceList?.toastrService.showSuccess(
                        'Record Updated Successfully',
                        'Success'
                    );
                }
            });
    }
    /**
     * Adds an attachment to the database.
     *
     * @param screenParameters - The screen parameters object.
     * @param mediaId - The media ID of the attachment.
     */
    addAttachment(screenParameters: any, mediaId: string) {
        const attachment = {
            MediaId: mediaId,
            RecordId: screenParameters.formIOObject.formio.data.RecordId,
        };
        screenParameters?.serviceList?.insertService
            ?.insertByTable('TABD_Attachments', attachment)
            ?.subscribe((res) => {
                if (!!res.Result && res.Result.length > 0) {
                    screenParameters?.serviceList?.toastrService.showSuccess(
                        'Attachment Added Successfully',
                        'Success'
                    );
                    // Deleting files
                    screenParameters.event.data =
                        screenParameters.formIOObject.formio.data;
                    this.updateFiles(screenParameters, []);
                } else {
                    screenParameters?.serviceList?.toastrService.showError(
                        res.Errors[0],
                        'File Upload Failed'
                    );
                }
            });
    }

    /**
     * Updates the files in the upload component.
     *
     * @param screenParameters - The screen parameters object.
     * @param files - The files to be updated in the screen.
     */
    updateFiles(screenParameters: any, files: any) {
        if (screenParameters.event) {
            if (files.length == 0) {
                // Removing loaded file from file Uploader component
                $('.fa-remove').trigger('click');
            }
            screenParameters.event.data.Files = files;
            // delete screenParameters.event.data.DocumentType;
            this.refreshForm(screenParameters, screenParameters?.event?.data);
        }
    }

    /**
     * Converts a data URI string to a Blob object.
     *
     * @param dataURI - The data URI string to convert.
     * @returns A Blob object representing the data URI.
     */
    dataURIToBlob(dataURI: string) {
        const splitDataURI = dataURI.split(',');
        const byteString =
            splitDataURI[0].indexOf('base64') >= 0
                ? atob(splitDataURI[1])
                : decodeURI(splitDataURI[1]);
        const mimeString = splitDataURI[0]?.split(':')[1]?.split(';')[0];

        const ia = new Uint8Array(byteString.length);
        for (let i = 0; i < byteString.length; i++)
            ia[i] = byteString.charCodeAt(i);

        return new Blob([ia], { type: mimeString });
    }

    /**
     * Refreshes the form with the updated screen parameters and data.
     *
     * @param screenParameters - The screen parameters object.
     * @param data - The updated data to be refreshed in the form.
     */
    refreshForm(screenParameters, data) {
        screenParameters?.formIOObject.refresh.emit({
            submission: {
                data: data,
            },
        });
    }

    /**
     * Determines a enum value representing the media type.
     * @param {object} file - The file object containing information about the file.
     * @returns {number} - A numerical value representing the file type.
     */
    getFileTypeEnumValue(file: any): number {
        if (file?.type?.startsWith('image/')) {
            return 1;
        } else if (file?.type?.startsWith('audio/')) {
            return 7;
        } else if (file?.type?.startsWith('video/')) {
            return 6;
        } else if (file?.type == 'application/pdf') {
            return 2;
        } else if (file?.type == 'text/csv') {
            return 3;
        } else if (
            file?.name?.toLowerCase().endsWith('.xlsx') ||
            file?.name?.toLowerCase().endsWith('.xls')
        ) {
            return 4;
        } else if (
            file?.name?.endsWith('.docx') ||
            file?.name?.endsWith('.doc')
        ) {
            return 5;
        }

        return 0;
    }
}
