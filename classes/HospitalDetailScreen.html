import { FormioUtils } from '@formio/angular';
import {
    IActionExtension,
    IScreenExtension,
} from '@techextensor/tab-interface';
import { TabConstant } from '@techextensor/tab-service';
import { take } from 'rxjs';

export class HospitalDetailScreen implements IScreenExtension {
    activeAction: IActionExtension;
    userData: any;
    constructor() {}

    screenId = 'd177be46-e1b2-4ff1-b3f2-106393da3547';

    onSubmit(screenParameters) {}

    onRender(screenParameters: any) {}

    async onBeforeSubmit(screenParameters: any) {
        if (
            screenParameters?.event?.data?.CaseHospitalMaster?.length > 0 &&
            screenParameters?.event?.data?.PolicySubtypeId?.ClaimType
                ?.EnumDetailID != 13
        ) {
            await this.uploadFiles(screenParameters);
        } else {
            screenParameters?.event?.callback(null, screenParameters?.event);
        }
    }

    onChange(screenParameters: any) {
        // this is for the dynamic reuqest header for file upload when file storage will URL
        this.setCustomheaders(screenParameters);
        if (
            screenParameters?.event?.changed?.component?.key ==
                'HospitalBillFile' &&
            screenParameters?.event?.changed?.value?.length == 0
        ) {
            const caseHospitalMaster =
                screenParameters?.event?.data.CaseHospitalMaster;
            for (let i = 0; i < caseHospitalMaster?.length; i++) {
                const hospital = caseHospitalMaster[i];
                if (hospital?.container?.HospitalBillFile?.length == 0) {
                    hospital.container['HospitalBill'] = null;
                }
            }
        }
    }
    setCustomheaders(screenParameters) {
        FormioUtils.eachComponent(
            screenParameters?.formIOObject?.formio?.component?.components,
            async (component, path) => {
                if (component.key == 'CaseHospitalMaster') {
                    FormioUtils.eachComponent(
                        component.components[0].components,
                        async (filecomponent, path) => {
                            if (
                                filecomponent.type == 'file' &&
                                filecomponent.storage == 'url'
                            ) {
                                const customOptions = {
                                    headers: {
                                        Accept: '*/*',
                                        Applicationcode: sessionStorage.getItem(
                                            TabConstant.Storage.applicationCode
                                        ),
                                        Authorization: sessionStorage.getItem(
                                            TabConstant.Storage.token
                                        ),
                                    },
                                };
                                filecomponent.options =
                                    JSON.stringify(customOptions);
                            }
                        }
                    );
                }
            }
        );
    }

    onInvalid(event) {}

    onNextPage(event) {}

    onPrevPage(event) {}

    onErrorChange(event) {}

    onFormLoad(screenParameters: any) {}

    onReady(screenParameters: any) {}

    onAfterSubmit(screenParameters: any) {}

    proceedObjectBeforeRender(screenParameters: any) {
        this.loadFiles(screenParameters);
    }

    updateCurrentActionInstance(actionData) {}

    onCustomEvent(screenParameters) {}

    /**
     * Uploads the files from the screen parameters to the server.
     *
     * @param screenParameters - The screen parameters object containing the files to be uploaded.
     */
    async uploadFiles(screenParameters: any) {
        const caseHospitalMaster =
            screenParameters?.event?.data?.CaseHospitalMaster;

        const toast = screenParameters?.serviceList?.toastrService.showInfo(
            'Uploading file... Please wait',
            'File Upload Progress',
            {
                disableTimeOut: true,
            }
        );
        for (let i = 0; i < caseHospitalMaster?.length; i++) {
            const hospital = caseHospitalMaster[i];
            if (!hospital?.container?.HospitalBill?.ID) {
                try {
                    const file = hospital.container.HospitalBillFile[0];
                    // commented this to prevent upload call on submit as we are using File storage mode URL
                    // const mediaInfo = await this.uploadFileAsync(screenParameters, file);
                    screenParameters.event.data.CaseHospitalMaster[i].container[
                        'HospitalBill'
                    ] = file?.data?.Result[0]?.MediaId;
                } catch (error) {
                    console.log('error while uploading file', error);
                    screenParameters?.serviceList?.toastrService.clearToast(
                        toast.toastId
                    );
                    screenParameters?.event?.callback(
                        {
                            message: 'upload failed',
                            component: null,
                        },
                        screenParameters?.event
                    );
                    return;
                }
            }
        }
        screenParameters?.serviceList?.toastrService.clearToast(toast.toastId);
        screenParameters?.event?.callback(null, screenParameters?.event);
    }

    /**
     * load the files and update screen
     *
     * @param screenParameters - The screen parameters object containing the files to be uploaded.
     */
    async loadFiles(screenParameters: any) {
        const caseHospitalMaster = screenParameters?.event?.CaseHospitalMaster;

        for (let i = 0; i < caseHospitalMaster?.length; i++) {
            const hospital = caseHospitalMaster[i];
            if (hospital?.container?.HospitalBill?.ID) {
                const fileData = {
                    name: hospital.container.HospitalBill.Title,
                    originalName: hospital.container.HospitalBill.Title,
                    storage: 'url',
                    size: 0,
                    type: 'text/png',
                    url: hospital.container.HospitalBill.URL,
                };
                try {
                    // comment this due to prevent the getmediainfo call
                    // const blob = await this.getFileAsync(screenParameters, hospital.container.HospitalBill.ID)
                    // fileData.size = blob.size;
                    // fileData.type = blob.type;
                } catch (error) {}
                screenParameters.event.CaseHospitalMaster[i].container[
                    'HospitalBillFile'
                ] = [fileData];
            }
        }

        screenParameters?.formIOObject.refresh.emit({
            submission: {
                data: screenParameters?.event,
            },
        });
    }

    /**
     * load the files and update screen
     *
     * @param screenParameters - The screen parameters object containing the files to be uploaded.
     * @param file - File which you want to upload
     */
    uploadFileAsync(screenParameters: any, file: any): Promise<any> {
        let formData = new FormData();
        const blobFile = this.dataURIToBlob(file.url);
        const uploadFile = new File([blobFile], file.originalName);
        formData.append('0', uploadFile);

        return new Promise((resolve, reject) => {
            return screenParameters?.serviceList?.mediaUploadService
                ?.uploadMedia(formData)
                .subscribe(
                    (res: any) => {
                        if (!!res.Result && res.Result.length > 0) {
                            resolve(res.Result[0]);
                        } else {
                            reject(res);
                        }
                    },
                    (error: any) => {
                        console.log('failed to upload file', error);
                        resolve(error?.error);
                    }
                );
        });
    }

    /**
     * get file from server
     *
     * @param screenParameters
     * @param mediaId - mediaId of file
     * @returns blob of file
     */
    getFileAsync(screenParameters: any, mediaId: string): Promise<Blob> {
        return new Promise((resolve, reject) => {
            screenParameters?.serviceList?.mediaUploadService
                ?.getFileByMediaId(mediaId)
                ?.pipe(take(1))
                .subscribe(
                    (blob: any) => resolve(blob),
                    (error) => reject(error?.error)
                );
        });
    }

    /**
     * Converts a data URI string to a Blob object.
     *
     * @param dataURI - The data URI string to convert.
     * @returns A Blob object representing the data URI.
     */
    dataURIToBlob(dataURI: string) {
        const splitDataURI = dataURI.split(',');
        const byteString =
            splitDataURI[0].indexOf('base64') >= 0
                ? atob(splitDataURI[1])
                : decodeURI(splitDataURI[1]);
        const mimeString = splitDataURI[0]?.split(':')[1]?.split(';')[0];

        const ia = new Uint8Array(byteString.length);
        for (let i = 0; i < byteString.length; i++)
            ia[i] = byteString.charCodeAt(i);

        return new Blob([ia], { type: mimeString });
    }

    /**
     * Converts a blob to base64
     *
     * @param blob - blob
     * @returns base64 string
     */
    convertBlobToBase64(blob: any) {
        return new Promise((resolve, reject) => {
            const reader = new FileReader();
            reader.onerror = reject;
            reader.onload = () => {
                resolve(reader.result);
            };
            reader.readAsDataURL(blob);
        });
    }
}
