import { ElementRef } from '@angular/core';
import { IScreenExtension } from '@techextensor/tab-interface';
import { TabDynamicDialogComponent } from 'app/shared/templates/tab-dynamic-dialog/tab-dynamic-dialog.component';

export class ClosureDocument implements IScreenExtension {
    screenId: string = '09745046-22db-487b-a255-384d4535952f';

    onSubmit(event: any) { }
    onRender(event: any) { }
    onBeforeSubmit(event: any) { }
    onChange(event: any) { }
    onInvalid(event: any) { }
    onNextPage(event: any) { }
    onPrevPage(event: any) { }
    ref;
    onCustomEvent(screenParameters: any) {
        const url = 'ExportData';
        const data = {
            DataReportId: 'E94392FB-B9A1-4E18-BDF7-615DA4939273',
            Reqtokens: {
                Id: screenParameters.event.data.Id,
            },
        };
        if (data.Reqtokens.Id != null) {
            screenParameters.serviceList.mediaUploadService
                .downloadPdfFiles(url, data)
                .subscribe(
                    (res) => {
                        const mediaId = res.Result;

                        if (screenParameters.event.type == 'onCustomEvent') {
                            if (mediaId) {
                                // Fetch the file (blob) using the obtained MediaId
                                screenParameters.serviceList.mediaUploadService
                                    .getFileByMediaId(mediaId)
                                    .subscribe(
                                        (blob) => {
                                            if (blob) {
                                                // Create a download link for the blob
                                                const url =
                                                    window.URL.createObjectURL(
                                                        blob
                                                    );
                                                const filename =
                                                    screenParameters.event.data
                                                        .ClaimNumber
                                                        ? `Closurepdf_${screenParameters.event.data.ClaimNumber}`
                                                        : mediaId.toString();
                                                const link =
                                                    document.createElement('a');
                                                link.href = url;
                                                // Set the file name for the download
                                                link.download = new File(
                                                    [blob],
                                                    filename
                                                ).name;
                                                // Trigger the download and clean up the object URL
                                                link.click();
                                                window.URL.revokeObjectURL(url);
                                            }
                                            // Show a success message using toastr service
                                            screenParameters.serviceList.toastrService.showSuccess(
                                                'Document Download Successfully',
                                                'Success'
                                            );
                                        },
                                        (error) => {
                                            // Handle blob retrieval error and display an error message
                                            screenParameters.serviceList.toastrService.showError(
                                                error.statusText,
                                                'Error'
                                            );
                                        }
                                    );
                            }
                        } else if (
                            screenParameters.event.type ==
                            'onCustomEventForView'
                        ) {
                            const apiData = {
                                AppObjectName: 'MediaInfo',
                                DSQName: 'Detail_MediaInfo',
                                Reqtokens: {
                                    Id: mediaId,
                                },
                            };
                            screenParameters.serviceList.getService
                                .executeRAWDSQWithName(apiData)
                                .subscribe((res) => {
                                    const MediaInfo = res.Result[0];
                                    // Change the fileType to 2
                                    MediaInfo.FileType = 2;
                                    const customEvent = new CustomEvent(
                                        'tabUtilsEvent',
                                        {
                                            detail: {
                                                parameters: MediaInfo ?? {},
                                                pageTitle: 'Document view',
                                                screenId:
                                                    '6279f630-ec9f-45f8-9a0c-e56a1c9d8f1f',
                                                screenMode: 'Add',
                                                showSubmit: false,
                                                eventName: 'openDefaultDialog',
                                            },
                                        }
                                    );
                                    window.dispatchEvent(customEvent);
                                });
                        }
                    },
                    (error) => {
                        // Handle TAB workflow execution error and display an error message
                        screenParameters.serviceList.toastrService.showError(
                            error.error.Errors[0],
                            'Error'
                        );
                    }
                );
        }
        if (screenParameters.event.type == 'onShareEvent') {
            let data = { parameters: {} };

            const shared = {
                AppObjectName: 'CaseWiseEmailIdsView',
                DSQName: 'closure_pdf_email_view',
                Reqtokens: {
                    caseId: screenParameters.event.data.Id,
                },
            };
            if (screenParameters?.event?.data?.Id) {
                screenParameters.serviceList.getService
                    .executeRAWDSQWithName(shared)
                    .subscribe((res) => {
                        data.parameters['emails'] = res.Result[0].EmailIds;
                        data.parameters['CaseId'] =
                            screenParameters.event.data.Id;
                        data.parameters['screenMode'] = 'Add';
                        const target = new ElementRef(
                            screenParameters.event.event.target
                        );
                        const rect =
                            target.nativeElement.getBoundingClientRect();

                        this.ref = screenParameters.dailogRef.open(
                            TabDynamicDialogComponent,
                            {
                                height: '200px',
                                width: '600px',
                                panelClass: 'custom-modalbox',
                                position: {
                                    left: `${rect.left - 550}px`,
                                    top: `${rect.top + 50}px`,
                                    right: `${rect.right}px`,
                                    bottom: `${rect.bottom}px`,
                                },
                                data: {
                                    screenID:
                                        'a099132d-6323-44f2-848c-53de09cc46e9',
                                    data: JSON.stringify(data),
                                    title: 'Share closure pdf',
                                },
                            }
                        );
                    });
            }
        }
    }
    onErrorChange(event: any) {}
    onFormLoad(event: any) {}
    onReady(event: any) {}
    proceedObjectBeforeRender(event: any) {}
    onAfterSubmit(event: any) {}
}
