import { FormioUtils } from '@formio/angular';
import {
    IActionExtension,
    IActionExtensionRegistry,
    IScreenExtension,
} from '@techextensor/tab-interface';
import { take } from 'rxjs';

export class PluginRenderer implements IScreenExtension {
    activeAction: IActionExtension;

    constructor() {}

    screenId = 'dfcb2c85-538c-43e2-bcaf-64eafce26886';

    onSubmit(event) {}

    onRender(screenParameters: any) {}

    onBeforeSubmit(screenParameters: any) {
        screenParameters?.event?.callback(null,screenParameters?.event);
    }

    onChange(screenParameters: any) {}

    onInvalid(event) {}

    onNextPage(event) {}

    onPrevPage(event) {}

    onErrorChange(event) {}

    onFormLoad(screenParameters: any) {}

    onReady(screenParameters: any) {}

    onAfterSubmit(screenParameters: any) {}

    proceedObjectBeforeRender(screenParameters: any) {
        var that = this;
        if (
            screenParameters?.formJson &&
            screenParameters?.formJson?.components
        ) {
            FormioUtils.eachComponent(
                screenParameters?.formJson.components,
                function (component: any) {
                    if (component.type.includes('file')) {
                        // Keeping FileName empty until we get file from server
                        screenParameters.event.APIFileName =
                            screenParameters.event.FileName;
                        screenParameters.event.FileName = [];
                        that.getFile(screenParameters);
                    }
                }
            );
        }
    }

    updateCurrentScreenInstance(actionData) {
        this.activeAction = null;
        var actions = IActionExtensionRegistry.GetImplementations();
        for (var x = 0; x < actions.length; x++) {
            const action = new actions[x]();
            if (actionData.$type.includes(action.actionType)) {
                this.activeAction = action;
            }
        }
    }

    onCustomEvent(event) {
        if (event?.event?.data?.FileName.length > 0) {
            const file = this.DataURIToBlob(
                event?.event?.data?.FileName[0].url
            );
            let formData = new FormData();
            event.event.data.File = event.event.data.FileName[0];
            // delete event.event.data.FileName;
            formData.append('File', file, event.event.data.File?.originalName);
            formData.append('DisplayName', event.event.data.DisplayName);
            if (
                event.event.data.PluginTypeId &&
                typeof event.event.data.PluginTypeId === 'object' &&
                event.event.data.PluginTypeId.hasOwnProperty('ID')
            ) {
                formData.append(
                    'PluginTypeId',
                    event.event.data.PluginTypeId.ID
                );
            } else {
                formData.append('PluginTypeId', event.event.data.PluginTypeId);
            }

            formData.append('IsActive', event.event.data.IsActive);
            if (event.event.data.ID) {
                formData.append('ID', event.event.data.ID);
            }

            event?.serviceList?.plugInService
                .uploadPlugin(formData)
                ?.pipe(take(1))
                ?.subscribe((res) => {
                    if (res?.StatusCode == '200') {
                        event?.serviceList?.toastrService.showSuccess(
                            'Plugin Uploaded Successfully',
                            'Success'
                        );
                    } else {
                        event?.serviceList?.toastrService.showError(
                            res.Errors[0],
                            'Error'
                        );
                    }
                });
        }
    }

    getFile(screenParameters: any) {
        if (screenParameters?.event?.ID) {
            screenParameters?.serviceList?.plugInService
                .getFile(screenParameters?.event?.ID)
                ?.pipe(take(1))
                ?.subscribe(async (res) => {
                    if (res) {
                        const fileName = screenParameters?.event?.APIFileName;
                        var blob = new Blob([res], {
                            type: 'application/javascript',
                        });
                        if (blob) {
                            const base64String: any =
                                await this.convertBlobToBase64(blob);
                            screenParameters.event.FileName = [];
                            screenParameters?.event.FileName.push({
                                hash: btoa(fileName),
                                name: fileName,
                                originalName: fileName,
                                size: blob.size,
                                storage: 'base64',
                                type: 'text/javascript',
                                url: base64String,
                            });

                            screenParameters?.formIOObject.refresh.emit({
                                submission: {
                                    data: screenParameters?.event,
                                },
                            });
                        }
                    }
                });
        }
    }

    private convertBlobToBase64 = (blob) =>
        new Promise((resolve, reject) => {
            const reader = new FileReader();
            reader.onerror = reject;
            reader.onload = () => {
                resolve(reader.result);
            };
            reader.readAsDataURL(blob);
        });

    private DataURIToBlob(dataURI: string) {
        const splitDataURI = dataURI.split(',');
        const byteString =
            splitDataURI[0].indexOf('base64') >= 0
                ? atob(splitDataURI[1])
                : decodeURI(splitDataURI[1]);
        const mimeString = splitDataURI[0].split(':')[1].split(';')[0];

        const ia = new Uint8Array(byteString.length);
        for (let i = 0; i < byteString.length; i++)
            ia[i] = byteString.charCodeAt(i);

        return new Blob([ia], { type: mimeString });
    }
}
