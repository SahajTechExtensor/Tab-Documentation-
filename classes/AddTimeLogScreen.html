import {
    IActionExtension,
    IActionExtensionRegistry,
    IScreenExtension,
} from '@techextensor/tab-interface';
import { cloneDeep } from 'lodash';
import moment from 'moment';

export class AddTimeLogScreen implements IScreenExtension {
    activeAction: IActionExtension;
    constructor() {}

    screenId = 'd75d8318-f604-4b66-99ee-30b718375de2';

    onSubmit(event) {}

    onRender(screenParameters: any) {}

    onBeforeSubmit(screenParameters: any) {
        screenParameters?.event?.callback(null,screenParameters?.event);
    }

    onChange(screenParameters: any) {}

    onInvalid(event) {}

    onNextPage(event) {}

    onPrevPage(event) {}

    onErrorChange(event) {}

    onFormLoad(screenParameters: any) {}

    onReady(screenParameters: any) {}

    onAfterSubmit(screenParameters: any) {

    }

    proceedObjectBeforeRender(screenParameters: any) {
        // Assign Formated FromTime and ToTime 
        if( screenParameters?.event){
           if(screenParameters?.event?.ToTime){
                const inputToTime = screenParameters.event.ToTime;
                const momentToTime = moment(inputToTime, 'DD-MM-YYYY hh:mm:ss A');
                const formatedTotime = momentToTime.format('YYYY-MM-DDTHH:mm:ssZ');
                screenParameters.event.To = formatedTotime;
           } 
            if(screenParameters?.event?.FromTime){
                const inputFromTime = screenParameters.event.FromTime;
                const momentFromTime = moment(inputFromTime, 'DD-MM-YYYY hh:mm:ss A');
                const formatedFromTime = momentFromTime.format('YYYY-MM-DDTHH:mm:ssZ');
                screenParameters.event.From = formatedFromTime;
            }
      }
     this.refreshForm(screenParameters, screenParameters.event);
 
    }
    refreshForm(screenParameters, data) {
            screenParameters?.formIOObject.refresh.emit({
                submission: {
                    data: data,
                },
            });
        }
    updateCurrentActionInstance(actionData) {
    }

    onCustomEvent(event) {}

}
