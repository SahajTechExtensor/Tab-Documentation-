import {
    IActionExtension,
    IScreenExtension,
} from '@techextensor/tab-interface';

export class AddNewCase implements IScreenExtension {
    activeAction: IActionExtension;
    constructor() {}

    screenId = 'e13d201b-4123-4222-acad-5f5ad930009b';

    onSubmit(event) {}

    onRender(screenParameters: any) {}

    onBeforeSubmit(screenParameters: any) {
        
        if (screenParameters?.event.data.SrNumberforCase == undefined || screenParameters.event.data.isFieldValidate == 1) {
            screenParameters?.event?.callback(null, screenParameters?.event);
        } else if (screenParameters.event.data.isFieldValidate == 0 && screenParameters?.event.data.SrNumberforCase != "") {
            screenParameters?.serviceList?.toastrService.showError(
                'Please conduct a reinvestigation before submitting',
                'Error'
            );         
               screenParameters?.event?.callback({
                message:"",
                component:null
               }, screenParameters?.event);

        }
        else if (screenParameters?.event.data.SrNumberforCase != "" && screenParameters.event.data.isFieldValidate == "") {
            screenParameters?.serviceList?.toastrService.showError(
                'Please conduct a reinvestigation before submitting',
                'Error'
            );
             screenParameters?.event?.callback({
                message:"",
                component:null
               }, screenParameters?.event);
        }
    }

    onChange(screenParameters: any) {
        if(screenParameters?.event?.changed?.component?.key == "SrNumberforCase" 
            && screenParameters?.event?.isModified == true) {
            screenParameters.event.data.isFieldValidate = 0;
        }
    }

    onInvalid(event) {}

    onNextPage(event) {}

    onPrevPage(event) {}

    onErrorChange(event) {}

    onFormLoad(screenParameters: any) {}

    onReady(screenParameters: any) {}

    onAfterSubmit(screenParameters: any) {
        var currentURL = window.location.href;

        // Remove add case path segment
        const lastSlashIndex = currentURL.lastIndexOf('/');
        const urlWithoutLastPath = currentURL.substring(0, lastSlashIndex);
        if (urlWithoutLastPath) {
            const viewCaseScreenId = '574d343a-480e-4b8b-86f1-697a3130fa7e';
            const editCaseScreenId = 'dc75bf7d-c1d9-483d-adf5-4690b50208db';
            const caseId = screenParameters?.event.res.Result;
            const newURL = `${urlWithoutLastPath}/${editCaseScreenId}?Id=${caseId}`;

            window.location.href = newURL;
        }
    }

    proceedObjectBeforeRender(screenParameters: any) {}

    updateCurrentActionInstance(actionData) {}

    onCustomEvent(screenParameters) {
        if (screenParameters?.event?.type == 'ReinvestigationCase') {
            const apiData = {
                AppObjectName: 'CaseMaster',
                DSQName: 'OldCaseCheck_caseMaster',
                Reqtokens: {
                    SrNumber: screenParameters.event.data.SrNumberforCase,
                },
            };
            screenParameters.serviceList.getService
                .executeRAWDSQWithName(apiData)
                .subscribe(
                    (res) => {
                        if ((res.StatusCode == 200 || !res.Errors.length) &&  res.Result?.length > 0) {
                            const caseId = res.Result[0].Id;
                            const SrNumber = res.Result[0].SrNumber;
                            const allowStatuses = [
                                '674B7C21-F87A-4AB0-90E3-50BDDEEBDA9D',
                                '2948147A-BC6D-41B6-99B9-906047F151A5',
                                '113A8867-176F-404A-A266-73C57D7C3C2A',
                                '4642F071-A17A-45FA-96E4-C6AA844DB43A',
                                'D7BB6C63-0669-4DF9-A422-969AD1EBF85E'
                            ].map(id => id.toLocaleLowerCase());
                            const caseStatusId = res.Result[0]?.RecordInfo?.BlueprintStatusId;
                            const addedSrNumber =
                                screenParameters.event.data.SrNumberforCase;
                            screenParameters.event.data.OldCaseId = caseId;
                            if (allowStatuses.includes(caseStatusId.toLocaleLowerCase())) {
                                if (
                                    addedSrNumber.toLowerCase() ===
                                    SrNumber.toLowerCase()
                                ) {
                                    if (caseId) {
                                        var currentURL = window.location.href;
                                        const lastSlashIndex =
                                            currentURL.lastIndexOf('/');
                                        const urlWithoutLastPath =
                                            currentURL.substring(0, lastSlashIndex);
                                        if (urlWithoutLastPath) {
                                            const viewCaseScreenId =
                                                '574d343a-480e-4b8b-86f1-697a3130fa7e';
                                            const editCaseScreenId =
                                                'fe4e24f0-9839-4228-9b85-af0dc0dc1e89';
                                            const newURL = `${urlWithoutLastPath}/${viewCaseScreenId}?Id=${caseId}`;

                                            window.open(newURL, '_blank');
                                            screenParameters.event.data.isFieldValidate = 1;
                                        }
                                    }
                                } else {
                                    screenParameters?.serviceList?.toastrService.showError(
                                        'Old case not found for the provided Service Request number. Please verify that you have entered the correct SR number or check if the case exists in our records.',
                                        'Error'
                                    );
                                }
                            } else {
                                screenParameters?.serviceList?.toastrService.showError(
                                    'The current status of the old case has not yet been disclosed.',
                                    'Error'
                                );
                            }
                        } else {
                            screenParameters.event.data.isFieldValidate = 0;
                            screenParameters?.serviceList?.toastrService.showError(
                                'Old case not found for the provided Service Request number. Please verify that you have entered the correct SR number or check if the case exists in our records.',
                                'Error'
                            );
                        }
                    },

                    (error) => {
                        if (error.status === 400) {
                            screenParameters?.serviceList?.toastrService.showError(
                                'Old case not found for the provided Service Request number. Please verify that you have entered the correct SR number or check if the case exists in our records.',
                                'Error'
                            );
                        }
                    }
                );
        }
    }
}
