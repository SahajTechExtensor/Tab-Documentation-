import {
    IActionExtension,
    IActionExtensionRegistry,
    IScreenExtension,
} from '@techextensor/tab-interface';
import { cloneDeep } from 'lodash';

export class AddBOScreen implements IScreenExtension {
    activeAction: IActionExtension;
    constructor() {}

    screenId = 'f9ee08a2-547a-446b-96a4-d8cff6e4b9cb';

    onSubmit(event) {}

    onRender(screenParameters: any) {}

    onBeforeSubmit(screenParameters: any) {
        this.RemoveKeyOfAspNetUserID(screenParameters);
        this.AspnetUserDataToSendInWorkflow(screenParameters?.event?.data);
        if (
            screenParameters?.event?.hasOwnProperty('isEdit') &&
            screenParameters?.event?.isEdit !== null &&
            screenParameters?.event?.isEdit !== undefined &&
            screenParameters?.event?.isEdit !== ''
        ) {
            screenParameters?.serviceList?.toastrService.showInfo(
                'Please note that there may be a delay in the user update in the listing view. Thank you for your patience.',
                'Info'
            );
            screenParameters.event.data['Id'] = screenParameters?.event?.isEdit;
            this.UpdateUser(screenParameters);
        }
        screenParameters?.event?.callback(null,screenParameters?.event);
    }

    private RemoveKeyOfAspNetUserID(screenParameters) {
        if (screenParameters.event.data.hasOwnProperty('AspNetUserId')) {
            screenParameters.event.data['AspNetUserData'] = cloneDeep(
                screenParameters?.event?.data['AspNetUserId']
            );
            delete screenParameters.event.data.AspNetUserId;
        }
    }

    onChange(screenParameters: any) {}

    onInvalid(event) {}

    onNextPage(event) {}

    onPrevPage(event) {}

    onErrorChange(event) {}

    onFormLoad(screenParameters: any) {}

    onReady(screenParameters: any) {}

    onAfterSubmit(screenParameters: any) {
        if (screenParameters?.event?.res?.StatusCode === '200'){
            this.AspnetUserDataToSendInWorkflow(
                screenParameters?.event?.formSubmission
            );
            if (
                screenParameters?.event?.event.hasOwnProperty('isEdit') &&
                screenParameters?.event?.event.isEdit !== null &&
                screenParameters?.event?.event.isEdit !== undefined &&
                screenParameters?.event?.event.isEdit !== ''
            ) {
                screenParameters.event.formSubmission['Id'] =
                    screenParameters?.event?.event?.isEdit;
            } else {
                screenParameters.event.formSubmission['Id'] =
                    screenParameters?.event?.res?.Result;
                if (
                    screenParameters?.event?.event?.data?.hasOwnProperty(
                        'UserName'
                    ) &&
                    screenParameters.event.event.data['UserName']
                ) {
                    screenParameters.event.formSubmission['PrimaryEmail'] =
                        screenParameters.event.event.data['UserName'];
                    screenParameters.event.formSubmission['Role'] = 'Back Officer';
                    screenParameters?.serviceList?.toastrService.showInfo(
                        'Please note that there may be a delay in the user appearing in the listing view. Thank you for your patience.',
                        'Info'
                    );
                    this.RegisterUser(screenParameters);
                }
            }
        } else {
            this.addKeyOfAspNetUserID(screenParameters);
            screenParameters?.formIOObject.refresh.emit({
                submission: {
                    data: screenParameters.event.formSubmission,
                },
            });
        }
    }

    private addKeyOfAspNetUserID(screenParameters) {
        if (screenParameters?.event?.formSubmission?.hasOwnProperty('AspNetUserData')) {
            screenParameters.event.formSubmission['AspNetUserId'] = cloneDeep(
                screenParameters?.event?.formSubmission['AspNetUserData']
            );
            delete screenParameters.event.formSubmission['AspNetUserData'];
        }
    }

    private AspnetUserDataToSendInWorkflow(formSubmission) {
        formSubmission['IsEnabled'] = formSubmission?.AspNetUserData?.IsEnabled;
        formSubmission['FirstName'] =
            formSubmission?.AspNetUserData?.PersonId?.FirstName;
        formSubmission['LastName'] =
            formSubmission?.AspNetUserData?.PersonId?.LastName;
        formSubmission['MiddleName'] =
            formSubmission?.AspNetUserData?.PersonId?.MiddleName;
        formSubmission['EducationQualification'] =
            formSubmission?.AspNetUserData?.PersonId?.EducationQualification;
        formSubmission['AdditionalQualification'] =
            formSubmission?.AspNetUserData?.PersonId?.AdditionalQualification;
        formSubmission['Age'] = formSubmission?.AspNetUserData?.PersonId?.Age;
    }

    private RegisterUser(event) {
        event?.serviceList?.tabWorkflowService
            .executeELSAWorkflow('RegisterUser', event?.event?.formSubmission)
            ?.subscribe();
    }

    proceedObjectBeforeRender(screenParameters: any) {}

    updateCurrentActionInstance(actionData) {
        this.activeAction = null;
        var actions = IActionExtensionRegistry.GetImplementations();
        for (var x = 0; x < actions.length; x++) {
            const action = new actions[x]();
            if (actionData.$type.includes(action.actionType)) {
                this.activeAction = action;
            }
        }
    }

    onCustomEvent(event) {}

    private UpdateUser(event) {
        event?.serviceList?.tabWorkflowService
            .executeELSAWorkflow('UsersUpdate', event?.event?.data)
            ?.subscribe();
    }
}
