import { ElementRef } from '@angular/core';
import { FormioUtils } from '@formio/angular';
import {
    IActionExtension,
    IActionExtensionRegistry,
    IScreenExtension,
} from '@techextensor/tab-interface';
import { TabDynamicDialogComponent } from 'app/shared/templates/tab-dynamic-dialog/tab-dynamic-dialog.component';

export class MBVCaseQuestionReOrderScreen implements IScreenExtension {
    activeAction: IActionExtension;
    constructor() {}

    screenId = '5ce9290e-ef96-47f9-a153-df8122a26bf3';

    onSubmit(event) {}

    onRender(screenParameters: any) {}

    onBeforeSubmit(screenParameters: any) {
        screenParameters?.event?.callback(null, screenParameters?.event);
    }

    onChange(screenParameters: any) {}

    onInvalid(event) {}

    onNextPage(event) {}

    onPrevPage(event) {}

    onErrorChange(event) {}

    onFormLoad(screenParameters: any) {}

    onReady(screenParameters: any) {}

    onAfterSubmit(screenParameters: any) {}

    proceedObjectBeforeRender(screenParameters: any) {}

    updateCurrentActionInstance(actionData) {
        this.activeAction = null;
        var actions = IActionExtensionRegistry.GetImplementations();
        for (var x = 0; x < actions.length; x++) {
            const action = new actions[x]();
            if (actionData.$type.includes(action.actionType)) {
                this.activeAction = action;
            }
        }
    }

    ref;
    onCustomEvent(screenParameters) {
        if (screenParameters.event.type == 'ReOrderQuestions') {
            let data = { parameters: {} };
            data.parameters['screenMode'] = 'Add';
            data.parameters['CaseSectionId'] =
                screenParameters?.event?.data?.CaseSectionMaster?.selectedRow?.Id;
            if (screenParameters?.event?.data?.isForFo == 1) {
                data.parameters['LanguageId'] = 1;
            }
            data.parameters['isForFo'] =
                screenParameters?.event?.data?.isForFo == 1;
            const target = new ElementRef(screenParameters?.event?.event?.target);
            const rect = target?.nativeElement?.getBoundingClientRect();
            this.ref = screenParameters.dailogRef.open(TabDynamicDialogComponent, {
                height: '500px',
                width: '600px',
                panelClass: 'custom-modalbox',
                position: {
                    left: `${rect?.left - 500}px`,
                    top: `${rect?.top + 50}px`,
                    right: `${rect?.right}px`,
                    bottom: `${rect?.bottom}px`,
                },
                data: {
                    screenID: 'dea9f4fa-dd84-4b9a-a23d-ee53f2103329',
                    data: JSON.stringify(data),
                    title: 'Question Order',
                },
            });
            this.ref.afterClosed().subscribe((result) => {
                const questionColumn = FormioUtils.getComponent(screenParameters?.formIOObject?.formio?.components, 'columns', true);
                if (questionColumn) {
                    const questionChildScreen = FormioUtils.getComponent(questionColumn.components,
                        screenParameters?.event?.data?.isForFo == 0 ? 'expertPrepareQuestions' : 'prepareQuestions'
                        , true);
                    if (questionChildScreen) questionChildScreen.triggerRedraw();
                }
            });
        }
    }
}
