import { ElementRef } from '@angular/core';
import { IActionExtension, IScreenExtension } from '@techextensor/tab-interface';
import { TabDynamicDialogComponent } from 'app/shared/templates/tab-dynamic-dialog/tab-dynamic-dialog.component';
import * as $ from 'jquery';

export class ClosurePdfShare implements IScreenExtension {
    activeAction: IActionExtension;
    shareData: any;
    screenId: string = 'a099132d-6323-44f2-848c-53de09cc46e9';

    onSubmit(event: any) {
    }
    onRender(event: any) {}
    onBeforeSubmit(screenParameters: any) {
        screenParameters?.event?.callback(null, screenParameters?.event);
    }
    onChange(event: any) {}
    onInvalid(event: any) {}
    onNextPage(event: any) {}
    onPrevPage(event: any) {}
    onCustomEvent(screenParameters: any) {
        let downloadUrl
        this.shareData = screenParameters.formIOObject.formio.data;
        const url = 'ExportData';
        const data = {
            DataReportId: 'E94392FB-B9A1-4E18-BDF7-615DA4939273',
            Reqtokens: {
                Id: this.shareData.CaseId,
            },
        };
        
        screenParameters.serviceList.mediaUploadService
            .downloadPdfFiles(url, data)
            .subscribe(
                (res) => {
                    const mediaId = res.Result;

                    if(mediaId){
                    const apiData = {
                        AppObjectName: 'MediaInfo',
                        DSQName: 'Detail_MediaInfo',
                        Reqtokens: {
                            Id: mediaId,
                        },
                    };
                    screenParameters.serviceList.getService
                        .executeRAWDSQWithName(apiData)
                        .subscribe((res) => {
                             downloadUrl = res.Result[0].URL
                                 // Call tabWorkflowService here, inside the subscribe block
                            if (downloadUrl) {
                                const workflowData = {
                                    CaseId: this.shareData.CaseId,
                                    Cc: this.shareData.Cc,
                                    Bcc: this.shareData.Bcc,
                                    DownloadURL: downloadUrl,
                                };

                                screenParameters?.serviceList?.toastrService.showInfo(
                                    ' Please wait, There may be a delay in receiving the email with the shared Closure PDF. Thank you for your patience.',
                                    'Info'
                                );
                                      this.sendClosurePdf(screenParameters,workflowData);
                                      $(".close-button").click();    


                            }

                        })
                    }
                })
        }
        
    private sendClosurePdf(event,data)
    {
        event?.serviceList?.tabWorkflowService
        .executeELSAWorkflow('SendClosurePDF',data)?.subscribe(res=>{
            event?.serviceList?.toastrService?.showSuccess('Closure Pdf shared successfully','Success')
        });
    }
    
    onErrorChange(event: any) {}
    onFormLoad(event: any) {}
    onReady(event: any) {}
    proceedObjectBeforeRender(event: any) {}
    onAfterSubmit(event: any) {
       
    }
}
