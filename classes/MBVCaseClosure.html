import {
    IActionExtension,
    IScreenExtension,
} from '@techextensor/tab-interface';

export class MBVCaseClosure implements IScreenExtension {
    activeAction: IActionExtension;
    constructor() {}

    screenId = '73902bfc-fdf1-439a-bca5-01deadffc700';

    onSubmit(event) {}

    onRender(screenParameters: any) {}

    onBeforeSubmit(screenParameters: any) {
        if (
            screenParameters?.event &&
            screenParameters?.event?.data &&
            screenParameters?.event?.data?.CaseClosureMaster?.length > 0
        ) {
            for (const closureObj of screenParameters.event.data
                .CaseClosureMaster) {
                if (
                    closureObj &&
                    closureObj.container?.ClosureGivenBy &&
                    closureObj.container.ClosureGivenBy?.Id
                ) {
                    closureObj.container.ClosureGivenBy =
                        closureObj.container.ClosureGivenBy.Id;
                }

                if (
                    closureObj.container.StatusId == 3 ||
                    closureObj.container.StatusId == 2
                ) {
                    closureObj.container.Conclusion = null;
                }
            }
        }
        screenParameters?.event?.callback(null, screenParameters?.event);
    }

    onChange(screenParameters: any) {
        if (
            screenParameters?.event?.changed?.component?.type == 'select' &&
            screenParameters?.event?.changed?.component?.key == 'StatusId'
        ) {
            if (
                screenParameters?.event?.changed?.value == 3 ||
                screenParameters?.event?.changed?.value == 2
            ) {
                screenParameters.event.data.CaseClosureMaster[0].container.Remark =
                    null;
                screenParameters.event.metadata.selectData.CaseClosureMaster[0].container[0].Remark =
                    null;
                screenParameters?.formIOObject.refresh.emit({
                    submission: {
                        data: screenParameters.event.data,
                    },
                });
            }
            if (screenParameters?.event?.changed?.value == 1) {
                screenParameters.event.data.CaseClosureMaster[0].container.Amount =
                    null;
                screenParameters.event.metadata.selectData.CaseClosureMaster[0].container[0].Amount =
                    null;
                screenParameters?.formIOObject.refresh.emit({
                    submission: {
                        data: screenParameters.event.data,
                    },
                });
            }
        }
        if (
            screenParameters?.event?.changed?.component?.type == 'select' &&
            screenParameters?.event?.changed?.component?.key == 'Conclusion'
        ) {
            if (
                screenParameters?.event?.changed?.value == 3 ||
                screenParameters?.event?.changed?.value == 2
            ) {
                const mergedObject =
                    screenParameters.event.data.CaseClosureMaster[0].container
                        .ClosureReason[0];
                mergedObject.container.Remark = '';
                mergedObject.container.ReasonId = '';
                screenParameters.event.data.CaseClosureMaster[0].container.ClosureReason =
                    mergedObject;
                screenParameters.event.metadata.selectData.CaseClosureMaster[0].container[0].ClosureReason =
                    mergedObject;
                screenParameters?.formIOObject.refresh.emit({
                    submission: {
                        data: screenParameters.event.data,
                    },
                });
            }
        }
    }

    onInvalid(event) {}

    onNextPage(event) {}

    onPrevPage(event) {}

    onErrorChange(event) {}

    onFormLoad(screenParameters: any) {}

    onReady(screenParameters: any) {}

    onAfterSubmit(screenParameters: any) {}

    proceedObjectBeforeRender(screenParameters: any) {}

    updateCurrentActionInstance(actionData) {}

    onCustomEvent(screenParameters) {}
}
