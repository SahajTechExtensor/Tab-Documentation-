    onChange(screenParameters: any) {
        if (
            screenParameters?.event?.changed?.component?.type == 'select' &&
            screenParameters?.event?.changed?.component?.key == 'status' &&
            screenParameters?.event?.isModified
        ) {
            const status =
                screenParameters?.event?.changed?.component.data.values.find(
                    (s) =>
                        s.StatusId?.toLowerCase() ==
                        screenParameters?.event.changed.value?.toLowerCase()
                );
            // console.log('updated status', status);
            if (status) {
                screenParameters.event.data.RecordInfo.BlueprintStatusId =
                    status?.StatusId;
                this.performTransactionAction(screenParameters, status);
            }
        }
    }

    updateStatus = (screenParameters, status) => {
        screenParameters.serviceList.appService.tabCoreHttpService.httpClient
            .get(
                TabConstant.API.endpoint +
                    'Transition/UpdateStatus?RecordId=' +
                    screenParameters?.event?.data?.RecordInfo?.RecordId +
                    '&NextPossibleStatusId=' +
                    status.StatusId,
                {
                    headers: {
                        ApplicationCode: sessionStorage.getItem(
                            TabConstant.Storage.applicationCode
                        ),
                        Authorization: sessionStorage.getItem(
                            TabConstant.Storage.token
                        ),
                    },
                }
            )
            ?.pipe(take(1))
            .subscribe((res) => {
                if (res?.StatusCode == '200') {
                    // Loading next statuses
                    FormioUtils.eachComponent(
                        screenParameters?.formIOObject?.formio?.component
                            ?.components,
                        (component: any, path: any) => {
                            if (
                                component.type == 'select' &&
                                component.key == 'status'
                            ) {
                                this.loadNextStatuses(
                                    screenParameters,
                                    screenParameters?.event?.data?.RecordInfo
                                        ?.RecordId,
                                    component,
                                    true
                                );
                            }
                        }
                    );
                    screenParameters?.serviceList?.toastrService.showSuccess(
                        'Status Updated successfully',
                        'Success'
                    );
                } else {
                    screenParameters?.serviceList?.toastrService.showError(
                        res.Errors[0],
                        'Error'
                    );
                }
            });
    };

    proceedObjectBeforeRender(screenParameters: any) {
        FormioUtils.eachComponent(
            screenParameters?.formIOObject?.formio?.component?.components,
            (component: any, path: any) => {
                if (component.type == 'select' && component.key == 'status') {
                    if (
                        screenParameters?.event?.RecordInfo &&
                        screenParameters?.event?.RecordInfo?.BlueprintId
                    ) {
                        this.loadNextStatuses(
                            screenParameters,
                            screenParameters?.event?.RecordInfo?.RecordId,
                            component,
                            true
                        );
                    }
                }
            }
        );
    }
    // on click on history and status timeline

        let sid = '';
        let data = { parameters: {} };

        if (screenParameters.event.tabHTMLEvent.detail.eventName == 'onShare') {
            sid = '001b0c09-4429-4c0e-a9e2-97917731e378';

            data.parameters['screenMode'] = 'Add';
            title = 'Share Record';
        } else if (

            title = 'History of Record';
            data.parameters['history'] = {

                        ?.RecordId,
                BlueprintId:
                    screenParameters?.formIOObject?.formio?.data?.RecordInfo

                SLASchemeID:
                    screenParameters?.formIOObject?.formio?.data?.RecordInfo
                        ?.SLASchemeID,
                SLAValue:
                    screenParameters?.formIOObject?.formio?.data?.RecordInfo
                        ?.SLAValue,
            };
            data.parameters['screenMode'] = 'Add';
        } else if (
            screenParameters.event.tabHTMLEvent.detail.eventName == 'onComment'
        ) {
            sid = '26cc7710-6e93-4aaa-aca8-2f6a8608435e';

                        Authorization: sessionStorage.getItem(
                            TabConstant.Storage.token
                        ),

                    StatusId: currentStatus.FromStatusId.toLowerCase(),
                });

                    statuses.push({
                        Name: status.ToStatusName,
                        StatusId: status.ToStatusId.toLowerCase(),
                        TransitionHooks_Before: status.TransitionHooks_Before,
                        TransitionHooks_During: status.TransitionHooks_During,
                        TransitionHooks_After: status.TransitionHooks_After,
                    });
