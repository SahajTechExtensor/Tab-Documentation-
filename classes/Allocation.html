import { FormioUtils } from '@formio/angular';
import {
    IActionExtension,
    IScreenExtension,
} from '@techextensor/tab-interface';
import { TabAppFunction, TabClasses } from '@techextensor/tab-service';
import { FormRenderSyncFieldLogic } from 'app/modules/admin/tab-form-renderer/tab-component-processor';
import { take } from 'rxjs';

export class Allocation implements IScreenExtension {
    activeAction: IActionExtension;
    mediaInfo: any;
    constructor() { }

    screenId = '972ef753-cb60-40b9-9489-1bf73cc69683';

    onSubmit(event) { }

    onRender(screenParameters: any) { }

    onBeforeSubmit(screenParameters: any) {
        screenParameters?.event?.callback(null, screenParameters?.event);
    }

    onChange(screenParameters: any) {
        if (
            screenParameters?.event?.changed?.component?.type == 'select' &&
            screenParameters?.event?.isModified
        ) {
            if (screenParameters?.event?.changed?.component?.key == 'AssignedFOId' &&
                screenParameters?.event?.data.IsFullyAssignedToFO == false) {
                this.updateScreenData(screenParameters);
            }
            if (
                screenParameters?.event?.changed?.component?.key == 'FOId' &&
                screenParameters?.event?.data?.IsFullyAssignedToFO == true) {
                const caseId = screenParameters.event?.data?.Id;
                const foId =
                    screenParameters?.event?.changed?.value;
                if (foId) {
                    this.UpdateCase(screenParameters, caseId, foId);
                    this.enableDisableComponent(screenParameters, 'FOId', 'select', true);
                    const columnComponent = FormioUtils.getComponent(screenParameters?.formIOObject?.formio?.components, 'columns', true);
                    if (columnComponent) {
                        const selectDropDown = FormioUtils.getComponent(columnComponent.components, 'FOId', true);
                        if (selectDropDown) selectDropDown.triggerRedraw();
                    }
                }
            }
        }
    }

    onInvalid(event) { }

    onNextPage(event) { }

    onPrevPage(event) { }

    onErrorChange(event) { }

    onFormLoad(screenParameters: any) { }

    onReady(screenParameters: any) { }

    onAfterSubmit(screenParameters: any) { }

    proceedObjectBeforeRender(screenParameters: any) {
        // this is for disable select after QC complete status of case
        const notAllowedStatuses = ['553A60BF-7AC1-4264-B391-6B28D880C8C2',
                                    '64F64228-7694-447A-A7AB-12C51AB4E3FE',
                                    '674B7C21-F87A-4AB0-90E3-50BDDEEBDA9D',
                                    '2948147A-BC6D-41B6-99B9-906047F151A5',
                                    '113A8867-176F-404A-A266-73C57D7C3C2A',
                                    'D7BB6C63-0669-4DF9-A422-969AD1EBF85E',
                                    '4642F071-A17A-45FA-96E4-C6AA844DB43A'
                                    ].map(a => a.toLowerCase());
        const isNowAllowed = notAllowedStatuses.find(s=> s == screenParameters?.event?.RecordInfo?.BlueprintStatusId?.toLowerCase());
        if (isNowAllowed || 
            (screenParameters?.event?.FOId != null &&
            screenParameters?.event?.IsFullyAssignedToFO == true)) {
            this.enableDisableComponent(screenParameters, 'FOId', 'select', true);
        }
        this.getFOFeeMasterId(screenParameters)
    }

    updateCurrentActionInstance(actionData) { }
    getFOFeeMasterId(screenParameters) {
        if(screenParameters.event){
        const InsuranceCompany = screenParameters.event.InsuranceCompany;
        const CaseType = screenParameters.event.PolicySubtypeId.ClaimType.EnumDetailID;

        const appInfo = {
            AppObjectName: 'FOFeeMaster',
            DSQName: 'GetId_FoFeeMaster',
            Reqtokens: {
                CaseType: CaseType,
                InsuranceCompanyId: InsuranceCompany
            },
        };
        screenParameters?.serviceList?.getService?.executeRAWDSQWithName(appInfo)
            .subscribe(res => {
                if (!!res.Result && res.Result.length > 0) {
                    const idList = res.Result.map(item => item.Id).join(',');
                    screenParameters.event['IdList'] = idList;
                    this.refreshForm(screenParameters, screenParameters.event);
                }
            })
            }

    }
    ref;
    async onCustomEvent(screenParameters) {
        if (screenParameters.event.type == 'PullBackSection') {
            if (
                screenParameters.event?.data?.IsDocumentSynced == false || screenParameters.event?.data?.IsDocumentSynced == 0
            ) {
                let text = "There are some documents which are not synched, Are you sure you want to Pull back case? Clicking on Yes, Case will be pull backed & Documents are lost.";
                if (confirm(text) == true) {
                    this.pullBackSection(screenParameters);
                } else {
                    return false;
                }
            } else {
                this.pullBackSection(screenParameters);
            }
        } else if (screenParameters.event.type == 'PullBackCase') {
            if (screenParameters.event?.data?.CaseSectionMaster.find(x => x.container?.IsDocumentSynced == false || x.container?.IsDocumentSynced == 0)) {
                let text = "There are some documents which are not synched, Are you sure you want to Pull back case? Clicking on Yes, Case will be pull backed & Documents are lost.";
                if (confirm(text) == true) {
                    this.pullBackCase(screenParameters);
                } else {
                    return false;
                }
            }
            else {
                this.pullBackCase(screenParameters);
            }

        }
    }

    private pullBackCase(screenParameters: any) {
        let FOId = screenParameters.event?.data?.FOId;
        let CaseId = screenParameters.event?.data?.Id;

        if (!!FOId && !!CaseId) {
            let data: any = {
                AppobjectId: "0E4FD907-F996-497A-AB5D-DB278C21A7F4",
                PrimaryKey: CaseId,
                PullBackUserId: FOId
            };
            screenParameters.serviceList.insertService
                .insertByTable('PullBackCase', data)
                ?.pipe(take(1))
                ?.subscribe((response) => {
                    if (
                        response.StatusCode == '200' ||
                        !response.Errors.length
                    ) {
                        screenParameters.event.data.FOId = null;
                        // delete screenParameters.event.data.FOId;
                        this.enableDisableComponent(screenParameters, 'FOId', 'select', false);
                        this.refreshForm(screenParameters, screenParameters.event.data);
                        const columnComponent = FormioUtils.getComponent(screenParameters?.formIOObject?.formio?.components, 'columns', true);
                        if (columnComponent) {
                            const selectDropDown = FormioUtils.getComponent(columnComponent.components, 'FOId', true);
                            if (selectDropDown) selectDropDown.triggerRedraw();
                        }
                        screenParameters?.serviceList?.toastrService.showSuccess(
                            'Case Pull Back Successfully',
                            'Success'
                        );
                        this.UpdateCase(screenParameters, CaseId, null);
                        this.RemoveCasessignFO(screenParameters, CaseId);
                    }
                });
        }
    }

    private pullBackSection(screenParameters: any) {
        let AssignedFOId = screenParameters.event?.data?.AssignedFOId;
        let CaseSectionId = screenParameters.event?.data?.Id;

        if (!!AssignedFOId && !!CaseSectionId) {
            let data: any = {
                AppobjectId: "4cfd04a2-ead8-4663-9901-13763a0806c4",
                PrimaryKey: CaseSectionId,
                PullBackUserId: AssignedFOId
            };
            screenParameters.serviceList.insertService
                .insertByTable('PullBackCase', data)
                ?.pipe(take(1))
                ?.subscribe((response) => {
                    if (
                        response.StatusCode == '200' ||
                        !response.Errors.length
                    ) {
                        screenParameters.event.data.AssignedFOId = null;
                        delete screenParameters.event.data.AssignedFOId;
                        screenParameters?.serviceList?.toastrService.showSuccess(
                            'Section Pull Back Successfully',
                            'Success'
                        );
                        this.RemoveCaseSectionAssignFO(screenParameters, CaseSectionId);
                    }
                });
        }
    }

    private UpdateCase(screenParameters: any, caseId: string, fOId: string) {
        let appObject =
            TabAppFunction.getAppObjectFromSystemDBTableName(
                'CaseMaster'
            );

        let payload = { data: {} };

        let data: any = {
            Id: caseId,
            FOId: fOId ?? null,
            IsFullyAssignedToFO: fOId != null,
        };
        payload.data = data;
        screenParameters.event.data.FOId = fOId;

        screenParameters.serviceList.updateService
            .updateUsingScreenId('', payload, appObject.ID)
            ?.pipe(take(1))
            ?.subscribe((response) => {
                if (
                    response.StatusCode == '200' ||
                    !response.Errors.length
                ) {
                    // screenParameters.formIOObject.formio.data.FOId = fOId;


                    screenParameters?.serviceList?.toastrService.showSuccess(
                        'Record Updated Successfully',
                        'Success'
                    );
                }
            });
    }

    private RemoveCaseSectionAssignFO(screenParameters: any, CaseSectionId: string) {
        let appObject =
            TabAppFunction.getAppObjectFromSystemDBTableName(
                'CaseSectionMaster'
            );

        let payload = { data: {} };

        let data: any = {
            Id: CaseSectionId,
            AssignedFOId: null,
            IsInvestigationCompleted: 0,
            IsDocumentSynced:null,
            SubFOId: null
        };

        payload.data = data;

        screenParameters.serviceList.updateService
            .updateUsingScreenId('', payload, appObject.ID)
            ?.pipe(take(1))
            ?.subscribe((response) => {
                if (
                    response.StatusCode == '200' ||
                    !response.Errors.length
                ) {
                    const section = screenParameters?.event?.data;
                    const caseMaster = screenParameters.formIOObject?.formio?.data?.CaseSectionMaster;

                    screenParameters.formIOObject.formio.data.CaseSectionMaster =
                        caseMaster.map(s => {
                            if (s?.container?.Id?.toLowerCase() == section?.Id?.toLowerCase()) {
                                delete s.container.AssignedFOId;
                            }
                            return s;
                        });
                    this.refreshForm(screenParameters, screenParameters.formIOObject.formio.data);


                    screenParameters?.serviceList?.toastrService.showSuccess(
                        'Record Updated Successfully',
                        'Success'
                    );
                }
            });
    }
    private RemoveCasessignFO(screenParameters: any, CaseId: string) {
        let queryFieldsArray: TabClasses.UpdateQueryParameter[] = [];
        queryFieldsArray.push({
            FieldName: 'CaseId',
            FieldValue: CaseId,
        });
        let payload = { data: {} };

        let data: any = {
            IsInvestigationCompleted: 0,
            IsDocumentSynced:null
        };

        payload.data = data;
        screenParameters.serviceList.updateService
            .updateUsingDSQ(
                '2f1d241e-16b2-4214-a6ef-fe973606fe47'
                , payload,
                'CaseSectionMaster'
                , queryFieldsArray)
            ?.pipe(take(1))
            ?.subscribe((response) => {
                if (
                    response.StatusCode == '200' ||
                    !response.Errors.length
                ) {
                    this.refreshForm(screenParameters, screenParameters.formIOObject.formio.data);
                }
            });
    }

    private updateScreenData(screenParameters) {
        FormRenderSyncFieldLogic.syncFields(
            screenParameters.event,
            screenParameters.formJson,
            this.screenId,
            screenParameters.serviceList
        );
        screenParameters.serviceList.updateService
            .updateUsingScreenId(this.screenId, screenParameters.event)
            ?.pipe(take(1))
            ?.subscribe(
                (res) => {
                    if (res?.StatusCode == '200') {
                        screenParameters?.serviceList?.toastrService.showSuccess(
                            'Record Updated Successfully',
                            'Success'
                        );
                    } else {
                        screenParameters?.serviceList?.toastrService.showError(
                            res.Errors[0],
                            'Error'
                        );
                    }
                }
            );
    }

    enableDisableComponent(screenParameters: any, key: string, type: string, disabled: boolean) {
        FormioUtils.eachComponent(
            screenParameters?.formIOObject?.formio?.component?.components,
            (component) => {
                if (component.key == key && component.type == type) {
                    component.disabled = disabled;
                }
            });
    }

    // Refresh the form
    refreshForm(screenParameters, data) {
        screenParameters?.formIOObject.refresh.emit({
            submission: {
                data: data,
            },
        });
    }
}
