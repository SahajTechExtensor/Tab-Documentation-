    constructor() {}

    screenId = '58e6d868-e01b-4ccc-bcd4-0680d748b485';

    onBeforeSubmit(screenParameters: any) {
        screenParameters?.event?.callback(null, screenParameters?.event);

    onChange(screenParameters: any) {
        if (
            screenParameters?.event?.changed?.component?.type == 'select' &&
            screenParameters?.event?.changed?.component?.key == 'select'
        ) {
            if (screenParameters?.event?.changed?.value == 'markInActive') {
                let udata = {
                    data: {
                        Id: screenParameters?.event?.data?.Id,
                        IsActive: false,
                    },
                };
                screenParameters?.serviceList?.updateService
                    .updateUsingScreenId(this.screenId, udata)
                    ?.pipe(take(1))
                    ?.subscribe((res) => {
                        if (res?.StatusCode == '200') {
                            screenParameters?.serviceList?.toastrService.showSuccess(
                                'This record is inactive now',
                                'Success'
                            );
                            // this._location.back();
                        } else {
                            screenParameters?.serviceList?.toastrService.showError(
                                res.Errors[0],
                                'Error'
                            );
                        }
                    });
            } else if (
                screenParameters?.event?.changed?.value == 'markActive'
            ) {
                let udata = {
                    data: {
                        Id: screenParameters?.event?.data?.Id,
                        IsActive: true,
                    },
                };
                screenParameters?.serviceList?.updateService
                    .updateUsingScreenId(this.screenId, udata)
                    ?.pipe(take(1))
                    ?.subscribe((res) => {
                        if (res?.StatusCode == '200') {
                            screenParameters?.serviceList?.toastrService.showSuccess(
                                'This record is active now',
                                'Success'
                            );
                            // this._location.back();
                        } else {
                            screenParameters?.serviceList?.toastrService.showError(
                                res.Errors[0],
                                'Error'
                            );
                        }
                    });
            } else if (
                screenParameters?.event?.changed?.value == 'markNotSystemRecord'
            ) {
                let udata = {
                    data: {
                        Id: screenParameters?.event?.data?.Id,
                        IsSystemRecord: false,
                    },
                };
                screenParameters?.serviceList?.updateService
                    .updateUsingScreenId(this.screenId, udata)
                    ?.pipe(take(1))
                    ?.subscribe((res) => {
                        if (res?.StatusCode == '200') {
                            screenParameters?.serviceList?.toastrService.showSuccess(
                                'This record is not system record now',
                                'Success'
                            );
                            // this._location.back();
                        } else {
                            screenParameters?.serviceList?.toastrService.showError(
                                res.Errors[0],
                                'Error'
                            );
                        }
                    });
            } else if (
                screenParameters?.event?.changed?.value == 'markSystemRecord'
            ) {
                let udata = {
                    data: {
                        Id: screenParameters?.event?.data?.Id,
                        IsSystemRecord: true,
                    },
                };
                screenParameters?.serviceList?.updateService
                    .updateUsingScreenId(this.screenId, udata)
                    ?.pipe(take(1))
                    ?.subscribe((res) => {
                        if (res?.StatusCode == '200') {
                            screenParameters?.serviceList?.toastrService.showSuccess(
                                'This record is system record now',
                                'Success'
                            );
                            // this._location.back();
                        } else {
                            screenParameters?.serviceList?.toastrService.showError(
                                res.Errors[0],
                                'Error'
                            );
                        }
                    });
            }
        }

        if (
            screenParameters?.event?.changed?.component?.type == 'select' &&
            screenParameters?.event?.changed?.component?.key == 'status' &&
            screenParameters?.event?.isModified
        ) {
            const status =
                screenParameters?.event?.changed?.component.data.values.find(
                    (s) =>
                        s.StatusId?.toLowerCase() ==
                        screenParameters?.event.changed.value?.toLowerCase()
                );
            console.log('updated status', status);
            if (status) {
                this.performTransactionAction(screenParameters, status);
            }
        }

        if (
            screenParameters?.event?.changed?.component?.type ==
                'customChips' &&
            screenParameters?.event?.changed?.component?.key == 'Tags'
        ) {
            let utags = screenParameters?.event?.changed?.value;
            let tagValueArray = [];
            if (
                typeof utags == 'object' &&
                utags &&
                utags?.selectedChips.length
            ) {
                utags?.selectedChips.forEach((element) => {
                    tagValueArray.push(element.value);
                });
                let udata = {
                    data: {
                        Id: screenParameters?.event?.data?.Id,
                        Tags: tagValueArray.join(),
                    },
                };
                screenParameters?.serviceList?.updateService
                    .updateUsingScreenId(this.screenId, udata)
                    ?.pipe(take(1))
                    ?.subscribe((res) => {
                        if (res?.StatusCode == '200') {
                            screenParameters?.serviceList?.toastrService.showSuccess(
                                'Tags Updated successfully',
                                'Success'
                            );
                            // this._location.back();
                        } else {
                            screenParameters?.serviceList?.toastrService.showError(
                                res.Errors[0],
                                'Error'
                            );
                        }
                    });
            }
        }
    }

    proceedObjectBeforeRender(screenParameters: any) {
        // this.handleCustomJsEvents(screenParameters);
        let displayName;
        let primaryKeyName = '';

        if (screenParameters?.event?.RecordInfo) {
            screenParameters.event.RecordInfo['isRecordExist'] = true;
        }

        // if RecordInfo doesn't exist then it's add screen
        if (screenParameters?.event?.RecordInfo?.isRecordExist) {
            let appObject = TabAppFunction.getAppObjectFromDataSourceQueryID(
                screenParameters?.event?.screen?.DataSourceQuery
            );
            if (appObject) {
                primaryKeyName = appObject.Fields.find(
                    (f) => f.IsPrimaryKey
                ).FieldName;
                displayName = appObject.Fields.filter(
                    (field) => field.IsDisplayField
                ).map((field) => field.FieldName);
            }
        }

        if (
            screenParameters.serviceList.route.snapshot.queryParams[
                primaryKeyName
            ] === screenParameters?.event?.RecordInfo?.PrimaryKey &&
            screenParameters.event.RecordInfo
        ) {
            screenParameters.event.RecordInfo['mainScreenRecordInfo'] =
                cloneDeep(screenParameters?.event?.RecordInfo);
            screenParameters.event.RecordInfo['mainScreen'] = cloneDeep(
                screenParameters?.event?.screen
            );

            FormioUtils.eachComponent(
                screenParameters?.formIOObject?.formio?.component?.components,
                function (component: any, path: any) {
                    // action menu data pass for pms workitem id custom code by hardik
                    if (
                        component.type == 'customWidget' &&
                        component.key == 'actionMenusPMS'
                    ) {
                        component.customOptions.dataSetting[
                            'Templatevariables'
                        ] = JSON.stringify(
                            screenParameters.event.RecordInfo.parentScreenData
                        );
                    }
                    //...................

                    if (
                        component.type == 'customWidget' &&
                        component.key == 'layoutdetails'
                    ) {
                        let val = {
                            Id: screenParameters?.event?.RecordInfo?.Id,
                        };
                        component.customOptions.dataSetting.parameters = val;
                    }

                    if (
                        component.type == 'customWidget' &&
                        (component.key == 'basicDetails' ||
                            component.key == 'basicDetails3' ||
                            component.key == 'actionMenus')
                    ) {
                        let val = {
                            RecordId:
                                screenParameters.event.RecordInfo.RecordId,
                        };
                        component.customOptions.dataSetting.parameters = val;
                    }

                    if (
                        component.type == 'customWidget' &&
                        component.key == 'name' &&
                        screenParameters?.event?.RecordInfo?.isRecordExist
                    ) {
                        component.customOptions.dataSetting['dataSourceMode'] =
                            'data';
                        component.customOptions.dataSetting[
                            'Templatevariables'
                        ] = JSON.stringify({
                            displayValue:
                                screenParameters.event.RecordInfo
                                    .parentScreenData[displayName[0]],
                        });
                    }
                }
            );

            FormioUtils.eachComponent(
                screenParameters?.formIOObject?.formio?.component?.components,
                function (component: any, path: any) {
                    if (
                        component.type == 'select' &&
                        component.key == 'select'
                    ) {
                        let dropdownOptionArray = [];
                        if (screenParameters?.event?.RecordInfo) {
                            if (
                                screenParameters?.event?.RecordInfo?.IsActive ==
                                true
                            ) {
                                dropdownOptionArray.push({
                                    label: 'Mark As Inactive',
                                    value: 'markInActive',
                                });
                            } else {
                                dropdownOptionArray.push({
                                    label: 'Mark As Active',
                                    value: 'markActive',
                                });
                            }

                            if (
                                screenParameters?.event?.RecordInfo
                                    ?.IsSystemRecord == true
                            ) {
                                dropdownOptionArray.push({
                                    label: 'Mark As Not System Record',
                                    value: 'markNotSystemRecord',
                                });
                            } else {
                                dropdownOptionArray.push({
                                    label: 'Mark As System Record',
                                    value: 'markSystemRecord',
                                });
                            }
                        }
                        if (
                            screenParameters?.event?.screen?.ScreenConfiguration
                        ) {
                            let scf = JSON.parse(
                                screenParameters?.event?.screen
                                    ?.ScreenConfiguration
                            );
                            if (scf && scf?.screenActionList?.length) {
                                for (let sf of scf?.screenActionList) {
                                    dropdownOptionArray.push({
                                        label: sf.options,
                                        value: sf.actionEvent,
                                    });
                                }
                            }
                        }
                        component.data.values = dropdownOptionArray;
                    }
                }
            );

            FormioUtils.eachComponent(
                screenParameters?.formIOObject?.formio?.component?.components,
                // screenParameters?.formIOObject.formio.components,
                (component: any, path: any) => {
                    if (
                        component.type == 'select' &&
                        component.key == 'status'
                    ) {
                        if (
                            screenParameters?.event?.RecordInfo &&
                            screenParameters?.event?.RecordInfo?.BlueprintId
                        ) {
                            this.loadNextStatuses(
                                screenParameters,
                                screenParameters?.event?.RecordInfo?.RecordId,
                                component,
                                true
                            );
                        }
                    }
                }
            );
        }

        if (!screenParameters.event.RecordInfo) {
            FormioUtils.eachComponent(
                screenParameters.formJson.components,
                function (component: any, path: any) {
                    if (
                        (component.type === 'customWidget' &&
                            (component.key === 'dropdownMenu3' ||
                                component.key === 'basicDetails')) ||
                        (component.type === 'button' &&
                            (component.key === 'share3' ||
                                component.key === 'history3' ||
                                component.key === 'comments3')) ||
                        (component.type === 'customChips' &&
                            component.key === 'Tags3')
                    ) {
                        component.hidden = true;
                        component.customClass = 'd-none remove-default-margin';
                    }

                    if (
                        component.type === 'button' &&
                        component.key === 'close3'
                    ) {
                        component.customClass =
                            'common-round-btn remove-default-margin';
                    }
                }
            );
        }
    }

    onCustomEvent(screenParameters) {
        console.log('onCustomEvent[screenParameters]', screenParameters);

        let sid = '';
        let data = { parameters: {} };
        let title = '';
        if (screenParameters.event.type == 'shareEvent') {
            sid = '001b0c09-4429-4c0e-a9e2-97917731e378';
            data.parameters['RecordID'] = screenParameters.event.data.Id;
            data.parameters['screenMode'] = 'Add';
            title = 'Share Record';
        } else if (screenParameters.event.type == 'historyClickEvent') {
            sid = '48f8e8a0-85cb-43c9-9928-1afa05e57ce8';
            // OLD CODE
            // data.parameters['AppObjectID'] =
            //     screenParameters.event.data.AppObjectID;

            title = 'History of Record';
            data.parameters['history'] = {
                DataSourceQueryID:
                    screenParameters.event.data.DataSourceQueryID,
                PrimaryKey: screenParameters.event.data.PrimaryKey,

                // FOR REFERENCE
                // DataSourceQueryID: '09835967-5A99-46B9-B0D9-8AA6E2037222',
                // PrimaryKey: '0d6d384f-bc4c-4030-8454-974eac418378',
            };

            data.parameters['screenMode'] = 'Add';
        } else if (screenParameters.event.type == 'commentsClickEvent') {
            sid = '26cc7710-6e93-4aaa-aca8-2f6a8608435e';
            title = 'Comments';
            data.parameters['comment'] = {
                RecordId: screenParameters.event.data?.RecordId,
            };
            data.parameters['screenMode'] = 'Add';
        }
        const target = new ElementRef(screenParameters.event.event.target);
        const rect = target.nativeElement.getBoundingClientRect();
        this.ref = screenParameters.dailogRef.open(TabDynamicDialogComponent, {
            height: '500px',
            width: '600px',
            panelClass: 'custom-modalbox',
            position: {
                left: `${rect.left - 550}px`,
                top: `${rect.top + 50}px`,
                right: `${rect.right}px`,
                bottom: `${rect.bottom}px`,
            },
            data: {
                screenID: sid,
                data: JSON.stringify(data),
                title: title,
            },
        });
    }

        screenParameters.serviceList.appService.tabCoreHttpService.httpClient
            .get(
                TabConstant.API.endpoint +
                    'Transition/GetNextStatus?RecordId=' +
                    recordId,

                {
                    headers: {
                        ApplicationCode: sessionStorage.getItem(
                            TabConstant.Storage.applicationCode
                        ),
                        Authorization: sessionStorage.getItem(
                            TabConstant.Storage.token
                        ),
                    },
                }
            )
            ?.pipe(take(1))
            .subscribe((res) => {
                const currentStatus = res.Result;
                const statuses: any = [];
                statuses.push({
                    Name: currentStatus.FromStatusName,
                    StatusId: currentStatus.FromStatusId.toLowerCase(),
                });

                currentStatus.ToStatusList?.forEach((status) => {
                    statuses.push({
                        Name: status.ToStatusName,
                        StatusId: status.ToStatusId.toLowerCase(),
                        TransitionHooks_Before: status.TransitionHooks_Before,
                        TransitionHooks_During: status.TransitionHooks_During,
                        TransitionHooks_After: status.TransitionHooks_After,
                    });
                });
                component.defaultValue =
                    currentStatus.FromStatusId.toLowerCase();
                component.data.values = statuses;
                const data: any = {
                    ...screenParameters?.formIOObject.formio.data,
                };
                data[component.key] = currentStatus.FromStatusId.toLowerCase();
                if (hardRefresh) {
                    const detailLayoutComponent = FormioUtils.getComponent(
                        screenParameters?.formIOObject?.formio?.components,
                        'DetailLayout',
                        true
                    );
                    const screenRender = FormioUtils.getComponent(
                        screenParameters?.formIOObject?.formio?.components,
                        'screenRender',
                        true
                    );
                    if (detailLayoutComponent) {
                        const columnComponent = FormioUtils.getComponent(
                            detailLayoutComponent.components,
                            'columns8',
                            true
                        );
                        if (columnComponent) {
                            const statusComponent = FormioUtils.getComponent(
                                columnComponent.components,
                                'status',
                                true
                            );
                            if (statusComponent)
                                statusComponent.triggerRedraw();
                        }
                    }
                    if (
                        screenRender &&
                        screenParameters?.event?.changed?.component?.type ==
                            'select' &&
                        screenParameters?.event?.changed?.component?.key ==
                            'status' &&
                        screenParameters?.event?.isModified
                    ) {
                        window.location.reload();
                    }
                } else {
                    this.refreshForm(screenParameters, data);
                }
            });
    };

                        ),
                        Authorization: sessionStorage.getItem(
                            TabConstant.Storage.token
                        ),
                    },
                }
            )
            ?.pipe(take(1))
            .subscribe((res) => {
                if (res?.StatusCode == '200') {
                    // Loading next statuses
                    FormioUtils.eachComponent(
                        screenParameters?.formIOObject?.formio?.component
                            ?.components,
                        (component: any, path: any) => {
                            if (
                                component.type == 'select' &&
                                component.key == 'status'
                            ) {
                                this.loadNextStatuses(
                                    screenParameters,
                                    screenParameters?.event?.data
                                        ?.mainScreenRecordInfo?.RecordId,
                                    component,
                                    true
                                );
                            }
                        }
                    );
                    screenParameters?.serviceList?.toastrService.showSuccess(
                        'Status Updated successfully',
                        'Success'
                    );
                } else {
                    screenParameters?.serviceList?.toastrService.showError(
                        res.Errors[0],
                        'Error'
                    );
                }
            });
    };
