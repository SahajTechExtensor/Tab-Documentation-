    insertDSQFields(event: any, insertService, dsqid, screenParameters) {
        event?.dataSource?.forEach((element) => {
            if (element.isChecked === 'true' || element.isChecked === true) {
                let data: any = {
                    DataSourceQueryID: dsqid,
                    AppFieldID: this.getAppFieldId(element.id),
                    FieldType: '1',
                };
                if (element.parentID) {
                    data['LookUpDetail'] = element.lookUPText;
                    data['FieldType'] = '2';
                }
                if (element && element.hasOwnProperty('isChildrenField')) {
                    if (
                        element.dsqForChildRelationship !== null &&
                        element.dsqForChildRelationship !== undefined
                    ) {
                        data['LookUpDetail'] =
                            element.name +
                            ':' +
                            element.dsqForChildRelationship;
                    } else {
                        data['LookUpDetail'] = element.lookUPText;
                    }

                    data['FieldType'] = '6';
                }
                insertService
                    .insertByTable('TABMD_DataSourceQueries_Fields', data)
                    ?.pipe(take(1))
                    ?.subscribe((response) => {
                        if (
                            response.StatusCode == '200' ||
                            !response.Errors.length
                        ) {
                            if (
                                element.hasChildren === 'true' ||
                                element.hasChildren === true
                            ) {
                                const childList =
                                    event.fields?.dataSource?.filter(
                                        (x) => x.parentID === element.id
                                    );
                                if (childList && childList.length > 0) {
                                    const childListDeleteSet = new Set(
                                        childList
                                    );

                                    event.fields.dataSource =
                                        event.fields.dataSource.filter(
                                            (ele) => {
                                                // return those elements not in the childListDeleteSet
                                                return !childListDeleteSet.has(
                                                    ele
                                                );
                                            }
                                        );
                                }
                            }
                            console.log('DSQ fields saved');
                        } else {
                            console.log('Error while saving DSQ fields');
                        }
                    });
            }
        });
        screenParameters?.serviceList?.toastrService.showSuccess(
            'Records Updated successfully',
            'Success'
        );
        // this.rebuildApp();
    }

    getAppFieldId(id: string): string {
        let returnId = '';

        if (id.includes('___')) {
            var splittedArray = id.split('___');

            if (splittedArray && splittedArray.length > 1) {
                returnId = splittedArray[splittedArray.length - 1].toString();
            }
        } else {
            returnId = id;
        }

        return returnId;
    }

    bindChildRelationshipToTreeView(appObjectData, pid, prefix) {
        if (
            appObjectData &&
            appObjectData.hasOwnProperty('ChildRelationShips') &&
            appObjectData.ChildRelationShips &&
            appObjectData.ChildRelationShips.length > 0
        ) {
            appObjectData.ChildRelationShips.forEach((chr) => {
                let childAppObjectDsqArray = [];
                if (
                    this.appObjectData?.SystemDBTableName !=
                    chr?.childDetails?.LookupObject
                ) {
                    if (
                        chr &&
                        chr.childDetails &&
                        chr.childDetails.LookupObject
                    ) {
                        let childObjectData =
                            TabAppFunction.getAppObjectFromSystemDBTableName(
                                chr.childDetails.LookupObject
                            );
                        if (
                            childObjectData &&
                            childObjectData.DataSourceQueries
                        ) {
                            childObjectData.DataSourceQueries.forEach(
                                (cdsq) => {
                                    childAppObjectDsqArray.push({
                                        ID: cdsq.ID,
                                        QueryName: cdsq.QueryName,
                                    });
                                }
                            );
                        }
                    }
                    let id: any = Guid.create();
                    let fieldData = {
                        id: id.value,
                        name: '#' + chr.childDetails.LookupObject,
                        childAppObjectDsqArray: childAppObjectDsqArray,
                        expanded: false,
                        hasChildren: false,
                        parentID: pid !== '' ? pid : undefined,
                        lookUPText: prefix
                            ? prefix + '.' + '#' + chr.childDetails.LookupObject
                            : '#' + chr.childDetails.LookupObject,
                        isChildrenField: true,
                    };
                    this.fieldData.push(fieldData);
                }
            });
        }
    }

    private bindAppObjectFieldsToTreeView(prefix, item, pid) {
        let fieldData = {
            id: prefix !== '' ? pid + '___' + item.ID : item.ID, // Create Unique ID
            name: item.SystemDBFieldName,
            expanded: false,
            hasChildren: false,
            parentID: pid !== '' ? pid : undefined,
            lookUPText: prefix
                ? prefix + '.' + item.SystemDBFieldName
                : item.SystemDBFieldName,
        };
        this.fieldData.push(fieldData);
        if (item.hasOwnProperty('LookUpDetails') && item.LookUpDetails) {
            fieldData.hasChildren = true;
            let lookupAppObjectData =
                TabAppFunction.getAppObjectFromSystemDBTableName(
                    item.LookUpDetails.LookupObject
                );
            if (lookupAppObjectData) {
                if (Object.keys(this.previousLookup).length > 0) {
                    if (
                        this.previousLookup.LookupObject ==
                            item.LookUpDetails.LookupObject &&
                        !item.LookUpDetails.LookupObject.toLowerCase().includes(
                            'tabd_recordinfo'
                        )
                    ) {
                        return;
                    }
                }
                this.previousLookup = item.LookUpDetails;
                let nprefix = fieldData.lookUPText;
                if (lookupAppObjectData?.Fields) {
                    each(lookupAppObjectData.Fields, (ff) => {
                        this.bindAppObjectFieldsToTreeView(
                            nprefix,
                            ff,
                            fieldData.id
                        );
                    });
                    this.bindChildRelationshipToTreeView(
                        lookupAppObjectData,
                        fieldData.id,
                        nprefix
                    );
                }
            }
        }
    }

    getDSQFieldsFromAPI(dataSourceQueryID, screenParameters?) {
        const apiData = {
            AppObjectName: 'TABMD_DataSourceQueries_Fields',
            DSQName: 'Default_TABMD_DataSourceQueries_Fields',
            Reqtokens: { DataSourceQueryID: dataSourceQueryID },
        };
        screenParameters.serviceList.getService
            .executeRAWDSQWithName(apiData)
            ?.pipe(take(1))
            ?.subscribe((res: any) => {
                if (res?.Result && res?.Result.length) {
                    // this below code can be optimized like make 3 array type wise and find in field data
                    // so it can be easy to manage now hardik is making for type 6 (childrelfield)
                    let result = res?.Result;
                    const childRelArray = result.filter(
                        (object) => object?.FieldType == 6
                    );
                    childRelArray.forEach((element) => {
                        if (
                            element?.LookUpDetail &&
                            element?.LookUpDetail.includes(':')
                        ) {
                            let devidedString = element.LookUpDetail.split(':');
                            element.LookUpDetail = devidedString[0];
                            element['dsqForChildRelationship'] =
                                devidedString[1];
                        }
                    });
                    this.fieldData.forEach((dsqField: any) => {
                        let field: any = {};
                        if (
                            dsqField.lookUPText &&
                            dsqField.lookUPText.includes('.')
                        ) {
                            field = result.find(
                                (a: any) =>
                                    a.LookUpDetail &&
                                    a.LookUpDetail.toLowerCase() ==
                                        dsqField.lookUPText.toLowerCase()
                            );
                        } else {
                            field = result.find(
                                (a: any) =>
                                    a.AppFieldID.toLowerCase() ==
                                    dsqField.id.toLowerCase()
                            );
                        }

                        if (field) {
                            dsqField.isChecked = true;
                            dsqField.dsqFieldID = field.ID;
                            this.checkedNodes.push(dsqField.id);
                        }
                        // manging for child-relationship keys
                        if (
                            dsqField.lookUPText &&
                            dsqField.lookUPText.includes('#')
                        ) {
                            let ff = childRelArray.find(
                                (a: any) =>
                                    a.LookUpDetail &&
                                    a.LookUpDetail.toLowerCase() ==
                                        dsqField.lookUPText.toLowerCase()
                            );
                            if (ff) {
                                dsqField.isChecked = true;
                                dsqField.dsqFieldID = ff.ID;
                                if (
                                    ff.hasOwnProperty('dsqForChildRelationship')
                                ) {
                                    dsqField.dsqForChildRelationship =
                                        ff.dsqForChildRelationship;
                                    let fff =
                                        dsqField?.childAppObjectDsqArray.find(
                                            (a: any) =>
                                                a.QueryName &&
                                                a.QueryName.toLowerCase() ==
                                                    ff?.dsqForChildRelationship.toLowerCase()
                                        );
                                    if (fff) {
                                        dsqField['dsqIDForChildRelationship'] =

    getDSQFilters(dsqId, screenParameters) {
        const reqtokens = { ID: dsqId };
        const apiData = {
            Reqtokens: reqtokens,
            AppObjectName: 'TABMD_DataSourceQueries_Filters',
            DSQName:
                'FK_TABMD_DataSourceQueries_Filters_TABMD_DataSourceQueries',
        };
        screenParameters.serviceList.getService
            .executeRAWDSQWithName(apiData)
            ?.pipe(take(1))
            .subscribe((response) => {
                if (!!response.Result && response.Result.length > 0) {
                    this.dsqFilterList = response.Result;
                }
                // setTimeout(() => {
                screenParameters.event.newDsqFilters = {
                    filterData: this.dsqFilterList,
                    isRefresh: true,
                    dataSourceQueryId: screenParameters.event.ID,
                };
                // }, 100);
                screenParameters?.formIOObject.refresh.emit({
                    submission: {
                        data: screenParameters?.event,
                    },
                });
            });
    }
