import { IScreenExtension } from '@techextensor/tab-interface';
import { TabAppFunction } from '@techextensor/tab-service';
import { each, filter } from 'lodash';
import { FormioUtils } from '@formio/angular';
import { take } from 'rxjs';

export class DSQSortingDetailScreen implements IScreenExtension {
    constructor() {}

    screenId = '120e4ab1-bfa2-4f51-982c-f30df083aafd';

    fieldData = [];
    previousLookup: any = {};
    checkedNodes: any = [];

    onSubmit(event) {}

    onRender(screenParameters: any) {}

    onBeforeSubmit(screenParameters: any) {
        if (screenParameters?.event?.data?.SortingAppFieldID) {
            // screenParameters.event.data['AppFieldID'].Id =
            //     screenParameters?.event?.data?.SortingAppFieldID;
            screenParameters.event.data['FieldType'] = '1';
            screenParameters.event.data['LookUpDetail'] =
                screenParameters?.event?.data?.SortingAppFieldID?.lookUPText;
            if (
                screenParameters?.event?.data?.SortingAppFieldID?.lookUPText &&
                screenParameters?.event?.data?.SortingAppFieldID?.lookUPText.indexOf(
                    '.'
                ) > -1
            ) {
                screenParameters.event.data['FieldType'] = '2';
            }
            screenParameters.event.data['AppFieldID'] = this.getAppFieldId(
                screenParameters?.event?.data?.SortingAppFieldID?.id
            );
        }
        screenParameters?.event?.callback(null, screenParameters?.event);
    }

    onChange(screenParameters: any) {}

    onInvalid(event) {}

    onNextPage(event) {}

    onPrevPage(event) {}

    onErrorChange(event) {}

    onFormLoad(screenParameters: any) {}

    onReady(screenParameters: any) {}

    onAfterSubmit(screenParameters: any) {}

    proceedObjectBeforeRender(screenParameters: any) {
        console.log(screenParameters);
        var that = this;
        FormioUtils.eachComponent(
            screenParameters?.formJson?.components,
            function (component: any, path: any) {
                if (
                    component.type == 'customDropdownTree' &&
                    component.key == 'SortingAppFieldID'
                ) {
                    that.checkForDropdownComponent(screenParameters, component);
                    component.customOptions.controlSetting['jsonData'] =
                        that.fieldData;
                }
            }
        );
    }

    async checkForDropdownComponent(screenParameters, component) {
        if (screenParameters?.event?.isEdit) {
            // await this.getAppObjectFields(
            //     screenParameters?.event?.DataSourceQueryID?.AppObject?.ID
            // );
            let appObjectDetails =
                TabAppFunction.getAppObjectFromDataSourceQueryID(
                    screenParameters?.event?.DataSourceQueryID?.ID
                );
            await this.getAppObjectFields(appObjectDetails?.ID);

            let fd = this.fieldData.find(
                (a: any) =>
                    this.getAppFieldId(a.id)?.toLowerCase() ===
                    // screenParameters?.event?.AppFieldID?.Id?.toLowerCase()
                    screenParameters?.event?.AppFieldID
            );
            if (fd) {
                fd.isSelected = true;
            }
            component.customOptions.controlSetting['jsonData'] = this.fieldData;
            screenParameters.event.SortingAppFieldID = {
                isRefresh: true,
                value: fd ? [fd.id] : [],
                fieldData: this.fieldData,
            };
        } else {
            await this.getAppObjectFields(
                screenParameters?.event?.parameters?.screenContext
                    ?.TABMD_DataSourceQueries?.activeRowsData?.AppObject
            );
            component.customOptions.controlSetting['jsonData'] = this.fieldData;
            screenParameters.event.parameters.SortingAppFieldID = {
                isRefresh: true,
                fieldData: this.fieldData,
            };
        }
    }

    onCustomEvent() {}

    private getAppObjectFields(appObjectId) {
        this.fieldData = [];
        let appObjectData = TabAppFunction.getAppObject(appObjectId);
        if (appObjectData?.Fields) {
            each(appObjectData.Fields, (f) => {
                this.bindAppObjectFieldsToTreeView('', f, '');
            });
        }
    }

    private bindAppObjectFieldsToTreeView(prefix, item, pid) {
        let fieldData = {
            id: prefix !== '' ? pid + '___' + item.ID : item.ID,
            name: item.SystemDBFieldName,
            lookUPText: prefix
                ? prefix + '.' + item.SystemDBFieldName
                : item.SystemDBFieldName,
        };
        if (pid) {
            fieldData['pid'] = pid;
        }
        this.fieldData.push(fieldData);
        if (item.hasOwnProperty('LookUpDetails') && item.LookUpDetails) {
            fieldData['hasChild'] = true;
            let lookupAppObjectData =
                TabAppFunction.getAppObjectFromSystemDBTableName(
                    item.LookUpDetails.LookupObject
                );
            if (lookupAppObjectData) {
                if (Object.keys(this.previousLookup).length > 0) {
                    if (
                        this.previousLookup.LookupObject ==
                            item.LookUpDetails.LookupObject &&
                        !item.LookUpDetails.LookupObject.toLowerCase().includes(
                            'tabd_recordinfo'
                        )
                    ) {
                        return;
                    }
                }
                this.previousLookup = item.LookUpDetails;
                let nprefix = fieldData.lookUPText;
                if (lookupAppObjectData?.Fields) {
                    each(lookupAppObjectData.Fields, (ff) => {
                        this.bindAppObjectFieldsToTreeView(
                            nprefix,
                            ff,
                            item.ID
                        );
                    });
                }
            }
        }
    }

    getSelectedFieldList(dataSourceQueryID, screenParameters?) {
        const apiData = {
            AppObjectName: 'TABMD_DataSourceQueries_Sort',
            DSQName: 'Default_TABMD_DataSourceQueries_Sort',
            Reqtokens: { DataSourceQueryID: dataSourceQueryID },
        };
        screenParameters.serviceList.getService
            .executeRAWDSQWithName(apiData)
            ?.pipe(take(1))
            ?.subscribe((res: any) => {
                if (res?.Result && res?.Result.length) {
                    let result = res?.Result;
                    this.fieldData.forEach((dsqField: any) => {
                        let d = result.find(
                            (a: any) =>
                                a.AppFieldID.toLowerCase() ==
                                dsqField.id.toLowerCase()
                        );
                        if (d) {
                            dsqField.isChecked = true;
                            dsqField.isSelected = true;
                            dsqField.dsqFieldID = d.ID;
                            this.checkedNodes.push(dsqField.id);
                        }
                    });
                    // screenParameters?.formIOObject.refresh.emit({
                    //     submission: {
                    //         data: screenParameters?.event,
                    //     },
                    // });
                }
            });
    }

    getAppFieldId(id: string): string {
        let returnId = '';

        if (id.includes('___')) {
            var splittedArray = id.split('___');

            if (splittedArray && splittedArray.length > 1) {
                returnId = splittedArray[splittedArray.length - 1].toString();
            }
        } else {
            returnId = id;
        }

        return returnId;
    }
}
